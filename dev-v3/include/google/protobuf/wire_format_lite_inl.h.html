226template <typename CType, enum WireFormatLite::FieldType DeclaredType>
227inline bool WireFormatLite::ReadRepeatedPrimitive(
228 int, // tag_size, unused.
229 uint32 tag,
230 io::CodedInputStream* input,
231 RepeatedField<CType>* values) {
232 CType value;
233 if (!ReadPrimitive<CType, DeclaredType>(input, &value)) return false;
234 values->Add(value);
235 int elements_already_reserved = values->Capacity() - values->size();
236 while (elements_already_reserved > 0 && input->ExpectTag(tag)) {
237 if (!ReadPrimitive<CType, DeclaredType>(input, &value)) return false;
238 values->AddAlreadyReserved(value);
239 elements_already_reserved--;
240 }
241 return true;
242}

244template <typename CType, enum WireFormatLite::FieldType DeclaredType>
245inline bool WireFormatLite::ReadRepeatedFixedSizePrimitive(
246 int tag_size,
247 uint32 tag,
248 io::CodedInputStream* input,
249 RepeatedField<CType>* values) {
250 GOOGLE_DCHECK_EQ(UInt32Size(tag), static_cast<size_t>(tag_size));
251 CType value;
252 if (!ReadPrimitive<CType, DeclaredType>(input, &value))
253 return false;
254 values->Add(value);
255
256 // For fixed size values, repeated values can be read more quickly by
257 // reading directly from a raw array.
258 //
259 // We can get a tight loop by only reading as many elements as can be
260 // added to the RepeatedField without having to do any resizing. Additionally,
261 // we only try to read as many elements as are available from the current
262 // buffer space. Doing so avoids having to perform boundary checks when
263 // reading the value: the maximum number of elements that can be read is
264 // known outside of the loop.
265 const void* void_pointer;
266 int size;
267 input->GetDirectBufferPointerInline(&void_pointer, &size);
268 if (size > 0) {
269 const uint8* buffer = reinterpret_cast<const uint8*>(void_pointer);
270 // The number of bytes each type occupies on the wire.
271 const int per_value_size = tag_size + static_cast<int>(sizeof(value));
272
273 // parentheses around (std::min) prevents macro expansion of min(...)
274 int elements_available =
275 (std::min)(values->Capacity() - values->size(), size / per_value_size);
276 int num_read = 0;
277 while (num_read < elements_available &&
278 (buffer = io::CodedInputStream::ExpectTagFromArray(
279 buffer, tag)) != NULL) {
280 buffer = ReadPrimitiveFromArray<CType, DeclaredType>(buffer, &value);
281 values->AddAlreadyReserved(value);
282 ++num_read;
283 }
284 const int read_bytes = num_read * per_value_size;
285 if (read_bytes > 0) {
286 input->Skip(read_bytes);
287 }
288 }
289 return true;
290}

326template <typename CType, enum WireFormatLite::FieldType DeclaredType>
327inline bool WireFormatLite::ReadPackedPrimitive(io::CodedInputStream* input,
328 RepeatedField<CType>* values) {
329 int length;
330 if (!input->ReadVarintSizeAsInt(&length)) return false;
331 io::CodedInputStream::Limit limit = input->PushLimit(length);
332 while (input->BytesUntilLimit() > 0) {
333 CType value;
334 if (!ReadPrimitive<CType, DeclaredType>(input, &value)) return false;
335 values->Add(value);
336 }
337 input->PopLimit(limit);
338 return true;
339}

341template <typename CType, enum WireFormatLite::FieldType DeclaredType>
342inline bool WireFormatLite::ReadPackedFixedSizePrimitive(
343 io::CodedInputStream* input, RepeatedField<CType>* values) {
344 int length;
345 if (!input->ReadVarintSizeAsInt(&length)) return false;
346 const int old_entries = values->size();
347 const int new_entries = length / static_cast<int>(sizeof(CType));
348 const int new_bytes = new_entries * static_cast<int>(sizeof(CType));
349 if (new_bytes != length) return false;
350 // We would *like* to pre-allocate the buffer to write into (for
351 // speed), but *must* avoid performing a very large allocation due
352 // to a malicious user-supplied "length" above. So we have a fast
353 // path that pre-allocates when the "length" is less than a bound.
354 // We determine the bound by calling BytesUntilTotalBytesLimit() and
355 // BytesUntilLimit(). These return -1 to mean "no limit set".
356 // There are four cases:
357 // TotalBytesLimit Limit
358 // -1 -1 Use slow path.
359 // -1 >= 0 Use fast path if length <= Limit.
360 // >= 0 -1 Use slow path.
361 // >= 0 >= 0 Use fast path if length <= min(both limits).
362 int64 bytes_limit = input->BytesUntilTotalBytesLimit();
363 if (bytes_limit == -1) {
364 bytes_limit = input->BytesUntilLimit();
365 } else {
366 // parentheses around (std::min) prevents macro expansion of min(...)
367 bytes_limit =
368 (std::min)(bytes_limit, static_cast<int64>(input->BytesUntilLimit()));
369 }
370 if (bytes_limit >= new_bytes) {
371 // Fast-path that pre-allocates *values to the final size.
372#if defined(PROTOBUF_LITTLE_ENDIAN)
373 values->Resize(old_entries + new_entries, 0);
374 // values->mutable_data() may change after Resize(), so do this after:
375 void* dest = reinterpret_cast<void*>(values->mutable_data() + old_entries);
376 if (!input->ReadRaw(dest, new_bytes)) {
377 values->Truncate(old_entries);
378 return false;
379 }
380#else
381 values->Reserve(old_entries + new_entries);
382 CType value;
383 for (int i = 0; i < new_entries; ++i) {
384 if (!ReadPrimitive<CType, DeclaredType>(input, &value)) return false;
385 values->AddAlreadyReserved(value);
386 }
387#endif
388 } else {
389 // This is the slow-path case where "length" may be too large to
390 // safely allocate. We read as much as we can into *values
391 // without pre-allocating "length" bytes.
392 CType value;
393 for (int i = 0; i < new_entries; ++i) {
394 if (!ReadPrimitive<CType, DeclaredType>(input, &value)) return false;
395 values->Add(value);
396 }
397 }
398 return true;
399}

429template<typename MessageType>
430inline bool WireFormatLite::ReadGroup(
431 int field_number, io::CodedInputStream* input,
432 MessageType* value) {
433 if (!input->IncrementRecursionDepth()) return false;
434 if (!value->MergePartialFromCodedStream(input)) return false;
435 input->UnsafeDecrementRecursionDepth();
436 // Make sure the last thing read was an end tag for this group.
437 if (!input->LastTagWas(MakeTag(field_number, WIRETYPE_END_GROUP))) {
438 return false;
439 }
440 return true;
441}
442template<typename MessageType>
443inline bool WireFormatLite::ReadMessage(
444 io::CodedInputStream* input, MessageType* value) {
445 int length;
446 if (!input->ReadVarintSizeAsInt(&length)) return false;
447 std::pair<io::CodedInputStream::Limit, int> p =
448 input->IncrementRecursionDepthAndPushLimit(length);
449 if (p.second < 0 || !value->MergePartialFromCodedStream(input)) return false;
450 // Make sure that parsing stopped when the limit was hit, not at an endgroup
451 // tag.
452 return input->DecrementRecursionDepthAndPopLimit(p.first);
453}

612template<typename T>
613inline uint8* WireFormatLite::WritePrimitiveNoTagToArray(
614 const RepeatedField<T>& value,
615 uint8* (*Writer)(T, uint8*), uint8* target) {
616 const int n = value.size();
617 GOOGLE_DCHECK_GT(n, 0);
618
619 const T* ii = value.unsafe_data();
620 int i = 0;
621 do {
622 target = Writer(ii[i], target);
623 } while (++i < n);
624
625 return target;
626}

628template<typename T>
629inline uint8* WireFormatLite::WriteFixedNoTagToArray(
630 const RepeatedField<T>& value,
631 uint8* (*Writer)(T, uint8*), uint8* target) {
632#if defined(PROTOBUF_LITTLE_ENDIAN)
633 (void) Writer;
634
635 const int n = value.size();
636 GOOGLE_DCHECK_GT(n, 0);
637
638 const T* ii = value.unsafe_data();
639 const int bytes = n * static_cast<int>(sizeof(ii[0]));
640 memcpy(target, ii, static_cast<size_t>(bytes));
641 return target + bytes;
642#else
643 return WritePrimitiveNoTagToArray(value, Writer, target);
644#endif
645}

789template<typename T>
790inline uint8* WireFormatLite::WritePrimitiveToArray(
791 int field_number,
792 const RepeatedField<T>& value,
793 uint8* (*Writer)(int, T, uint8*), uint8* target) {
794 const int n = value.size();
795 if (n == 0) {
796 return target;
797 }
798
799 const T* ii = value.unsafe_data();
800 int i = 0;
801 do {
802 target = Writer(field_number, ii[i], target);
803 } while (++i < n);
804
805 return target;
806}

983inline size_t WireFormatLite::LengthDelimitedSize(size_t length) {
984 // The static_cast here prevents an error in certain compiler configurations
985 // but is not technically correct--if length is too large to fit in a uint32
986 // then it will be silently truncated. We will need to fix this if we ever
987 // decide to start supporting serialized messages greater than 2 GiB in size.
988 return length + io::CodedOutputStream::VarintSize32(
989 static_cast<uint32>(length));
