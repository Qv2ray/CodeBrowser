19#ifndef GRPCPP_IMPL_SERVER_BUILDER_PLUGIN_H
20#define GRPCPP_IMPL_SERVER_BUILDER_PLUGIN_H
21
22#include <memory>
23
24#include <grpcpp/support/config.h>
25
26namespace grpc {
27
28class ServerBuilder;
29class ServerInitializer;
30class ChannelArguments;
31
32/// This interface is meant for internal usage only. Implementations of this
33/// interface should add themselves to a \a ServerBuilder instance through the
34/// \a InternalAddPluginFactory method.
35class ServerBuilderPlugin {
36 public:
37 virtual ~ServerBuilderPlugin() {}
38 virtual grpc::string name() = 0;
39
40 /// UpdateServerBuilder will be called at an early stage in
41 /// ServerBuilder::BuildAndStart(), right after the ServerBuilderOptions have
42 /// done their updates.
43 virtual void UpdateServerBuilder(ServerBuilder* builder) {}
44
45 /// InitServer will be called in ServerBuilder::BuildAndStart(), after the
46 /// Server instance is created.
47 virtual void InitServer(ServerInitializer* si) = 0;
48
49 /// Finish will be called at the end of ServerBuilder::BuildAndStart().
50 virtual void Finish(ServerInitializer* si) = 0;
51
52 /// ChangeArguments is an interface that can be used in
53 /// ServerBuilderOption::UpdatePlugins
54 virtual void ChangeArguments(const grpc::string& name, void* value) = 0;
55
56 /// UpdateChannelArguments will be called in ServerBuilder::BuildAndStart(),
57 /// before the Server instance is created.
58 virtual void UpdateChannelArguments(ChannelArguments* args) {}
59
60 virtual bool has_sync_methods() const { return false; }
61 virtual bool has_async_methods() const { return false; }
62};
63
64} // namespace grpc
65
66#endif // GRPCPP_IMPL_SERVER_BUILDER_PLUGIN_H
