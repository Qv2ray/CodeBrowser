19#ifndef GRPCPP_CREATE_CHANNEL_H
20#define GRPCPP_CREATE_CHANNEL_H
21
22#include <memory>
23
24#include <grpcpp/channel.h>
25#include <grpcpp/impl/codegen/client_interceptor.h>
26#include <grpcpp/security/credentials.h>
27#include <grpcpp/support/channel_arguments.h>
28#include <grpcpp/support/config.h>
29
30namespace grpc {
31
32/// Create a new \a Channel pointing to \a target.
33///
34/// \param target The URI of the endpoint to connect to.
35/// \param creds Credentials to use for the created channel. If it does not
36/// hold an object or is invalid, a lame channel (one on which all operations
37/// fail) is returned.
38std::shared_ptr<Channel> CreateChannel(
39 const grpc::string& target,
40 const std::shared_ptr<ChannelCredentials>& creds);
41
42/// Create a new \em custom \a Channel pointing to \a target.
43///
44/// \warning For advanced use and testing ONLY. Override default channel
45/// arguments only if necessary.
46///
47/// \param target The URI of the endpoint to connect to.
48/// \param creds Credentials to use for the created channel. If it does not
49/// hold an object or is invalid, a lame channel (one on which all operations
50/// fail) is returned.
51/// \param args Options for channel creation.
52std::shared_ptr<Channel> CreateCustomChannel(
53 const grpc::string& target,
54 const std::shared_ptr<ChannelCredentials>& creds,
55 const ChannelArguments& args);
56
57namespace experimental {
58/// Create a new \em custom \a Channel pointing to \a target with \a
59/// interceptors being invoked per call.
60///
61/// \warning For advanced use and testing ONLY. Override default channel
62/// arguments only if necessary.
63///
64/// \param target The URI of the endpoint to connect to.
65/// \param creds Credentials to use for the created channel. If it does not
66/// hold an object or is invalid, a lame channel (one on which all operations
67/// fail) is returned.
68/// \param args Options for channel creation.
69std::shared_ptr<Channel> CreateCustomChannelWithInterceptors(
70 const grpc::string& target,
71 const std::shared_ptr<ChannelCredentials>& creds,
72 const ChannelArguments& args,
73 std::unique_ptr<std::vector<
74 std::unique_ptr<experimental::ClientInterceptorFactoryInterface>>>
75 interceptor_creators);
76} // namespace experimental
77} // namespace grpc
78
79#endif // GRPCPP_CREATE_CHANNEL_H
