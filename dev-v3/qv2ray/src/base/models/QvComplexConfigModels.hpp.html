58 struct OutboundObjectMeta
59 {
60 MetaOutboundObjectType metaType;
61 QString displayName;
62 //
63 ConnectionId connectionId;
64 QList<QString> outboundTags;
65 QString strategyType;
66 int chainPortAllocation = QV2RAY_CHAINED_OUTBOUND_PORT_ALLOCATION;
67 //
68 safetype::OUTBOUND realOutbound;
69 QString getDisplayName() const
70 {
71 if (metaType == METAOUTBOUND_ORIGINAL)
72 return realOutbound["tag"].toString();
73 else
74 return displayName;
75 }
76 static OutboundObjectMeta loadFromOutbound(const safetype::OUTBOUND &out)
77 {
78 OutboundObjectMeta meta;
79 meta.loadJson(out[META_OUTBOUND_KEY_NAME].toObject());
80 meta.realOutbound = out;
81 return meta;
82 }
83 OutboundObjectMeta() : metaType(METAOUTBOUND_ORIGINAL){};
84 QJS_PLAIN_JSON(metaType, displayName, connectionId, outboundTags, chainPortAllocation, strategyType)
85 };
