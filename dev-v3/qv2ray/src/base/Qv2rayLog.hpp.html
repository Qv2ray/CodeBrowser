54 template<QvLogType t, typename... T>
55 inline void log_internal(T... v)
56 {
57 ((logStream << v << " "), ...);
58 ((tempStream << v << " "), ...);
59 logStream << NEWLINE;
60#ifndef QT_DEBUG
61 // We only process DEBUG log in Release mode
62 // Prevent QvCoreApplication nullptr
63 // TODO: Move log function inside QvCoreApplication
64 if (t == QV2RAY_LOG_DEBUG && QvCoreApplication && !QvCoreApplication->StartupArguments.debugLog)
65 {
66 // Discard debug log in non-debug Qv2ray version with
67 // no-debugLog mode.
68 return;
69 }
70#endif
71
72 const auto logString = tempStream.readAll();
73#ifdef Q_OS_ANDROID
74 __android_log_write(ANDROID_LOG_INFO, "Qv2ray", logString.toStdString().c_str());
75#else
76 std::cout << logString.toStdString() << std::endl;
77#endif
78 }

92template<typename TKey, typename TVal>

95 stream << "{ ";
