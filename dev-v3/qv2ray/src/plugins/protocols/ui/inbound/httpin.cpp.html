48void HTTPInboundEditor::on_httpRemoveUserBtn_clicked()
49{
50 PLUGIN_EDITOR_LOADING_GUARD
51 if (httpAccountListBox->currentRow() < 0)
52 {
53 InternalProtocolSupportPluginInstance->PluginErrorMessageBox(tr("Removing a user"), tr("You haven't selected a user yet."));
54 return;
55 }
56 const auto item = httpAccountListBox->currentItem();
57 auto list = content["accounts"].toArray();
58
59 for (int i = 0; i < list.count(); i++)
60 {
61 const auto user = list[i].toObject();
62 const auto entry = user["user"].toString() + ":" + user["pass"].toString();
63 if (entry == item->text().trimmed())
64 {
65 list.removeAt(i);
66 content["accounts"] = list;
67 httpAccountListBox->takeItem(httpAccountListBox->currentRow());
68 return;
69 }
70 }
71}

73void HTTPInboundEditor::on_httpAddUserBtn_clicked()
74{
75 PLUGIN_EDITOR_LOADING_GUARD
76 const auto user = httpAddUserTxt->text();
77 const auto pass = httpAddPasswordTxt->text();
78 //
79 auto list = content["accounts"].toArray();
80
81 for (int i = 0; i < list.count(); i++)
82 {
83 const auto _user = list[i].toObject();
84 if (_user["user"].toString() == user)
85 {
86 InternalProtocolSupportPluginInstance->PluginErrorMessageBox(tr("Add a user"), tr("This user exists already."));
87 return;
88 }
89 }
90
91 httpAddUserTxt->clear();
92 httpAddPasswordTxt->clear();
93 list.append(QJsonObject{ { "user", user }, { "pass", pass } });
94 httpAccountListBox->addItem(user + ":" + pass);
95 content["accounts"] = list;
96}
