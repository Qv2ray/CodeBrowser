10const OutboundInfoObject BuiltinSerializer::GetOutboundInfo(const QString &protocol, const QJsonObject &outbound) const
11{
12 OutboundInfoObject obj;
13 obj[INFO_PROTOCOL] = protocol;
14 if (protocol == "http")
15 {
16 HttpServerObject http;
17 http.loadJson(outbound["servers"].toArray().first());
18 obj[INFO_SERVER] = *http.address;
19 obj[INFO_PORT] = *http.port;
20 }
21 else if (protocol == "socks")
22 {
23 SocksServerObject socks;
24 socks.loadJson(outbound["servers"].toArray().first());
25 obj[INFO_SERVER] = *socks.address;
26 obj[INFO_PORT] = *socks.port;
27 }
28 else if (protocol == "vmess")
29 {
30 VMessServerObject vmess;
31 vmess.loadJson(outbound["vnext"].toArray().first());
32 obj[INFO_SERVER] = *vmess.address;
33 obj[INFO_PORT] = *vmess.port;
34 }
35 else if (protocol == "vless")
36 {
37 VLESSServerObject vless;
38 vless.loadJson(outbound["vnext"].toArray().first());
39 obj[INFO_SERVER] = *vless.address;
40 obj[INFO_PORT] = *vless.port;
41 }
42 else if (protocol == "shadowsocks")
43 {
44 ShadowSocksServerObject ss;
45 ss.loadJson(outbound["servers"].toArray().first());
46 obj[INFO_SERVER] = *ss.address;
47 obj[INFO_PORT] = *ss.port;
48 }
49 return obj;
50}

52void BuiltinSerializer::SetOutboundInfo(const QString &protocol, const OutboundInfoObject &info, QJsonObject &outbound) const
53{
54 if ((QStringList{ "http", "socks", "shadowsocks" }).contains(protocol))
55 {
56 QJsonIO::SetValue(outbound, info[INFO_SERVER].toString(), "servers", 0, "address");
57 QJsonIO::SetValue(outbound, info[INFO_PORT].toInt(), "servers", 0, "port");
58 }
59 else if ((QStringList{ "vless", "vmess" }).contains(protocol))
60 {
61 QJsonIO::SetValue(outbound, info[INFO_SERVER].toString(), "vnext", 0, "address");
62 QJsonIO::SetValue(outbound, info[INFO_PORT].toInt(), "vnext", 0, "port");
63 }
64}

66const QString BuiltinSerializer::SerializeOutbound(const QString &protocol, const QString &alias, const QString &, const QJsonObject &obj,
67 const QJsonObject &objStream) const
68{
69 if (protocol == "http" || protocol == "socks")
70 {
71 QUrl url;
72 url.setScheme(protocol);
73 url.setHost(QJsonIO::GetValue(obj, { "servers", 0, "address" }).toString());
74 url.setPort(QJsonIO::GetValue(obj, { "servers", 0, "port" }).toInt());
75 if (QJsonIO::GetValue(obj, { "servers", 0 }).toObject().contains("users"))
76 {
77 url.setUserName(QJsonIO::GetValue(obj, { "servers", 0, "users", 0, "user" }).toString());
78 url.setPassword(QJsonIO::GetValue(obj, { "servers", 0, "users", 0, "pass" }).toString());
79 }
80 return url.toString();
81 }
82 if (protocol == "vless")
83 {
84 QUrl url;
85 url.setFragment(QUrl::toPercentEncoding(alias));
86 url.setScheme(protocol);
87 url.setHost(QJsonIO::GetValue(obj, { "vnext", 0, "address" }).toString());
88 url.setPort(QJsonIO::GetValue(obj, { "vnext", 0, "port" }).toInt());
89 url.setUserName(QJsonIO::GetValue(obj, { "vnext", 0, "users", 0, "id" }).toString());
90
91 // -------- COMMON INFORMATION --------
92 QUrlQuery query;
93 const auto encryption = QJsonIO::GetValue(obj, { "vnext", 0, "users", 0, "encryption" }).toString("none");
94 if (encryption != "none")
95 query.addQueryItem("encryption", encryption);
96
97 const auto network = QJsonIO::GetValue(objStream, "network").toString("tcp");
98 if (network != "tcp")
99 query.addQueryItem("type", network);
100
101 const auto security = QJsonIO::GetValue(objStream, "security").toString("none");
102 if (security != "none")
103 query.addQueryItem("security", security);
104
105 // -------- TRANSPORT RELATED --------
106 if (network == "kcp")
107 {
108 const auto seed = QJsonIO::GetValue(objStream, { "kcpSettings", "seed" }).toString();
109 if (!seed.isEmpty())
110 query.addQueryItem("seed", QUrl::toPercentEncoding(seed));
111
112 const auto headerType = QJsonIO::GetValue(objStream, { "kcpSettings", "header", "type" }).toString("none");
113 if (headerType != "none")
114 query.addQueryItem("headerType", headerType);
115 }
116 else if (network == "http")
117 {
118 const auto path = QJsonIO::GetValue(objStream, { "httpSettings", "path" }).toString("/");
119 query.addQueryItem("path", QUrl::toPercentEncoding(path));
120
121 const auto hosts = QJsonIO::GetValue(objStream, { "httpSettings", "host" }).toArray();
122 QStringList hostList;
123 for (const auto item : hosts)
124 {
125 const auto host = item.toString();
126 if (!host.isEmpty())
127 hostList << host;
128 }
129 query.addQueryItem("host", QUrl::toPercentEncoding(hostList.join(",")));
130 }
131 else if (network == "ws")
132 {
133 const auto path = QJsonIO::GetValue(objStream, { "wsSettings", "path" }).toString("/");
134 query.addQueryItem("path", QUrl::toPercentEncoding(path));
135
136 const auto host = QJsonIO::GetValue(objStream, { "wsSettings", "headers", "Host" }).toString();
137 query.addQueryItem("host", host);
138 }
139 else if (network == "quic")
140 {
141 const auto quicSecurity = QJsonIO::GetValue(objStream, { "quicSettings", "security" }).toString("none");
142 if (quicSecurity != "none")
143 {
144 query.addQueryItem("quicSecurity", quicSecurity);
145
146 const auto key = QJsonIO::GetValue(objStream, { "quicSettings", "key" }).toString();
147 query.addQueryItem("key", QUrl::toPercentEncoding(key));
148
149 const auto headerType = QJsonIO::GetValue(objStream, { "quicSettings", "header", "type" }).toString("none");
150 if (headerType != "none")
151 query.addQueryItem("headerType", headerType);
152 }
153 }
154 else if (network == "grpc")
155 {
156 const auto serviceName = QJsonIO::GetValue(objStream, { "grpcSettings", "serviceName" }).toString("GunService");
157 if (serviceName != "GunService")
158 query.addQueryItem("serviceName", QUrl::toPercentEncoding(serviceName));
159
160 const auto multiMode = QJsonIO::GetValue(objStream, { "grpcSettings", "multiMode" }).toBool(false);
161 if (multiMode)
162 query.addQueryItem("mode", "multi");
163 }
164 // -------- TLS RELATED --------
165 const auto tlsKey = security == "xtls" ? "xtlsSettings" : "tlsSettings";
166
167 const auto sni = QJsonIO::GetValue(objStream, { tlsKey, "serverName" }).toString();
168 if (!sni.isEmpty())
169 query.addQueryItem("sni", sni);
170
171 // ALPN
172 const auto alpnArray = QJsonIO::GetValue(objStream, { tlsKey, "alpn" }).toArray();
173 QStringList alpnList;
174 for (const auto v : alpnArray)
175 {
176 const auto alpn = v.toString();
177 if (!alpn.isEmpty())
178 alpnList << alpn;
179 }
180 query.addQueryItem("alpn", QUrl::toPercentEncoding(alpnList.join(",")));
181
182 // -------- XTLS Flow --------
183 if (security == "xtls")
184 {
185 const auto flow = QJsonIO::GetValue(obj, "vnext", 0, "users", 0, "flow").toString();
186 query.addQueryItem("flow", flow);
187 }
188
189 // ======== END OF QUERY ========
190 url.setQuery(query);
191 return url.toString(QUrl::FullyEncoded);
192 }
193 return "(Unsupported)";
194}

196const QPair<QString, QJsonObject> BuiltinSerializer::DeserializeOutbound(const QString &link, QString *alias, QString *errorMessage) const
197{
198 Q_UNUSED(alias);
199 Q_UNUSED(errorMessage);
200 if (link.startsWith("http://") || link.startsWith("socks://"))
201 {
202 const QUrl url = link;
203 QJsonObject root;
204 QJsonIO::SetValue(root, url.host(), "servers", 0, "address");
205 QJsonIO::SetValue(root, url.port(), "servers", 0, "port");
206 if (url.userName().isEmpty() && url.password().isEmpty())
207 {
208 QJsonIO::SetValue(root, url.userName(), "servers", 0, "users", 0, "user");
209 QJsonIO::SetValue(root, url.password(), "servers", 0, "users", 0, "pass");
210 }
211 return { url.scheme(), root };
212 }
213 return {};
214}
