18QStringList Qv2rayPlatformApplication::CheckPrerequisites()
19{
20 QStringList errors;
21 if (!QSslSocket::supportsSsl())
22 {
23 // Check OpenSSL version for auto-update and subscriptions
24 const auto osslReqVersion = QSslSocket::sslLibraryBuildVersionString();
25 const auto osslCurVersion = QSslSocket::sslLibraryVersionString();
26 LOG("Current OpenSSL version: " + osslCurVersion);
27 LOG("Required OpenSSL version: " + osslReqVersion);
28 errors << "Qv2ray cannot run without OpenSSL.";
29 errors << "This is usually caused by using the wrong version of OpenSSL";
30 errors << "Required=" + osslReqVersion + "Current=" + osslCurVersion;
31 }
32 return errors + checkPrerequisitesInternal();
33}

35bool Qv2rayPlatformApplication::Initialize()
36{
37 QString errorMessage;
38 bool canContinue;
39 const auto hasError = parseCommandLine(&errorMessage, &canContinue);
40 if (hasError)
41 {
42 LOG("Command line:" A(errorMessage));
43 if (!canContinue)
44 {
45 LOG("Fatal, Qv2ray cannot continue.");
46 return false;
47 }
48 else
49 {
50 LOG("Non-fatal error, continue starting up.");
51 }
52 }
53
54#ifdef Q_OS_WIN
55 const auto appPath = QDir::toNativeSeparators(applicationFilePath());
56 const auto regPath = "HKEY_CURRENT_USER\\Software\\Classes\\" + QV2RAY_URL_SCHEME;
57 QSettings reg(regPath, QSettings::NativeFormat);
58 reg.setValue("Default", "Qv2ray");
59 reg.setValue("URL Protocol", "");
60 reg.beginGroup("DefaultIcon");
61 reg.setValue("Default", QString("%1,1").arg(appPath));
62 reg.endGroup();
63 reg.beginGroup("shell");
64 reg.beginGroup("open");
65 reg.beginGroup("command");
66 reg.setValue("Default", appPath + " %1");
67#endif
68
69 connect(this, &Qv2rayPlatformApplication::aboutToQuit, this, &Qv2rayPlatformApplication::quitInternal);
70#ifndef QV2RAY_NO_SINGLEAPPLICATON
71 connect(this, &SingleApplication::receivedMessage, this, &Qv2rayPlatformApplication::onMessageReceived, Qt::QueuedConnection);
72 if (isSecondary())
73 {
74 StartupArguments.version = QV2RAY_VERSION_STRING;
75 StartupArguments.buildVersion = QV2RAY_VERSION_BUILD;
76 StartupArguments.fullArgs = arguments();
77 if (StartupArguments.arguments.isEmpty())
78 StartupArguments.arguments << Qv2rayStartupArguments::NORMAL;
79 bool status = sendMessage(JsonToString(StartupArguments.toJson(), QJsonDocument::Compact).toUtf8());
80 if (!status)
81 LOG("Cannot send message.");
82 SetExitReason(EXIT_SECONDARY_INSTANCE);
83 return false;
84 }
85#endif
86
87#ifdef QV2RAY_GUI
88#ifdef Q_OS_LINUX
89#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
90 setFallbackSessionManagementEnabled(false);
91#endif
92 connect(this, &QGuiApplication::commitDataRequest, [] {
93 RouteManager->SaveRoutes();
94 ConnectionManager->SaveConnectionConfig();
95 PluginHost->SavePluginSettings();
96 SaveGlobalSettings();
97 });
98#endif
99
100#ifdef Q_OS_WIN
101 SetCurrentDirectory(applicationDirPath().toStdWString().c_str());
102 // Set special font in Windows
103 QFont font;
104 font.setPointSize(9);
105 font.setFamily("Microsoft YaHei");
106 setFont(font);
107#endif
108#endif
109
110 // Install a default translater. From the OS/DE
111 Qv2rayTranslator = std::make_unique<QvTranslator>();
112 Qv2rayTranslator->InstallTranslation(QLocale::system().name());
113 const auto allTranslations = Qv2rayTranslator->GetAvailableLanguages();
114 const auto osLanguage = QLocale::system().name();
115 //
116 LocateConfiguration();
117 if (!allTranslations.contains(GlobalConfig.uiConfig->language))
118 {
119 // If we need to reset the language.
120 if (allTranslations.contains(osLanguage))
121 {
122 GlobalConfig.uiConfig->language = osLanguage;
123 }
124 else if (!allTranslations.isEmpty())
125 {
126 GlobalConfig.uiConfig->language = allTranslations.first();
127 }
128 }
129
130 if (!Qv2rayTranslator->InstallTranslation(GlobalConfig.uiConfig->language))
131 {
132 QvMessageBoxWarn(nullptr, "Translation Failed",
133 "Cannot load translation for " + GlobalConfig.uiConfig->language + NEWLINE + //
134 "English is now used." + NEWLINE + NEWLINE + //
135 "Please go to Preferences Window to change language or open an Issue");
136 GlobalConfig.uiConfig->language = "en_US";
137 }
138 return true;
139}

166bool Qv2rayPlatformApplication::parseCommandLine(QString *errorMessage, bool *canContinue)
167{
168 *canContinue = true;
169 QStringList filteredArgs;
170 for (const auto &arg : arguments())
171 {
172#ifdef Q_OS_MACOS
173 if (arg.contains("-psn"))
174 continue;
175#endif
176 filteredArgs << arg;
177 }
178 QCommandLineParser parser;
179 //
180 QCommandLineOption noAPIOption("noAPI", QObject::tr("Disable gRPC API subsystem"));
181 QCommandLineOption noPluginsOption("noPlugin", QObject::tr("Disable plugins feature"));
182 QCommandLineOption debugLogOption("debug", QObject::tr("Enable debug output"));
183 QCommandLineOption noAutoConnectionOption("noAutoConnection", QObject::tr("Do not automatically connect"));
184 QCommandLineOption disconnectOption("disconnect", QObject::tr("Stop current connection"));
185 QCommandLineOption reconnectOption("reconnect", QObject::tr("Reconnect last connection"));
186 QCommandLineOption exitOption("exit", QObject::tr("Exit Qv2ray"));
187 //
188 parser.setApplicationDescription(QObject::tr("Qv2ray - A cross-platform Qt frontend for V2Ray."));
189 parser.setSingleDashWordOptionMode(QCommandLineParser::ParseAsLongOptions);
190 //
191 parser.addOption(noAPIOption);
192 parser.addOption(noPluginsOption);
193 parser.addOption(debugLogOption);
194 parser.addOption(noAutoConnectionOption);
195 parser.addOption(disconnectOption);
196 parser.addOption(reconnectOption);
197 parser.addOption(exitOption);
198 //
199 const auto helpOption = parser.addHelpOption();
200 const auto versionOption = parser.addVersionOption();
201
202 if (!parser.parse(filteredArgs))
203 {
204 *canContinue = true;
205 *errorMessage = parser.errorText();
206 return false;
207 }
208
209 if (parser.isSet(versionOption))
210 {
211 parser.showVersion();
212 return true;
213 }
214
215 if (parser.isSet(helpOption))
216 {
217 parser.showHelp();
218 return true;
219 }
220
221 for (const auto &arg : parser.positionalArguments())
222 {
223 if (arg.startsWith(QV2RAY_URL_SCHEME + "://"))
224 {
225 StartupArguments.arguments << Qv2rayStartupArguments::QV2RAY_LINK;
226 StartupArguments.links << arg;
227 }
228 }
229
230 if (parser.isSet(exitOption))
231 {
232 DEBUG("disconnectOption is set.");
233 StartupArguments.arguments << Qv2rayStartupArguments::EXIT;
234 }
235
236 if (parser.isSet(disconnectOption))
237 {
238 DEBUG("disconnectOption is set.");
239 StartupArguments.arguments << Qv2rayStartupArguments::DISCONNECT;
240 }
241
242 if (parser.isSet(reconnectOption))
243 {
244 DEBUG("reconnectOption is set.");
245 StartupArguments.arguments << Qv2rayStartupArguments::RECONNECT;
246 }
247
248#define ProcessExtraStartupOptions(option) \
249 DEBUG("Startup Options:" A(parser.isSet(option##Option))); \
250 StartupArguments.option = parser.isSet(option##Option);
251
252 ProcessExtraStartupOptions(noAPI);
253 ProcessExtraStartupOptions(debugLog);
254 ProcessExtraStartupOptions(noAutoConnection);
255 ProcessExtraStartupOptions(noPlugins);
256 return true;
257}
