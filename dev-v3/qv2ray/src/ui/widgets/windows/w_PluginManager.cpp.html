10PluginManageWindow::PluginManageWindow(QWidget *parent) : QvDialog("PluginManager", parent)
11{
12 addStateOptions("width", { [&] { return width(); }, [&](QJsonValue val) { resize(val.toInt(), size().height()); } });
13 addStateOptions("height", { [&] { return height(); }, [&](QJsonValue val) { resize(size().width(), val.toInt()); } });
14 addStateOptions("x", { [&] { return x(); }, [&](QJsonValue val) { move(val.toInt(), y()); } });
15 addStateOptions("y", { [&] { return y(); }, [&](QJsonValue val) { move(x(), val.toInt()); } });
16
17 setupUi(this);
18 for (auto &plugin : PluginHost->AllPlugins())
19 {
20 const auto &info = PluginHost->GetPlugin(plugin)->metadata;
21 auto item = new QListWidgetItem(pluginListWidget);
22 item->setCheckState(PluginHost->GetPluginEnabled(info.InternalName) ? Qt::Checked : Qt::Unchecked);
23 item->setData(Qt::UserRole, info.InternalName);
24 item->setText(info.Name + " (" + (PluginHost->GetPlugin(info.InternalName)->isLoaded ? tr("Loaded") : tr("Not loaded")) + ")");
25 pluginListWidget->addItem(item);
26 }
27 pluginListWidget->sortItems();
28 isLoading = false;
29 if (pluginListWidget->count() > 0)
30 on_pluginListWidget_currentItemChanged(pluginListWidget->item(0), nullptr);
31}

40void PluginManageWindow::on_pluginListWidget_currentItemChanged(QListWidgetItem *current, QListWidgetItem *previous)
41{
42 Q_UNUSED(previous)
43 if (currentPluginInfo && currentSettingsWidget)
44 {
45 currentPluginInfo->pluginInterface->UpdateSettings(currentSettingsWidget->GetSettings());
46 pluginSettingsLayout->removeWidget(currentSettingsWidget.get());
47 currentSettingsWidget.reset();
48 }
49 pluginIconLabel->clear();
50 if (!current)
51 return;
52
53 currentPluginInfo = PluginHost->GetPlugin(current->data(Qt::UserRole).toString());
54 auto &info = currentPluginInfo->metadata;
55
56 pluginNameLabel->setText(info.Name);
57 pluginAuthorLabel->setText(info.Author);
58 pluginDescriptionLabel->setText(info.Description);
59 pluginLibPathLabel->setText(currentPluginInfo->libraryPath);
60 pluginStateLabel->setText(currentPluginInfo->isLoaded ? tr("Loaded") : tr("Not loaded"));
61 pluginComponentsLabel->setText(GetPluginComponentsString(info.Components).join(NEWLINE));
62
63 if (!currentPluginInfo->isLoaded)
64 {
65 pluginUnloadLabel->setVisible(true);
66 pluginUnloadLabel->setText(tr("Plugin Not Loaded"));
67 return;
68 }
69
70 if (currentPluginInfo->hasComponent(COMPONENT_GUI))
71 {
72 const auto pluginUIInterface = currentPluginInfo->pluginInterface->GetGUIInterface();
73 pluginGuiComponentsLabel->setText(GetPluginComponentsString(pluginUIInterface->GetComponents()).join(NEWLINE));
74 pluginIconLabel->setPixmap(pluginUIInterface->Icon().pixmap(pluginIconLabel->size() * devicePixelRatio()));
75 if (pluginUIInterface->GetComponents().contains(GUI_COMPONENT_SETTINGS))
76 {
77 currentSettingsWidget = pluginUIInterface->GetSettingsWidget();
78 currentSettingsWidget->SetSettings(currentPluginInfo->pluginInterface->GetSettngs());
79 pluginUnloadLabel->setVisible(false);
80 pluginSettingsLayout->addWidget(currentSettingsWidget.get());
81 }
82 else
83 {
84 pluginUnloadLabel->setVisible(true);
85 pluginUnloadLabel->setText(tr("Plugin does not have settings widget."));
86 }
87 }
88 else
89 {
90 pluginGuiComponentsLabel->setText(tr("None"));
91 }
92}

100void PluginManageWindow::on_pluginListWidget_itemChanged(QListWidgetItem *item)
101{
102 if (isLoading)
103 return;
104 bool isEnabled = item->checkState() == Qt::Checked;
105 const auto pluginInternalName = item->data(Qt::UserRole).toString();
106 PluginHost->SetPluginEnabled(pluginInternalName, isEnabled);
107 const auto info = PluginHost->GetPlugin(pluginInternalName);
108 item->setText(info->metadata.Name + " (" + (info->isLoaded ? tr("Loaded") : tr("Not loaded")) + ")");
109}

111void PluginManageWindow::on_pluginEditSettingsJsonBtn_clicked()
112{
113 if (const auto &current = pluginListWidget->currentItem(); current != nullptr)
114 {
115 const auto &info = PluginHost->GetPlugin(current->data(Qt::UserRole).toString());
116 if (!info->isLoaded)
117 {
118 QvMessageBoxWarn(this, tr("Plugin not loaded"), tr("This plugin is not loaded, please enable or reload the plugin to continue."));
119 return;
120 }
121 JsonEditor w(info->pluginInterface->GetSettngs());
122 auto newConf = w.OpenEditor();
123 if (w.result() == QDialog::Accepted)
124 {
125 info->pluginInterface->UpdateSettings(newConf);
126 }
127 }
128}
