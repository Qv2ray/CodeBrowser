57void MainWindow::ReloadRecentConnectionList()
58{
59 QList<ConnectionGroupPair> newRecentConnections;
60 const auto iterateRange = std::min(*GlobalConfig.uiConfig->maxJumpListCount, (int) GlobalConfig.uiConfig->recentConnections->count());
61 for (auto i = 0; i < iterateRange; i++)
62 {
63 const auto &item = GlobalConfig.uiConfig->recentConnections->at(i);
64 if (newRecentConnections.contains(item) || item.isEmpty())
65 continue;
66 newRecentConnections << item;
67 }
68 GlobalConfig.uiConfig->recentConnections = newRecentConnections;
69}

364void MainWindow::ProcessCommand(QString command, QStringList commands, QMap<QString, QString> args)
365{
366 if (commands.isEmpty())
367 return;
368 if (command == "open")
369 {
370 const auto subcommand = commands.takeFirst();
371 QvDialog *w;
372 if (subcommand == "preference")
373 w = new PreferencesWindow();
374 else if (subcommand == "plugin")
375 w = new PluginManageWindow();
376 else if (subcommand == "group")
377 w = new GroupManager();
378 else if (subcommand == "import")
379 w = new ImportConfigWindow();
380 else
381 return;
382 w->processCommands(command, commands, args);
383 w->exec();
384 delete w;
385 }

400void MainWindow::keyPressEvent(QKeyEvent *e)
401{
402 if (focusWidget() == connectionTreeView)
403 {
404 CheckCurrentWidget;
405 if (e->key() == Qt::Key_Enter || e->key() == Qt::Key_Return)
406 {
407 // If pressed enter or return on connectionListWidget. Try to connect to the selected connection.
408 if (widget->IsConnection())
409 {
410 widget->BeginConnection();
411 }
412 else
413 {
414 connectionTreeView->expand(connectionTreeView->currentIndex());
415 }
416 }
417 else if (e->key() == Qt::Key_F2)
418 {
419 widget->BeginRename();
420 }
421 else if (e->key() == Qt::Key_Delete)
422 {
423 Action_DeleteConnections();
424 }
425 }
426
427 if (e->key() == Qt::Key_Escape)
428 {
429 auto widget = GetIndexWidget(connectionTreeView->currentIndex());
430 // Check if this key was accpted by the ConnectionItemWidget
431 if (widget && widget->IsRenaming())
432 {
433 widget->CancelRename();
434 return;
435 }
436 else if (this->isActiveWindow())
437 {
438 this->close();
439 }
440 }
441 // Ctrl + Q = Exit
442 else if (e->modifiers() & Qt::ControlModifier && e->key() == Qt::Key_Q)
443 {
444 if (QvMessageBoxAsk(this, tr("Quit Qv2ray"), tr("Are you sure to exit Qv2ray?")) == Yes)
445 Action_Exit();
446 }
447 // Control + W = Close Window
448 else if (e->modifiers() & Qt::ControlModifier && e->key() == Qt::Key_W)
449 {
450 if (this->isActiveWindow())
451 this->close();
452 }
453}

572void MainWindow::Action_DeleteConnections()
573{
574 QList<ConnectionGroupPair> connlist;
575 QList<GroupId> groupsList;
576
577 for (const auto &item : connectionTreeView->selectionModel()->selectedIndexes())
578 {
579 const auto widget = GetIndexWidget(item);
580 if (!widget)
581 continue;
582
583 const auto identifier = widget->Identifier();
584 if (widget->IsConnection())
585 {
586 // Simply add the connection id
587 connlist.append(identifier);
588 continue;
589 }
590
591 for (const auto &conns : ConnectionManager->GetConnections(identifier.groupId))
592 {
593 connlist.append(ConnectionGroupPair{ conns, identifier.groupId });
594 }
595
596 const auto message = tr("Do you want to remove this group as well?") + NEWLINE + tr("Group: ") + GetDisplayName(identifier.groupId);
597 if (QvMessageBoxAsk(this, tr("Removing Connection"), message) == Yes)
598 {
599 groupsList << identifier.groupId;
600 }
601 }
602
603 const auto strRemoveConnTitle = tr("Removing Connection(s)", "", connlist.count());
604 const auto strRemoveConnContent = tr("Are you sure to remove selected connection(s)?", "", connlist.count());
605 if (QvMessageBoxAsk(this, strRemoveConnTitle, strRemoveConnContent) != Yes)
606 {
607 return;
608 }
609
610 for (const auto &conn : connlist)
611 {
612 ConnectionManager->RemoveConnectionFromGroup(conn.connectionId, conn.groupId);
613 }
614
615 for (const auto &group : groupsList)
616 {
617 ConnectionManager->DeleteGroup(group);
618 }

711void MainWindow::OnStatsAvailable(const ConnectionGroupPair &id, const QMap<StatisticsType, QvStatsSpeedData> &data)
712{
713 if (!ConnectionManager->IsConnected(id))
714 return;
715 // This may not be, or may not precisely be, speed per second if the backend
716 // has "any" latency. (Hope not...)
717 //
718 QMap<SpeedWidget::GraphType, long> pointData;
719 bool isOutbound = GlobalConfig.uiConfig->graphConfig->useOutboundStats;
720 bool hasDirect = isOutbound && GlobalConfig.uiConfig->graphConfig->hasDirectStats;
721 for (const auto &[type, data] : data.toStdMap())
722 {
723 const auto upSpeed = data.first.first;
724 const auto downSpeed = data.first.second;
725 switch (type)
726 {
727 case API_INBOUND:
728 if (!isOutbound)
729 {
730 pointData[SpeedWidget::INBOUND_UP] = upSpeed;
731 pointData[SpeedWidget::INBOUND_DOWN] = downSpeed;
732 }
733 break;
734 case API_OUTBOUND_PROXY:
735 if (isOutbound)
736 {
737 pointData[SpeedWidget::OUTBOUND_PROXY_UP] = upSpeed;
738 pointData[SpeedWidget::OUTBOUND_PROXY_DOWN] = downSpeed;
739 }
740 break;
741 case API_OUTBOUND_DIRECT:
742 if (hasDirect)
743 {
744 pointData[SpeedWidget::OUTBOUND_DIRECT_UP] = upSpeed;
745 pointData[SpeedWidget::OUTBOUND_DIRECT_DOWN] = downSpeed;
746 }
747 break;
748 case API_OUTBOUND_BLACKHOLE: break;
749 }
750 }
751
752 speedChartWidget->AddPointData(pointData);
753 //
754 const auto upSpeed = data[CurrentStatAPIType].first.first;
755 const auto downSpeed = data[CurrentStatAPIType].first.second;
756 auto totalSpeedUp = FormatBytes(upSpeed) + "/s";
757 auto totalSpeedDown = FormatBytes(downSpeed) + "/s";
758 auto totalDataUp = FormatBytes(data[CurrentStatAPIType].second.first);
759 auto totalDataDown = FormatBytes(data[CurrentStatAPIType].second.second);
760 //
761 netspeedLabel->setText(totalSpeedUp + NEWLINE + totalSpeedDown);
762 dataamountLabel->setText(totalDataUp + NEWLINE + totalDataDown);
763 //
764 qvAppTrayIcon->setToolTip(TRAY_TOOLTIP_PREFIX NEWLINE + tr("Connected: ") + GetDisplayName(id.connectionId) + //
765 NEWLINE "Up: " + totalSpeedUp + " Down: " + totalSpeedDown);
766}

768void MainWindow::OnVCoreLogAvailable(const ConnectionGroupPair &id, const QString &log)
769{
770 Q_UNUSED(id);
771 FastAppendTextDocument(log.trimmed(), vCoreLogDocument);
772 // vCoreLogDocument->setPlainText(vCoreLogDocument->toPlainText() + log);
773 // From https://gist.github.com/jemyzhang/7130092
774 auto maxLines = *GlobalConfig.uiConfig->maximumLogLines;
775 auto block = vCoreLogDocument->begin();
776
777 while (block.isValid())
778 {
779 if (vCoreLogDocument->blockCount() > maxLines)
780 {
781 QTextCursor cursor(block);
782 block = block.next();
783 cursor.select(QTextCursor::BlockUnderCursor);
784 cursor.movePosition(QTextCursor::NextCharacter, QTextCursor::KeepAnchor);
785 cursor.removeSelectedText();
786 continue;
787 }
788
789 break;
790 }
791}

1053void MainWindow::Action_CopyRecentLogs()
1054{
1055 const auto lines = SplitLines(masterLogBrowser->document()->toPlainText());
1056 bool accepted = false;
1057 const auto line = QInputDialog::getInt(this, tr("Copy latest logs"), tr("Number of lines of logs to copy"), 20, 0, 2500, 1, &accepted);

1060 const auto totalLinesCount = lines.count();
1061 const auto linesToCopy = std::min((int) totalLinesCount, line);

1064 {

1066 }
1067 qApp->clipboard()->setText(result.join(NEWLINE));
