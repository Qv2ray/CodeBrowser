29void MainWindow::MWShowWindow()
30{
31#if QV2RAY_FEATURE(ui_has_store_state)
32 RestoreState();
33#endif
34 this->show();
35#ifdef Q_OS_WIN
36 setWindowState(Qt::WindowNoState);
37 SetWindowPos(HWND(this->winId()), HWND_TOPMOST, 0, 0, 0, 0, SWP_NOMOVE | SWP_NOSIZE | SWP_SHOWWINDOW);
38 QThread::msleep(20);
39 SetWindowPos(HWND(this->winId()), HWND_NOTOPMOST, 0, 0, 0, 0, SWP_NOMOVE | SWP_NOSIZE | SWP_SHOWWINDOW);
40#endif
41#ifdef Q_OS_MAC
42 ProcessSerialNumber psn = { 0, kCurrentProcess };
43 TransformProcessType(&psn, kProcessTransformToForegroundApplication);
44#endif
45 MWToggleVisibilitySetText();
46}

61void MainWindow::MWSetSystemProxy()
62{
63 const auto inboundInfo = KernelInstance->GetCurrentConnectionInboundInfo();
64
65 ProtocolSettingsInfoObject httpInboundInfo;
66 ProtocolSettingsInfoObject socksInboundInfo;
67
68 for (const auto &info : inboundInfo)
69 {
70 if (info.protocol == "http")
71 httpInboundInfo = info;
72 if (info.protocol == "socks")
73 socksInboundInfo = info;
74 }
75
76 const bool httpEnabled = httpInboundInfo.port > 0;
77 const bool socksEnabled = socksInboundInfo.port > 0;
78 QString proxyAddress;
79
80 if (httpEnabled || socksEnabled)
81 {
82 proxyAddress = httpEnabled ? httpInboundInfo.address : socksInboundInfo.address;
83 if (proxyAddress == "0.0.0.0")
84 proxyAddress = "127.0.0.1";
85
86 if (proxyAddress == "::")
87 proxyAddress = "::1";
88
89 LOG("ProxyAddress: " + proxyAddress);
90 SetSystemProxy(proxyAddress, httpInboundInfo.port, socksInboundInfo.port);
91 qvAppTrayIcon->setIcon(Q_TRAYICON("tray-systemproxy"));
92 if (!GlobalConfig.uiConfig->quietMode)
93 {
94 QvWidgetApplication->ShowTrayMessage(tr("System proxy configured."));
95 }
96 }
97 else
98 {
99 LOG("Neither of HTTP nor SOCKS is enabled, cannot set system proxy.");
100 QvMessageBoxWarn(this, tr("Cannot set system proxy"), tr("Both HTTP and SOCKS inbounds are not enabled"));
101 }
102}

127void MainWindow::CheckSubscriptionsUpdate()
128{
129 QList<std::pair<QString, GroupId>> updateList;
130 QStringList updateNamesList;
131
132 for (const auto &entry : ConnectionManager->Subscriptions())
133 {
134 const auto info = ConnectionManager->GetGroupMetaObject(entry);
135 //
136 // The update is ignored.
137 if (info.subscriptionOption->updateInterval == 0.0f)
138 continue;
139 //
140 const auto lastRenewDate = QDateTime::fromSecsSinceEpoch(info.lastUpdatedDate);
141 const auto renewTime = lastRenewDate.addSecs(info.subscriptionOption->updateInterval * 86400);
142
143 if (renewTime <= QDateTime::currentDateTime())
144 {
145 updateList << std::pair{ *info.displayName, entry };
146 updateNamesList << info.displayName;
147 LOG(QString("Subscription update \"%1\": L=%2 R=%3 I=%4")
148 .arg(info.displayName)
149 .arg(lastRenewDate.toString())
150 .arg(QSTRN(info.subscriptionOption->updateInterval))
151 .arg(renewTime.toString()));
152 }
153 }
154
155 if (!updateList.isEmpty())
156 {
157 const auto result = GlobalConfig.uiConfig->quietMode ? Yes :
158 QvMessageBoxAsk(this, tr("Update Subscriptions"), //
159 tr("Do you want to update these subscriptions?") + NEWLINE + //
160 updateNamesList.join(NEWLINE), //
161 { Yes, No, Ignore });
162
163 for (const auto &[name, id] : updateList)
164 {
165 if (result == Yes)
166 {
167 LOG("Updating subscription: " + name);
168 ConnectionManager->UpdateSubscriptionAsync(id);
169 }
170 else if (result == Ignore)
171 {
172 LOG("Ignored subscription update: " + name);
173 ConnectionManager->IgnoreSubscriptionUpdate(id);
174 }
175 }
176 }
