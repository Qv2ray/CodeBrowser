10OutboundNodeModel::OutboundNodeModel(std::shared_ptr<NodeDispatcher> _dispatcher, std::shared_ptr<node_data_t> data) : NodeDataModel()
11{
12 dataptr = data;
13 dispatcher = _dispatcher;
14 switch (data->metaType)
15 {
16 case complex::METAOUTBOUND_ORIGINAL:
17 case complex::METAOUTBOUND_EXTERNAL:
18 {
19 widget = new InboundOutboundWidget(NODE_OUTBOUND, dispatcher);
20 ((InboundOutboundWidget *) widget)->setValue(data);
21 break;
22 }
23 case complex::METAOUTBOUND_BALANCER:
24 {
25 widget = new BalancerWidget(dispatcher);
26 ((BalancerWidget *) widget)->setValue(data);
27 break;
28 }
29 case complex::METAOUTBOUND_CHAIN:
30 {
31 widget = new ChainWidget(dispatcher);
32 ((ChainWidget *) widget)->setValue(data);
33 break;
34 }
35 }
36 connect(widget, &QvNodeWidget::OnSizeUpdated, this, &OutboundNodeModel::embeddedWidgetSizeUpdated);
37 widget->setWindowFlags(Qt::FramelessWindowHint);
38 widget->setAttribute(Qt::WA_TranslucentBackground);
39}

45void OutboundNodeModel::setInData(std::vector<std::shared_ptr<NodeData>> indata, PortIndex)
46{
47 if (dispatcher->IsNodeConstructing())
48 return;
49 for (const auto &d : indata)
50 {
51 if (!d)
52 {
53 LOG("Invalid inbound nodedata to rule.");
54 continue;
55 }
56 const auto rule = static_cast<RuleNodeData *>(d.get());
57 if (!rule)
58 {
59 LOG("Invalid rule nodedata to outbound.");
60 return;
61 }
62 const auto rulePtr = rule->GetData();
63 //
64 if (dataptr->metaType == METAOUTBOUND_BALANCER)
65 rulePtr->balancerTag = dataptr->getDisplayName();
66 else
67 rulePtr->outboundTag = dataptr->getDisplayName();
68 //
69 DEBUG("Connecting rule:", *rulePtr->QV2RAY_RULE_TAG, "to", dataptr->getDisplayName());
70 }
71}

74void OutboundNodeModel::onNodeHoverEnter()
75{
76 if (dataptr->metaType == METAOUTBOUND_ORIGINAL)
77 {
78 emit dispatcher->OnInboundOutboundNodeHovered(dataptr->getDisplayName(), GetOutboundInfo(dataptr->realOutbound));
79 }
80 else if (dataptr->metaType == METAOUTBOUND_EXTERNAL)
81 {
82 emit dispatcher->OnInboundOutboundNodeHovered(dataptr->getDisplayName(), GetConnectionInfo(dataptr->connectionId));
83 }
84}
