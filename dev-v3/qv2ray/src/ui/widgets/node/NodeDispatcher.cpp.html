21std::tuple<QMap<QString, INBOUND>, QMap<QString, RuleObject>, QMap<QString, OutboundObjectMeta>> NodeDispatcher::GetData() const
22{
23 QMap<QString, INBOUND> _inbounds;
24 QMap<QString, RuleObject> _rules;
25 QMap<QString, OutboundObjectMeta> _outbounds;
26 for (const auto &[key, val] : inbounds.toStdMap())
27 _inbounds[key] = *val;
28 for (const auto &[key, val] : rules.toStdMap())
29 _rules[key] = *val;
30 for (const auto &[key, val] : outbounds.toStdMap())
31 _outbounds[key] = *val;
32 return std::make_tuple(_inbounds, _rules, _outbounds);
33}

35void NodeDispatcher::LoadFullConfig(const CONFIGROOT &root)
36{
37 isOperationLocked = true;
38 // Show connections in the node graph
39 for (const auto &in : root["inbounds"].toArray())
40 {
41 auto inObject = in.toObject();
42 if (inObject["tag"].toString().isEmpty())
43 inObject["tag"] = GenerateRandomString();
44 auto _ = CreateInbound(INBOUND(in.toObject()));
45 }
46
47 for (const auto &out : root["outbounds"].toArray())
48 {
49 const auto meta = OutboundObjectMeta::loadFromOutbound(OUTBOUND(out.toObject()));
50 if (meta.metaType == METAOUTBOUND_ORIGINAL && meta.realOutbound["tag"].toString().isEmpty())
51 meta.realOutbound["tag"] = GenerateRandomString();
52 auto _ = CreateOutbound(meta);
53 }
54
55 for (const auto &item : root["routing"].toObject()["rules"].toArray())
56 {
57 auto _ = CreateRule(RuleObject(item.toObject()));
58 }
59
60 for (const auto &balancer : root["routing"].toObject()["balancers"].toArray())
61 {
62 const auto array = balancer.toObject()["selector"].toArray();
63 QStringList selector;
64 for (const auto &item : array)
65 {
66 selector << item.toString();
67 }
68 const auto meta = make_balancer_outbound(selector, balancer.toObject()["tag"].toString());
69 auto _ = CreateOutbound(meta);
70 }
71
72 for (const auto &rule : rules)
73 {
74 if (!ruleNodes.contains(rule->QV2RAY_RULE_TAG))
75 {
76 LOG("Could not find rule: ", *rule->QV2RAY_RULE_TAG);
77 continue;
78 }
79 const auto ruleNodeId = ruleNodes[rule->QV2RAY_RULE_TAG];
80 // Process inbounds.
81 for (const auto &inboundTag : *rule->inboundTag)
82 {
83 if (!inboundNodes.contains(inboundTag))
84 {
85 LOG("Could not find inbound: ", inboundTag);
86 continue;
87 }
88 const auto inboundNodeId = inboundNodes.value(inboundTag);
89 ruleScene->createConnection(*ruleScene->node(ruleNodeId), 0, *ruleScene->node(inboundNodeId), 0);
90 }
91
92 const auto &outboundTag = rule->outboundTag->isEmpty() ? rule->balancerTag : rule->outboundTag;
93 if (!outboundNodes.contains(outboundTag))
94 {
95 LOG("Could not find outbound: ", *outboundTag);
96 continue;
97 }
98 const auto &outboundNodeId = outboundNodes[outboundTag];
99 ruleScene->createConnection(*ruleScene->node(outboundNodeId), 0, *ruleScene->node(ruleNodeId), 0);
100 }
101 isOperationLocked = false;
102 emit OnFullConfigLoadCompleted();
103}

105void NodeDispatcher::OnNodeDeleted(const QtNodes::Node &node)
106{
107 if (isOperationLocked)
108 return;
109 const auto nodeId = node.id();
110 bool isInbound = inboundNodes.values().contains(nodeId);
111 bool isOutbound = outboundNodes.values().contains(nodeId);
112 bool isRule = ruleNodes.values().contains(nodeId);
113 // Lots of duplicated checks.
114 {
115 Q_ASSERT_X(isInbound || isOutbound || isRule, "Delete Node", "Node type error.");
116 if (isInbound)
117 Q_ASSERT_X(!isOutbound && !isRule, "Delete Node", "Node Type Error");
118 if (isOutbound)
119 Q_ASSERT_X(!isInbound && !isRule, "Delete Node", "Node Type Error");
120 if (isRule)
121 Q_ASSERT_X(!isInbound && !isOutbound, "Delete Node", "Node Type Error");
122 }
123
124#define CLEANUP(type) \
125 if (!type##Nodes.values().contains(nodeId)) \
126 { \
127 LOG("Could not find a " #type " with id: " + nodeId.toString()); \
128 return; \
129 } \
130 const auto type##Tag = type##Nodes.key(nodeId); \
131 const auto type = type##s.value(type##Tag); \
132 \
133 type##Nodes.remove(type##Tag); \
134 type##s.remove(type##Tag);
135
136 if (isInbound)
137 {
138 CLEANUP(inbound);
139 }
140 else if (isOutbound)
141 {
142 CLEANUP(outbound);
143 const auto object = *outbound;
144 if (object.metaType == METAOUTBOUND_CHAIN)
145 {
146 emit OnChainedOutboundDeleted(object);
147 }
148 emit OnOutboundDeleted(object);
149 }
150 else if (isRule)
151 {
152 CLEANUP(rule);
153 emit OnRuleDeleted(*rule);
154 }
155 else
156 {
157 Q_UNREACHABLE();
158 }
159#undef CLEANUP
160}

162QString NodeDispatcher::CreateInbound(INBOUND in)
163{
164 auto tag = getTag(in);
165 while (inbounds.contains(tag))
166 {
167 tag += "_" + GenerateRandomString(5);
168 }
169 in["tag"] = tag;
170 auto dataPtr = std::make_shared<INBOUND>(in);
171 inbounds.insert(tag, dataPtr);
172 // Create node and emit signals.
173 {
174 auto nodeData = std::make_unique<InboundNodeModel>(shared_from_this(), dataPtr);
175 auto &node = ruleScene->createNode(std::move(nodeData));
176 inboundNodes.insert(tag, node.id());
177 emit OnInboundCreated(dataPtr, node);
178 }
179 return tag;
180}

182QString NodeDispatcher::CreateOutbound(OutboundObjectMeta out)
183{
184 QString tag = out.getDisplayName();
185 // In case the tag is duplicated:
186 while (outbounds.contains(tag))
187 {
188 tag += "_" + GenerateRandomString(5);
189 // It's ok to set them directly without checking.
190 out.displayName = tag;
191 out.realOutbound["tag"] = tag;
192 }
193 auto dataPtr = std::make_shared<OutboundObjectMeta>(out);
194 outbounds.insert(tag, dataPtr);
195 // Create node and emit signals.
196 {
197 auto nodeData = std::make_unique<OutboundNodeModel>(shared_from_this(), dataPtr);
198 auto &node = ruleScene->createNode(std::move(nodeData));
199 outboundNodes.insert(tag, node.id());
200 emit OnOutboundCreated(dataPtr, node);
201 }
202
203 // Create node and emit signals.
204 if (dataPtr->metaType == METAOUTBOUND_EXTERNAL || dataPtr->metaType == METAOUTBOUND_ORIGINAL)
205 {
206 auto nodeData = std::make_unique<ChainOutboundNodeModel>(shared_from_this(), dataPtr);
207 auto &node = chainScene->createNode(std::move(nodeData));
208 emit OnChainedOutboundCreated(dataPtr, node);
209 }
210 else if (dataPtr->metaType == METAOUTBOUND_CHAIN)
211 {
212 emit OnChainedCreated(dataPtr);
213 }
214 else
215 {
216 LOG("Ignored non-connection outbound for Chain Editor.");
217 }
218
219 return tag;
220}
