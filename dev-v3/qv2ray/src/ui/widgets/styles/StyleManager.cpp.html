23 void QvStyleManager::ReloadStyles()
24 {
25 styles.clear();
26 styles.insert(QV2RAY_BUILT_IN_DARK_MODE_NAME, {});
27 for (const auto &key : QStyleFactory::keys())
28 {
29 LOG("Found factory style: " + key);
30 QvStyle style;
31 style.Name = key;
32 style.Type = QvStyle::QVSTYLE_FACTORY;
33 styles.insert(key, style);
34 }
35
36 for (const auto &styleDir : QvCoreApplication->GetAssetsPaths("uistyles"))
37 {
38 for (const auto &file : GetFileList(QDir(styleDir)))
39 {
40 QFileInfo fileInfo(styleDir + "/" + file);
41 if (fileInfo.suffix() == "css" || fileInfo.suffix() == "qss" || fileInfo.suffix() == "qvstyle")
42 {
43 LOG("Found QSS style at: \"" + fileInfo.absoluteFilePath() + "\"");
44 QvStyle style;
45 style.Name = fileInfo.baseName();
46 style.qssPath = fileInfo.absoluteFilePath();
47 style.Type = QvStyle::QVSTYLE_QSS;
48 styles.insert(style.Name, style);
49 }
50 }
51 }
52 }

54 bool QvStyleManager::ApplyStyle(const QString &style)
55 {
56 if (!styles.contains(style))
57 return false;
58 qApp->setStyle("fusion");
59 if (style == QV2RAY_BUILT_IN_DARK_MODE_NAME)
60 {
61 LOG("Applying built-in darkmode theme.");
62 // From https://forum.qt.io/topic/101391/windows-10-dark-theme/4
63 static const QColor darkColor(45, 45, 45);
64 static const QColor disabledColor(70, 70, 70);
65 static const QColor defaultTextColor(210, 210, 210);
66 //
67 QPalette darkPalette;
68 darkPalette.setColor(QPalette::Window, darkColor);
69 darkPalette.setColor(QPalette::Button, darkColor);
70 darkPalette.setColor(QPalette::AlternateBase, darkColor);
71 //
72 darkPalette.setColor(QPalette::Text, defaultTextColor);
73 darkPalette.setColor(QPalette::ButtonText, defaultTextColor);
74 darkPalette.setColor(QPalette::WindowText, defaultTextColor);
75 darkPalette.setColor(QPalette::ToolTipBase, defaultTextColor);
76 darkPalette.setColor(QPalette::ToolTipText, defaultTextColor);
77 //
78 darkPalette.setColor(QPalette::Disabled, QPalette::Text, disabledColor);
79 darkPalette.setColor(QPalette::Disabled, QPalette::WindowText, disabledColor);
80 darkPalette.setColor(QPalette::Disabled, QPalette::ButtonText, disabledColor);
81 darkPalette.setColor(QPalette::Disabled, QPalette::HighlightedText, disabledColor);
82 //
83 darkPalette.setColor(QPalette::Base, QColor(18, 18, 18));
84 darkPalette.setColor(QPalette::Link, QColor(42, 130, 218));
85 darkPalette.setColor(QPalette::Highlight, QColor(42, 130, 218));
86 //
87 darkPalette.setColor(QPalette::BrightText, Qt::red);
88 darkPalette.setColor(QPalette::HighlightedText, Qt::black);
89 qApp->setPalette(darkPalette);
90 qApp->setStyleSheet("QToolTip { color: #ffffff; background-color: #2a82da; border: 1px solid white; }");
91 return true;
92 }
93
94 const auto &s = styles[style];
95 switch (s.Type)
96 {
97 case QvStyle::QVSTYLE_QSS:
98 {
99 LOG("Applying UI QSS style: " + s.qssPath);
100 const auto content = StringFromFile(s.qssPath);
101 qApp->setStyleSheet(content);
102 break;
103 }
104 case QvStyle::QVSTYLE_FACTORY:
105 {
106 LOG("Applying UI style: " + s.Name);
107 const auto &_style = QStyleFactory::create(s.Name);
108 qApp->setPalette(_style->standardPalette());
109 qApp->setStyle(_style);
110 qApp->setStyleSheet("");
111 break;
112 }
113 default:
114 {
115 return false;
116 }
117 }
118 qApp->processEvents();
119 return true;
120 }
