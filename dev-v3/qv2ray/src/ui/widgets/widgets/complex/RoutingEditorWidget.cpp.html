9RoutingEditorWidget::RoutingEditorWidget(std::shared_ptr<NodeDispatcher> dispatcher, QWidget *parent) : QWidget(parent), dispatcher(dispatcher)
10{
11 setupUi(this);
12 QvMessageBusConnect(RoutingEditorWidget);
13 //
14 scene = new QtNodes::FlowScene(this);
15 connect(dispatcher.get(), &NodeDispatcher::OnInboundCreated, this, &RoutingEditorWidget::OnDispatcherInboundCreated);
16 connect(dispatcher.get(), &NodeDispatcher::OnOutboundCreated, this, &RoutingEditorWidget::OnDispatcherOutboundCreated);
17 connect(dispatcher.get(), &NodeDispatcher::OnRuleCreated, this, &RoutingEditorWidget::OnDispatcherRuleCreated);
18 //
19 view = new QtNodes::FlowView(scene, this);
20 view->scaleDown();
21
22 if (!viewWidget->layout())
23 {
24 // The QWidget will take ownership of layout.
25 viewWidget->setLayout(new QVBoxLayout());
26 }
27 auto l = viewWidget->layout();
28 l->addWidget(view);
29 l->setContentsMargins(0, 0, 0, 0);
30 l->setSpacing(0);
31}

85void RoutingEditorWidget::on_delBtn_clicked()
86{
87 if (scene->selectedNodes().empty())
88 {
89 QvMessageBoxWarn(this, tr("Remove Items"), tr("Please select a node from the graph to continue."));
90 return;
91 }
92
93 const auto selecteNodes = scene->selectedNodes();
94 if (selecteNodes.empty())
95 {
96 QvMessageBoxWarn(this, tr("Deleting a node"), tr("You need to select a node first"));
97 return;
98 }
99 scene->removeNode(*selecteNodes.front());
100}
