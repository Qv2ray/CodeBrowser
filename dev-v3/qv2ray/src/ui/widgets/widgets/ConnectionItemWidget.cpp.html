21ConnectionItemWidget::ConnectionItemWidget(const ConnectionGroupPair &id, QWidget *parent) : ConnectionItemWidget(parent)
22{
23 connectionId = id.connectionId;
24 groupId = id.groupId;
25 originalItemName = GetDisplayName(id.connectionId);
26 //
27 indentSpacer->changeSize(10, indentSpacer->sizeHint().height());
28 //
29 auto latency = GetConnectionLatency(id.connectionId);
30 latencyLabel->setText(latency == LATENCY_TEST_VALUE_NODATA ? //
31 tr("Not Tested") : //
32 (latency == LATENCY_TEST_VALUE_ERROR ? //
33 tr("Error") : //
34 (QSTRN(latency) + " ms"))); //
35 //
36 connTypeLabel->setText(GetConnectionProtocolString(id.connectionId).toUpper());
37 auto [uplink, downlink] = GetConnectionUsageAmount(connectionId);
38 dataLabel->setText(FormatBytes(uplink) + " / " + FormatBytes(downlink));
39 //
40 if (ConnectionManager->IsConnected(id))
41 {
42 emit RequestWidgetFocus(this);
43 }
44 // Fake trigger
45 OnConnectionItemRenamed(id.connectionId, "", originalItemName);
46 connect(ConnectionManager, &QvConfigHandler::OnConnectionRenamed, this, &ConnectionItemWidget::OnConnectionItemRenamed);
47 //
48 // Rename events
49 connect(renameTxt, &QLineEdit::returnPressed, this, &ConnectionItemWidget::on_doRenameBtn_clicked);
50 connect(ConnectionManager, &QvConfigHandler::OnConnectionModified, this, &ConnectionItemWidget::OnConnectionModified);
51}

54ConnectionItemWidget::ConnectionItemWidget(const GroupId &id, QWidget *parent) : ConnectionItemWidget(parent)
55{
56 layout()->removeWidget(connTypeLabel);
57 layout()->removeWidget(dataLabel);
58 delete connTypeLabel;
59 delete dataLabel;
60 //
61 delete doRenameBtn;
62 delete renameTxt;
63 //
64 groupId = id;
65 originalItemName = GetDisplayName(id);
66 RecalculateConnectionsCount();
67 //
68 auto font = connNameLabel->font();
69 font.setBold(true);
70 connNameLabel->setFont(font);
71 //
72 OnGroupItemRenamed(id, "", originalItemName);
73 connect(ConnectionManager, &QvConfigHandler::OnConnectionCreated, this, &ConnectionItemWidget::RecalculateConnectionsCount);
74 connect(ConnectionManager, &QvConfigHandler::OnConnectionModified, this, &ConnectionItemWidget::RecalculateConnectionsCount);
75 connect(ConnectionManager, &QvConfigHandler::OnConnectionLinkedWithGroup, this, &ConnectionItemWidget::RecalculateConnectionsCount);
76 connect(ConnectionManager, &QvConfigHandler::OnSubscriptionAsyncUpdateFinished, this, &ConnectionItemWidget::RecalculateConnectionsCount);
77 connect(ConnectionManager, &QvConfigHandler::OnConnectionRemovedFromGroup, this, &ConnectionItemWidget::RecalculateConnectionsCount);
78 //
79 connect(ConnectionManager, &QvConfigHandler::OnGroupRenamed, this, &ConnectionItemWidget::OnGroupItemRenamed);
80}

94bool ConnectionItemWidget::NameMatched(const QString &arg) const
95{
96 auto searchString = arg.toLower();
97 auto isGroupNameMatched = GetDisplayName(groupId).toLower().contains(arg);
98
99 if (IsConnection())
100 {
101 return isGroupNameMatched || GetDisplayName(connectionId).toLower().contains(searchString);
102 }
103 else
104 {
105 return isGroupNameMatched;
106 }
107}

109void ConnectionItemWidget::RecalculateConnectionsCount()

112 latencyLabel->setText(QSTRN(connectionCount) + " " + (connectionCount < 2 ? tr("connection") : tr("connections")));

184void ConnectionItemWidget::on_doRenameBtn_clicked()
185{
186 if (renameTxt->text().isEmpty())
187 return;
188 if (connectionId == NullConnectionId)
189 ConnectionManager->RenameGroup(groupId, renameTxt->text());
190 else
191 ConnectionManager->RenameConnection(connectionId, renameTxt->text());
192 stackedWidget->setCurrentIndex(0);
193}

195void ConnectionItemWidget::OnConnectionItemRenamed(const ConnectionId &id, const QString &, const QString &newName)
196{
197 if (id == connectionId)
198 {
199 connNameLabel->setText((ConnectionManager->IsConnected({ connectionId, groupId }) ? " " : "") + newName);
200 originalItemName = newName;
201 const auto conn = ConnectionManager->GetConnectionMetaObject(connectionId);
202 this->setToolTip(newName + //
203 NEWLINE + tr("Last Connected: ") + timeToString(conn.lastConnected) + //
204 NEWLINE + tr("Last Updated: ") + timeToString(conn.lastUpdatedDate));
205 }
206}
