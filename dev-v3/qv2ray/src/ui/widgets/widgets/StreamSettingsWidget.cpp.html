31void StreamSettingsWidget::SetStreamObject(const StreamSettingsObject &sso)
32{
33 stream = sso;
34 transportCombo->setCurrentText(stream.network);
35 // TLS XTLS
36 {
37 const static QMap<QString, int> securityIndexMap{ { "none", 0 }, { "tls", 1 }, { "xtls", 2 } };
38 if (securityIndexMap.contains(stream.security))
39 securityTypeCB->setCurrentIndex(securityIndexMap[stream.security]);
40 else
41 LOG("Unsupported Security Type:", *stream.security);
42
43#define tls_xtls_process(prefix) \
44 { \
45 serverNameTxt->setText(stream.prefix##Settings->serverName); \
46 disableSessionResumptionCB->setChecked(stream.prefix##Settings->disableSessionResumption); \
47 disableSystemRoot->setChecked(stream.prefix##Settings->disableSystemRoot); \
48 alpnTxt->setText(stream.prefix##Settings->alpn->join("|")); \
49 }
50
51 tls_xtls_process(tls);
52
53 if (stream.security == "xtls")
54 tls_xtls_process(xtls);
55 }
56 // TCP
57 {
58 tcpHeaderTypeCB->setCurrentText(stream.tcpSettings->header->type);
59 tcpRequestTxt->setPlainText(JsonToString(stream.tcpSettings->header->request.toJson()));
60 tcpRespTxt->setPlainText(JsonToString(stream.tcpSettings->header->response.toJson()));
61 }
62 // HTTP
63 {
64 httpHostTxt->setPlainText(stream.httpSettings->host->join(NEWLINE));
65 httpPathTxt->setText(stream.httpSettings->path);
66 }
67 // WS
68 {
69 wsPathTxt->setText(stream.wsSettings->path);
70 QString wsHeaders;
71 for (const auto &[key, value] : stream.wsSettings->headers->toStdMap())
72 {
73 wsHeaders = wsHeaders % key % "|" % value % NEWLINE;
74 }
75 wsHeadersTxt->setPlainText(wsHeaders);
76 wsEarlyDataSB->setValue(stream.wsSettings->maxEarlyData);
77 wsBrowserForwardCB->setChecked(stream.wsSettings->useBrowserForwarding);
78 }
79 // mKCP
80 {
81 kcpMTU->setValue(stream.kcpSettings->mtu);
82 kcpTTI->setValue(stream.kcpSettings->tti);
83 kcpHeaderType->setCurrentText(stream.kcpSettings->header->type);
84 kcpCongestionCB->setChecked(stream.kcpSettings->congestion);
85 kcpReadBufferSB->setValue(stream.kcpSettings->readBufferSize);
86 kcpUploadCapacSB->setValue(stream.kcpSettings->uplinkCapacity);
87 kcpDownCapacitySB->setValue(stream.kcpSettings->downlinkCapacity);
88 kcpWriteBufferSB->setValue(stream.kcpSettings->writeBufferSize);
89 kcpSeedTxt->setText(stream.kcpSettings->seed);
90 }
91 // DS
92 {
93 dsPathTxt->setText(stream.dsSettings->path);
94 }
95 // QUIC
96 {
97 quicKeyTxt->setText(stream.quicSettings->key);
98 quicSecurityCB->setCurrentText(stream.quicSettings->security);
99 quicHeaderTypeCB->setCurrentText(stream.quicSettings->header->type);
100 }
101 // gRPC
102 {
103 grpcServiceNameTxt->setText(stream.grpcSettings->serviceName);
104 }
105 // SOCKOPT
106 {
107 tProxyCB->setCurrentText(stream.sockopt->tproxy);
108 tcpFastOpenCB->setChecked(stream.sockopt->tcpFastOpen);
109 soMarkSpinBox->setValue(stream.sockopt->mark);
110 }
111}

118void StreamSettingsWidget::on_httpHostTxt_textChanged()
119{
120 const auto hosts = httpHostTxt->toPlainText().replace("\r", "").split("\n");
121 stream.httpSettings->host->clear();
122 for (const auto &host : hosts)
123 {
124 if (!host.trimmed().isEmpty())
125 stream.httpSettings->host->push_back(host.trimmed());
126 }
127}

129void StreamSettingsWidget::on_wsHeadersTxt_textChanged()
130{
131 const auto headers = SplitLines(wsHeadersTxt->toPlainText());
132 stream.wsSettings->headers->clear();
133 for (const auto &header : headers)
134 {
135 if (header.isEmpty())
136 continue;
137
138 if (!header.contains("|"))
139 {
140 LOG("Header missing '|' separator");
141 RED(wsHeadersTxt);
142 return;
143 }
144
145 const auto index = header.indexOf("|");
146 auto key = header.left(index);
147 auto value = header.right(header.length() - index - 1);
148 stream.wsSettings->headers->insert(key, value);
149 }
150 BLACK(wsHeadersTxt);
151}
