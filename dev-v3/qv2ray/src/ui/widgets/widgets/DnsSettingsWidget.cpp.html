39DnsSettingsWidget::DnsSettingsWidget(QWidget *parent) : QWidget(parent)
40{
41 setupUi(this);
42 QvMessageBusConnect(DnsSettingsWidget);
43 //
44 auto sourceStringsDomain = ReadGeoSiteFromFile(GlobalConfig.kernelConfig->AssetsPath() + "/geosite.dat");
45 auto sourceStringsIP = ReadGeoSiteFromFile(GlobalConfig.kernelConfig->AssetsPath() + "/geoip.dat");
46 //
47 domainListTxt = new AutoCompleteTextEdit("geosite", sourceStringsDomain, this);
48 ipListTxt = new AutoCompleteTextEdit("geoip", sourceStringsIP, this);
49 connect(domainListTxt, &AutoCompleteTextEdit::textChanged,
50 [&]() { (*dns.servers)[currentServerIndex].domains = SplitLines(domainListTxt->toPlainText()); });
51 connect(ipListTxt, &AutoCompleteTextEdit::textChanged,
52 [&]() { (*dns.servers)[currentServerIndex].expectIPs = SplitLines(ipListTxt->toPlainText()); });
53
54 domainsLayout->addWidget(domainListTxt);
55 expectedIPsLayout->addWidget(ipListTxt);
56 detailsSettingsGB->setCheckable(true);
57 detailsSettingsGB->setChecked(false);
58 UPDATE_UI_ENABLED_STATE;
59 updateColorScheme();
60}

77void DnsSettingsWidget::SetDNSObject(const DNSObject &_dns, const FakeDNSObject &_fakeDNS)
78{
79 this->dns = _dns;
80 this->fakeDNS = _fakeDNS;
81
82 dnsClientIPTxt->setText(dns.clientIp);
83 dnsTagTxt->setText(dns.tag);
84
85 serversListbox->clear();
86 std::for_each(dns.servers->begin(), dns.servers->end(), [&](const auto &dns) { serversListbox->addItem(dns.address); });
87
88 if (serversListbox->count() > 0)
89 {
90 serversListbox->setCurrentRow(0);
91 ShowCurrentDnsServerDetails();
92 }
93
94 staticResolvedDomainsTable->clearContents();
95 for (const auto &[host, ip] : dns.hosts->toStdMap())
96 {
97 const auto rowId = staticResolvedDomainsTable->rowCount();
98 staticResolvedDomainsTable->insertRow(rowId);
99 staticResolvedDomainsTable->setItem(rowId, 0, new QTableWidgetItem(host));
100 staticResolvedDomainsTable->setItem(rowId, 1, new QTableWidgetItem(ip));
101 }
102 staticResolvedDomainsTable->resizeColumnsToContents();
103
104 dnsQueryStrategyCB->setCurrentText(dns.queryStrategy);
105 dnsDisableFallbackCB->setChecked(dns.disableFallback);
106 dnsDisableCacheCB->setChecked(dns.disableCache);
107
108 fakeDNSIPPool->setCurrentText(fakeDNS.ipPool);
109 fakeDNSIPPoolSize->setValue(fakeDNS.poolSize);
110 UPDATE_UI_ENABLED_STATE
111}

113bool DnsSettingsWidget::CheckIsValidDNS() const
114{
115 if (!dns.clientIp->isEmpty() && !IsValidIPAddress(dns.clientIp))
116 return false;
117 for (const auto &server : *dns.servers)
118 {
119 if (!IsValidDNSServer(server.address))
120 return false;
121 }
122 return true;
123}

155std::pair<DNSObject, FakeDNSObject> DnsSettingsWidget::GetDNSObject()
156{
157 dns.hosts->clear();
158 for (auto i = 0; i < staticResolvedDomainsTable->rowCount(); i++)
159 {
160 const auto &item1 = staticResolvedDomainsTable->item(i, 0);
161 const auto &item2 = staticResolvedDomainsTable->item(i, 1);
162 if (item1 && item2)
163 dns.hosts->insert(item1->text(), item2->text());
164 }
165 return { dns, fakeDNS };
166}

189void DnsSettingsWidget::on_removeServerBtn_clicked()
190{
191 dns.servers->removeAt(currentServerIndex);
192 // Block the signals
193 serversListbox->blockSignals(true);
194 auto item = serversListbox->item(currentServerIndex);
195 serversListbox->removeItemWidget(item);
196 delete item;
197 serversListbox->blockSignals(false);
198 UPDATE_UI_ENABLED_STATE
199
200 if (serversListbox->count() > 0)
201 {
202 if (currentServerIndex < 0)
203 serversListbox->setCurrentRow(0);
204 ShowCurrentDnsServerDetails();
205 }
206}
