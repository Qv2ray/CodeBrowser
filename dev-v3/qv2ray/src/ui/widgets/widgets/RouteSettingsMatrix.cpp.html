61void RouteSettingsMatrixWidget::SetRouteConfig(const QvConfig_Route &conf)
62{
63 domainStrategyCombo->setCurrentText(conf.domainStrategy);
64 domainMatcherCombo->setCurrentIndex(conf.domainMatcher == "mph" ? 1 : 0);
65 //
66 directDomainTxt->setPlainText(conf.domains->direct->join(NEWLINE));
67 proxyDomainTxt->setPlainText(conf.domains->proxy->join(NEWLINE));
68 blockDomainTxt->setPlainText(conf.domains->block->join(NEWLINE));
69 //
70 blockIPTxt->setPlainText(conf.ips->block->join(NEWLINE));
71 directIPTxt->setPlainText(conf.ips->direct->join(NEWLINE));
72 proxyIPTxt->setPlainText(conf.ips->proxy->join(NEWLINE));
73}

75QvConfig_Route RouteSettingsMatrixWidget::GetRouteConfig() const
76{
77 QvConfig_Route conf;
78 // Workaround for translation
79 const auto index = domainMatcherCombo->currentIndex();
80 conf.domainMatcher = index == 0 ? "" : "mph";
81 conf.domainStrategy = domainStrategyCombo->currentText();
82 conf.domains->block = SplitLines(blockDomainTxt->toPlainText().replace(" ", ""));
83 conf.domains->direct = SplitLines(directDomainTxt->toPlainText().replace(" ", ""));
84 conf.domains->proxy = SplitLines(proxyDomainTxt->toPlainText().replace(" ", ""));
85 //
86 conf.ips->block = SplitLines(blockIPTxt->toPlainText().replace(" ", ""));
87 conf.ips->direct = SplitLines(directIPTxt->toPlainText().replace(" ", ""));
88 conf.ips->proxy = SplitLines(proxyIPTxt->toPlainText().replace(" ", ""));
89 return conf;
90}

101void RouteSettingsMatrixWidget::on_importSchemeBtn_clicked()
102{
103 try
104 {
105 // open up the file dialog and choose a file.
106 auto filePath = this->openFileDialog();
107 if (!filePath)
108 return;
109
110 // read the file and parse back to struct.
111 // if error occurred on parsing, an exception will be thrown.
112 auto content = StringFromFile(*filePath);
113 auto scheme = Qv2rayRouteScheme(JsonFromString(content));
114
115 // show the information of this scheme to user,
116 // and ask user if he/she wants to import and apply this.
117 auto strPrompt = tr("Import scheme '%1' made by '%2'? \r\n Description: %3").arg(scheme.name, scheme.author, scheme.description);
118 auto decision = QvMessageBoxAsk(this, tr("Importing Scheme"), strPrompt);
119
120 // if user don't want to import, just leave.
121 if (decision != Yes)
122 return;
123
124 // write the scheme onto the window
125 this->SetRouteConfig(static_cast<QvConfig_Route>(scheme));
126
127 // done
128 LOG("Imported route config: " + scheme.name + " by: " + scheme.author);
129 }
130 catch (std::exception &e)
131 {
132 LOG("Exception: ", e.what());
133 // TODO: Give some error as Notification
134 }
135}

137/**
138 * @brief RouteSettingsMatrixWidget::on_exportSchemeBtn_clicked
139 * @author DuckSoft <realducksoft@gmail.com>
140 */
141void RouteSettingsMatrixWidget::on_exportSchemeBtn_clicked()
142{
143 try
144 {
145 // parse the config back from the window components
146 auto config = this->GetRouteConfig();
147
148 // init some constants
149 const auto dialogTitle = tr("Exporting Scheme");
150
151 // scheme name?
152 bool ok = false;
153 auto schemeName = QInputDialog::getText(this, dialogTitle, tr("Scheme name:"), QLineEdit::Normal, tr("Unnamed Scheme"), &ok);
154 if (!ok)
155 return;
156
157 // scheme author?
158 auto schemeAuthor = QInputDialog::getText(this, dialogTitle, tr("Author:"), QLineEdit::Normal, "Anonymous <mystery@example.com>", &ok);
159 if (!ok)
160 return;
161
162 // scheme description?
163 auto schemeDescription =
164 QInputDialog::getText(this, dialogTitle, tr("Description:"), QLineEdit::Normal, tr("The author is too lazy to leave a comment"));
165 if (!ok)
166 return;
167
168 // where to save?
169 auto savePath = this->saveFileDialog();
170 if (!savePath)
171 return;
172
173 // construct the data structure
174 Qv2rayRouteScheme scheme;
175 scheme.name = schemeName;
176 scheme.author = schemeAuthor;
177 scheme.description = schemeDescription;
178 scheme.domainStrategy = config.domainStrategy;
179 scheme.ips = config.ips;
180 scheme.domains = config.domains;
181
182 // serialize and write out
183 auto content = JsonToString(scheme.toJson());
184 StringToFile(content, *savePath);
185
186 // done
187 // TODO: Give some success as Notification
188 QvMessageBoxInfo(this, dialogTitle, tr("Your route scheme has been successfully exported!"));
189 }
190 catch (...)
191 {
192 // TODO: Give some error as Notification
193 }
194}

200 */
201std::optional<QString> RouteSettingsMatrixWidget::saveFileDialog()
202{
203 QFileDialog dialog;
204 dialog.setFileMode(QFileDialog::AnyFile);
205 dialog.setOption(QFileDialog::Option::DontConfirmOverwrite, false);
206 dialog.setNameFilter(tr("QvRoute Schemes(*.json)"));
207 dialog.setAcceptMode(QFileDialog::AcceptMode::AcceptSave);
208 if (!dialog.exec() || dialog.selectedFiles().length() != 1)
209 {
210 return std::nullopt;
211 }
212 return dialog.selectedFiles().first();
213}
