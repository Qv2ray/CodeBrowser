17InboundEditor::InboundEditor(INBOUND source, QWidget *parent) : QDialog(parent), original(source)
18{
19 QvMessageBusConnect(InboundEditor);
20 setupUi(this);
21 streamSettingsWidget = new StreamSettingsWidget(this);
22 streamSettingsWidget->SetStreamObject({});
23 if (!transportFrame->layout())
24 {
25 auto l = new QGridLayout();
26 l->setHorizontalSpacing(0);
27 l->setVerticalSpacing(0);
28 transportFrame->setLayout(l);
29 }
30 transportFrame->layout()->addWidget(streamSettingsWidget);
31 this->current = source;
32 inboundProtocol = current["protocol"].toString("http");
33 allocateSettings = current["allocate"].toObject();
34 sniffingSettings = current["sniffing"].toObject();
35
36 isLoading = true;
37 for (const auto &name : PluginHost->UsablePlugins())
38 {
39 const auto &plugin = PluginHost->GetPlugin(name);
40 if (!plugin->hasComponent(COMPONENT_GUI))
41 continue;
42
43 const auto guiInterface = plugin->pluginInterface->GetGUIInterface();
44
45 if (!guiInterface)
46 {
47 LOG("Found a plugin with COMPONENT_GUI but returns an invalid GUI interface: ", plugin->metadata.Name);
48 continue;
49 }
50
51 if (!guiInterface->GetComponents().contains(GUI_COMPONENT_INBOUND_EDITOR))
52 continue;
53
54 const auto editors = guiInterface->GetInboundEditors();
55 for (const auto &editorInfo : editors)
56 {
57 inboundProtocolCombo->addItem(editorInfo.first.displayName, editorInfo.first.protocol);
58 stackedWidget->addWidget(editorInfo.second);
59 pluginWidgets.insert(editorInfo.first.protocol, editorInfo.second);
60 }
61 }
62 inboundProtocolCombo->model()->sort(0);
63 isLoading = false;
64 loadUI();
65}

84INBOUND InboundEditor::getResult()
85{
86 INBOUND newRoot = current;
87 for (const auto &[protocol, widget] : pluginWidgets.toStdMap())
88 {
89 if (protocol == inboundProtocol)
90 {
91 newRoot["settings"] = INBOUNDSETTING(widget->GetContent());
92 break;
93 }
94 }
95 if (streamSettingsWidget->isEnabled())
96 {
97 newRoot["streamSettings"] = streamSettingsWidget->GetStreamSettings().toJson();
98 }
99 newRoot["protocol"] = inboundProtocol;
100 newRoot["sniffing"] = sniffingSettings;
101 newRoot["allocate"] = allocateSettings;
102 return newRoot;
103}

105void InboundEditor::loadUI()
106{
107 isLoading = true;
108 streamSettingsWidget->SetStreamObject(StreamSettingsObject(original["streamSettings"].toObject()));
109 {
110 inboundTagTxt->setText(current["tag"].toString());
111 inboundHostTxt->setText(current["listen"].toString());
112 inboundPortTxt->setText(current["port"].toVariant().toString());
113 }
114 {
115 const auto allocationStrategy = allocateSettings["strategy"].toString();
116 allocateSettings["strategy"] = allocationStrategy.isEmpty() ? "always" : allocationStrategy;
117 strategyCombo->setCurrentText(allocationStrategy);
118 refreshNumberBox->setValue(allocateSettings["refresh"].toInt());
119 concurrencyNumberBox->setValue(allocateSettings["concurrency"].toInt());
120 }
121 {
122 sniffingGroupBox->setChecked(sniffingSettings["enabled"].toBool());
123 sniffMetaDataOnlyCB->setChecked(sniffingSettings["metadataOnly"].toBool());
124 const auto data = sniffingSettings["destOverride"].toArray();
125 sniffHTTPCB->setChecked(data.contains("http"));
126 sniffTLSCB->setChecked(data.contains("tls"));
127 sniffFakeDNSCB->setChecked(data.contains("fakedns"));
128 sniffFakeDNSOtherCB->setChecked(data.contains("fakedns+others"));
129 }
130 bool processed = false;
131 const auto settings = current["settings"].toObject();
132 for (const auto &[protocol, widget] : pluginWidgets.toStdMap())
133 {
134 if (protocol == inboundProtocol)
135 {
136 widget->SetContent(settings);
137 inboundProtocolCombo->setCurrentIndex(inboundProtocolCombo->findData(protocol));
138 processed = true;
139 break;
140 }
141 }
142
143 if (!processed)
144 {
145 LOG("Inbound protocol: " + inboundProtocol + " is not supported.");
146 QvMessageBoxWarn(this, tr("Unknown inbound."),
147 tr("The specified inbound type is invalid, this may be caused by a plugin failure.") + NEWLINE +
148 tr("Please use the JsonEditor or reload the plugin."));
149 reject();
150 }
151 isLoading = false;
152 on_stackedWidget_currentChanged(0);
153}
