218CONFIGROOT RouteEditor::OpenEditor()
219{
220 auto result = this->exec();
221 if (result != QDialog::Accepted)
222 return original;
223
224 const auto &[inbounds, rules, outbounds] = nodeDispatcher->GetData();
225 //
226 // Inbounds
227 QJsonArray inboundsJson;
228 for (const auto &in : inbounds)
229 {
230 inboundsJson << in;
231 }
232 root["inbounds"] = inboundsJson;
233 //
234 // QJsonArray rules
235 QJsonArray rulesJsonArray;
236 for (auto i = 0; i < ruleListWidget->count(); i++)
237 {
238 // Sorted
239 const auto ruleTag = ruleListWidget->item(i)->text();
240 if (rules.contains(ruleTag))
241 {
242 const auto &ruleObject = rules[ruleTag];
243 auto ruleJson = ruleObject.toJson();
244 if (ruleJson["outboundTag"].toString().isEmpty())
245 ruleJson.remove("outboundTag");
246 else
247 ruleJson.remove("balancerTag");
248 rulesJsonArray << ruleJson;
249 }
250 else
251 {
252 LOG("Could not find rule tag:", ruleTag);
253 }
254 }
255 //
256 // QJsonArray balancers
257 QJsonArray balancersArray;
258 for (const auto &out : outbounds)
259 {
260 if (out.metaType != METAOUTBOUND_BALANCER)
261 continue;
262 BalancerObject o;
263 o.tag = out.getDisplayName();
264 o.selector = out.outboundTags;
265 balancersArray << o.toJson();
266 o.strategy->type = out.strategyType;
267 }
268
269 QJsonObject routingObject;
270 routingObject["domainStrategy"] = domainStrategy;
271 routingObject["rules"] = rulesJsonArray;
272 routingObject["balancers"] = balancersArray;
273 root["routing"] = routingObject;
274
275 // QJsonArray Outbounds
276 QJsonArray outboundsArray;
277
278 for (const auto &out : outbounds)
279 {
280 QJsonObject outboundJsonObject;
281 if (out.metaType == METAOUTBOUND_BALANCER)
282 continue;
283 if (out.metaType == METAOUTBOUND_ORIGINAL)
284 {
285 outboundJsonObject = out.realOutbound;
286 }
287 outboundJsonObject[META_OUTBOUND_KEY_NAME] = out.toJson();
288 const auto displayName = out.getDisplayName();
289 if (displayName == defaultOutboundTag)
290 outboundsArray.prepend(outboundJsonObject);
291 else
292 outboundsArray.append(outboundJsonObject);
293 }
294 root["outbounds"] = outboundsArray;
295 // Process DNS
296 const auto &[dns, fakedns] = dnsWidget->GetDNSObject();
297 root["dns"] = GenerateDNS(dns);
298 root["fakedns"] = fakedns.toJson();
299 {
300 // Process Browser Forwarder
301 QJsonObject browserForwarder;
302 browserForwarder["listenAddr"] = bfListenIPTxt->text();
303 browserForwarder["listenPort"] = bfListenPortTxt->value();
304 root["browserForwarder"] = browserForwarder;
305 }
306 {
307 // Process Observatory
308 QJsonObject observatory;
309 observatory["subjectSelector"] = QJsonArray::fromStringList(SplitLines(obSubjectSelectorTxt->toPlainText()));
310 root["observatory"] = observatory;
311 }
312 return root;
313}

331void RouteEditor::on_addDefaultBtn_clicked()
332{
333 LOADINGCHECK
334 // Add default connection from GlobalConfig
335 //
336 const auto &inboundConfig = GlobalConfig.inboundConfig;
337 const static QJsonObject sniffingOff{ { "enabled", false } };
338 const static QJsonObject sniffingOn{ { "enabled", true }, { "destOverride", QJsonArray{ "http", "tls" } } };
339 //
340 if (inboundConfig->useHTTP)
341 {
342 const auto httpSettings = GenerateHTTPIN(inboundConfig->httpSettings->useAuth, { inboundConfig->httpSettings->account });
343 const auto httpConfig = GenerateInboundEntry("GlobalConfig-HTTP", "http", //
344 inboundConfig->listenip, //
345 inboundConfig->httpSettings->port, //
346 httpSettings, //
347 inboundConfig->httpSettings->sniffing ? sniffingOn : sniffingOff);
348 const auto _ = nodeDispatcher->CreateInbound(httpConfig);
349 Q_UNUSED(_)
350 }
351 if (inboundConfig->useSocks)
352 {
353 const auto socks = GenerateSocksIN((inboundConfig->socksSettings->useAuth ? "password" : "noauth"), //
354 { inboundConfig->socksSettings->account }, //
355 inboundConfig->socksSettings->enableUDP, //
356 inboundConfig->socksSettings->localIP);
357 const auto socksConfig = GenerateInboundEntry("GlobalConfig-Socks", "socks", //
358 inboundConfig->listenip, //
359 inboundConfig->socksSettings->port, //
360 socks, //
361 (inboundConfig->socksSettings->sniffing ? sniffingOn : sniffingOff));
362 const auto _ = nodeDispatcher->CreateInbound(socksConfig);
363 }
364
365 if (inboundConfig->useTPROXY)
366 {
367 QList<QString> networks;
368#define ts inboundConfig->tProxySettings
369 if (ts->hasTCP)
370 networks << "tcp";
371 if (ts->hasUDP)
372 networks << "udp";
373 const auto tproxy_network = networks.join(",");
374 auto tproxyInSettings = GenerateDokodemoIN("", 0, tproxy_network, 0, true);
375 //
376 const static QJsonObject tproxy_sniff{ { "enabled", true }, { "destOverride", QJsonArray{ "http", "tls" } } };
377 const QJsonObject tproxy_streamSettings{ { "sockopt", QJsonObject{ { "tproxy", *ts->mode } } } };
378 {
379 auto tProxyIn = GenerateInboundEntry("tProxy IPv4", "dokodemo-door", ts->tProxyIP, ts->port, tproxyInSettings);
380 tProxyIn.insert("sniffing", tproxy_sniff);
381 tProxyIn.insert("streamSettings", tproxy_streamSettings);
382 auto _ = nodeDispatcher->CreateInbound(tProxyIn);
383 Q_UNUSED(_)
384 }
385 if (!ts->tProxyV6IP->isEmpty())
386 {
387 auto tProxyV6In = GenerateInboundEntry("tProxy IPv6", "dokodemo-door", ts->tProxyV6IP, ts->port, tproxyInSettings);
388 tProxyV6In.insert("sniffing", tproxy_sniff);
389 tProxyV6In.insert("streamSettings", tproxy_streamSettings);
390 auto _ = nodeDispatcher->CreateInbound(tProxyV6In);
391 Q_UNUSED(_)
392 }
393#undef ts
394 }
395}

468 ImportConnection(connId);

471 }
472 else
473 {
474 ImportConnection(cid);
475 }

538void RouteEditor::on_importOutboundBtn_clicked()
539{
540 LOADINGCHECK
541 ImportConfigWindow w(this);
542 // True here for not keep the inbounds.
543 auto configs = w.SelectConnection(true);
544
545 for (auto i = 0; i < configs.count(); i++)
546 {
547 const auto conf = configs.values()[i];
548 const auto name = configs.key(conf, "");
549
550 if (name.isEmpty())
551 continue;
552
553 // conf is rootObject, needs to unwrap it.
554 const auto confList = conf["outbounds"].toArray();
555
556 for (int i = 0; i < confList.count(); i++)
557 {
558 auto _ = nodeDispatcher->CreateOutbound(make_normal_outbound(OUTBOUND(confList[i].toObject())));
559 Q_UNUSED(_)
