6JsonEditor::JsonEditor(QJsonObject rootObject, QWidget *parent) : QDialog(parent)
7{
8 setupUi(this);
9 QvMessageBusConnect(JsonEditor);
10 //
11 original = rootObject;
12 final = rootObject;
13 QString jsonString = JsonToString(rootObject);
14
15 if (VerifyJsonString(jsonString).isEmpty())
16 {
17 jsonTree->setModel(&model);
18 model.loadJson(QJsonDocument(rootObject).toJson());
19 }
20 else
21 {
22 QvMessageBoxWarn(this, tr("Json Contains Syntax Errors"), tr("Original Json may contain syntax errors. Json tree is disabled."));
23 }
24
25 jsonEditor->setText(JsonToString(rootObject));
26 jsonTree->expandAll();
27 jsonTree->resizeColumnToContents(0);
28}

41QJsonObject JsonEditor::OpenEditor()
42{
43 int resultCode = this->exec();
44 auto string = jsonEditor->toPlainText();
45
46 while (resultCode == QDialog::Accepted && !VerifyJsonString(string).isEmpty())
47 {
48 QvMessageBoxWarn(this, tr("Json Contains Syntax Errors"), tr("You must correct these errors before continuing."));
49 resultCode = this->exec();
50 string = jsonEditor->toPlainText();
51 }
52
53 return resultCode == QDialog::Accepted ? final : original;
54}

60void JsonEditor::on_jsonEditor_textChanged()
61{
62 auto string = jsonEditor->toPlainText();
63 auto VerifyResult = VerifyJsonString(string);
64 jsonValidateStatus->setText(VerifyResult);
65
66 if (VerifyResult.isEmpty())
67 {
68 BLACK(jsonEditor);
69 final = JsonFromString(string);
70 model.loadJson(QJsonDocument(final).toJson());
71 jsonTree->expandAll();
72 jsonTree->resizeColumnToContents(0);
73 }
74 else
75 {
76 RED(jsonEditor);

80void JsonEditor::on_formatJsonBtn_clicked()
81{
82 auto string = jsonEditor->toPlainText();
83 auto VerifyResult = VerifyJsonString(string);
84 jsonValidateStatus->setText(VerifyResult);
85
86 if (VerifyResult.isEmpty())
87 {
88 BLACK(jsonEditor);
89 jsonEditor->setPlainText(JsonToString(JsonFromString(string)));
90 model.loadJson(QJsonDocument(JsonFromString(string)).toJson());
91 jsonTree->setModel(&model);
92 jsonTree->expandAll();
93 jsonTree->resizeColumnToContents(0);
94 }
95 else
96 {
97 RED(jsonEditor);
98 QvMessageBoxWarn(this, tr("Syntax Errors"), tr("Please fix the JSON errors or remove the comments before continue"));
99 }
