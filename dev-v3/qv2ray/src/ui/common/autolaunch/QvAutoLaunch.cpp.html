31 bool GetLaunchAtLoginStatus()
32 {
33#ifdef Q_OS_WIN
34 QString appName = QCoreApplication::applicationName();
35 QSettings reg("HKEY_CURRENT_USER\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run", QSettings::NativeFormat);
36 return reg.contains(appName);
37 }
38
39#elif defined Q_OS_MAC
40 // From
41 // https://github.com/nextcloud/desktop/blob/master/src/common/utility_mac.cpp
42 // this is quite some duplicate code with setLaunchOnStartup, at some
43 // point we should fix this FIXME.
44 bool returnValue = false;
45 QString filePath = QDir(QCoreApplication::applicationDirPath() + QLatin1String("/../..")).absolutePath();
46 CFStringRef folderCFStr = CFStringCreateWithCString(0, filePath.toUtf8().data(), kCFStringEncodingUTF8);
47 CFURLRef urlRef = CFURLCreateWithFileSystemPath(0, folderCFStr, kCFURLPOSIXPathStyle, true);
48 LSSharedFileListRef loginItems = LSSharedFileListCreate(0, kLSSharedFileListSessionLoginItems, 0);
49
50 if (loginItems)
51 {
52 // We need to iterate over the items and check which one is "ours".
53 UInt32 seedValue;
54 CFArrayRef itemsArray = LSSharedFileListCopySnapshot(loginItems, &seedValue);
55 CFStringRef appUrlRefString = CFURLGetString(urlRef); // no need for release
56
57 for (int i = 0; i < CFArrayGetCount(itemsArray); i++)
58 {
59 LSSharedFileListItemRef item = (LSSharedFileListItemRef) CFArrayGetValueAtIndex(itemsArray, i);
60 CFURLRef itemUrlRef = NULL;
61
62 if (LSSharedFileListItemResolve(item, 0, &itemUrlRef, NULL) == noErr && itemUrlRef)
63 {
64 CFStringRef itemUrlString = CFURLGetString(itemUrlRef);
65
66 if (CFStringCompare(itemUrlString, appUrlRefString, 0) == kCFCompareEqualTo)
67 {
68 returnValue = true;
69 }
70
71 CFRelease(itemUrlRef);
72 }
73 }
74
75 CFRelease(itemsArray);
76 }
77
78 CFRelease(loginItems);
79 CFRelease(folderCFStr);
80 CFRelease(urlRef);
81 return returnValue;
82 }

85 QString appName = QCoreApplication::applicationName();
86 QString desktopFileLocation = getUserAutostartDir_private() + appName + QLatin1String(".desktop");
87 return QFile::exists(desktopFileLocation);
88 }

91 void SetLaunchAtLoginStatus(bool enable)
92 {
93#ifdef Q_OS_WIN
94 QString appName = QCoreApplication::applicationName();
95 QSettings reg("HKEY_CURRENT_USER\\SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run", QSettings::NativeFormat);
96
97 if (enable)
98 {
99 QString strAppPath = QDir::toNativeSeparators(QCoreApplication::applicationFilePath());
100 reg.setValue(appName, strAppPath);
101 }
102 else
103 {
104 reg.remove(appName);
105 }
106 }
107
108#elif defined Q_OS_MAC
109 // From
110 // https://github.com/nextcloud/desktop/blob/master/src/common/utility_mac.cpp
111 QString filePath = QDir(QCoreApplication::applicationDirPath() + QLatin1String("/../..")).absolutePath();
112 CFStringRef folderCFStr = CFStringCreateWithCString(0, filePath.toUtf8().data(), kCFStringEncodingUTF8);
113 CFURLRef urlRef = CFURLCreateWithFileSystemPath(0, folderCFStr, kCFURLPOSIXPathStyle, true);
114 LSSharedFileListRef loginItems = LSSharedFileListCreate(0, kLSSharedFileListSessionLoginItems, 0);
115
116 if (loginItems && enable)
117 {
118 // Insert an item to the list.
119 LSSharedFileListItemRef item = LSSharedFileListInsertItemURL(loginItems, kLSSharedFileListItemLast, 0, 0, urlRef, 0, 0);
120
121 if (item)
122 CFRelease(item);
123
124 CFRelease(loginItems);
125 }
126 else if (loginItems && !enable)
127 {
128 // We need to iterate over the items and check which one is "ours".
129 UInt32 seedValue;
130 CFArrayRef itemsArray = LSSharedFileListCopySnapshot(loginItems, &seedValue);
131 CFStringRef appUrlRefString = CFURLGetString(urlRef);
132
133 for (int i = 0; i < CFArrayGetCount(itemsArray); i++)
134 {
135 LSSharedFileListItemRef item = (LSSharedFileListItemRef) CFArrayGetValueAtIndex(itemsArray, i);
136 CFURLRef itemUrlRef = NULL;
137
138 if (LSSharedFileListItemResolve(item, 0, &itemUrlRef, NULL) == noErr && itemUrlRef)
139 {
140 CFStringRef itemUrlString = CFURLGetString(itemUrlRef);
141
142 if (CFStringCompare(itemUrlString, appUrlRefString, 0) == kCFCompareEqualTo)
143 {
144 LSSharedFileListItemRemove(loginItems, item); // remove it!
145 }
146
147 CFRelease(itemUrlRef);
148 }
149 }
150
151 CFRelease(itemsArray);
152 CFRelease(loginItems);
153 }
154
155 CFRelease(folderCFStr);
156 CFRelease(urlRef);
157 }
158
159#elif defined Q_OS_LINUX
160 //
161 // For AppImage packaging.
162 auto binPath = qEnvironmentVariableIsSet("APPIMAGE") ? qEnvironmentVariable("APPIMAGE") : QCoreApplication::applicationFilePath();
163 //
164 // From https://github.com/nextcloud/desktop/blob/master/src/common/utility_unix.cpp
165 QString appName = QCoreApplication::applicationName();
166 QString userAutoStartPath = getUserAutostartDir_private();
167 QString desktopFileLocation = userAutoStartPath + appName + QLatin1String(".desktop");
168 QStringList appCmdList = QCoreApplication::arguments();
169 appCmdList.replace(0, binPath);
170
171 if (enable)
172 {
173 if (!QDir().exists(userAutoStartPath) && !QDir().mkpath(userAutoStartPath))
174 {
175 // qCWarning(lcUtility) << "Could not create autostart folder"
176 // << userAutoStartPath;
177 return;
178 }
179
180 QFile iniFile(desktopFileLocation);
181
182 if (!iniFile.open(QIODevice::WriteOnly))
183 {
184 // qCWarning(lcUtility) << "Could not write auto start entry" <<
185 // desktopFileLocation;
186 return;
187 }
188
189 QTextStream ts(&iniFile);
190#if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
191 ts.setCodec("UTF-8");
192#endif
193 ts << QLatin1String("[Desktop Entry]") << NEWLINE //
194 << QLatin1String("Name=") << appName << NEWLINE //
195 << QLatin1String("GenericName=") << QLatin1String("V2Ray Frontend") << NEWLINE //
196 << QLatin1String("Exec=") << appCmdList.join(" ") << NEWLINE //
197 << QLatin1String("Terminal=") << "false" << NEWLINE //
198 << QLatin1String("Icon=") << "qv2ray" << NEWLINE //
199 << QLatin1String("Categories=") << "Network" << NEWLINE //
200 << QLatin1String("Type=") << "Application" << NEWLINE //
201 << QLatin1String("StartupNotify=") << "false" << NEWLINE //
202 << QLatin1String("X-GNOME-Autostart-enabled=") << "true" << NEWLINE;
203 ts.flush();
204 iniFile.close();
205 }
206 else
207 {
208 QFile::remove(desktopFileLocation);
209 QFile::remove(desktopFileLocation.replace("qv2ray", "Qv2ray"));
210 }
211 }
212#endif
