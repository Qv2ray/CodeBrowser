28#pragma once
29
30#include <QGraphicsView>
31#include <QMap>
32#include <QPen>
33
34class SpeedWidget : public QGraphicsView
35{
36 Q_OBJECT
37 public:
38 enum GraphType
39 {
40 INBOUND_UP,
41 INBOUND_DOWN,
42 OUTBOUND_PROXY_UP,
43 OUTBOUND_PROXY_DOWN,
44 OUTBOUND_DIRECT_UP,
45 OUTBOUND_DIRECT_DOWN,
46 OUTBOUND_BLOCK_UP,
47 OUTBOUND_BLOCK_DOWN,
48 NB_GRAPHS,
49 };
50 struct PointData
51 {
52 qint64 x;
53 quint64 y[NB_GRAPHS];
54 PointData()
55 {
56 for (auto i = 0; i < NB_GRAPHS; i++)
57 y[i] = 0;
58 }
59 };
60
61 explicit SpeedWidget(QWidget *parent = nullptr);
62 void UpdateSpeedPlotSettings();
63 void AddPointData(QMap<SpeedWidget::GraphType, long> data);
64 void Clear();
65 void replot();
66
67 protected:
68 void paintEvent(QPaintEvent *event) override;
69
70 private:
71 struct GraphProperties
72 {
73 GraphProperties(){};
74 GraphProperties(const QString &name, const QPen &pen) : name(name), pen(pen){};
75 QString name;
76 QPen pen;
77 };
78
79 quint64 maxYValue();
80 QList<PointData> dataCollection;
81
82 QMap<GraphType, GraphProperties> m_properties;
83};
