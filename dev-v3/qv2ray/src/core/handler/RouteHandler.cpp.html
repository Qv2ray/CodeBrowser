52 bool RouteHandler::ExpandChainedOutbounds(CONFIGROOT &root) const
53 {
54 // Proxy Chain Expansion
55 const auto outbounds = root["outbounds"].toArray();
56 const auto inbounds = root["inbounds"].toArray();
57 const auto rules = root["routing"].toObject()["rules"].toArray();
58
59 QJsonArray newOutbounds, newInbounds, newRules;
60
61 QMap<QString, QJsonObject> outboundCache;
62 // First pass - Resolve Indexes (tags), build cache
63 for (const auto &singleOutboundVal : outbounds)
64 {
65 const auto outbound = singleOutboundVal.toObject();
66 const auto meta = OutboundObjectMeta::loadFromOutbound(OUTBOUND(outbound));
67 if (meta.metaType != METAOUTBOUND_CHAIN)
68 outboundCache[outbound["tag"].toString()] = outbound;
69 }
70
71 // Second pass - Build Chains
72 for (const auto &singleOutboundVal : outbounds)
73 {
74 const auto outbound = singleOutboundVal.toObject();
75 const auto meta = OutboundObjectMeta::loadFromOutbound(OUTBOUND(outbound));
76 if (meta.metaType != METAOUTBOUND_CHAIN)
77 {
78 newOutbounds << outbound;
79 continue;
80 }
81
82 if (meta.outboundTags.isEmpty())
83 {
84 LOG("Trying to expand an empty chain.");
85 continue;
86 }
87
88 int nextInboundPort = meta.chainPortAllocation;
89 const auto firstOutboundTag = meta.getDisplayName();
90 const auto lastOutboundTag = meta.outboundTags.first();
91
92 OutboundInfoObject lastOutbound;
93
94 const auto outbountTagCount = meta.outboundTags.count();
95
96 for (auto i = outbountTagCount - 1; i >= 0; i--)
97 {
98 const auto chainOutboundTag = meta.outboundTags[i];
99
100 const auto isFirstOutbound = i == outbountTagCount - 1;
101 const auto isLastOutbound = i == 0;
102 const auto newOutboundTag = [&]() {
103 if (isFirstOutbound)
104 return firstOutboundTag;
105 else if (isLastOutbound)
106 return lastOutboundTag;
107 else
108 return (firstOutboundTag + "_" + chainOutboundTag + "_" + QSTRN(nextInboundPort));
109 }();
110
111 if (!outboundCache.contains(chainOutboundTag))
112 {
113 LOG("Cannot build outbound chain: Missing tag: " + firstOutboundTag);
114 return false;
115 }
116 auto newOutbound = outboundCache[chainOutboundTag];
117
118 // Create Inbound
119 if (!isFirstOutbound)
120 {
121 const auto inboundTag = firstOutboundTag + ":" + QSTRN(nextInboundPort) + "->" + newOutboundTag;
122 const auto inboundSettings = GenerateDokodemoIN(lastOutbound[INFO_SERVER].toString(), lastOutbound[INFO_PORT].toInt(), "tcp,udp");
123 const auto newInbound = GenerateInboundEntry(inboundTag, "dokodemo-door", "127.0.0.1", nextInboundPort, inboundSettings);
124 nextInboundPort++;
125 newInbounds << newInbound;
126 //
127 QJsonObject ruleObject;
128 ruleObject["type"] = "field";
129 ruleObject["inboundTag"] = QJsonArray{ inboundTag };
130 ruleObject["outboundTag"] = newOutboundTag;
131 newRules.prepend(ruleObject);
132 }
133
134 if (!isLastOutbound)
135 {
136 // Begin process outbound.
137 const auto outboundProtocol = newOutbound["protocol"].toString();
138 auto outboundSettings = newOutbound["settings"].toObject();
139 // Get Outbound Info for next Inbound
140 bool getOutboundInfoStatus = false;
141 lastOutbound = PluginHost->GetOutboundInfo(outboundProtocol, outboundSettings, getOutboundInfoStatus);
142 if (!getOutboundInfoStatus)
143 {
144 LOG("Cannot get outbound info for: " + chainOutboundTag);
145 return false;
146 }
147
148 // Update allocated port as outbound server/port
149 OutboundInfoObject newOutboundInfo;
150 newOutboundInfo[INFO_SERVER] = "127.0.0.1";
151 newOutboundInfo[INFO_PORT] = nextInboundPort;
152 // For those kernels deducing SNI from the server name.
153 if (!lastOutbound.contains(INFO_SNI) || lastOutbound[INFO_SNI].toString().trimmed().isEmpty())
154 newOutboundInfo[INFO_SNI] = lastOutbound[INFO_SERVER];
155 //
156 PluginHost->SetOutboundInfo(outboundProtocol, newOutboundInfo, outboundSettings);
157 newOutbound.insert("settings", outboundSettings);
158
159 // Create new outbound
160 newOutbound.insert("tag", newOutboundTag);
161 }
162 newOutbounds << newOutbound;
163 }
164 }
165
166 //
167 // Finalize
168 {
169 QJsonArray _newInbounds = inbounds, _newRules = rules;
170 for (const auto &in : newInbounds)
171 _newInbounds << in;
172 for (const auto &rule : newRules)
173 _newRules.prepend(rule);
174
175 root["outbounds"] = newOutbounds;
176 root["inbounds"] = _newInbounds;
177 QJsonIO::SetValue(root, _newRules, "routing", "rules");
178 }
179 return true;
180 }

182 OUTBOUNDS RouteHandler::ExpandExternalConnection(const OUTBOUNDS &outbounds) const
183 {
184 OUTBOUNDS result;
185 for (const auto &out : outbounds)
186 {
187 auto outObject = out.toObject();
188 const auto meta = OutboundObjectMeta::loadFromOutbound(OUTBOUND(outObject));
189 if (meta.metaType == METAOUTBOUND_EXTERNAL)
190 {
191 outObject = ConnectionManager->GetConnectionRoot(meta.connectionId)["outbounds"].toArray().first().toObject();
192 outObject["tag"] = meta.getDisplayName();
193 }
194 result << outObject;
195 }
196 return result;
197 }

204 //
205 // BEGIN RUNTIME CONFIG GENERATION
206 // We need copy construct here
207 CONFIGROOT RouteHandler::GenerateFinalConfig(CONFIGROOT root, const GroupRoutingId &routingId, bool hasAPI) const
208 {
209 const auto &config = configs.contains(routingId) ? configs[routingId] : *GlobalConfig.defaultRouteConfig;
210 //
211 const auto connConf = *(config.overrideConnectionConfig ? config.connectionConfig : GlobalConfig.defaultRouteConfig->connectionConfig);
212 const auto dnsConf = *(config.overrideDNS ? config.dnsConfig : GlobalConfig.defaultRouteConfig->dnsConfig);
213 const auto fakeDNSConf = *(config.overrideDNS ? config.fakeDNSConfig : GlobalConfig.defaultRouteConfig->fakeDNSConfig);
214 const auto routeConf = *(config.overrideRoute ? config.routeConfig : GlobalConfig.defaultRouteConfig->routeConfig);
215 const auto fpConf = *(config.overrideForwardProxyConfig ? config.forwardProxyConfig : GlobalConfig.defaultRouteConfig->forwardProxyConfig);
216 //
217 //
218 // Note: The part below always makes the whole functionality in
219 // trouble...... BE EXTREME CAREFUL when changing these code
220 // below...
221 //
222 // Check if is complex BEFORE adding anything.
223 bool isComplex = IsComplexConfig(root);
224
225 if (isComplex)
226 {
227 // For some config files that has routing entries already.
228 // We DO NOT add extra routings.
229 //
230 // HOWEVER, we need to verify the QV2RAY_RULE_ENABLED entry.
231 // And what's more, process (by removing unused items) from a
232 // rule object.
233 ROUTING routing(root["routing"].toObject());
234 QJsonArray newRules;
235 LOG("Processing an existing routing table.");
236
237 for (const auto &_rule : routing["rules"].toArray())
238 {
239 auto rule = _rule.toObject();
240
241 // For backward compatibility
242 if (rule.contains("QV2RAY_RULE_USE_BALANCER"))
243 {
244 // We use balancer, or the normal outbound
245 rule.remove(rule["QV2RAY_RULE_USE_BALANCER"].toBool(false) ? "outboundTag" : "balancerTag");
246 }
247 else
248 {
249 LOG("We found a rule without QV2RAY_RULE_USE_BALANCER, so didn't process it.");
250 }
251
252 // If this entry has been disabled.
253 if (rule.contains("QV2RAY_RULE_ENABLED") && rule["QV2RAY_RULE_ENABLED"].toBool() == false)
254 {
255 LOG("Discarded a rule as it's been set DISABLED");
256 }
257 else
258 {
259 newRules.append(rule);
260 }
261 }
262
263 routing["rules"] = newRules;
264 root["routing"] = routing;
265 root["outbounds"] = ExpandExternalConnection(OUTBOUNDS(root["outbounds"].toArray()));
266 const auto result = ExpandChainedOutbounds(root);
267 if (!result)
268 LOG("Cannot expand chained outbounds!");
269 }
270 else
271 {
272 LOG("Processing a simple connection config");
273 if (root["outbounds"].toArray().count() != 1)
274 {
275 // There are no ROUTING but 2 or more outbounds.... This is rare, but possible.
276 LOG("WARN: This message usually indicates the config file has logic errors:");
277 LOG("WARN: --> The config file has NO routing section, however more than 1 outbounds are detected.");
278 }
279 //
280 auto tag = QJsonIO::GetValue(root, "outbounds", 0, "tag").toString();
281 if (tag.isEmpty())
282 {
283 LOG("Applying workaround when an outbound tag is empty");
284 tag = GenerateRandomString(15);
285 QJsonIO::SetValue(root, tag, "outbounds", 0, "tag");
286 }
287 root["routing"] = GenerateRoutes(connConf.enableProxy, connConf.bypassCN, connConf.bypassLAN, tag, routeConf);
288
289 //
290 // Forward proxy
291 if (fpConf.enableForwardProxy)
292 {
293 if (QJsonIO::GetValue(root, "outbounds", 0, QV2RAY_USE_FPROXY_KEY).toBool(false))
294 {
295 if (fpConf.type->isEmpty())
296 {
297 DEBUG("WARNING: Empty forward proxy type.");
298 }
299 else if (fpConf.type->toLower() != "http" && fpConf.type->toLower() != "socks")
300 {
301 DEBUG("WARNING: Unsupported forward proxy type: " + fpConf.type);
302 }
303 else
304 {
305 const static QJsonObject proxySettings{ { "tag", OUTBOUND_TAG_FORWARD_PROXY } };
306 LOG("Applying forward proxy to current connection.");
307 QJsonIO::SetValue(root, proxySettings, "outbounds", 0, "proxySettings");
308 const auto forwardProxySettings = GenerateHTTPSOCKSOut(fpConf.serverAddress, //
309 fpConf.port, //
310 fpConf.useAuth, //
311 fpConf.username, //
312 fpConf.password);
313 const auto forwardProxyOutbound = GenerateOutboundEntry(OUTBOUND_TAG_FORWARD_PROXY, //
314 fpConf.type->toLower(), //
315 forwardProxySettings, {});
316 auto outboundArray = root["outbounds"].toArray();
317 outboundArray.push_back(forwardProxyOutbound);
318 root["outbounds"] = outboundArray;
319 }
320 }
321 else
322 {
323 // Remove proxySettings from first outbound
324 QJsonIO::SetValue(root, QJsonIO::Undefined, "outbounds", 0, "proxySettings");
325 }
326 }
327
328 //
329 // Process FREEDOM and BLACKHOLE outbound
330 {
331 OUTBOUNDS outbounds(root["outbounds"].toArray());
332 const auto freeDS = (connConf.v2rayFreedomDNS) ? "UseIP" : "AsIs";
333 outbounds.append(GenerateOutboundEntry(OUTBOUND_TAG_DIRECT, "freedom", GenerateFreedomOUT(freeDS, ":0"), {}));
334 outbounds.append(GenerateOutboundEntry(OUTBOUND_TAG_BLACKHOLE, "blackhole", GenerateBlackHoleOUT(false), {}));
335 root["outbounds"] = outbounds;
336 }
337
338 //
339 // Connection Filters
340 {
341 if (GlobalConfig.defaultRouteConfig->connectionConfig->dnsIntercept)
342 {
343 const auto hasTProxy = GlobalConfig.inboundConfig->useTPROXY && GlobalConfig.inboundConfig->tProxySettings->hasUDP;
344 const auto hasIPv6 = hasTProxy && (!GlobalConfig.inboundConfig->tProxySettings->tProxyV6IP->isEmpty());
345 const auto hasSocksUDP = GlobalConfig.inboundConfig->useSocks && GlobalConfig.inboundConfig->socksSettings->enableUDP;
346 DNSInterceptFilter(root, hasTProxy, hasIPv6, hasSocksUDP);
347 }
348
349 if (GlobalConfig.inboundConfig->useTPROXY && GlobalConfig.outboundConfig->mark > 0)
350 OutboundMarkSettingFilter(root, GlobalConfig.outboundConfig->mark);
351
352 // Process bypass bitTorrent
353 if (connConf.bypassBT)
354 BypassBTFilter(root);
355
356 // Process mKCP seed
357 mKCPSeedFilter(root);
358
359 // Remove empty Mux object from settings
360 RemoveEmptyMuxFilter(root);
361 }
362 }
363
364 //
365 // Process Log
366 QJsonIO::SetValue(root, V2RayLogLevel[GlobalConfig.logLevel], "log", "loglevel");
367
368 //
369 // Process DNS
370 const auto hasDNS = root.contains("dns") && !root.value("dns").toObject().isEmpty();
371 if (!hasDNS)
372 {
373 root.insert("dns", GenerateDNS(dnsConf));
374 root.insert("fakedns", fakeDNSConf.toJson());
375 LOG("Added global DNS config");
376 }
377
378 //
379 // If inbounds list is empty, we append our global configured inbounds to the config.
380 // The setting applies to BOTH complex config AND simple config.
381 // Just to ensure there's AT LEAST 1 possible inbound is being configured.
382 if (!root.contains("inbounds") || root.value("inbounds").toArray().empty())
383 {
384 root["inbounds"] = GenerateDefaultInbounds();
385 DEBUG("Added global inbound config");
386 }
387
388 //
389 // API 0 speed issue occured when no tag is configured.
390 FillupTagsFilter(root, "inbounds");
391 FillupTagsFilter(root, "outbounds");
392
393 //
394 // Let's process some api features.
395 if (hasAPI && GlobalConfig.kernelConfig->enableAPI)
396 {
397 //
398 // Stats
399 root.insert("stats", QJsonObject());
400 //
401 // Routes
402 QJsonObject routing = root["routing"].toObject();
403 QJsonArray routingRules = routing["rules"].toArray();
404 const static QJsonObject APIRouteRoot{ { "type", "field" }, //
405 { "outboundTag", API_TAG_DEFAULT }, //
406 { "inboundTag", QJsonArray{ API_TAG_INBOUND } } };
407 routingRules.push_front(APIRouteRoot);
408 routing["rules"] = routingRules;
409 root["routing"] = routing;
410 //
411 // Policy
412 QJsonIO::SetValue(root, true, "policy", "system", "statsInboundUplink");
413 QJsonIO::SetValue(root, true, "policy", "system", "statsInboundDownlink");
414 QJsonIO::SetValue(root, true, "policy", "system", "statsOutboundUplink");
415 QJsonIO::SetValue(root, true, "policy", "system", "statsOutboundDownlink");
416 //
417 // Inbounds
418 INBOUNDS inbounds(root["inbounds"].toArray());
419 static const QJsonObject fakeDocodemoDoor{ { "address", "127.0.0.1" } };
420 const auto apiInboundsRoot = GenerateInboundEntry(API_TAG_INBOUND, "dokodemo-door", //
421 "127.0.0.1", //
422 GlobalConfig.kernelConfig->statsPort, //
423 INBOUNDSETTING(fakeDocodemoDoor));
424 inbounds.push_front(apiInboundsRoot);
425 root["inbounds"] = inbounds;
426 //
427 // API
428 root["api"] = GenerateAPIEntry(API_TAG_DEFAULT);
429 }
430
431 return root;
432 }
