13 KernelInstanceHandler::KernelInstanceHandler(QObject *parent) : QObject(parent)
14 {
15 KernelInstance = this;
16 vCoreInstance = new V2RayKernelInstance(this);
17 connect(vCoreInstance, &V2RayKernelInstance::OnNewStatsDataArrived, this, &KernelInstanceHandler::OnV2RayStatsDataRcvd_p);
18 connect(vCoreInstance, &V2RayKernelInstance::OnProcessOutputReadyRead, this, &KernelInstanceHandler::OnV2RayKernelLog_p);
19 connect(vCoreInstance, &V2RayKernelInstance::OnProcessErrored, this, &KernelInstanceHandler::OnKernelCrashed_p);
20 //
21 auto kernelList = PluginHost->UsablePlugins();
22 for (const auto &internalName : kernelList)
23 {
24 const auto info = PluginHost->GetPlugin(internalName);
25 if (!info->hasComponent(COMPONENT_KERNEL))
26 continue;
27 auto kernel = info->pluginInterface->GetKernel();
28 for (const auto &protocol : kernel->GetKernelProtocols())
29 {
30 if (kernelMap.contains(protocol))
31 {
32 LOG("Found multiple kernel providers for a protocol: " + protocol);
33 continue;
34 }
35 kernelMap.insert(protocol, internalName);
36 }
37 }

45 std::optional<QString> KernelInstanceHandler::CheckPort(const QMap<QString, ProtocolSettingsInfoObject> &info, int plugins)
46 {
47 QStringList portDetectionErrorMessage;
48 auto portDetectionMsg = tr("There are other processes occupying the ports necessary to start the connection:") + NEWLINE + NEWLINE;
49 for (const auto &key : info.keys())
50 {
51 const auto result = components::port::CheckTCPPortStatus(info[key].address, info[key].port);
52 if (!result)
53 portDetectionErrorMessage << tr("Endpoint: %1:%2 for inbound: \"%3\"").arg(info[key].address).arg(info[key].port).arg(key);
54 }
55 if (GlobalConfig.pluginConfig->v2rayIntegration)
56 {
57 for (auto i = 0; i < plugins; i++)
58 {
59 const auto thisPort = GlobalConfig.pluginConfig->portAllocationStart + i;
60 const auto result = components::port::CheckTCPPortStatus("127.0.0.1", thisPort);
61 if (!result)
62 portDetectionErrorMessage << tr("Local port: %1 for plugin integration.").arg(thisPort);
63 }
64 }
65 if (!portDetectionErrorMessage.isEmpty())
66 {
67 portDetectionMsg += portDetectionErrorMessage.join(NEWLINE);
68 return portDetectionMsg;
69 }
70 else
71 {
72 return std::nullopt;
73 }

76 std::optional<QString> KernelInstanceHandler::StartConnection(const ConnectionGroupPair &id, CONFIGROOT fullConfig)
77 {
78 StopConnection();
79 inboundInfo = GetInboundInfo(fullConfig);
80 //
81 const auto inboundPorts = GetInboundPorts();
82 const auto inboundHosts = GetInboundHosts();
83 PluginHost->SendEvent({ GetDisplayName(id.connectionId), inboundPorts, Connectivity::Connecting });
84 // QList<std::tuple<QString, int, QString>> inboundInfo;
85 // for (const auto &inbound_v : fullConfig["inbounds"].toArray())
86 //{
87 // const auto &inbound = inbound_v.toObject();
88 // inboundInfo.push_back({ inbound["protocol"].toString(), inbound["port"].toInt(), inbound["tag"].toString() });
89 //}
90 //
91 if (GlobalConfig.pluginConfig->v2rayIntegration)
92 {
93 // Process outbounds.
94 OUTBOUNDS processedOutbounds;
95 auto pluginPort = *GlobalConfig.pluginConfig->portAllocationStart;
96 //
97 for (auto i = 0; i < fullConfig["outbounds"].toArray().count(); i++)
98 {
99 auto outbound = QJsonIO::GetValue(fullConfig, "outbounds", i).toObject();
100 const auto outProtocol = outbound["protocol"].toString();
101 //
102 if (!kernelMap.contains(outProtocol))
103 {
104 // Normal outbound, or the one without a plugin supported.
105 // Marked as processed.
106 processedOutbounds.push_back(outbound);
107 LOG("Outbound protocol " + outProtocol + " is not a registered plugin outbound.");
108 continue;
109 }
110 LOG("Creating kernel plugin instance for protocol" + outProtocol);
111 auto kernel = PluginHost->GetPlugin(kernelMap[outProtocol])->pluginInterface->GetKernel()->CreateKernel();
112 // New object does not need disconnect?
113 // disconnect(kernel, &QvPluginKernel::OnKernelStatsAvailable, this, &KernelInstanceHandler::OnStatsDataArrived_p);
114 //
115 QMap<KernelOptionFlags, QVariant> _inboundSettings;
116
117 LOG("V2RayIntegration: " + QSTRN(pluginPort) + " = " + outProtocol);
118 _inboundSettings[KERNEL_HTTP_ENABLED] = false;
119 _inboundSettings[KERNEL_SOCKS_ENABLED] = true;
120 _inboundSettings[KERNEL_SOCKS_PORT] = pluginPort;
121 _inboundSettings[KERNEL_SOCKS_UDP_ENABLED] = *GlobalConfig.inboundConfig->socksSettings->enableUDP;
122 _inboundSettings[KERNEL_SOCKS_LOCAL_ADDRESS] = *GlobalConfig.inboundConfig->socksSettings->localIP;
123 _inboundSettings[KERNEL_LISTEN_ADDRESS] = "127.0.0.1";
124 LOG("Sending connection settings to kernel.");
125 kernel->SetConnectionSettings(_inboundSettings, outbound["settings"].toObject());
126 activeKernels.push_back({ outProtocol, std::move(kernel) });
127 //
128 const auto pluginOutSettings = GenerateHTTPSOCKSOut("127.0.0.1", pluginPort, false, "", "");
129 //
130 const auto pluginOut = GenerateOutboundEntry(outbound["tag"].toString(), "socks", pluginOutSettings, {});
131 //
132 // Add the integration outbound to the list.
133 processedOutbounds.push_back(pluginOut);
134 pluginPort++;
135 }
136 LOG("Applying new outbound settings.");
137 fullConfig["outbounds"] = processedOutbounds;
138 RemoveEmptyMuxFilter(fullConfig);
139 }
140
141 //
142 // ======================================================================= Start Kernels
143 //
144 {
145 const auto portResult = CheckPort(inboundInfo, activeKernels.size());
146 if (portResult)
147 {
148 LOG(*portResult);
149 return portResult;
150 }
151 auto firstOutbound = fullConfig["outbounds"].toArray().first().toObject();
152 const auto firstOutboundProtocol = firstOutbound["protocol"].toString();
153 if (GlobalConfig.pluginConfig->v2rayIntegration)
154 {
155 LOG("Starting kernels with V2RayIntegration.");
156 bool hasAllKernelStarted = true;
157 for (auto &[outboundProtocol, kernelObject] : activeKernels)
158 {
159 LOG("Starting kernel for protocol: " + outboundProtocol);
160 bool status = kernelObject->StartKernel();
161 connect(kernelObject.get(), &PluginKernel::OnKernelCrashed, this, &KernelInstanceHandler::OnKernelCrashed_p,
162 Qt::QueuedConnection);
163 connect(kernelObject.get(), &PluginKernel::OnKernelLogAvailable, this, &KernelInstanceHandler::OnPluginKernelLog_p,
164 Qt::QueuedConnection);
165 hasAllKernelStarted = hasAllKernelStarted && status;
166 if (!status)
167 {
168 LOG("Plugin Kernel: " + outboundProtocol + " failed to start.");
169 break;
170 }
171 }
172 if (!hasAllKernelStarted)
173 {
174 StopConnection();
175 return tr("A plugin kernel failed to start. Please check the outbound settings.");
176 }
177 currentId = id;
178 //
179 // Also start V2Ray-core.
180 auto result = vCoreInstance->StartConnection(fullConfig);
181 //
182 if (result.has_value())
183 {
184 StopConnection();
185 PluginHost->SendEvent({ GetDisplayName(id.connectionId), inboundPorts, Connectivity::Disconnected });
186 return result;
187 }
188 else
189 {
190 emit OnConnected(id);
191 PluginHost->SendEvent({ GetDisplayName(id.connectionId), inboundPorts, Connectivity::Connected });
192 }
193 }
194 else if (kernelMap.contains(firstOutboundProtocol))
195 {
196 // Connections without V2Ray Integration will have and ONLY have ONE kernel.
197 LOG("Starting kernel " + firstOutboundProtocol + " without V2Ray Integration");
198 {
199 auto kernel =
200 PluginHost->GetPlugin(kernelMap[firstOutbound["protocol"].toString()])->pluginInterface->GetKernel()->CreateKernel();
201 activeKernels.push_back({ firstOutboundProtocol, std::move(kernel) });
202 }
203 Q_ASSERT(activeKernels.size() == 1);
204#define theKernel (activeKernels.front().second.get())
205 connect(theKernel, &PluginKernel::OnKernelStatsAvailable, this, &KernelInstanceHandler::OnPluginStatsDataRcvd_p,
206 Qt::QueuedConnection);
207 connect(theKernel, &PluginKernel::OnKernelCrashed, this, &KernelInstanceHandler::OnKernelCrashed_p, Qt::QueuedConnection);
208 connect(theKernel, &PluginKernel::OnKernelLogAvailable, this, &KernelInstanceHandler::OnPluginKernelLog_p, Qt::QueuedConnection);
209 currentId = id;
210 //
211 QMap<KernelOptionFlags, QVariant> pluginSettings;
212
213 for (const auto &v : inboundInfo)
214 {
215 if (v.protocol != "http" && v.protocol != "socks")
216 continue;
217 pluginSettings[KERNEL_HTTP_ENABLED] = pluginSettings[KERNEL_HTTP_ENABLED].toBool() || v.protocol == "http";
218 pluginSettings[KERNEL_SOCKS_ENABLED] = pluginSettings[KERNEL_SOCKS_ENABLED].toBool() || v.protocol == "socks";
219 pluginSettings.insert(v.protocol.toLower() == "http" ? KERNEL_HTTP_PORT : KERNEL_SOCKS_PORT, v.port);
220 }
221
222 pluginSettings[KERNEL_SOCKS_UDP_ENABLED] = *GlobalConfig.inboundConfig->socksSettings->enableUDP;
223 pluginSettings[KERNEL_SOCKS_LOCAL_ADDRESS] = *GlobalConfig.inboundConfig->socksSettings->localIP;
224 pluginSettings[KERNEL_LISTEN_ADDRESS] = *GlobalConfig.inboundConfig->listenip;
225 //
226 theKernel->SetConnectionSettings(pluginSettings, firstOutbound["settings"].toObject());
227 bool kernelStarted = theKernel->StartKernel();
228#undef theKernel
229 if (kernelStarted)
230 {
231 emit OnConnected(id);
232 PluginHost->SendEvent({ GetDisplayName(id.connectionId), inboundPorts, Connectivity::Connected });
233 }
234 else
235 {
236 return tr("A plugin kernel failed to start. Please check the outbound settings.");
237 StopConnection();
238 }
239 }
240 else
241 {
242 LOG("Starting V2Ray without plugin.");
243 currentId = id;
244 auto result = vCoreInstance->StartConnection(fullConfig);
245 if (result.has_value())
246 {
247 PluginHost->SendEvent({ GetDisplayName(id.connectionId), inboundPorts, Connectivity::Disconnected });
248 StopConnection();
249 return result;
250 }
251 else
252 {
253 emit OnConnected(id);
254 PluginHost->SendEvent({ GetDisplayName(id.connectionId), inboundPorts, Connectivity::Connected });
255 }
256 }
257 }
258 // Return
259 return std::nullopt;
260 }

291 void KernelInstanceHandler::StopConnection()
292 {
293 if (isConnected)
294 {
295 const auto inboundPorts = GetInboundPorts();
296 PluginHost->SendEvent({ GetDisplayName(currentId.connectionId), inboundPorts, Connectivity::Disconnecting });
297 if (vCoreInstance->IsKernelRunning())
298 {
299 vCoreInstance->StopConnection();
300 }
301 for (const auto &[kernel, kernelObject] : activeKernels)
302 {
303 LOG("Stopping plugin kernel: " + kernel);
304 kernelObject->StopKernel();
305 }
306 pluginLogPrefixPadding = 0;
307 emit OnDisconnected(currentId);
308 PluginHost->SendEvent({ GetDisplayName(currentId.connectionId), inboundPorts, Connectivity::Disconnected });
309 }
310 else
311 {
312 LOG("Cannot disconnect when there's nothing connected.");
313 }
314 currentId.clear();
315 activeKernels.clear();
316 }
