14 INBOUNDSETTING GenerateHTTPIN(bool enableAuth, const QList<AccountObject> &_accounts, int timeout, bool allowTransparent)
15 {
16 INBOUNDSETTING root;
17 QJsonArray accounts;
18
19 for (const auto &account : _accounts)
20 {
21 if (account.user->isEmpty() && account.pass->isEmpty())
22 continue;
23 accounts.append(account.toJson());
24 }
25
26 if (enableAuth && !accounts.isEmpty())
27 JADD(accounts)
28
29 JADD(timeout, allowTransparent)
30 return root;
31 }

33 INBOUNDSETTING GenerateSocksIN(const QString &auth, const QList<AccountObject> &_accounts, bool udp, const QString &ip)
34 {
35 INBOUNDSETTING root;
36 QJsonArray accounts;
37 for (const auto &acc : _accounts)
38 {
39 if (acc.user->isEmpty() && acc.pass->isEmpty())
40 continue;
41 accounts.append(acc.toJson());
42 }
43
44 if (!accounts.isEmpty())
45 JADD(accounts)
46
47 if (udp)
48 {
49 JADD(auth, udp, ip)
50 }
51 else
52 {
53 JADD(auth)
54 }
55 return root;
56 }

58 INBOUND GenerateInboundEntry(const QString &tag, const QString &protocol, const QString &listen, int port, const INBOUNDSETTING &settings,

61 INBOUND root;
62 DEBUG("Allocation is not used here, Not Implemented");

65 return root;

68 QJsonObject GenerateSniffingObject(bool enabled, QList<QString> destOverride, bool metadataOnly)
69 {
70 QJsonObject root;
71 QStringList list;
72 const auto size = destOverride.size();
73 if (!enabled)
74 return root;
75
76 root.insert("enabled", enabled);
77 root["metadataOnly"] = metadataOnly;
78
79 if (!destOverride.isEmpty())
80 {
81 for (auto i = 0; i < size; ++i)
82 list << destOverride.at(i);
83 root.insert("destOverride", QJsonArray::fromStringList(list));
84 }
85 return root;
86 }

88 INBOUNDS GenerateDefaultInbounds()
89 {
90#define INCONF GlobalConfig.inboundConfig
91 INBOUNDS inboundsList;
92
93 // HTTP Inbound
94 if (GlobalConfig.inboundConfig->useHTTP)
95 {
96 const auto httpInSettings = GenerateHTTPIN(INCONF->httpSettings->useAuth, { INCONF->httpSettings->account });
97 const auto httpSniffingObject = GenerateSniffingObject(INCONF->httpSettings->sniffing, //
98 INCONF->httpSettings->destOverride);
99 const auto httpInboundObject = GenerateInboundEntry("http_IN", "http", //
100 INCONF->listenip, //
101 INCONF->httpSettings->port, //
102 httpInSettings, //
103 httpSniffingObject);
104 inboundsList.append(httpInboundObject);
105 }
106
107 // SOCKS Inbound
108 if (INCONF->useSocks)
109 {
110 const auto socksInSettings = GenerateSocksIN(INCONF->socksSettings->useAuth ? "password" : "noauth", //
111 { INCONF->socksSettings->account }, //
112 INCONF->socksSettings->enableUDP, //
113 INCONF->socksSettings->localIP);
114 const auto socksSniffingObject = GenerateSniffingObject(INCONF->socksSettings->sniffing, //
115 INCONF->socksSettings->destOverride);
116 const auto socksInboundObject = GenerateInboundEntry("socks_IN", "socks", //
117 INCONF->listenip, //
118 INCONF->socksSettings->port, //
119 socksInSettings, //
120 socksSniffingObject);
121 inboundsList.append(socksInboundObject);
122 }
123
124 // TPROXY
125 if (INCONF->useTPROXY)
126 {
127 QList<QString> networks;
128 if (INCONF->tProxySettings->hasTCP)
129 networks << "tcp";
130 if (INCONF->tProxySettings->hasUDP)
131 networks << "udp";
132 const auto tproxy_network = networks.join(",");
133 const auto tProxySettings = GenerateDokodemoIN("", 0, tproxy_network, 0, true);
134 const auto tproxySniffingObject = GenerateSniffingObject(INCONF->tProxySettings->sniffing, //
135 INCONF->tProxySettings->destOverride);
136 // tProxy IPv4 Settings
137 {
138 LOG("Processing tProxy IPv4 inbound");
139 auto tProxyIn = GenerateInboundEntry("tproxy_IN", "dokodemo-door", //
140 INCONF->tProxySettings->tProxyIP, //
141 INCONF->tProxySettings->port, //
142 tProxySettings, //
143 tproxySniffingObject);
144 tProxyIn.insert("streamSettings", QJsonObject{ { "sockopt", QJsonObject{ { "tproxy", *INCONF->tProxySettings->mode } } } });
145 inboundsList.append(tProxyIn);
146 }
147 if (!INCONF->tProxySettings->tProxyV6IP->isEmpty())
148 {
149 LOG("Processing tProxy IPv6 inbound");
150 auto tProxyIn = GenerateInboundEntry("tproxy_IN_V6", "dokodemo-door", //
151 INCONF->tProxySettings->tProxyV6IP, //
152 INCONF->tProxySettings->port, //
153 tProxySettings, //
154 tproxySniffingObject);
155 tProxyIn.insert("streamSettings", QJsonObject{ { "sockopt", QJsonObject{ { "tproxy", *INCONF->tProxySettings->mode } } } });
156 inboundsList.append(tProxyIn);
157 }
158 }
159
160#undef INCONF
161 return inboundsList;
162 }
