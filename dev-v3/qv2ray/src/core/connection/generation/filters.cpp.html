25 void DNSInterceptFilter(CONFIGROOT &root, const bool have_tproxy, const bool have_tproxy_v6, const bool have_socks)
26 {
27 // Static DNS Objects
28 static const QJsonObject dnsOutboundObj{ { "protocol", "dns" }, { "tag", "dns-out" } };
29
30 QJsonArray dnsRouteInTag;
31
32 if (have_tproxy)
33 dnsRouteInTag.append("tproxy_IN");
34 if (have_tproxy_v6)
35 dnsRouteInTag.append("tproxy_IN_V6");
36 if (have_socks)
37 dnsRouteInTag.append("socks_IN");
38
39 // If no UDP inbound, then DNS outbound is useless.
40 if (dnsRouteInTag.isEmpty())
41 return;
42
43 const QJsonObject dnsRoutingRuleObj{ { "outboundTag", "dns-out" }, { "port", "53" }, { "type", "field" }, { "inboundTag", dnsRouteInTag } };
44
45 // DNS Outbound
46 QJsonIO::SetValue(root, dnsOutboundObj, "outbounds", root["outbounds"].toArray().count());
47 // DNS Route
48 auto _rules = QJsonIO::GetValue(root, "routing", "rules").toArray();
49 _rules.prepend(dnsRoutingRuleObj);
50 QJsonIO::SetValue(root, _rules, "routing", "rules");
51 }

63 void mKCPSeedFilter(CONFIGROOT &root)
64 {
65 for (auto i = 0; i < root["outbounds"].toArray().count(); i++)
66 {
67 const auto seedItem = QJsonIO::GetValue(root, "outbounds", i, "streamSettings", "kcpSettings", "seed");
68 bool shouldProcess = !seedItem.isNull() && !seedItem.isUndefined();
69 bool isEmptySeed = seedItem.toString().isEmpty();
70 if (shouldProcess && isEmptySeed)
71 QJsonIO::SetValue(root, QJsonIO::Undefined, "outbounds", i, "streamSettings", "kcpSettings", "seed");
72 }
73 }
