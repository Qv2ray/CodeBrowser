12 CONFIGROOT Deserialize(const QString &ssUri, QString *alias, QString *errMessage)
13 {
14 ShadowSocksServerObject server;
15 QString d_name;
16
17 // auto ssUri = _ssUri.toStdString();
18 if (ssUri.length() < 5)
19 {
20 LOG("ss:// string too short");
21 *errMessage = QObject::tr("SS URI is too short");
22 }
23
24 auto uri = ssUri.mid(5);
25 auto hashPos = uri.lastIndexOf("#");
26 DEBUG("Hash sign position: " + QSTRN(hashPos));
27
28 if (hashPos >= 0)
29 {
30 // Get the name/remark
31 d_name = uri.mid(uri.lastIndexOf("#") + 1);
32 uri.truncate(hashPos);
33 }
34
35 auto atPos = uri.indexOf('@');
36 DEBUG("At sign position: " + QSTRN(atPos));
37
38 if (atPos < 0)
39 {
40 // Old URI scheme
41 QString decoded = QByteArray::fromBase64(uri.toUtf8(), QByteArray::Base64Option::OmitTrailingEquals);
42 auto colonPos = decoded.indexOf(':');
43 DEBUG("Colon position: " + QSTRN(colonPos));
44
45 if (colonPos < 0)
46 {
47 *errMessage = QObject::tr("Can't find the colon separator between method and password");
48 }
49
50 server.method = decoded.left(colonPos);
51 decoded.remove(0, colonPos + 1);
52 atPos = decoded.lastIndexOf('@');
53 DEBUG("At sign position: " + QSTRN(atPos));
54
55 if (atPos < 0)
56 {
57 *errMessage = QObject::tr("Can't find the at separator between password and hostname");
58 }
59
60 server.password = decoded.mid(0, atPos);
61 decoded.remove(0, atPos + 1);
62 colonPos = decoded.lastIndexOf(':');
63 DEBUG("Colon position: " + QSTRN(colonPos));
64
65 if (colonPos < 0)
66 {
67 *errMessage = QObject::tr("Can't find the colon separator between hostname and port");
68 }
69
70 server.address = decoded.mid(0, colonPos);
71 server.port = decoded.mid(colonPos + 1).toInt();
72 }
73 else
74 {
75 // SIP002 URI scheme
76 auto x = QUrl::fromUserInput(uri);
77 server.address = x.host();
78 server.port = x.port();
79 const auto userInfo = SafeBase64Decode(x.userName());
80 const auto userInfoSp = userInfo.indexOf(':');
81 //
82 DEBUG("Userinfo splitter position: " + QSTRN(userInfoSp));
83
84 if (userInfoSp < 0)
85 {
86 *errMessage = QObject::tr("Can't find the colon separator between method and password");
87 return CONFIGROOT{};
88 }
89
90 const auto method = userInfo.mid(0, userInfoSp);
91 server.method = method;
92 server.password = userInfo.mid(userInfoSp + 1);
93 }
94
95 d_name = QUrl::fromPercentEncoding(d_name.toUtf8());
96 CONFIGROOT root;
97 OUTBOUNDS outbounds;
98 outbounds.append(GenerateOutboundEntry(OUTBOUND_TAG_PROXY, "shadowsocks", GenerateShadowSocksOUT({ server }), {}));
99 JADD(outbounds)
100 *alias = alias->isEmpty() ? d_name : *alias + "_" + d_name;
101 LOG("Deduced alias: " + *alias);
102 return root;
103 }

105 const QString Serialize(const ShadowSocksServerObject &server, const QString &alias)
106 {
107 QUrl url;
108 const auto plainUserInfo = server.method + ":" + server.password;
109 const auto userinfo = plainUserInfo.toUtf8().toBase64(QByteArray::Base64UrlEncoding | QByteArray::OmitTrailingEquals);
110 url.setUserInfo(userinfo);
111 url.setScheme("ss");
112 url.setHost(server.address);
113 url.setPort(server.port);
114 url.setFragment(alias);
115 return url.toString(QUrl::ComponentFormattingOption::FullyEncoded);
116 }
