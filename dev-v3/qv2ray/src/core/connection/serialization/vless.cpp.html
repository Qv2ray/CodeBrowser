8namespace Qv2ray::core::connection
9{
10 namespace serialization::vless
11 {
12 CONFIGROOT Deserialize(const QString &str, QString *alias, QString *errMessage)
13 {
14 // must start with vless://
15 if (!str.startsWith("vless://"))
16 {
17 *errMessage = QObject::tr("VLESS link should start with vless://");
18 return CONFIGROOT();
19 }
20
21 // parse url
22 QUrl url(str);
23 if (!url.isValid())
24 {
25 *errMessage = QObject::tr("link parse failed: %1").arg(url.errorString());
26 return CONFIGROOT();
27 }
28
29 // fetch host
30 const auto hostRaw = url.host();
31 if (hostRaw.isEmpty())
32 {
33 *errMessage = QObject::tr("empty host");
34 return CONFIGROOT();
35 }
36 const auto host = (hostRaw.startsWith('[') && hostRaw.endsWith(']')) ? hostRaw.mid(1, hostRaw.length() - 2) : hostRaw;
37
38 // fetch port
39 const auto port = url.port();
40 if (port == -1)
41 {
42 *errMessage = QObject::tr("missing port");
43 return CONFIGROOT();
44 }
45
46 // fetch remarks
47 const auto remarks = url.fragment();
48 if (!remarks.isEmpty())
49 {
50 *alias = remarks;
51 }
52
53 // fetch uuid
54 const auto uuid = url.userInfo();
55 if (uuid.isEmpty())
56 {
57 *errMessage = QObject::tr("missing uuid");
58 return CONFIGROOT();
59 }
60
61 // initialize QJsonObject with basic info
62 QJsonObject outbound;
63 QJsonObject stream;
64
65 QJsonIO::SetValue(outbound, "vless", "protocol");
66 QJsonIO::SetValue(outbound, host, { "settings", "vnext", 0, "address" });
67 QJsonIO::SetValue(outbound, port, { "settings", "vnext", 0, "port" });
68 QJsonIO::SetValue(outbound, uuid, { "settings", "vnext", 0, "users", 0, "id" });
69
70 // parse query
71 QUrlQuery query(url.query());
72
73 // handle type
74 const auto hasType = query.hasQueryItem("type");
75 const auto type = hasType ? query.queryItemValue("type") : "tcp";
76 if (type != "tcp")
77 QJsonIO::SetValue(stream, type, "network");
78
79 // handle encryption
80 const auto hasEncryption = query.hasQueryItem("encryption");
81 const auto encryption = hasEncryption ? query.queryItemValue("encryption") : "none";
82 QJsonIO::SetValue(outbound, encryption, { "settings", "vnext", 0, "users", 0, "encryption" });
83
84 // type-wise settings
85 if (type == "kcp")
86 {
87 const auto hasSeed = query.hasQueryItem("seed");
88 if (hasSeed)
89 QJsonIO::SetValue(stream, query.queryItemValue("seed"), { "kcpSettings", "seed" });
90
91 const auto hasHeaderType = query.hasQueryItem("headerType");
92 const auto headerType = hasHeaderType ? query.queryItemValue("headerType") : "none";
93 if (headerType != "none")
94 QJsonIO::SetValue(stream, headerType, { "kcpSettings", "header", "type" });
95 }
96 else if (type == "http")
97 {
98 const auto hasPath = query.hasQueryItem("path");
99 const auto path = hasPath ? QUrl::fromPercentEncoding(query.queryItemValue("path").toUtf8()) : "/";
100 if (path != "/")
101 QJsonIO::SetValue(stream, path, { "httpSettings", "path" });
102
103 const auto hasHost = query.hasQueryItem("host");
104 if (hasHost)
105 {
106 const auto hosts = QJsonArray::fromStringList(query.queryItemValue("host").split(","));
107 QJsonIO::SetValue(stream, hosts, { "httpSettings", "host" });
108 }
109 }
110 else if (type == "ws")
111 {
112 const auto hasPath = query.hasQueryItem("path");
113 const auto path = hasPath ? QUrl::fromPercentEncoding(query.queryItemValue("path").toUtf8()) : "/";
114 if (path != "/")
115 QJsonIO::SetValue(stream, path, { "wsSettings", "path" });
116
117 const auto hasHost = query.hasQueryItem("host");
118 if (hasHost)
119 {
120 QJsonIO::SetValue(stream, query.queryItemValue("host"), { "wsSettings", "headers", "Host" });
121 }
122 }
123 else if (type == "quic")
124 {
125 const auto hasQuicSecurity = query.hasQueryItem("quicSecurity");
126 if (hasQuicSecurity)
127 {
128 const auto quicSecurity = query.queryItemValue("quicSecurity");
129 QJsonIO::SetValue(stream, quicSecurity, { "quicSettings", "security" });
130
131 if (quicSecurity != "none")
132 {
133 const auto key = query.queryItemValue("key");
134 QJsonIO::SetValue(stream, key, { "quicSettings", "key" });
135 }
136
137 const auto hasHeaderType = query.hasQueryItem("headerType");
138 const auto headerType = hasHeaderType ? query.queryItemValue("headerType") : "none";
139 if (headerType != "none")
140 QJsonIO::SetValue(stream, headerType, { "quicSettings", "header", "type" });
141 }
142 }
143 else if (type == "grpc")
144 {
145 const auto hasServiceName = query.hasQueryItem("serviceName");
146 if (hasServiceName)
147 {
148 const auto serviceName = QUrl::fromPercentEncoding(query.queryItemValue("serviceName").toUtf8());
149 if (serviceName != "GunService")
150 {
151 QJsonIO::SetValue(stream, serviceName, { "grpcSettings", "serviceName" });
152 }
153 }
154 }
155
156 // tls-wise settings
157 const auto hasSecurity = query.hasQueryItem("security");
158 const auto security = hasSecurity ? query.queryItemValue("security") : "none";
159 const auto tlsKey = security == "xtls" ? "xtlsSettings" : "tlsSettings";
160 if (security != "none")
161 {
162 QJsonIO::SetValue(stream, security, "security");
163 }
164 // sni
165 const auto hasSNI = query.hasQueryItem("sni");
166 if (hasSNI)
167 {
168 const auto sni = query.queryItemValue("sni");
169 QJsonIO::SetValue(stream, sni, { tlsKey, "serverName" });
170 }
171 // alpn
172 const auto hasALPN = query.hasQueryItem("alpn");
173 if (hasALPN)
174 {
175 const auto alpnRaw = QUrl::fromPercentEncoding(query.queryItemValue("alpn").toUtf8());
176 const auto alpnArray = QJsonArray::fromStringList(alpnRaw.split(","));
177 QJsonIO::SetValue(stream, alpnArray, { tlsKey, "alpn" });
178 }
179 // xtls-specific
180 if (security == "xtls")
181 {
182 const auto flow = query.queryItemValue("flow");
183 QJsonIO::SetValue(outbound, flow, { "settings", "vnext", 0, "users", 0, "flow" });
184 }
185
186 // assembling config
187 CONFIGROOT root;
188 outbound["streamSettings"] = stream;
189 root["outbounds"] = QJsonArray{ outbound };
190
191 // return
192 return root;
193 }
194 } // namespace serialization::vless
195} // namespace Qv2ray::core::connection
