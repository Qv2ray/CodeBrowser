13 // From https://github.com/2dust/v2rayN/wiki/分享链接格式说明(ver-2)
14 const QString Serialize(const StreamSettingsObject &transfer, const VMessServerObject &server, const QString &alias)
15 {
16 QJsonObject vmessUriRoot;
17 // Constant
18 vmessUriRoot["v"] = 2;
19 vmessUriRoot["ps"] = alias;
20 vmessUriRoot["add"] = *server.address;
21 vmessUriRoot["port"] = *server.port;
22 vmessUriRoot["id"] = *server.users->front().id;
23 vmessUriRoot["aid"] = *server.users->front().alterId;
24 vmessUriRoot["net"] = *transfer.network == "http" ? "h2" : *transfer.network;
25 vmessUriRoot["tls"] = *transfer.security;
26
27 if (transfer.network == "tcp")
28 {
29 vmessUriRoot["type"] = *transfer.tcpSettings->header->type;
30 }
31 else if (transfer.network == "kcp")
32 {
33 vmessUriRoot["type"] = *transfer.kcpSettings->header->type;
34 }
35 else if (transfer.network == "quic")
36 {
37 vmessUriRoot["type"] = *transfer.quicSettings->header->type;
38 vmessUriRoot["host"] = *transfer.quicSettings->security;
39 vmessUriRoot["path"] = *transfer.quicSettings->key;
40 }
41 else if (transfer.network == "ws")
42 {
43 auto x = *transfer.wsSettings->headers;
44 auto host = x.contains("host");
45 auto CapHost = x.contains("Host");
46 auto realHost = host ? x["host"] : (CapHost ? x["Host"] : "");
47 //
48 vmessUriRoot["host"] = realHost;
49 vmessUriRoot["path"] = *transfer.wsSettings->path;
50 }
51 else if (transfer.network == "h2" || transfer.network == "http")
52 {
53 vmessUriRoot["host"] = transfer.httpSettings->host->join(",");
54 vmessUriRoot["path"] = *transfer.httpSettings->path;
55 }
56
57 if (!vmessUriRoot.contains("type") || vmessUriRoot["type"].toString().isEmpty())
58 {
59 vmessUriRoot["type"] = "none";
60 }
61
62 //
63 auto vmessPart = Base64Encode(JsonToString(vmessUriRoot, QJsonDocument::JsonFormat::Compact));
64 return "vmess://" + vmessPart;
65 }

67 // This generates global config containing only one outbound....
68 CONFIGROOT Deserialize(const QString &vmessStr, QString *alias, QString *errMessage)
69 {
70#define default CONFIGROOT()
71 QString vmess = vmessStr;
72
73 if (vmess.trimmed() != vmess)
74 {
75 LOG("VMess string has some prefix/postfix spaces, trimming.");
76 vmess = vmessStr.trimmed();
77 }
78
79 // Reset errMessage
80 *errMessage = "";
81
82 if (!qAsConst(vmess).toLower().startsWith("vmess://"))
83 {
84 *errMessage = QObject::tr("VMess string should start with 'vmess://'");
85 return default;
86 }
87
88 const auto b64Str = vmess.mid(8, vmess.length() - 8);
89 if (b64Str.isEmpty())
90 {
91 *errMessage = QObject::tr("VMess string should be a valid base64 string");
92 return default;
93 }
94
95 auto vmessString = SafeBase64Decode(b64Str);
96 auto jsonErr = VerifyJsonString(vmessString);
97
98 if (!jsonErr.isEmpty())
99 {
100 *errMessage = jsonErr;
101 return default;
102 }
103
104 auto vmessConf = JsonFromString(vmessString);
105
106 if (vmessConf.isEmpty())
107 {
108 *errMessage = QObject::tr("JSON should not be empty");
109 return default;
110 }
111
112 // --------------------------------------------------------------------------------------
113 CONFIGROOT root;
114 QString ps, add, id, net, type, host, path, tls;
115 int port, aid;
116 //
117 // __vmess_checker__func(key, values)
118 //
119 // - Key = Key in JSON and the variable name.
120 // - Values = Candidate variable list, if not match, the first one is used as default.
121 //
122 // - [[val.size() <= 1]] is used when only the default value exists.
123 //
124 // - It can be empty, if so, if the key is not in the JSON, or the value is empty, report an error.
125 // - Else if it contains one thing. if the key is not in the JSON, or the value is empty, use that one.
126 // - Else if it contains many things, when the key IS in the JSON but not within the THINGS, use the first in the THINGS
127 // - Else -------------------------------------------->>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>> use the JSON value
128 //
129#define __vmess_checker__func(key, values) \
130 { \
131 auto val = QVector<QString>() values; \
132 if (vmessConf.contains(#key) && !vmessConf[#key].toVariant().toString().trimmed().isEmpty() && \
133 (val.size() <= 1 || val.contains(vmessConf[#key].toVariant().toString()))) \
134 { \
135 key = vmessConf[#key].toVariant().toString(); \
136 } \
137 else if (!val.isEmpty()) \
138 { \
139 key = val.first(); \
140 DEBUG("Using key \"" #key "\" from the first candidate list: " + key); \
141 } \
142 else \
143 { \
144 *errMessage = QObject::tr(#key " does not exist."); \
145 LOG("Cannot process \"" #key "\" since it's not included in the json object."); \
146 LOG(" --> values: " + val.join(";")); \
147 LOG(" --> PS: " + ps); \
148 } \
149 }
150
151 // vmess v1 upgrader
152 if (!vmessConf.contains("v"))
153 {
154 LOG("Detected deprecated vmess v1. Trying to upgrade...");
155 if (const auto network = vmessConf["net"].toString(); network == "ws" || network == "h2")
156 {
157 const QStringList hostComponents = vmessConf["host"].toString().replace(" ", "").split(";");
158 if (const auto nParts = hostComponents.length(); nParts == 1)
159 vmessConf["path"] = hostComponents[0], vmessConf["host"] = "";
160 else if (nParts == 2)
161 vmessConf["path"] = hostComponents[0], vmessConf["host"] = hostComponents[1];
162 else
163 vmessConf["path"] = "/", vmessConf["host"] = "";
164 }
165 }
166
167 // Strict check of VMess protocol, to check if the specified value
168 // is in the correct range.
169 //
170 // Get Alias (AKA ps) from address and port.
171 {
172 // Some idiot vmess:// links are using alterId...
173 aid = vmessConf.value(vmessConf.contains("aid") ? "aid" : "alterId").toVariant().toInt();
174 __vmess_checker__func(ps, << vmessConf["add"].toVariant().toString() + ":" + vmessConf["port"].toVariant().toString());
175 __vmess_checker__func(add, nothing);
176 __vmess_checker__func(id, nothing);
177 __vmess_checker__func(type, << "none"
178 << "http"
179 << "srtp"
180 << "utp"
181 << "wechat-video");
182
183 __vmess_checker__func(net, << "tcp"
184 << "http"
185 << "h2"
186 << "ws"
187 << "kcp"
188 << "domainsocket"
189 << "quic");
190
191 __vmess_checker__func(tls, << "none"
192 << "tls");
193 path = vmessConf.contains("path") ? vmessConf["path"].toVariant().toString() : (net == "quic" ? "" : "/");
194 host = vmessConf.contains("host") ? vmessConf["host"].toVariant().toString() : (net == "quic" ? "none" : "");
195 }
196
197 // Repect connection type rather than obfs type
198 if (QStringList{ "srtp", "utp", "wechat-video" }.contains(type))
199 {
200 if (net != "quic" && net != "kcp")
201 {
202 LOG("Reset obfs settings from " + type + " to none");
203 type = "none";
204 }
205 }
206
207 port = vmessConf["port"].toVariant().toInt();
208 //
209 // Apply the settings.
210 // User
211 VMessUserObject user;
212 user.id = id;
213 user.alterId = aid;
214 //
215 // Server
216 VMessServerObject serv;
217 serv.port = port;
218 serv.address = add;
219 serv.users->push_back(user);
220 //
221 //
222 // Stream Settings
223 StreamSettingsObject streaming;
224
225 if (net == "tcp")
226 {
227 streaming.tcpSettings->header->type = type;
228 }
229 else if (net == "http" || net == "h2")
230 {
231 // Fill hosts for HTTP
232 for (const auto &_host : host.split(','))
233 {
234 if (!_host.isEmpty())
235 {
236 streaming.httpSettings->host << _host.trimmed();
237 }
238 }
239
240 streaming.httpSettings->path = path;
241 }
242 else if (net == "ws")
243 {
244 if (!host.isEmpty())
245 (*streaming.wsSettings->headers)["Host"] = host;
246 streaming.wsSettings->path = path;
247 }
248 else if (net == "kcp")
249 {
250 streaming.kcpSettings->header->type = type;
251 }
252 else if (net == "domainsocket")
253 {
254 streaming.dsSettings->path = path;
255 }
256 else if (net == "quic")
257 {
258 streaming.quicSettings->security = host;
259 streaming.quicSettings->header->type = type;
260 streaming.quicSettings->key = path;
261 }
262
263 // FIXME: makeshift patch for #290.
264 // to be rewritten after refactoring.
265 if (tls == "tls" && host != "" && (net == "tcp" || net == "ws"))
266 {
267 streaming.tlsSettings->serverName = host;
268 // streaming.tlsSettings.allowInsecure = false;
269 }
270
271 streaming.security = tls;
272 //
273 // Network type
274 // NOTE(DuckSoft): Damn vmess:// just don't write 'http' properly
275 if (net == "h2")
276 net = "http";
277 streaming.network = net;
278 //
279 // VMess root config
280 OUTBOUNDSETTING vConf;
281 vConf["vnext"] = QJsonArray{ serv.toJson() };
282 const auto outbound = GenerateOutboundEntry(OUTBOUND_TAG_PROXY, "vmess", vConf, streaming.toJson());
283 root["outbounds"] = QJsonArray{ outbound };
284 // If previous alias is empty, just the PS is needed, else, append a "_"
285 *alias = alias->trimmed().isEmpty() ? ps : *alias + "_" + ps;
286 return root;
287#undef default
288 }
