10 QList<std::pair<QString, CONFIGROOT>> ConvertConfigFromString(const QString &link, QString *aliasPrefix, QString *errMessage,
11 QString *newGroup)
12 {
13
14 QList<std::pair<QString, CONFIGROOT>> connectionConf;
15 if (link.startsWith("vmess://") && link.contains("@"))
16 {
17 auto conf = vmess_new::Deserialize(link, aliasPrefix, errMessage);
18 connectionConf << std::pair{ *aliasPrefix, conf };
19 }
20 else if (link.startsWith("vless://"))
21 {
22 auto conf = vless::Deserialize(link, aliasPrefix, errMessage);
23 connectionConf << std::pair{ *aliasPrefix, conf };
24 }
25 else if (link.startsWith("vmess://"))
26 {
27 auto conf = vmess::Deserialize(link, aliasPrefix, errMessage);
28 connectionConf << std::pair{ *aliasPrefix, conf };
29 }
30 else if (link.startsWith("ss://") && !link.contains("plugin="))
31 {
32 auto conf = ss::Deserialize(link, aliasPrefix, errMessage);
33 connectionConf << std::pair{ *aliasPrefix, conf };
34 }
35 else if (link.startsWith("ssd://"))
36 {
37 QStringList errMessageList;
38 connectionConf << ssd::Deserialize(link, newGroup, &errMessageList);
39 *errMessage = errMessageList.join(NEWLINE);
40 }
41 else
42 {
43 bool ok = false;
44 const auto configs = PluginHost->TryDeserializeShareLink(link, aliasPrefix, errMessage, newGroup, ok);
45 if (ok)
46 {
47 errMessage->clear();
48 for (const auto &[_alias, _protocol, _outbound] : configs)
49 {
50 CONFIGROOT root;
51 auto outbound = GenerateOutboundEntry(OUTBOUND_TAG_PROXY, _protocol, OUTBOUNDSETTING(_outbound), {});
52 QJsonIO::SetValue(root, outbound, "outbounds", 0);
53 connectionConf << std::pair{ _alias, root };
54 }
55 }
56 else if (errMessage->isEmpty())
57 {
58 *errMessage = QObject::tr("Unsupported share link format.");
59 }
60 }
61
62 return connectionConf;
63 }

65 const QString ConvertConfigToString(const ConnectionGroupPair &identifier)
66 {
67 auto alias = GetDisplayName(identifier.connectionId);
68 if (IsComplexConfig(identifier.connectionId))
69 {
70 return QV2RAY_SERIALIZATION_COMPLEX_CONFIG_PLACEHOLDER;
71 }
72 auto server = ConnectionManager->GetConnectionRoot(identifier.connectionId);
73 return ConvertConfigToString(alias, GetDisplayName(identifier.groupId), server);
74 }

76 const QString ConvertConfigToString(const QString &alias, const QString &groupName, const CONFIGROOT &server)
77 {
78 const auto outbound = OUTBOUND(server["outbounds"].toArray().first().toObject());
79 const auto type = outbound["protocol"].toString();
80 const auto settings = outbound["settings"].toObject();
81 const auto streamSettings = outbound["streamSettings"].toObject();
82
83 QString sharelink;
84
85 if (type.isEmpty())
86 {
87 return "";
88 }
89
90 if (type == "vmess")
91 {
92 const auto vmessServer = VMessServerObject(settings["vnext"].toArray().first().toObject());
93 const auto transport = StreamSettingsObject(streamSettings);
94 sharelink = vmess_new::Serialize(transport, vmessServer, alias);
95 }
96 else if (type == "shadowsocks")
97 {
98 auto ssServer = ShadowSocksServerObject(settings["servers"].toArray().first().toObject());
99 sharelink = ss::Serialize(ssServer, alias);
100 }
101 else
102 {
103 bool ok = false;
104 sharelink = PluginHost->SerializeOutbound(type, settings, streamSettings, alias, groupName, &ok);
105 Q_UNUSED(ok)
106 }
107
108 return sharelink;
109 }
