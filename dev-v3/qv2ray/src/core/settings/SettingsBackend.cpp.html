27 bool CheckSettingsPathAvailability(const QString &_path, bool checkExistingConfig)
28 {
29 auto path = _path;
30
31 if (!path.endsWith("/"))
32 path.append("/");
33
34 // Does not exist.
35 if (!QDir(path).exists())
36 return false;
37
38 {
39 // A temp file used to test file permissions in that folder.
40 QFile testFile(path + ".qv2ray_test_file" + QSTRN(QTime::currentTime().msecsSinceStartOfDay()));
41
42 if (!testFile.open(QFile::OpenModeFlag::ReadWrite))
43 {
44 LOG("Directory at: " + path + " cannot be used as a valid config file path.");
45 LOG("---> Cannot create a new file or open a file for writing.");
46 return false;
47 }
48
49 testFile.write("Qv2ray test file, feel free to remove.");
50 testFile.flush();
51 testFile.close();
52
53 if (!testFile.remove())
54 {
55 // This is rare, as we can create a file but failed to remove it.
56 LOG("Directory at: " + path + " cannot be used as a valid config file path.");
57 LOG("---> Cannot remove a file.");
58 return false;
59 }
60 }
61
62 if (!checkExistingConfig)
63 {
64 // Just pass the test
65 return true;
66 }
67
68 // Check if an existing config is found.
69 QFile configFile(path + "Qv2ray.conf");
70
71 // No such config file.
72 if (!configFile.exists())
73 return false;
74
75 if (!configFile.open(QIODevice::ReadWrite))
76 {
77 LOG("File: " + configFile.fileName() + " cannot be opened!");
78 return false;
79 }
80
81 const auto err = VerifyJsonString(StringFromFile(configFile));
82 if (!err.isEmpty())
83 {
84 LOG("Json parse returns:", err);
85 return false;
86 }
87
88 // If the file format is valid.
89 const auto conf = JsonFromString(StringFromFile(configFile));
90 LOG("Found a config file," A(conf["config_version"].toString()) A(path));
91 configFile.close();
92 return true;
93 }

95 bool LocateConfiguration()
96 {
97 LOG("Application exec path: " + qApp->applicationDirPath());
98 // Non-standard paths needs special handing for "_debug"
99 const auto currentPathConfig = qApp->applicationDirPath() + "/config" QV2RAY_CONFIG_DIR_SUFFIX;
100 const auto homeQv2ray = QDir::homePath() + "/.qv2ray" QV2RAY_CONFIG_DIR_SUFFIX;
101 //
102 // Standard paths already handles the "_debug" suffix for us.
103 const auto configQv2ray = QStandardPaths::writableLocation(QStandardPaths::AppConfigLocation);
104 //
105 //
106 // Some built-in search paths for Qv2ray to find configs. (load the first one if possible).
107 const auto useManualConfigPath = qEnvironmentVariableIsSet(QV2RAY_CONFIG_PATH_ENV_NAME);
108 const auto manualConfigPath = qEnvironmentVariable(QV2RAY_CONFIG_PATH_ENV_NAME);
109 //
110 QStringList configFilePaths;
111 if (useManualConfigPath)
112 {
113 LOG("Using config path from env: " + manualConfigPath);
114 configFilePaths << manualConfigPath;
115 }
116 else
117 {
118 configFilePaths << currentPathConfig;
119 configFilePaths << configQv2ray;
120 configFilePaths << homeQv2ray;
121 }
122
123 QString configPath = "";
124 bool hasExistingConfig = false;
125 for (const auto &path : qAsConst(configFilePaths))
126 {
127 // Verify the config path, check if the config file exists and in the
128 // correct JSON format. True means we check for config existence as
129 // well. ---------------------------------------------|HERE|
130 bool isValidConfigPath = CheckSettingsPathAvailability(path, true);
131
132 // If we already found a valid config file. just simply load it...
133 if (hasExistingConfig)
134 break;
135
136 if (isValidConfigPath)
137 {
138 DEBUG("Path:", path, " is valid.");
139 configPath = path;
140 hasExistingConfig = true;
141 }
142 else
143 {
144 LOG("Path:", path, "does not contain a valid config file.");
145 }
146 }
147
148 if (hasExistingConfig)
149 {
150 // Use the config path found by the checks above
151 SetConfigDirPath(configPath);
152 LOG("Using ", QV2RAY_CONFIG_DIR, " as the config path.");
153 }
154 else
155 {
156 // If there's no existing config.
157 //
158 // Create new config at these dirs, these are default values for each platform.
159 if (useManualConfigPath)
160 {
161 configPath = manualConfigPath;
162 }
163 else
164 {
165#if defined(Q_OS_WIN) && !defined(QV2RAY_NO_ASIDECONFIG)
166 configPath = currentPathConfig;
167#else
168 configPath = configQv2ray;
169#endif
170 }
171
172 bool hasPossibleNewLocation = QDir().mkpath(configPath) && CheckSettingsPathAvailability(configPath, false);
173 // Check if the dirs are write-able
174 if (!hasPossibleNewLocation)
175 {
176 // None of the path above can be used as a dir for storing config.
177 // Even the last folder failed to pass the check.
178 LOG("FATAL");
179 LOG(" ---> CANNOT find a proper place to store Qv2ray config files.");
180 QvMessageBoxWarn(nullptr, QObject::tr("Cannot Start Qv2ray"),
181 QObject::tr("Cannot find a place to store config files.") + NEWLINE + //
182 QObject::tr("Qv2ray has searched these paths below:") + NEWLINE + NEWLINE + //
183 configFilePaths.join(NEWLINE) + NEWLINE + //
184 QObject::tr("It usually means you don't have the write permission to all of those locations.") + NEWLINE + //
185 QObject::tr("Qv2ray will now exit.")); //
186 return false;
187 }
188
189 // Found a valid config dir, with write permission, but assume no config is located in it.
190 LOG("Set " + configPath + " as the config path.");
191 SetConfigDirPath(configPath);
192
193 if (QFile::exists(QV2RAY_CONFIG_FILE))
194 {
195 // As we already tried to load config from every possible dir.
196 //
197 // This condition branch (!hasExistingConfig check) holds the fact that current config dir,
198 // should NOT contain any valid file (at least in the same name)
199 //
200 // It usually means that QV2RAY_CONFIG_FILE here has a corrupted JSON format.
201 //
202 // Otherwise Qv2ray would have loaded this config already instead of notifying to create a new config in this folder.
203 //
204 LOG("This should not occur: Qv2ray config exists but failed to load.");
205 QvMessageBoxWarn(nullptr, QObject::tr("Failed to initialise Qv2ray"),
206 QObject::tr("Failed to determine the location of config file:") + NEWLINE + //
207 QObject::tr("Qv2ray has found a config file, but it failed to be loaded due to some errors.") + NEWLINE + //
208 QObject::tr("A workaround is to remove the this file and restart Qv2ray:") + NEWLINE + //
209 QV2RAY_CONFIG_FILE + NEWLINE + //
210 QObject::tr("Qv2ray will now exit.") + NEWLINE + //
211 QObject::tr("Please report if you think it's a bug.")); //
212 return false;
213 }
214
215 GlobalConfig.kernelConfig->KernelPath(QV2RAY_DEFAULT_VCORE_PATH);
216 GlobalConfig.kernelConfig->AssetsPath(QV2RAY_DEFAULT_VASSETS_PATH);
217 GlobalConfig.logLevel = 3;
218 GlobalConfig.uiConfig->language = QLocale::system().name();
219 GlobalConfig.defaultRouteConfig->dnsConfig->servers->append({ "1.1.1.1" });
220 GlobalConfig.defaultRouteConfig->dnsConfig->servers->append({ "8.8.8.8" });
221 GlobalConfig.defaultRouteConfig->dnsConfig->servers->append({ "8.8.4.4" });
222
223 // Save initial config.
224 SaveGlobalSettings();
225 LOG("Created initial config file.");
226 }
227
228 if (!QDir(QV2RAY_GENERATED_DIR).exists())
229 {
230 // The dir used to generate final config file, for V2Ray interaction.
231 QDir().mkdir(QV2RAY_GENERATED_DIR);
232 LOG("Created config generation dir at: " + QV2RAY_GENERATED_DIR);
233 }
234 //
235 // BEGIN LOAD CONFIGURATIONS
236 //
237 {
238 // Load the config for upgrade, but do not parse it to the struct.
239 auto conf = JsonFromString(StringFromFile(QV2RAY_CONFIG_FILE));
240 const auto configVersion = conf["config_version"].toInt();
241
242 if (configVersion > QV2RAY_CONFIG_VERSION)
243 {
244 // Config version is larger than the current version...
245 // This is rare but it may happen....
246 QvMessageBoxWarn(nullptr, QObject::tr("Qv2ray Cannot Continue"), //
247 QObject::tr("You are running a lower version of Qv2ray compared to the current config file.") + NEWLINE + //
248 QObject::tr("Please check if there's an issue explaining about it.") + NEWLINE + //
249 QObject::tr("Or submit a new issue if you think this is an error.") + NEWLINE + NEWLINE + //
250 QObject::tr("Qv2ray will now exit."));
251 return false;
252 }
253 else if (configVersion < QV2RAY_CONFIG_VERSION)
254 {
255 // That is the config file needs to be upgraded.
256 conf = Qv2ray::UpgradeSettingsVersion(configVersion, QV2RAY_CONFIG_VERSION, conf);
257 }
258
259 // Let's save the config.
260 GlobalConfig.loadJson(conf);
261 SaveGlobalSettings();
262 return true;
263 }
264 }
