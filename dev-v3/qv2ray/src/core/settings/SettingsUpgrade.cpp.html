2// This file handles some important migration
3// from old to newer versions of Qv2ray.

6#include "base/Qv2rayBase.hpp"
7#include "utils/QvHelpers.hpp"
8
9#define QV_MODULE_NAME "SettingsUpgrade"
10
11#define UPGRADELOG(msg) LOG("[" + QSTRN(fromVersion) + "-" + QSTRN(fromVersion + 1) + "] --> " + msg)
12
13namespace Qv2ray
14{
15 // Private member
16 QJsonObject UpgradeConfig_Inc(int fromVersion, const QJsonObject &original)
17 {
18 auto root = original;
19 switch (fromVersion)
20 {
21 default:
22 {
23 //
24 // Due to technical issue, we cannot maintain all of those upgrade processes anymore. Check
25 // https://github.com/Qv2ray/Qv2ray/issues/353#issuecomment-586117507
26 // https://github.com/Qv2ray/Qv2ray/issues/1284#issuecomment-778776959
27 // for more information, see commit 2f716a9a443b71ddb96aaab081de73c0095cb637
28 //
29 QvMessageBoxWarn(nullptr, QObject::tr("Configuration Upgrade Failed"),
30 QObject::tr("Unsupported config version number: ") + QSTRN(fromVersion) + NEWLINE + NEWLINE +
31 QObject::tr("Please go to https://github.com/Qv2ray/Qv2ray/issues to check for related announcements."));
32 LOG("The configuration version of your old Qv2ray installation is out-of-date and that"
33 " version is not supported anymore, please try to update to an intermediate version of Qv2ray first.");
34 qApp->exit(1);
35 }
36 }
37 root["config_version"] = root["config_version"].toInt() + 1;
38 return root;
39 }
40
41 // Exported function
42 QJsonObject UpgradeSettingsVersion(int fromVersion, int toVersion, const QJsonObject &original)
43 {
44 auto root = original;
45 LOG("Migrating config from version ", fromVersion, "to", toVersion);
46
47 for (int i = fromVersion; i < toVersion; i++)
48 {
49 root = UpgradeConfig_Inc(i, root);
50 }
51
52 return root;
53 }
54} // namespace Qv2ray
