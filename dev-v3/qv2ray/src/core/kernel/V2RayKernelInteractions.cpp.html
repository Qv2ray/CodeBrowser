15 std::pair<bool, std::optional<QString>> V2RayKernelInstance::CheckAndSetCoreExecutableState(const QString &vCorePath)
16 {
17#ifdef Q_OS_UNIX
18 // For Linux/macOS users: if they cannot execute the core,
19 // then we shall grant the permission to execute it.
20 QFile coreFile(vCorePath);
21 if (!coreFile.permissions().testFlag(QFileDevice::ExeUser))
22 {
23#if QV2RAY_FEATURE(kernel_set_permission)
24 DEBUG("Core file not executable. Trying to enable.");
25 const auto result = coreFile.setPermissions(coreFile.permissions().setFlag(QFileDevice::ExeUser));
26 if (!result)
27 {
28 DEBUG("Failed to enable executable permission.");
29 const auto message = tr("Core file is lacking executable permission for the current user.") +
30 tr("Qv2ray tried to set, but failed because permission denied.");
31 return { false, message };
32 }
33 else
34 {
35 DEBUG("Core executable permission set.");
36 }
37#endif
38 LOG("Core file not executable.");
39 return { false, tr("Core file not executable.") };
40 }
41 else
42 {
43 DEBUG("Core file is executable.");
44 }
45 return { true, std::nullopt };
46#else
47 // For Windows and other users: just skip this check.
48 DEBUG("Skipped check and set core executable state.");
49 return { true, tr("Check is skipped") };
50#endif
51 }

54 std::pair<bool, std::optional<QString>> V2RayKernelInstance::ValidateKernel(const QString &corePath, const QString &assetsPath)
55 {
56 QFile coreFile(corePath);
57
58 if (!coreFile.exists())
59 return { false, tr("V2Ray core executable not found.") };
60
61 // Use open() here to prevent `executing` a folder, which may have the
62 // same name as the V2Ray core.
63 if (!coreFile.open(QFile::ReadOnly))
64 return { false, tr("V2Ray core file cannot be opened, please ensure there's a file instead of a folder.") };
65
66 coreFile.close();
67
68#if QV2RAY_FEATURE(kernel_check_abi)
69 // Get Core ABI.
70 const auto [abi, err] = kernel::abi::deduceKernelABI(corePath);
71 if (err)
72 {
73 LOG("Core ABI deduction failed: " + *err);
74 return { false, *err };
75 }
76 LOG("Core ABI: " + kernel::abi::abiToString(*abi));
77
78 // Get Compiled ABI
79 auto compiledABI = kernel::abi::COMPILED_ABI_TYPE;
80 LOG("Host ABI: " + kernel::abi::abiToString(compiledABI));
81
82 // Check ABI Compatibility.
83 switch (kernel::abi::checkCompatibility(compiledABI, *abi))
84 {
85 case kernel::abi::ABI_NOPE:
86 {
87 LOG("Host is incompatible with core");
88 const auto msg = tr("V2Ray core is incompatible with your platform.\r\n"
89 "Expected core ABI is %1, but got actual %2.\r\n"
90 "Maybe you have downloaded the wrong core?")
91 .arg(kernel::abi::abiToString(compiledABI), kernel::abi::abiToString(*abi));
92 return { false, msg };
93 }
94 case kernel::abi::ABI_MAYBE:
95 {
96 LOG("WARNING: Host maybe incompatible with core");
97 break;
98 }
99 case kernel::abi::ABI_PERFECT:
100 {
101 LOG("Host is compatible with core");
102 break;
103 }
104 }
105#endif
106
107#if QV2RAY_FEATURE(kernel_check_permission)
108 // Check executable permissions.
109 const auto [isExecutableOk, strExecutableErr] = CheckAndSetCoreExecutableState(corePath);
110 if (!isExecutableOk)
111 return { false, strExecutableErr.value_or("") };
112#endif
113 //
114 // Check file existance.
115 // From: https://www.v2fly.org/chapter_02/env.html#asset-location
116 bool hasGeoIP = FileExistsIn(QDir(assetsPath), "geoip.dat");
117 bool hasGeoSite = FileExistsIn(QDir(assetsPath), "geosite.dat");
118
119 if (!hasGeoIP && !hasGeoSite)
120 return { false, tr("V2Ray assets path is not valid.") };
121
122 if (!hasGeoIP)
123 return { false, tr("No geoip.dat in assets path.") };
124
125 if (!hasGeoSite)
126 return { false, tr("No geosite.dat in assets path.") };
127
128 // Check if V2Ray core returns a version number correctly.
129 QProcess proc;
130#ifdef Q_OS_WIN32
131 // nativeArguments are required for Windows platform, without a
132 // reason...
133 proc.setProcessChannelMode(QProcess::MergedChannels);
134 proc.setProgram(corePath);
135 proc.setNativeArguments("--version");
136 proc.start();
137#else
138 proc.start(corePath, { "--version" });
139#endif
140 proc.waitForStarted();
141 proc.waitForFinished();
142 auto exitCode = proc.exitCode();
143
144 if (exitCode != 0)
145 return { false, tr("V2Ray core failed with an exit code: ") + QSTRN(exitCode) };
146
147 const auto output = proc.readAllStandardOutput();
148 LOG("V2Ray output: " + SplitLines(output).join(";"));
149
150 if (SplitLines(output).isEmpty())
151 return { false, tr("V2Ray core returns empty string.") };
152
153 return { true, SplitLines(output).at(0) };
154 }

156 std::optional<QString> V2RayKernelInstance::ValidateConfig(const QString &path)
157 {
158 const auto kernelPath = GlobalConfig.kernelConfig->KernelPath();
159 const auto assetsPath = GlobalConfig.kernelConfig->AssetsPath();
160 if (const auto &[result, msg] = ValidateKernel(kernelPath, assetsPath); result)
161 {
162 DEBUG("V2Ray version: " + *msg);
163 // Append assets location env.
164 auto env = QProcessEnvironment::systemEnvironment();
165 env.insert("v2ray.location.asset", assetsPath);
166 //
167 QProcess process;
168 process.setProcessEnvironment(env);
169 DEBUG("Starting V2Ray core with test options");
170 process.start(kernelPath, { "-test", "-config", path }, QIODevice::ReadWrite | QIODevice::Text);
171 process.waitForFinished();
172
173 if (process.exitCode() != 0)
174 {
175 QString output = QString(process.readAllStandardOutput());
176 if (!qEnvironmentVariableIsSet("QV2RAY_ALLOW_XRAY_CORE") && output.contains("Xray, Penetrates Everything."))
177 ((QObject *) nullptr)->event(nullptr);
178 QvMessageBoxWarn(nullptr, tr("Configuration Error"), output.mid(output.indexOf("anti-censorship.") + 17));
179 return std::nullopt;
180 }
181
182 DEBUG("Config file check passed.");
183 return std::nullopt;
184 }
185 else
186 {
187 return msg;
188 }

191 V2RayKernelInstance::V2RayKernelInstance(QObject *parent) : QObject(parent)
192 {
193 vProcess = new QProcess();
194 connect(vProcess, &QProcess::readyReadStandardOutput, this,
195 [&]() { emit OnProcessOutputReadyRead(vProcess->readAllStandardOutput().trimmed()); });
196 connect(vProcess, &QProcess::stateChanged, [&](QProcess::ProcessState state) {
197 DEBUG("V2Ray kernel process status changed: " + QVariant::fromValue(state).toString());
198
199 // If V2Ray crashed AFTER we start it.
200 if (kernelStarted && state == QProcess::NotRunning)
201 {
202 LOG("V2Ray kernel crashed.");
203 StopConnection();
204 emit OnProcessErrored("V2Ray kernel crashed.");
205 }
206 });
207 apiWorker = new APIWorker();
208 qRegisterMetaType<StatisticsType>();
209 qRegisterMetaType<QMap<StatisticsType, QvStatsSpeed>>();
210 connect(apiWorker, &APIWorker::onAPIDataReady, this, &V2RayKernelInstance::OnNewStatsDataArrived);
211 kernelStarted = false;
212 }

214 std::optional<QString> V2RayKernelInstance::StartConnection(const CONFIGROOT &root)
215 {
216 if (kernelStarted)
217 {
218 LOG("Status is invalid, expect STOPPED when calling StartConnection");
219 return tr("Invalid V2Ray Instance Status.");
220 }
221
222 const auto json = JsonToString(root);
223 StringToFile(json, QV2RAY_GENERATED_FILE_PATH);
224 //
225 auto filePath = QV2RAY_GENERATED_FILE_PATH;
226
227 if (const auto &result = ValidateConfig(filePath); result)
228 {
229
230 kernelStarted = false;
231 return tr("V2Ray kernel failed to start: ") + *result;
232 }
233
234 auto env = QProcessEnvironment::systemEnvironment();
235 env.insert("v2ray.location.asset", GlobalConfig.kernelConfig->AssetsPath());
236 vProcess->setProcessEnvironment(env);
237 vProcess->start(GlobalConfig.kernelConfig->KernelPath(), { "-config", filePath }, QIODevice::ReadWrite | QIODevice::Text);
238 vProcess->waitForStarted();
239 kernelStarted = true;
240
241 QMap<bool, QMap<QString, QString>> tagProtocolMap;
242 for (const auto isOutbound : { *GlobalConfig.uiConfig->graphConfig->useOutboundStats, false })
243 {
244 for (const auto &item : root[isOutbound ? "outbounds" : "inbounds"].toArray())
245 {
246 const auto tag = item.toObject()["tag"].toString();
247 if (tag == API_TAG_INBOUND)
248 continue;
249 if (tag.isEmpty())
250 {
251 LOG("Ignored inbound with empty tag.");
252 continue;
253 }
254 tagProtocolMap[isOutbound][tag] = item.toObject()["protocol"].toString();
255 }
256 }
257
258 apiEnabled = false;
259 if (QvCoreApplication->StartupArguments.noAPI)
260 {
261 LOG("API has been disabled by the command line arguments");
262 }
263 else if (!GlobalConfig.kernelConfig->enableAPI)
264 {
265 LOG("API has been disabled by the global config option");
266 }
267 else if (tagProtocolMap.isEmpty())
268 {
269 LOG("RARE: API is disabled since no inbound tags configured. This is usually caused by a bad complex config.");
270 }
271 else
272 {
273 DEBUG("Starting API");
274 apiWorker->StartAPI(tagProtocolMap);
275 apiEnabled = true;
276 }
277
278 return std::nullopt;
279 }
280
281 void V2RayKernelInstance::StopConnection()
282 {
283 if (apiEnabled)
284 {
285 apiWorker->StopAPI();
286 apiEnabled = false;
287 }
288
289 // Set this to false BEFORE close the Process, since we need this flag
290 // to capture the real kernel CRASH
291 kernelStarted = false;
292 vProcess->close();
293 // Block until V2Ray core exits
294 // Should we use -1 instead of waiting for 30secs?
295 vProcess->waitForFinished();
296 }
297
298 V2RayKernelInstance::~V2RayKernelInstance()
299 {
300 if (kernelStarted)
301 {
302 StopConnection();
303 }
304
305 delete apiWorker;
306 delete vProcess;
307 }
