14 const QString GenerateRandomString(int len)
15 {
16 const QString possibleCharacters("abcdefghijklmnopqrstuvwxyz");
17 QString randomString;
18
19 for (int i = 0; i < len; ++i)
20 {
21 uint rand = QRandomGenerator::system()->generate();
22 uint max = static_cast<uint>(possibleCharacters.length());
23 QChar nextChar = possibleCharacters[rand % max];
24 randomString.append(nextChar);
25 }
26
27 return randomString;
28 }

36 QString StringFromFile(QFile &source)
37 {
38 bool wasOpened = source.isOpen();
39 if (!wasOpened)
40 source.open(QFile::ReadOnly);
41 auto byteArray = source.readAll();
42 if (!wasOpened)
43 source.close();
44#if QT_VERSION >= QT_VERSION_CHECK(6, 0, 0)
45 for (const auto &encoding : { QStringDecoder::Utf8, QStringDecoder::Utf16, QStringDecoder::System })
46 {
47 auto converter = QStringDecoder(encoding, QStringConverter::Flag::ConvertInvalidToNull);
48 const auto data = converter(byteArray).data;
49 if (!data.contains('\0'))
50 return data;
51 }
52 Q_ASSERT_X(false, Q_FUNC_INFO, "Unsupported File Encoding");
53 return "";
54#else
55 QTextCodec *codec = QTextCodec::codecForName("UTF-8");
56 QTextCodec::ConverterState state;
57 const QString text = codec->toUnicode(byteArray.constData(), byteArray.size(), &state);
58 if (state.invalidChars > 0)
59 {
60 LOG("Not a valid UTF-8 sequence: " + source.fileName());
61 }
62 return state.invalidChars > 0 ? byteArray : text;
63#endif
64 }

67 {

75 QSaveFile f{ targetpath };
76 f.open(QIODevice::WriteOnly);
77 f.write(text.toUtf8());
78 f.commit();
79 return override;

141 QString SafeBase64Encode(const QString &string, bool trim)
142 {
143 const auto base64 = QString(string.toUtf8().toBase64()).replace(QChar('+'), QChar('-')).replace(QChar('/'), QChar('_'));
144 if (!trim)
145 {
146 return base64;
147 }
148 auto tmp = base64;
149 auto crbedin = tmp.crbegin();
150 auto idx = tmp.length();
151 while (crbedin != tmp.crend() && (*crbedin) == '=')
152 idx -= 1, crbedin++;
153 return idx != tmp.length() ? tmp.remove(idx, tmp.length() - idx) : tmp;
154 }

187 QString FormatBytes(const int64_t b)
188 {
189 const static char *sizes[5] = { "B", "KB", "MB", "GB", "TB" };
190 auto _bytes = b;
191 char str[64];
192 int i;
193
194 double dblByte = _bytes;
195 for (i = 0; i < 5 && _bytes >= 1000; i++, _bytes /= 1000)
196 dblByte = _bytes / 1000.0;
197
198 sprintf(str, "%.2f", dblByte);
199 return QString(str) + " " + QString(sizes[i]);
200 }

207 QString RemoveInvalidFileName(const QString &fileName)
208 {
209 const static QString pattern = R"("/\?%&^*;:|><)";
210 const static auto find_func = [](QChar c) { return pattern.contains(c); };
211 auto s = fileName;
212 std::replace_if(s.begin(), s.end(), find_func, '_');
213 return s;
214 }

216 /// This returns a file name without extensions.
217 void DeducePossibleFileName(const QString &baseDir, QString *fileName, const QString &extension)
218 {
219 int i = 1;
220
221 if (!QDir(baseDir).exists())
222 {
223 QDir(baseDir).mkpath(baseDir);
224 LOG("Making path: " + baseDir);
225 }
226
227 while (true)
228 {
229 if (!QFile(baseDir + "/" + *fileName + "_" + QSTRN(i) + extension).exists())
230 {
231 *fileName = *fileName + "_" + QSTRN(i);
232 return;
233 }
234 else
235 {
236 DEBUG("File with name: " + *fileName + "_" + QSTRN(i) + extension + " already exists");
237 }
238
239 i++;
240 }
