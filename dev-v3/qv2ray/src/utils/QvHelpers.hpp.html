74 inline bool IsIPv6Address(const QString &addr)
75 {
76 QHostAddress address(addr);
77 return QAbstractSocket::IPv6Protocol == address.protocol();
78 }
79
80 inline bool IsValidIPAddress(const QString &addr)

85 inline bool IsValidDNSServer(const QString &addr)
86 {

88 || IsIPv6Address(addr) //
89 || addr.startsWith("https://") //
90 || addr.startsWith("https+local://") //
91 || addr.startsWith("quic+local://") //
92 || addr == "localhost" //
93 || addr == "fakedns";
