49const QString SayLastWords() noexcept
50{
51 QStringList msg;
52 msg << "------- BEGIN QV2RAY CRASH REPORT -------";
53
54 {
55#ifdef Q_OS_WIN
56 void *stack[1024];
57 HANDLE process = GetCurrentProcess();
58 SymInitialize(process, NULL, TRUE);
59 SymSetOptions(SYMOPT_LOAD_ANYTHING);
60 WORD numberOfFrames = CaptureStackBackTrace(0, 1024, stack, NULL);
61 SYMBOL_INFO *symbol = (SYMBOL_INFO *) malloc(sizeof(SYMBOL_INFO) + (512 - 1) * sizeof(TCHAR));
62 symbol->MaxNameLen = 512;
63 symbol->SizeOfStruct = sizeof(SYMBOL_INFO);
64 DWORD displacement;
65 IMAGEHLP_LINE64 *line = (IMAGEHLP_LINE64 *) malloc(sizeof(IMAGEHLP_LINE64));
66 line->SizeOfStruct = sizeof(IMAGEHLP_LINE64);
67 //
68 for (int i = 0; i < numberOfFrames; i++)
69 {
70 const auto address = (DWORD64) stack[i];
71 SymFromAddr(process, address, NULL, symbol);
72 if (SymGetLineFromAddr64(process, address, &displacement, line))
73 {
74 msg << QString("[%1]: %2 (%3:%4)").arg(symbol->Address).arg(symbol->Name).arg(line->FileName).arg(line->LineNumber);
75 }
76 else
77 {
78 msg << QString("[%1]: %2 SymGetLineFromAddr64[%3]").arg(symbol->Address).arg(symbol->Name).arg(GetLastError());
79 }
80 }
81#endif
82 }
83
84 if (KernelInstance)
85 {
86 msg << "Active Kernel Instances:";
87 const auto kernels = KernelInstance->GetActiveKernelProtocols();
88 msg << JsonToString(JsonStructHelper::Serialize(static_cast<QList<QString>>(kernels)).toArray(), QJsonDocument::Compact);
89 msg << "Current Connection:";
90 //
91 const auto currentConnection = KernelInstance->CurrentConnection();
92 msg << JsonToString(currentConnection.toJson(), QJsonDocument::Compact);
93 msg << NEWLINE;
94 //
95 if (ConnectionManager && !currentConnection.isEmpty())
96 {
97 msg << "Active Connection Settings:";
98 const auto connection = ConnectionManager->GetConnectionMetaObject(currentConnection.connectionId);
99 auto group = ConnectionManager->GetGroupMetaObject(currentConnection.groupId);
100 //
101 // Do not collect private data.
102 // msg << NEWLINE;
103 // msg << JsonToString(ConnectionManager->GetConnectionRoot(currentConnection.connectionId));
104 group.subscriptionOption->address = "HIDDEN";
105 //
106 msg << JsonToString(connection.toJson(), QJsonDocument::Compact);
107 msg << NEWLINE;
108 msg << "Group:";
109 msg << JsonToString(group.toJson(), QJsonDocument::Compact);
110 msg << NEWLINE;
111 }
112 }
113
114 if (PluginHost)
115 {
116 msg << "Plugins:";
117 const auto plugins = PluginHost->AllPlugins();
118 for (const auto &plugin : plugins)
119 {
120 const auto data = PluginHost->GetPlugin(plugin)->metadata;
121 QList<QString> dataList;
122 dataList << data.Name;
123 dataList << data.Author;
124 dataList << data.InternalName;
125 dataList << data.Description;
126 msg << JsonToString(JsonStructHelper::Serialize(dataList).toArray(), QJsonDocument::Compact);
127 }
128 msg << NEWLINE;
129 }
130
131 if (QvCoreApplication)
132 {
133 msg << "GlobalConfig:";
134 msg << JsonToString(GlobalConfig.toJson(), QJsonDocument::Compact);
135 }
136
137 msg << "------- END OF QV2RAY CRASH REPORT -------";
138 return msg.join(NEWLINE);
139}

141void signalHandler(int signum)
142{
143#ifndef Q_OS_WIN
144 if (signum == SIGTRAP)
145 {
146 exit(-99);
147 return;
148 }
149#endif
150 std::cout << "Qv2ray: Interrupt signal (" << signum << ") received." << std::endl;
151
152 if (signum == SIGTERM)
153 {
154 if (qApp)
155 qApp->exit();
156 return;
157 }
158 std::cout << "Collecting StackTrace" << std::endl;
159 const auto msg = "Signal: " + QSTRN(signum) + NEWLINE + SayLastWords();
160 std::cout << msg.toStdString() << std::endl;
161
162 if (qApp && QvCoreApplication)
163 {
164 QDir().mkpath(QV2RAY_CONFIG_DIR + "bugreport/");
165 const auto filePath = QV2RAY_CONFIG_DIR + "bugreport/QvBugReport_" + QSTRN(system_clock::to_time_t(system_clock::now())) + ".stacktrace";
166 StringToFile(msg, filePath);
167 std::cout << "Backtrace saved in: " + filePath.toStdString() << std::endl;
168 const auto message = QObject::tr("Qv2ray has encountered an uncaught exception: ") + NEWLINE + //
169 QObject::tr("Please report a bug via Github with the file located here: ") + NEWLINE + //
170 NEWLINE + filePath;
171 BootstrapMessageBox("UNCAUGHT EXCEPTION", message);
172 }
173
174#if defined Q_OS_WIN || defined QT_DEBUG
175 exit(-99);
176#else
177 kill(getpid(), SIGTRAP);
178#endif
179}

189int main(int argc, char *argv[])
190{
191 globalArgc = argc;
192 globalArgv = argv;
193 // Register signal handlers.
194 signal(SIGABRT, signalHandler);
195 signal(SIGSEGV, signalHandler);
196 signal(SIGTERM, signalHandler);
197#ifndef Q_OS_WIN
198 signal(SIGHUP, signalHandler);
199 signal(SIGKILL, signalHandler);
200#else
201 // AddVectoredExceptionHandler(0, TopLevelExceptionHandler);
202#endif
203 //
204 // This line must be called before any other ones, since we are using these
205 // values to identify instances.
206 QCoreApplication::setApplicationVersion(QV2RAY_VERSION_STRING);
207
208#ifdef QT_DEBUG
209 QCoreApplication::setApplicationName("qv2ray_debug");
210#else
211 QCoreApplication::setApplicationName("qv2ray");
212#endif
213
214#ifdef QV2RAY_GUI
215 QApplication::setApplicationDisplayName("Qv2ray");
216#endif
217
218#ifdef QT_DEBUG
219 std::cerr << "WARNING: ================ This is a debug build, many features are not stable enough. ================" << std::endl;
220#else
221// #error "Do not use Qv2ray v3.0 in Production"
222#endif
223
224 if (qEnvironmentVariableIsSet("QV2RAY_NO_SCALE_FACTORS"))
225 {
226 LOG("Force set QT_SCALE_FACTOR to 1.");
227 DEBUG("UI", "Original QT_SCALE_FACTOR was:", qEnvironmentVariable("QT_SCALE_FACTOR"));
228 qputenv("QT_SCALE_FACTOR", "1");
229 }
230 else
231 {
232 DEBUG("High DPI scaling is enabled.");
233#if QT_VERSION >= QT_VERSION_CHECK(5, 14, 0)
234#ifdef QV2RAY_GUI
235 QGuiApplication::setHighDpiScaleFactorRoundingPolicy(Qt::HighDpiScaleFactorRoundingPolicy::PassThrough);
236#endif
237#endif
238 }
239
240 Qv2rayApplication app(argc, argv);
241 if (const auto list = app.CheckPrerequisites(); !list.isEmpty())
242 {
243 BootstrapMessageBox("Qv2ray Prerequisites Check Failed", list.join(NEWLINE));
244 return Qv2rayExitReason::EXIT_PRECONDITION_FAILED;
245 }
246
247 if (!app.Initialize())
248 {
249 const auto reason = app.GetExitReason();
250 if (reason == EXIT_INITIALIZATION_FAILED)
251 {
252 BootstrapMessageBox("Qv2ray Initialization Failed", "PreInitialization Failed." NEWLINE "For more information, please see the log.");
253 LOG("Qv2ray initialization failed:", reason);
254 }
255 return reason;
256 }
257
258#ifndef Q_OS_WIN
259 signal(SIGUSR1, [](int) { ConnectionManager->RestartConnection(); });
260 signal(SIGUSR2, [](int) { ConnectionManager->StopConnection(); });
261#endif
262
263 app.RunQv2ray();
264 const auto reason = app.GetExitReason();
265 if (reason == EXIT_NEW_VERSION_TRIGGER)
266 {
267 LOG("Starting new version of Qv2ray: " + app.StartupArguments._qvNewVersionPath);
268 QProcess::startDetached(app.StartupArguments._qvNewVersionPath, {});
269 }
270 return reason;
271}
