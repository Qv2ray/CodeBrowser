26 NtpTimestamp NtpTimestamp::fromDateTime(const QDateTime &dateTime)
27 {
28 qint64 ntpMSecs = dateTime.toMSecsSinceEpoch() - january_1_1900;
29
30 quint32 seconds = ntpMSecs / 1000;
31 quint32 fraction = 0x100000000ll * (ntpMSecs % 1000) / 1000;
32
33 NtpTimestamp result;
34 result.seconds = qToBigEndian(seconds);
35 result.fraction = qToBigEndian(fraction);
36 return result;
37 }
38
39 QDateTime NtpTimestamp::toDateTime(const NtpTimestamp &ntpTime)
40 {
41 /* Convert to local-endian. */
42 quint32 seconds = qFromBigEndian(ntpTime.seconds);
43 quint32 fraction = qFromBigEndian(ntpTime.fraction);
44
45 /* Convert NTP timestamp to number of milliseconds passed since Jan 1 1900. */
46 qint64 ntpMSecs = seconds * 1000ll + fraction * 1000ll / 0x100000000ll;
47
48 /* Construct Qt date time. */
49 return QDateTime::fromMSecsSinceEpoch(ntpMSecs + january_1_1900);
50 }

159 void NtpClient::init(const QHostAddress &bindAddress, quint16 bindPort)
160 {

172 bool NtpClient::sendRequest(const QHostAddress &address, quint16 port)
173 {
174 if (mSocket->state() != QAbstractSocket::BoundState)
175 return false;
176
177 /* Initialize the NTP packet. */
178 NtpPacket packet;
179 memset(&packet, 0, sizeof(packet));
180 packet.flags.mode = ClientMode;
181 packet.flags.versionNumber = 4;
182 packet.transmitTimestamp = NtpTimestamp::fromDateTime(QDateTime::currentDateTimeUtc());
183
184 /* Send it. */
185 if (mSocket->writeDatagram(reinterpret_cast<const char *>(&packet), sizeof(packet), address, port) < 0)
186 return false;
187
188 return true;
189 }
190
191 void NtpClient::readPendingDatagrams()
192 {
193 while (mSocket->hasPendingDatagrams())
194 {
195 NtpFullPacket packet;
196 memset(&packet, 0, sizeof(packet));
197
198 QHostAddress address;
199 quint16 port;
200
201 if (mSocket->readDatagram(reinterpret_cast<char *>(&packet), sizeof(packet), &address, &port) < (qint64) sizeof(NtpPacket))
202 continue;
203
204 QDateTime now = QDateTime::currentDateTime();
205
206 /* Prepare reply. */
207 NtpReplyPrivate *replyPrivate = new NtpReplyPrivate();
208 replyPrivate->packet = packet;
209 replyPrivate->destinationTime = now;
210 NtpReply reply(replyPrivate);
211
212 /* Notify. */
213 Q_EMIT replyReceived(address, port, reply);
214 }
