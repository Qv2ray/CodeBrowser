29 int QvPluginHost::refreshPluginList()
30 {
31 clearPlugins();
32 LOG("Reloading plugin list");
33 for (const auto &pluginDirPath : QvCoreApplication->GetAssetsPaths("plugins"))
34 {
35 const QStringList entries = QDir(pluginDirPath).entryList(QDir::Files);
36 for (const auto &fileName : entries)
37 {
38 DEBUG("Loading plugin: " + fileName + " from: " + pluginDirPath);
39 //
40 QvPluginInfo info;
41 auto pluginFullPath = QDir(pluginDirPath).absoluteFilePath(fileName);
42 info.libraryPath = pluginFullPath;
43 info.pluginLoader = new QPluginLoader(pluginFullPath, this);
44 // auto meta = pluginLoader.metaData();
45 // You should not call "delete" on this object, it's handled by the QPluginLoader
46 QObject *plugin = info.pluginLoader->instance();
47 if (plugin == nullptr)
48 {
49 const auto errMessage = info.pluginLoader->errorString();
50 LOG(errMessage);
51 QvMessageBoxWarn(nullptr, tr("Failed to load plugin"), errMessage);
52 continue;
53 }
54 info.pluginInterface = qobject_cast<Qv2rayInterface *>(plugin);
55 if (info.pluginInterface == nullptr)
56 {
57 LOG("Failed to cast from QObject to Qv2rayPluginInterface");
58 info.pluginLoader->unload();
59 continue;
60 }
61
62 if (info.pluginInterface->QvPluginInterfaceVersion != QV2RAY_PLUGIN_INTERFACE_VERSION)
63 {
64 // The plugin was built for a not-compactable version of Qv2ray. Don't load the plugin by default.
65 LOG(info.libraryPath + " is built with an older Interface, ignoring");
66 QvMessageBoxWarn(nullptr, tr("Cannot load plugin"),
67 tr("The plugin cannot be loaded: ") + NEWLINE + info.libraryPath + NEWLINE NEWLINE +
68 tr("This plugin was built against a different version of the Plugin Interface.") + NEWLINE +
69 tr("Please contact the plugin provider or report the issue to Qv2ray Workgroup."));
70 info.pluginLoader->unload();
71 continue;
72 }
73 info.metadata = info.pluginInterface->GetMetadata();
74 if (plugins.contains(info.metadata.InternalName))
75 {
76 LOG("Found another plugin with the same internal name: " + info.metadata.InternalName + ". Skipped");
77 continue;
78 }
79 connect(plugin, SIGNAL(PluginLog(const QString &)), this, SLOT(QvPluginLog(const QString &)));
80 connect(plugin, SIGNAL(PluginErrorMessageBox(const QString &, const QString &)), this,
81 SLOT(QvPluginMessageBox(const QString &, const QString &)));
82 LOG("Loaded plugin: \"" + info.metadata.Name + "\" made by: \"" + info.metadata.Author + "\"");
83 plugins.insert(info.metadata.InternalName, info);
84 }
85 }
86 return plugins.count();
87 }

148 bool QvPluginHost::initializePlugin(const QString &internalName)
149 {
150 const auto &plugin = plugins[internalName];
151 if (plugin.isLoaded)
152 {
153 LOG("The plugin: \"" + internalName + "\" has already been loaded.");
154 return true;
155 }
156 if (!GlobalConfig.pluginConfig->pluginStates->contains(internalName))
157 {
158 // If not contained, default to enabled.
159 (*GlobalConfig.pluginConfig->pluginStates)[internalName] = true;
160 }
161 // If the plugin is disabled
162 if (!GlobalConfig.pluginConfig->pluginStates->value(internalName))
163 {
164 LOG("Cannot load a plugin that's been disabled.");
165 return false;
166 }
167
168 auto conf = JsonFromString(StringFromFile(QV2RAY_PLUGIN_SETTINGS_DIR + internalName + ".conf"));
169 plugins[internalName].pluginInterface->InitializePlugin(QV2RAY_PLUGIN_SETTINGS_DIR + internalName + "/", conf);
170 plugins[internalName].isLoaded = true;
171 return true;
172 }

227 const QList<std::tuple<QString, QString, QJsonObject>> QvPluginHost::TryDeserializeShareLink(const QString &sharelink, //
228 QString *aliasPrefix, //
229 QString *errMessage, //
230 QString *newGroupName, //
231 bool &ok) const
232 {
233 Q_UNUSED(newGroupName)
234 QList<std::tuple<QString, QString, QJsonObject>> data;
235 ok = false;
236 for (const auto &plugin : plugins)
237 {
238 if (plugin.isLoaded && plugin.metadata.Components.contains(COMPONENT_OUTBOUND_HANDLER))
239 {
240 auto serializer = plugin.pluginInterface->GetOutboundHandler();
241 bool thisPluginCanHandle = false;
242 for (const auto &prefix : serializer->SupportedLinkPrefixes())
243 {
244 thisPluginCanHandle = thisPluginCanHandle || sharelink.startsWith(prefix);
245 }
246 if (thisPluginCanHandle)
247 {
248 // Populate Plugin Options
249 // Currently there're no options
250 //{
251 // auto opt = plugin.pluginLoader->instance()->property(QV2RAY_PLUGIN_INTERNAL_PROPERTY_KEY).value<Qv2rayPluginOption>();
252 // plugin.pluginLoader->instance()->setProperty(QV2RAY_PLUGIN_INTERNAL_PROPERTY_KEY, QVariant::fromValue(opt));
253 //}
254 const auto &[protocol, outboundSettings] = serializer->DeserializeOutbound(sharelink, aliasPrefix, errMessage);
255 if (errMessage->isEmpty())
256 {
257 data << std::tuple{ *aliasPrefix, protocol, outboundSettings };
258 ok = true;
259 }
260 break;
261 }
262 }
263 }
264 return data;
265 }
