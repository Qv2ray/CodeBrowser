7SCENARIO("Test RealPing get proxy address", "[RealPing]")
8{
9 QvTestApplication app;
10 GIVEN("A realping object")
11 {
12 auto loop = uvw::Loop::create();
13 LatencyTestHost *p;
14 LatencyTestRequest req;
15 auto realping = std::make_shared<Qv2ray::components::latency::realping::RealPing>(loop, req, p);
16 WHEN("test IPv4 any address")
17 {
18 GlobalConfig.inboundConfig->listenip = "0.0.0.0";
19 GlobalConfig.inboundConfig->useHTTP = true;
20 GlobalConfig.inboundConfig->httpSettings->useAuth = true;
21 GlobalConfig.inboundConfig->httpSettings->port = 9090;
22 GlobalConfig.inboundConfig->httpSettings->account->user = "usr";
23 GlobalConfig.inboundConfig->httpSettings->account->pass = "pp";
24 REQUIRE(realping->getProxyAddress() == "http://usr:pp@127.0.0.1:9090");
25 }
26 WHEN("test IPv6 any address")
27 {
28 GlobalConfig.inboundConfig->listenip = "::";
29 GlobalConfig.inboundConfig->useHTTP = false;

33 GlobalConfig.inboundConfig->socksSettings->account->user = "ausr";
34 GlobalConfig.inboundConfig->socksSettings->account->pass = "";

36 }
