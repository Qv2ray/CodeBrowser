8TEST_CASE("QJsonIO Get Simple Value")
9{
10 const auto obj = QJsonObject{
11 { "string", "a quick brown fox jumps over the lazy dog" }, //
12 { "integer", -32767 }, //
13 { "boolean", true }, //
14 { "decimal", 0.618 }, //
15 };
16 const auto arr = QJsonArray{ obj };
17
18 SECTION("Get Array Object")
19 {
20 const auto val = QJsonIO::GetValue(arr, 0);
21 REQUIRE(val.isObject());
22 REQUIRE(val.toObject() == obj);
23 }
24
25 SECTION("Get Object String")
26 {
27 const auto val = QJsonIO::GetValue(obj, "string");
28 REQUIRE(val.isString());
29 REQUIRE(val.toString() == obj["string"].toString());
30 }
31
32 SECTION("Get Object Integer")
33 {
34 const auto val = QJsonIO::GetValue(obj, "integer");
35 REQUIRE(val.toInt() == obj["integer"].toInt());
36 }
37
38 SECTION("Get Object Boolean")
39 {
40 const auto val = QJsonIO::GetValue(obj, "boolean");
41 REQUIRE(val.isBool());
42 REQUIRE(val.toBool() == obj["boolean"].toBool());

47 const auto val = QJsonIO::GetValue(obj, "decimal");

98TEST_CASE("QJsonIO Set Simple Value")
99{
100 auto obj = QJsonObject{};
101 auto arr = QJsonArray{ obj };
102
103 SECTION("Set Object String")
104 {
105 QJsonIO::SetValue(obj, "qv2ray_test", "string");
106 const auto val = QJsonIO::GetValue(obj, "string");
107 REQUIRE(val.isString());
108 REQUIRE(val.toString().toStdString() == "qv2ray_test");
109 }
110
111 SECTION("Set Object Decimal")
112 {
113 QJsonIO::SetValue(obj, 13.14, "decimal");

115 REQUIRE(val.isDouble());
116 REQUIRE(val.toDouble() == 13.14);
117 }

135TEST_CASE("QJsonIO Set Path")
136{
137
138 SECTION("Path Test 1")
139 {
140 auto obj = QJsonObject{};
141 QJsonIO::SetValue(obj, "vmess", "outbounds", 0, "protocol");
142 QJsonIO::SetValue(obj, "0.0.0.0", "outbounds", 0, "sendThrough");
143 QJsonIO::SetValue(obj, "114.51.41.191", "outbounds", 0, "settings", "vnext", 0, "address");
144 QJsonIO::SetValue(obj, 9810, "outbounds", 0, "settings", "vnext", 0, "port");
145 INFO("QJsonIO Result:" << QJsonDocument(obj).toJson().toStdString());
146
147 const auto ans = QJsonObject{
148 { "outbounds",
149 QJsonArray{ QJsonObject{
150 { "protocol", "vmess" },
151 { "sendThrough", "0.0.0.0" },
152 { "settings", QJsonObject{ { "vnext", QJsonArray{ QJsonObject{ { "address", "114.51.41.191" }, { "port", 9810 } } } } } } } } }
153 };
154 INFO("QJsonObject Result: " << QJsonDocument(ans).toJson().toStdString())
155
156 REQUIRE(ans == obj);
157 }
