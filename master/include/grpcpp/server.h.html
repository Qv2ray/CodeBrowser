42namespace grpc {
43
44class AsyncGenericService;
45class HealthCheckServiceInterface;
46class ServerContext;
47class ServerInitializer;

49/// Represents a gRPC server.
50///
51/// Use a \a grpc::ServerBuilder to create, configure, and start
52/// \a Server instances.
53class Server : public ServerInterface, private GrpcLibraryCodegen {
54 public:
55 ~Server();
56
57 /// Block until the server shuts down.
58 ///
59 /// \warning The server must be either shutting down or some other thread must
60 /// call \a Shutdown for this function to ever return.
61 void Wait() override;
62
63 /// Global callbacks are a set of hooks that are called when server
64 /// events occur. \a SetGlobalCallbacks method is used to register
65 /// the hooks with gRPC. Note that
66 /// the \a GlobalCallbacks instance will be shared among all
67 /// \a Server instances in an application and can be set exactly
68 /// once per application.
69 class GlobalCallbacks {
70 public:
71 virtual ~GlobalCallbacks() {}
72 /// Called before server is created.
73 virtual void UpdateArguments(ChannelArguments* args) {}
74 /// Called before application callback for each synchronous server request
75 virtual void PreSynchronousRequest(ServerContext* context) = 0;
76 /// Called after application callback for each synchronous server request
77 virtual void PostSynchronousRequest(ServerContext* context) = 0;
78 /// Called before server is started.
79 virtual void PreServerStart(Server* server) {}
80 /// Called after a server port is added.
81 virtual void AddPort(Server* server, const grpc::string& addr,
82 ServerCredentials* creds, int port) {}
83 };
84 /// Set the global callback object. Can only be called once per application.
85 /// Does not take ownership of callbacks, and expects the pointed to object
86 /// to be alive until all server objects in the process have been destroyed.
87 /// The same \a GlobalCallbacks object will be used throughout the
88 /// application and is shared among all \a Server objects.
89 static void SetGlobalCallbacks(GlobalCallbacks* callbacks);
90
91 /// Returns a \em raw pointer to the underlying \a grpc_server instance.
92 /// EXPERIMENTAL: for internal/test use only
93 grpc_server* c_server();
94
95 /// Returns the health check service.
96 HealthCheckServiceInterface* GetHealthCheckService() const {
97 return health_check_service_.get();
98 }
99
100 /// Establish a channel for in-process communication
101 std::shared_ptr<Channel> InProcessChannel(const ChannelArguments& args);
102
103 /// NOTE: class experimental_type is not part of the public API of this class.
104 /// TODO(yashykt): Integrate into public API when this is no longer
105 /// experimental.
106 class experimental_type {
107 public:
108 explicit experimental_type(Server* server) : server_(server) {}
109
110 /// Establish a channel for in-process communication with client
111 /// interceptors
112 std::shared_ptr<Channel> InProcessChannelWithInterceptors(
113 const ChannelArguments& args,
114 std::unique_ptr<std::vector<
115 std::unique_ptr<experimental::ClientInterceptorFactoryInterface>>>
116 interceptor_creators);
117
118 private:
119 Server* server_;
120 };
121
122 /// NOTE: The function experimental() is not stable public API. It is a view
123 /// to the experimental components of this class. It may be changed or removed
124 /// at any time.
125 experimental_type experimental() { return experimental_type(this); }
126
127 protected:
128 /// Register a service. This call does not take ownership of the service.
129 /// The service must exist for the lifetime of the Server instance.
130 bool RegisterService(const grpc::string* host, Service* service) override;
131
132 /// Try binding the server to the given \a addr endpoint
133 /// (port, and optionally including IP address to bind to).
134 ///
135 /// It can be invoked multiple times. Should be used before
136 /// starting the server.
137 ///
138 /// \param addr The address to try to bind to the server (eg, localhost:1234,
139 /// 192.168.1.1:31416, [::1]:27182, etc.).
140 /// \param creds The credentials associated with the server.
141 ///
142 /// \return bound port number on success, 0 on failure.
143 ///
144 /// \warning It is an error to call this method on an already started server.
145 int AddListeningPort(const grpc::string& addr,
146 ServerCredentials* creds) override;
147
148 /// NOTE: This is *NOT* a public API. The server constructors are supposed to
149 /// be used by \a ServerBuilder class only. The constructor will be made
150 /// 'private' very soon.
151 ///
152 /// Server constructors. To be used by \a ServerBuilder only.
153 ///
154 /// \param max_message_size Maximum message length that the channel can
155 /// receive.
156 ///
157 /// \param args The channel args
158 ///
159 /// \param sync_server_cqs The completion queues to use if the server is a
160 /// synchronous server (or a hybrid server). The server polls for new RPCs on
161 /// these queues
162 ///
163 /// \param min_pollers The minimum number of polling threads per server
164 /// completion queue (in param sync_server_cqs) to use for listening to
165 /// incoming requests (used only in case of sync server)
166 ///
167 /// \param max_pollers The maximum number of polling threads per server
168 /// completion queue (in param sync_server_cqs) to use for listening to
169 /// incoming requests (used only in case of sync server)
170 ///
171 /// \param sync_cq_timeout_msec The timeout to use when calling AsyncNext() on
172 /// server completion queues passed via sync_server_cqs param.
173 Server(int max_message_size, ChannelArguments* args,
174 std::shared_ptr<std::vector<std::unique_ptr<ServerCompletionQueue>>>
175 sync_server_cqs,
176 int min_pollers, int max_pollers, int sync_cq_timeout_msec,
177 grpc_resource_quota* server_rq = nullptr);
178
179 /// Start the server.
180 ///
181 /// \param cqs Completion queues for handling asynchronous services. The
182 /// caller is required to keep all completion queues live until the server is
183 /// destroyed.
184 /// \param num_cqs How many completion queues does \a cqs hold.
185 void Start(ServerCompletionQueue** cqs, size_t num_cqs) override;
186
187 grpc_server* server() override { return server_; };
188
189 private:
190 friend class AsyncGenericService;
191 friend class ServerBuilder;
192 friend class ServerInitializer;
193
194 class SyncRequest;
195 class UnimplementedAsyncRequest;
196 class UnimplementedAsyncResponse;
197
198 /// SyncRequestThreadManager is an implementation of ThreadManager. This class
199 /// is responsible for polling for incoming RPCs and calling the RPC handlers.
200 /// This is only used in case of a Sync server (i.e a server exposing a sync
201 /// interface)
202 class SyncRequestThreadManager;
203
204 /// Register a generic service. This call does not take ownership of the
205 /// service. The service must exist for the lifetime of the Server instance.
206 void RegisterAsyncGenericService(AsyncGenericService* service) override;
207
208 void PerformOpsOnCall(internal::CallOpSetInterface* ops,
209 internal::Call* call) override;
210
211 void ShutdownInternal(gpr_timespec deadline) override;
212
213 int max_receive_message_size() const override {
214 return max_receive_message_size_;
215 };
216
217 ServerInitializer* initializer();
218
219 const int max_receive_message_size_;
220
221 /// The following completion queues are ONLY used in case of Sync API
222 /// i.e. if the server has any services with sync methods. The server uses
223 /// these completion queues to poll for new RPCs
224 std::shared_ptr<std::vector<std::unique_ptr<ServerCompletionQueue>>>
225 sync_server_cqs_;
226
227 /// List of \a ThreadManager instances (one for each cq in
228 /// the \a sync_server_cqs)
229 std::vector<std::unique_ptr<SyncRequestThreadManager>> sync_req_mgrs_;
230
231 // Server status
232 std::mutex mu_;
233 bool started_;
234 bool shutdown_;
235 bool shutdown_notified_; // Was notify called on the shutdown_cv_
236
237 std::condition_variable shutdown_cv_;
238
239 std::shared_ptr<GlobalCallbacks> global_callbacks_;
240
241 std::vector<grpc::string> services_;
242 bool has_generic_service_;
243
244 // Pointer to the wrapped grpc_server.
245 grpc_server* server_;
246
247 std::unique_ptr<ServerInitializer> server_initializer_;
248
249 std::unique_ptr<HealthCheckServiceInterface> health_check_service_;
250 bool health_check_service_disabled_;
251
252 // A special handler for resource exhausted in sync case
253 std::unique_ptr<internal::MethodHandler> resource_exhausted_handler_;
254};
255
256} // namespace grpc

258#endif // GRPCPP_SERVER_H
