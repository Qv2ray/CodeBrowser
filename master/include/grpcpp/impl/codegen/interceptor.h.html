19#ifndef GRPCPP_IMPL_CODEGEN_INTERCEPTOR_H
20#define GRPCPP_IMPL_CODEGEN_INTERCEPTOR_H
21
22namespace grpc {
23namespace experimental {
24class InterceptedMessage {
25 public:
26 template <class M>
27 bool Extract(M* msg); // returns false if definitely invalid extraction
28 template <class M>
29 M* MutableExtract();
30 uint64_t length(); // length on wire
31};
32
33enum class InterceptionHookPoints {
34 /* The first two in this list are for clients and servers */
35 PRE_SEND_INITIAL_METADATA,
36 PRE_SEND_MESSAGE,
37 PRE_SEND_STATUS /* server only */,
38 /* The following three are for hijacked clients only and can only be
39 registered by the global interceptor */
40 PRE_RECV_INITIAL_METADATA,
41 PRE_RECV_MESSAGE,
42 PRE_RECV_STATUS,
43 /* The following two are for all clients and servers */
44 POST_RECV_INITIAL_METADATA,
45 POST_RECV_MESSAGE,
46 POST_RECV_STATUS /* client only */,
47 POST_RECV_CLOSE /* server only */,
48 NUM_INTERCEPTION_HOOKS
49};
50
51class InterceptorBatchMethods {
52 public:
53 virtual ~InterceptorBatchMethods();
54 // Queries to check whether the current batch has an interception hook point
55 // of type \a type
56 virtual bool QueryInterceptionHookPoint(InterceptionHookPoints type) = 0;
57 // Calling this will signal that the interceptor is done intercepting the
58 // current batch of the RPC
59 virtual void Proceed() = 0;
60 // Calling this indicates that the interceptor has hijacked the RPC (only
61 // valid if the batch contains send_initial_metadata on the client side)
62 virtual void Hijack() = 0;
63};
64} // namespace experimental
65} // namespace grpc

67#endif // GRPCPP_IMPL_CODEGEN_INTERCEPTOR_H
