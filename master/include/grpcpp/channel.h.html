31struct grpc_channel;

33namespace grpc {
34
35namespace experimental {
36/// Resets the channel's connection backoff.
37/// TODO(roth): Once we see whether this proves useful, either create a gRFC
38/// and change this to be a method of the Channel class, or remove it.
39void ChannelResetConnectionBackoff(Channel* channel);
40} // namespace experimental
41
42/// Channels represent a connection to an endpoint. Created by \a CreateChannel.
43class Channel final : public ChannelInterface,
44 public internal::CallHook,
45 public std::enable_shared_from_this<Channel>,
46 private GrpcLibraryCodegen {
47 public:
48 ~Channel();
49
50 /// Get the current channel state. If the channel is in IDLE and
51 /// \a try_to_connect is set to true, try to connect.
52 grpc_connectivity_state GetState(bool try_to_connect) override;
53
54 /// Returns the LB policy name, or the empty string if not yet available.
55 grpc::string GetLoadBalancingPolicyName() const;
56
57 /// Returns the service config in JSON form, or the empty string if
58 /// not available.
59 grpc::string GetServiceConfigJSON() const;
60
61 private:
62 template <class InputMessage, class OutputMessage>
63 friend class internal::BlockingUnaryCallImpl;
64 friend void experimental::ChannelResetConnectionBackoff(Channel* channel);
65 friend std::shared_ptr<Channel> CreateChannelInternal(
66 const grpc::string& host, grpc_channel* c_channel,
67 std::unique_ptr<std::vector<
68 std::unique_ptr<experimental::ClientInterceptorFactoryInterface>>>
69 interceptor_creators);
70 Channel(const grpc::string& host, grpc_channel* c_channel,
71 std::unique_ptr<std::vector<
72 std::unique_ptr<experimental::ClientInterceptorFactoryInterface>>>
73 interceptor_creators);
74
75 internal::Call CreateCall(const internal::RpcMethod& method,
76 ClientContext* context,
77 CompletionQueue* cq) override;
78 void PerformOpsOnCall(internal::CallOpSetInterface* ops,
79 internal::Call* call) override;
80 void* RegisterMethod(const char* method) override;
81
82 void NotifyOnStateChangeImpl(grpc_connectivity_state last_observed,
83 gpr_timespec deadline, CompletionQueue* cq,
84 void* tag) override;
85 bool WaitForStateChangeImpl(grpc_connectivity_state last_observed,
86 gpr_timespec deadline) override;
87
88 CompletionQueue* CallbackCQ() override;
89
90 const grpc::string host_;
91 grpc_channel* const c_channel_; // owned
92
93 // mu_ protects callback_cq_ (the per-channel callbackable completion queue)
94 std::mutex mu_;
95
96 // callback_cq_ references the callbackable completion queue associated
97 // with this channel (if any). It is set on the first call to CallbackCQ().
98 // It is _not owned_ by the channel; ownership belongs with its internal
99 // shutdown callback tag (invoked when the CQ is fully shutdown).
100 CompletionQueue* callback_cq_ = nullptr;
101
102 std::vector<std::unique_ptr<experimental::ClientInterceptorFactoryInterface>>
103 interceptor_creators_;
104};
105
106} // namespace grpc
