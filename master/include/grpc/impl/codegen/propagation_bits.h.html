19#ifndef GRPC_IMPL_CODEGEN_PROPAGATION_BITS_H
20#define GRPC_IMPL_CODEGEN_PROPAGATION_BITS_H
21
22#include <grpc/impl/codegen/port_platform.h>
23
24#ifdef __cplusplus
25extern "C" {
26#endif
27
28/** Propagation bits: this can be bitwise or-ed to form propagation_mask for
29 * grpc_call */
30/** Propagate deadline */
31#define GRPC_PROPAGATE_DEADLINE ((uint32_t)1)
32/** Propagate census context */
33#define GRPC_PROPAGATE_CENSUS_STATS_CONTEXT ((uint32_t)2)
34#define GRPC_PROPAGATE_CENSUS_TRACING_CONTEXT ((uint32_t)4)
35/** Propagate cancellation */
36#define GRPC_PROPAGATE_CANCELLATION ((uint32_t)8)
37
38/** Default propagation mask: clients of the core API are encouraged to encode
39 deltas from this in their implementations... ie write:
40 GRPC_PROPAGATE_DEFAULTS & ~GRPC_PROPAGATE_DEADLINE to disable deadline
41 propagation. Doing so gives flexibility in the future to define new
42 propagation types that are default inherited or not. */
43#define GRPC_PROPAGATE_DEFAULTS \
44 ((uint32_t)(( \
45 0xffff | GRPC_PROPAGATE_DEADLINE | GRPC_PROPAGATE_CENSUS_STATS_CONTEXT | \
46 GRPC_PROPAGATE_CENSUS_TRACING_CONTEXT | GRPC_PROPAGATE_CANCELLATION)))
47
48#ifdef __cplusplus
49}
50#endif
51
52#endif /* GRPC_IMPL_CODEGEN_PROPAGATION_BITS_H */
