31#ifndef GOOGLE_PROTOBUF_PLATFORM_MACROS_H_
32#define GOOGLE_PROTOBUF_PLATFORM_MACROS_H_
33
34#define GOOGLE_PROTOBUF_PLATFORM_ERROR \
35#error "Host platform was not detected as supported by protobuf"
36
37// Processor architecture detection. For more info on what's defined, see:
38// http://msdn.microsoft.com/en-us/library/b0084kay.aspx
39// http://www.agner.org/optimize/calling_conventions.pdf
40// or with gcc, run: "echo | gcc -E -dM -"
41#if defined(_M_X64) || defined(__x86_64__)
42#define GOOGLE_PROTOBUF_ARCH_X64 1
43#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
44#elif defined(_M_IX86) || defined(__i386__)
45#define GOOGLE_PROTOBUF_ARCH_IA32 1
46#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
47#elif defined(__QNX__)
48#define GOOGLE_PROTOBUF_ARCH_ARM_QNX 1
49#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
50#elif defined(_M_ARM) || defined(__ARMEL__)
51#define GOOGLE_PROTOBUF_ARCH_ARM 1
52#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
53#elif defined(_M_ARM64)
54#define GOOGLE_PROTOBUF_ARCH_ARM 1
55#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
56#elif defined(__aarch64__)
57#define GOOGLE_PROTOBUF_ARCH_AARCH64 1
58#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
59#elif defined(__MIPSEL__)
60#if defined(__LP64__)
61#define GOOGLE_PROTOBUF_ARCH_MIPS64 1
62#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
63#else
64#define GOOGLE_PROTOBUF_ARCH_MIPS 1
65#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
66#endif
67#elif defined(__pnacl__)
68#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
69#elif defined(sparc)
70#define GOOGLE_PROTOBUF_ARCH_SPARC 1
71#if defined(__sparc_v9__) || defined(__sparcv9) || defined(__arch64__)
72#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
73#else
74#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
75#endif
76#elif defined(_POWER) || defined(__powerpc64__) || defined(__PPC64__)
77#define GOOGLE_PROTOBUF_ARCH_POWER 1
78#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
79#elif defined(__PPC__)
80#define GOOGLE_PROTOBUF_ARCH_PPC 1
81#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
82#elif defined(__s390__) || defined(__s390x__)
83#define GOOGLE_PROTOBUF_ARCH_S390 1
84#if defined(__s390x__)
85#define GOOGLE_PROTOBUF_ARCH_64_BIT 1
86#else
87#define GOOGLE_PROTOBUF_ARCH_32_BIT 1
88#endif
89#elif defined(__GNUC__)
90# if (((__GNUC__ == 4) && (__GNUC_MINOR__ >= 7)) || (__GNUC__ > 4))
91// We fallback to the generic Clang/GCC >= 4.7 implementation in atomicops.h
92# elif defined(__clang__)
93# if !__has_extension(c_atomic)
94GOOGLE_PROTOBUF_PLATFORM_ERROR
95# endif
96// We fallback to the generic Clang/GCC >= 4.7 implementation in atomicops.h
97# endif
98# if __LP64__
99# define GOOGLE_PROTOBUF_ARCH_64_BIT 1
100# else
101# define GOOGLE_PROTOBUF_ARCH_32_BIT 1
102# endif
103#else
104GOOGLE_PROTOBUF_PLATFORM_ERROR
105#endif
106
107#if defined(__APPLE__)
108#define GOOGLE_PROTOBUF_OS_APPLE
109#include <TargetConditionals.h>
110#if TARGET_OS_IPHONE
111#define GOOGLE_PROTOBUF_OS_IPHONE
112#endif
113#elif defined(__EMSCRIPTEN__)
114#define GOOGLE_PROTOBUF_OS_EMSCRIPTEN
115#elif defined(__native_client__)
116#define GOOGLE_PROTOBUF_OS_NACL
117#elif defined(sun)
118#define GOOGLE_PROTOBUF_OS_SOLARIS
119#elif defined(_AIX)
120#define GOOGLE_PROTOBUF_OS_AIX
121#elif defined(__ANDROID__)
122#define GOOGLE_PROTOBUF_OS_ANDROID
123#endif
124
125#undef GOOGLE_PROTOBUF_PLATFORM_ERROR
126
127#if defined(GOOGLE_PROTOBUF_OS_ANDROID) || defined(GOOGLE_PROTOBUF_OS_IPHONE) || defined(__OpenBSD__)
128// Android ndk does not support the __thread keyword very well yet. Here
129// we use pthread_key_create()/pthread_getspecific()/... methods for
130// TLS support on android.
131// iOS and OpenBSD also do not support the __thread keyword.
132#define GOOGLE_PROTOBUF_NO_THREADLOCAL
133#endif

135#endif // GOOGLE_PROTOBUF_PLATFORM_MACROS_H_
