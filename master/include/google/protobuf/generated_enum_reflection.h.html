53// Returns the EnumDescriptor for enum type E, which must be a
54// proto-declared enum type. Code generated by the protocol compiler
55// will include specializations of this template for each enum type declared.
56template <typename E>
57const EnumDescriptor* GetEnumDescriptor();
58
59namespace internal {
60
61// Helper for EnumType_Parse functions: try to parse the string 'name' as an
62// enum name of the given type, returning true and filling in value on success,
63// or returning false and leaving value unchanged on failure.
64LIBPROTOBUF_EXPORT bool ParseNamedEnum(const EnumDescriptor* descriptor,
65 const string& name,
66 int* value);
67
68template<typename EnumType>
69bool ParseNamedEnum(const EnumDescriptor* descriptor,
70 const string& name,
71 EnumType* value) {
72 int tmp;
73 if (!ParseNamedEnum(descriptor, name, &tmp)) return false;
74 *value = static_cast<EnumType>(tmp);
75 return true;
76}
77
78// Just a wrapper around printing the name of a value. The main point of this
79// function is not to be inlined, so that you can do this without including
80// descriptor.h.
81LIBPROTOBUF_EXPORT const string& NameOfEnum(const EnumDescriptor* descriptor, int value);
