8 #define EOL_STRING "\n"
9 #else
10 #define EOL_STRING "\r\n"

13std::string RemoveComment(const std::string &source)
14{
15 std::istringstream source_stream(source);
16 std::string targetText = "";
17 bool isInBlockComment = false;
18 std::string text;
19
20 while (std::getline(source_stream, text))
21 {
22 bool isInLineComment = false;
23 bool isInDoubleQoute = false;
24 bool isInSingleQoute = false;
25 std::string currentLineParsed;
26 //
27 // 0 = not a fslash, 1 = first one. 2 = second one.
28 bool willNextEscape = false;
29
30 for (size_t i = 0; i < text.length(); i++)
31 {
32 if (isInLineComment)
33 continue;
34
35 const auto current = text.at(i);
36 const auto priv = (i == 0) ? '\0' : text.at(i - 1);
37 const auto next = (i == text.length() - 1) ? '\0' : text.at(i + 1);
38 //
39 bool currentIsBeingEscaped = false;
40
41 // If the next char will not be escape
42 if (!willNextEscape)
43 {
44 // but find a metachar....
45 if (current == '\\')
46 {
47 // Meaning the next char will be escaped.
48 willNextEscape = true;
49 }
50 }
51 else
52 {
53 // the current char will be escaped.
54 willNextEscape = false;
55 currentIsBeingEscaped = true;
56 }
57
58 // Not in the comment.
59 if (!isInBlockComment && !isInLineComment)
60 {
61 // If the current one will NOT be escaped and it's a qoute. Like \" and \'
62 if (!currentIsBeingEscaped && !isInSingleQoute && (current == '"'))
63 {
64 isInDoubleQoute = !isInDoubleQoute;
65 }
66 if (!currentIsBeingEscaped && !isInDoubleQoute && (current == '\''))
67 {
68 isInSingleQoute = !isInSingleQoute;
69 }
70 }
71
72 if (!isInDoubleQoute && !isInSingleQoute)
73 {
74 // First try to close the block comment.
75 if (current == '/' && priv == '*')
76 {
77 isInBlockComment = false;
78 continue;
79 }
80
81 if (current == '/' && next == '/')
82 {
83 isInLineComment = true;
84 continue;
85 }
86
87 if (current == '/' && next == '*')
88 {
89 isInBlockComment = true;
90 continue;
91 }
92 }
93
94 if (!isInLineComment && !isInBlockComment)
95 {
96 currentLineParsed.push_back(current);
97 // Reset escape char state.
98 }
99 }
100
101 if (!currentLineParsed.empty())
102 {
103 targetText.append(currentLineParsed + EOL_STRING);
104 }
105 }
106
107 return targetText;
108}
