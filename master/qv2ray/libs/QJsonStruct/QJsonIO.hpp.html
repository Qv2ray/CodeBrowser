7class QJsonIO
8{
9 public:
10 const static inline auto Null = QJsonValue(QJsonValue::Null);
11 const static inline auto Undefined = QJsonValue(QJsonValue::Undefined);
12 template<typename current_key_type, typename... t_other_types>
13 static QJsonValue GetValue(const QJsonValue &parent, const current_key_type &current, const t_other_types &... other)
14 {
15 if constexpr (sizeof...(t_other_types) == 0)
16 return parent[current];
17 else if constexpr (std::is_same<current_key_type, QJsonArray::size_type>::value)
18 return GetValue(parent.toArray()[current], other...);
19 else
20 return GetValue(parent.toObject()[current], other...);
21 }
22
23 template<typename... key_types_t>
24 static QJsonValue GetValue(QJsonValue value, const std::tuple<key_types_t...> &keys, const QJsonValue &defaultValue = Undefined)
25 {
26 std::apply([&](auto &&... args) -> void { ((value = value[args]), ...); }, keys);
27 return value.isUndefined() ? defaultValue : value;
28 }
29
30 //
31 // ========= Set Values =========
32 //
33 template<typename parent_type, typename t_value_type, typename current_key_type, typename... t_other_key_types>
34 static void SetValue(parent_type &parent, const t_value_type &val, const current_key_type &current, const t_other_key_types &... other)
35 {
36 // If current parent is an array, increase its size to fit the "key"
37 if constexpr (std::is_same<current_key_type, QJsonArray::size_type>::value)
38 for (auto i = parent.size(); i <= current; i++) parent.insert(i, {});
39
40 // If the t_other_key_types has nothing....
41 // Means we have reached the end of recursion.
42 if constexpr (sizeof...(t_other_key_types) == 0)
43 parent[current] = val;
44 else if constexpr (std::is_same<typename std::tuple_element<0, std::tuple<t_other_key_types...>>::type, QJsonArray::size_type>::value)
45 {
46 // Means we still have many keys
47 // So this element is an array.
48 auto _array = parent[current].toArray();
49 SetValue(_array, val, other...);
50 parent[current] = _array;
51 }
52 else
53 {
54 auto _object = parent[current].toObject();
55 SetValue(_object, val, other...);
56 parent[current] = _object;
57 }
58 }
59};
