31 enum NtpLeapIndicator
32 {
33 NoWarning = 0, /**< No warning. */
34 LastMinute61Warning = 1, /**< Last minute has 61 seconds. */
35 LastMinute59Warning = 2, /**< Last minute has 59 seconds. */
36 UnsynchronizedWarning = 3, /**< Alarm condition (clock not synchronized). */
37 };
38
39 enum NtpMode
40 {
41 ReservedMode = 0, /**< Reserved. */
42 SymmetricActiveMode = 1, /**< Symmetric active. */
43 SymmetricPassiveMode = 2, /**< Symmetric passive. */
44 ClientMode = 3, /**< Client. */
45 ServerMode = 4, /**< Server. */
46 BroadcastMode = 5, /**< Broadcast. */
47 ControlMode = 6, /**< NTP control message. */
48 PrivateMode = 7, /**< Reserved for private use. */
49 };
50
51 enum NtpStratum
52 {
53 UnspecifiedStratum = 0, /**< Unspecified or unavailable. */
54 PrimaryStratum = 1, /**< Primary reference (e.g. radio-clock). */
55 SecondaryStratumFirst = 2, /**< Secondary reference (via NTP or SNTP). */
56 SecondaryStratumLast = 15,
57 UnsynchronizedStratum = 16, /**< Unsynchronized. */
58 /* 17-255 are reserved. */
59 };
60
61 struct NtpPacketFlags
62 {
63 unsigned char mode : 3;
64 unsigned char versionNumber : 3;
65 unsigned char leapIndicator : 2;
66 };

70 struct NtpTimestamp
71 {
72 quint32 seconds;
73 quint32 fraction;
74 static inline NtpTimestamp fromDateTime(const QDateTime &dateTime);
75 static inline QDateTime toDateTime(const NtpTimestamp &ntpTime);
76 };

78 struct NtpPacket
79 {
80 NtpPacketFlags flags;
81 quint8 stratum;
82 qint8 poll;
83 qint8 precision;
84 qint32 rootDelay;
85 qint32 rootDispersion;
86 qint8 referenceID[4];
87 NtpTimestamp referenceTimestamp;
88 NtpTimestamp originateTimestamp;
89 NtpTimestamp receiveTimestamp;
90 NtpTimestamp transmitTimestamp;
91 };
92
93 struct NtpAuthenticationInfo
94 {
95 quint32 keyId;
96 quint8 messageDigest[16];
97 };
98
99 struct NtpFullPacket
100 {
101 NtpPacket basic;
102 NtpAuthenticationInfo auth;
103 };

114 class NtpReply
115 {
116 public:
117 NtpReply();
118 NtpReply(const NtpReply &other);
119 ~NtpReply();
120 NtpReply &operator=(const NtpReply &other);
121 NtpLeapIndicator leapIndicator() const;
122 quint8 versionNumber() const;
123 NtpMode mode() const;
124 quint8 stratum() const;
125 qreal pollInterval() const;
126 qreal precision() const;
127 QDateTime referenceTime() const;
128 QDateTime originTime() const;
129 QDateTime receiveTime() const;
130 QDateTime transmitTime() const;
131 QDateTime destinationTime() const;
132 qint64 roundTripDelay() const;
133 qint64 localClockOffset() const;
134 bool isNull() const;
135
136 protected:
137 friend class NtpClient;
138 NtpReply(NtpReplyPrivate *dd);
139
140 private:
141 QSharedDataPointer<NtpReplyPrivate> d;
142 };
