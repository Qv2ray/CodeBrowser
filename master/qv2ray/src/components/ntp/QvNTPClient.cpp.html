25 NtpTimestamp NtpTimestamp::fromDateTime(const QDateTime &dateTime)
26 {
27 qint64 ntpMSecs = dateTime.toMSecsSinceEpoch() - january_1_1900;
28
29 quint32 seconds = ntpMSecs / 1000;
30 quint32 fraction = 0x100000000ll * (ntpMSecs % 1000) / 1000;
31
32 NtpTimestamp result;
33 result.seconds = qToBigEndian(seconds);
34 result.fraction = qToBigEndian(fraction);
35 return result;
36 }
37
38 QDateTime NtpTimestamp::toDateTime(const NtpTimestamp &ntpTime)
39 {
40 /* Convert to local-endian. */
41 quint32 seconds = qFromBigEndian(ntpTime.seconds);
42 quint32 fraction = qFromBigEndian(ntpTime.fraction);
43
44 /* Convert NTP timestamp to number of milliseconds passed since Jan 1 1900. */
45 qint64 ntpMSecs = seconds * 1000ll + fraction * 1000ll / 0x100000000ll;
46
47 /* Construct Qt date time. */
48 return QDateTime::fromMSecsSinceEpoch(ntpMSecs + january_1_1900);
49 }

158 void NtpClient::init(const QHostAddress &bindAddress, quint16 bindPort)
159 {
160 mSocket = new QUdpSocket(this);
161 mSocket->bind(bindAddress, bindPort);
162
163 connect(mSocket, SIGNAL(readyRead()), this, SLOT(readPendingDatagrams()));

171 bool NtpClient::sendRequest(const QHostAddress &address, quint16 port)
172 {
173 if (mSocket->state() != QAbstractSocket::BoundState)
174 return false;
175
176 /* Initialize the NTP packet. */
177 NtpPacket packet;
178 memset(&packet, 0, sizeof(packet));
179 packet.flags.mode = ClientMode;
180 packet.flags.versionNumber = 4;
181 packet.transmitTimestamp = NtpTimestamp::fromDateTime(QDateTime::currentDateTimeUtc());
182
183 /* Send it. */
184 if (mSocket->writeDatagram(reinterpret_cast<const char *>(&packet), sizeof(packet), address, port) < 0)
185 return false;
186
187 return true;
188 }
189
190 void NtpClient::readPendingDatagrams()
191 {
192 while (mSocket->hasPendingDatagrams())
193 {
194 NtpFullPacket packet;
195 memset(&packet, 0, sizeof(packet));
196
197 QHostAddress address;
198 quint16 port;
199
200 if (mSocket->readDatagram(reinterpret_cast<char *>(&packet), sizeof(packet), &address, &port) < (qint64) sizeof(NtpPacket))
201 continue;
202
203 QDateTime now = QDateTime::currentDateTime();
204
205 /* Prepare reply. */
206 NtpReplyPrivate *replyPrivate = new NtpReplyPrivate();
207 replyPrivate->packet = packet;
208 replyPrivate->destinationTime = now;
209 NtpReply reply(replyPrivate);
210
211 /* Notify. */
212 Q_EMIT replyReceived(address, port, reply);
213 }
