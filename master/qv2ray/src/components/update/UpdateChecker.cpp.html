29 void QvUpdateChecker::CheckUpdate()
30 {
31#ifndef DISABLE_AUTO_UPDATE
32 if (QFile(QV2RAY_CONFIG_DIR + "QV2RAY_FEATURE_DISABLE_AUTO_UPDATE").exists())
33 return;
34 const auto &updateChannel = GlobalConfig.updateConfig.updateChannel;
35 LOG(MODULE_NETWORK, "Start checking update for channel ID: " + QSTRN(updateChannel))
36 requestHelper->AsyncHttpGet(UpdateChannelLink[updateChannel], &QvUpdateChecker::VersionUpdate);
37#endif
38 }

40 void QvUpdateChecker::VersionUpdate(const QByteArray &data)
41 {
42 // Version update handler.
43 const auto doc = QJsonDocument::fromJson(data);
44 const auto root = doc.isArray() ? doc.array().first().toObject() : doc.object();
45 if (root.isEmpty())
46 return;
47
48 const auto newVersionStr = root["tag_name"].toString("v").mid(1);
49 const auto currentVersionStr = QString(QV2RAY_VERSION_STRING);
50 const auto ignoredVersionStr = GlobalConfig.updateConfig.ignoredVersion.isEmpty() ? "0.0.0" : GlobalConfig.updateConfig.ignoredVersion;
51 //
52 bool hasUpdate = false;
53 try
54 {
55 const auto newVersion = semver::version::from_string(newVersionStr.toStdString());
56 const auto currentVersion = semver::version::from_string(currentVersionStr.toStdString());
57 const auto ignoredVersion = semver::version::from_string(ignoredVersionStr.toStdString());
58 //
59 LOG(MODULE_UPDATE, QString("Received update info:") + NEWLINE + //
60 " --> Latest: " + newVersionStr + NEWLINE + //
61 " --> Current: " + currentVersionStr + NEWLINE + //
62 " --> Ignored: " + ignoredVersionStr)
63 // If the version is newer than us.
64 // And new version is newer than the ignored version.
65 hasUpdate = (newVersion > currentVersion && newVersion > ignoredVersion);
66 }
67 catch (...)
68 {
69 LOG(MODULE_UPDATE, "Some strange exception occured, cannot check update.")
70 }
71 if (hasUpdate)
72 {
73 const auto name = root["name"].toString("");
74 if (name.contains("NO_RELEASE"))
75 {
76 LOG(MODULE_UPDATE, "Found the recent release title with NO_RELEASE tag. Ignoring")
77 return;
78 }
79 const auto link = root["html_url"].toString("");
80 auto result = QvMessageBoxAsk(nullptr, tr("Qv2ray Update"),
81 tr("A new version of Qv2ray has been found:") + //
82 "v" + newVersionStr + NEWLINE + NEWLINE + //
83 name + NEWLINE "------------" NEWLINE + //
84 root["body"].toString(""),
85 QMessageBox::Ignore);
86
87 if (result == QMessageBox::Yes)
88 {
89 QDesktopServices::openUrl(link);
90 }
91 else if (result == QMessageBox::Ignore)
92 {
93 // Set and save ingored version.
94 GlobalConfig.updateConfig.ignoredVersion = newVersionStr;
95 SaveGlobalSettings();
96 }
97 }
98 else
99 {
100 LOG(MODULE_UPDATE, "No suitable updates found on channel " + QSTRN(GlobalConfig.updateConfig.updateChannel))
101 }
102 }
