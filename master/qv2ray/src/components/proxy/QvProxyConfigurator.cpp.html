16 QStringList macOSgetNetworkServices()
17 {
18 QProcess p;
19 p.setProgram("/usr/sbin/networksetup");
20 p.setArguments(QStringList{ "-listallnetworkservices" });
21 p.start();
22 p.waitForStarted();
23 p.waitForFinished();
24 LOG(MODULE_PROXY, p.errorString())
25 auto str = p.readAllStandardOutput();
26 auto lines = SplitLines(str);
27 QStringList result;
28
29 // Start from 1 since first line is unneeded.
30 for (auto i = 1; i < lines.count(); i++)
31 {
32 // * means disabled.
33 if (!lines[i].contains("*"))
34 {
35 result << lines[i];
36 }
37 }
38
39 LOG(MODULE_PROXY, "Found " + QSTRN(result.size()) + " network services: " + result.join(";"))
40 return result;
41 }

49 bool __QueryProxyOptions()
50 {
51 INTERNET_PER_CONN_OPTION_LIST List;
52 INTERNET_PER_CONN_OPTION Option[5];
53 //
54 unsigned long nSize = sizeof(INTERNET_PER_CONN_OPTION_LIST);
55 Option[0].dwOption = INTERNET_PER_CONN_AUTOCONFIG_URL;
56 Option[1].dwOption = INTERNET_PER_CONN_AUTODISCOVERY_FLAGS;
57 Option[2].dwOption = INTERNET_PER_CONN_FLAGS;
58 Option[3].dwOption = INTERNET_PER_CONN_PROXY_BYPASS;
59 Option[4].dwOption = INTERNET_PER_CONN_PROXY_SERVER;
60 //
61 List.dwSize = sizeof(INTERNET_PER_CONN_OPTION_LIST);
62 List.pszConnection = nullptr; // NO_CONST(DEFAULT_CONNECTION_NAME);
63 List.dwOptionCount = 5;
64 List.dwOptionError = 0;
65 List.pOptions = Option;
66
67 if (!InternetQueryOption(nullptr, INTERNET_OPTION_PER_CONNECTION_OPTION, &List, &nSize))
68 {
69 LOG(MODULE_PROXY, "InternetQueryOption failed, GLE=" + QSTRN(GetLastError()))
70 }
71
72 LOG(MODULE_PROXY, "System default proxy info:")
73
74 if (Option[0].Value.pszValue != nullptr)
75 {
76 LOG(MODULE_PROXY, QString::fromWCharArray(Option[0].Value.pszValue))
77 }
78
79 if ((Option[2].Value.dwValue & PROXY_TYPE_AUTO_PROXY_URL) == PROXY_TYPE_AUTO_PROXY_URL)
80 {
81 LOG(MODULE_PROXY, "PROXY_TYPE_AUTO_PROXY_URL")
82 }
83
84 if ((Option[2].Value.dwValue & PROXY_TYPE_AUTO_DETECT) == PROXY_TYPE_AUTO_DETECT)
85 {
86 LOG(MODULE_PROXY, "PROXY_TYPE_AUTO_DETECT")
87 }
88
89 if ((Option[2].Value.dwValue & PROXY_TYPE_DIRECT) == PROXY_TYPE_DIRECT)
90 {
91 LOG(MODULE_PROXY, "PROXY_TYPE_DIRECT")
92 }
93
94 if ((Option[2].Value.dwValue & PROXY_TYPE_PROXY) == PROXY_TYPE_PROXY)
95 {
96 LOG(MODULE_PROXY, "PROXY_TYPE_PROXY")
97 }
98
99 if (!InternetQueryOption(nullptr, INTERNET_OPTION_PER_CONNECTION_OPTION, &List, &nSize))
100 {
101 LOG(MODULE_PROXY, "InternetQueryOption failed,GLE=" + QSTRN(GetLastError()))
102 }
103
104 if (Option[4].Value.pszValue != nullptr)
105 {
106 LOG(MODULE_PROXY, QString::fromStdWString(Option[4].Value.pszValue))
107 }
108
109 INTERNET_VERSION_INFO Version;
110 nSize = sizeof(INTERNET_VERSION_INFO);
111 InternetQueryOption(nullptr, INTERNET_OPTION_VERSION, &Version, &nSize);
112
113 if (Option[0].Value.pszValue != nullptr)
114 {
115 GlobalFree(Option[0].Value.pszValue);
116 }
117
118 if (Option[3].Value.pszValue != nullptr)
119 {
120 GlobalFree(Option[3].Value.pszValue);
121 }
122
123 if (Option[4].Value.pszValue != nullptr)
124 {
125 GlobalFree(Option[4].Value.pszValue);
126 }
127
128 return false;
129 }
130 bool __SetProxyOptions(LPWSTR proxy_full_addr, bool isPAC)
131 {
132 INTERNET_PER_CONN_OPTION_LIST list;
133 BOOL bReturn;
134 DWORD dwBufSize = sizeof(list);
135 // Fill the list structure.
136 list.dwSize = sizeof(list);
137 // NULL == LAN, otherwise connectoid name.
138 list.pszConnection = nullptr;
139
140 if (isPAC)
141 {
142 LOG(MODULE_PROXY, "Setting system proxy for PAC")
143 //
144 list.dwOptionCount = 2;
145 list.pOptions = new INTERNET_PER_CONN_OPTION[2];
146
147 // Ensure that the memory was allocated.
148 if (nullptr == list.pOptions)
149 {
150 // Return FALSE if the memory wasn't allocated.
151 return FALSE;
152 }
153
154 // Set flags.
155 list.pOptions[0].dwOption = INTERNET_PER_CONN_FLAGS;
156 list.pOptions[0].Value.dwValue = PROXY_TYPE_DIRECT | PROXY_TYPE_AUTO_PROXY_URL;
157 // Set proxy name.
158 list.pOptions[1].dwOption = INTERNET_PER_CONN_AUTOCONFIG_URL;
159 list.pOptions[1].Value.pszValue = proxy_full_addr;
160 }
161 else
162 {
163 LOG(MODULE_PROXY, "Setting system proxy for Global Proxy")
164 //
165 list.dwOptionCount = 3;
166 list.pOptions = new INTERNET_PER_CONN_OPTION[3];
167
168 if (nullptr == list.pOptions)
169 {
170 return false;
171 }
172
173 // Set flags.
174 list.pOptions[0].dwOption = INTERNET_PER_CONN_FLAGS;
175 list.pOptions[0].Value.dwValue = PROXY_TYPE_DIRECT | PROXY_TYPE_PROXY;
176 // Set proxy name.
177 list.pOptions[1].dwOption = INTERNET_PER_CONN_PROXY_SERVER;
178 list.pOptions[1].Value.pszValue = proxy_full_addr;
179 // Set proxy override.
180 list.pOptions[2].dwOption = INTERNET_PER_CONN_PROXY_BYPASS;
181 auto localhost = L"localhost";
182 list.pOptions[2].Value.pszValue = NO_CONST(localhost);
183 }
184
185 // Set the options on the connection.
186 bReturn = InternetSetOption(nullptr, INTERNET_OPTION_PER_CONNECTION_OPTION, &list, dwBufSize);
187 delete[] list.pOptions;
188 InternetSetOption(nullptr, INTERNET_OPTION_SETTINGS_CHANGED, nullptr, 0);
189 InternetSetOption(nullptr, INTERNET_OPTION_REFRESH, nullptr, 0);
190 return bReturn;
191 }

194 void SetSystemProxy(const QString &address, int httpPort, int socksPort)
195 {
196 LOG(MODULE_PROXY, "Setting up System Proxy")
197 bool hasHTTP = (httpPort > 0);
198 bool hasSOCKS = (socksPort > 0);
199
200 if (!hasHTTP && !hasSOCKS)
201 {
202 LOG(MODULE_PROXY, "Nothing?")
203 return;
204 }
205
206 if (hasHTTP)
207 {
208 LOG(MODULE_PROXY, "Qv2ray will set system proxy to use HTTP")
209 }
210
211 if (hasSOCKS)
212 {
213 LOG(MODULE_PROXY, "Qv2ray will set system proxy to use SOCKS")
214 }
215
216#ifdef Q_OS_WIN
217 const auto scheme = (hasHTTP ? "" : "socks5://");
218 QString __a = scheme + address + ":" + QSTRN(hasHTTP ? httpPort : socksPort);
219
220 LOG(MODULE_PROXY, "Windows proxy string: " + __a)
221 auto proxyStrW = new WCHAR[__a.length() + 1];
222 wcscpy(proxyStrW, __a.toStdWString().c_str());
223 //
224 __QueryProxyOptions();
225
226 if (!__SetProxyOptions(proxyStrW, false))
227 {
228 LOG(MODULE_PROXY, "Failed to set proxy.")
229 }
230
231 __QueryProxyOptions();
232#elif defined(Q_OS_LINUX)
233 QList<ProcessArgument> actions;
234 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy", "mode", "manual" } };
235 //
236 bool isKDE = qEnvironmentVariable("XDG_SESSION_DESKTOP") == "KDE" || qEnvironmentVariable("XDG_SESSION_DESKTOP") == "plasma";
237 bool isDDE = !isKDE && qEnvironmentVariable("XDG_CURRENT_DESKTOP").toLower() == "deepin";
238 const auto configPath = QStandardPaths::writableLocation(QStandardPaths::ConfigLocation);
239
240 //
241 // Configure HTTP Proxies for HTTP, FTP and HTTPS
242 if (hasHTTP)
243 {
244 // iterate over protocols...
245 for (const auto &protocol : QStringList{ "http", "ftp", "https" })
246 {
247 // for GNOME:
248 {
249 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy." + protocol, "host", address } };
250 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy." + protocol, "port", QSTRN(httpPort) } };
251 }
252
253 // for KDE:
254 if (isKDE)
255 {
256 actions << ProcessArgument{ "kwriteconfig5",
257 { "--file", configPath + "/kioslaverc", //
258 "--group", "Proxy Settings", //
259 "--key", protocol + "Proxy", //
260 "http://" + address + " " + QSTRN(httpPort) } };
261 }
262 }
263 }
264
265 // Configure SOCKS5 Proxies
266 if (hasSOCKS)
267 {
268 // for GNOME:
269 {
270 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy.socks", "host", address } };
271 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy.socks", "port", QSTRN(socksPort) } };
272
273 // for KDE:
274 if (isKDE)
275 {
276 actions << ProcessArgument{ "kwriteconfig5",
277 { "--file", configPath + "/kioslaverc", //
278 "--group", "Proxy Settings", //
279 "--key", "socksProxy", //
280 "socks://" + address + " " + QSTRN(socksPort) } };
281 }
282 }
283 }
284 // Setting Proxy Mode to Manual
285 {
286 // for GNOME:
287 {
288 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy", "mode", "manual" } };
289 }
290
291 // for KDE:
292 if (isKDE)
293 {
294 actions << ProcessArgument{ "kwriteconfig5",
295 { "--file", configPath + "/kioslaverc", //
296 "--group", "Proxy Settings", //
297 "--key", "ProxyType", "1" } };
298 }
299 }
300
301 // Notify kioslaves to reload system proxy configuration.
302 if (isKDE)
303 {
304 actions << ProcessArgument{ "dbus-send",
305 { "--type=signal", "/KIO/Scheduler", //
306 "org.kde.KIO.Scheduler.reparseSlaveConfiguration", //
307 "string:''" } };
308 }
309 // Execute them all!
310 //
311 // note: do not use std::all_of / any_of / none_of,
312 // because those are short-circuit and cannot guarantee atomicity.
313 QList<bool> results;
314 for (const auto &action : actions)
315 {
316 // execute and get the code
317 const auto returnCode = QProcess::execute(action.first, action.second);
318 // print out the commands and result codes
319 DEBUG(MODULE_PROXY, QString("[%1] Program: %2, Args: %3").arg(returnCode).arg(action.first).arg(action.second.join(";")))
320 // give the code back
321 results << (returnCode == QProcess::NormalExit);
322 }
323
324 if (results.count(true) != actions.size())
325 {
326 LOG(MODULE_PROXY, "Something wrong when setting proxies.")
327 }
328
329 // Post-Actions for HTTP on Deepin Desktop Environment.
330 if (isDDE && hasHTTP)
331 {
332 if (!RuntimeConfig.deepinHorribleProxyHint)
333 {
334 RuntimeConfig.deepinHorribleProxyHint = true;
335 const auto deepinWarnTitle = QObject::tr("Deepin Detected");
336 const auto deepinWarnMessage =
337 QObject::tr("Deepin plays smart and sets you the wrong HTTPS_PROXY, FTP_PROXY environment variable.") + NEWLINE + //
338 QObject::tr("The origin scheme http is wrongly replaced by https and ftp, causing the problem.") + NEWLINE + //
339 QObject::tr("Qv2ray cannot help you change them back. Please don't blame us if things go wrong."); //
340 QvMessageBoxWarn(nullptr, deepinWarnTitle, deepinWarnMessage);
341 }
342
343 // set them back! - NOPE. setenv only works within your little program.
344 // const auto httpProxyURL = QString("http://%1:%2").arg(address, QSTRN(httpPort)).toStdString();
345 // setenv("https_proxy", httpProxyURL.c_str(), true);
346 // setenv("ftp_proxy", httpProxyURL.c_str(), true);
347 }
348#else
349
350 for (const auto &service : macOSgetNetworkServices())
351 {
352 LOG(MODULE_PROXY, "Setting proxy for interface: " + service)
353
354 if (hasHTTP)
355 {
356 QProcess::execute("/usr/sbin/networksetup", { "-setwebproxystate", service, "on" });
357 QProcess::execute("/usr/sbin/networksetup", { "-setsecurewebproxystate", service, "on" });
358 QProcess::execute("/usr/sbin/networksetup", { "-setwebproxy", service, address, QSTRN(httpPort) });
359 QProcess::execute("/usr/sbin/networksetup", { "-setsecurewebproxy", service, address, QSTRN(httpPort) });
360 }
361
362 if (hasSOCKS)
363 {
364 QProcess::execute("/usr/sbin/networksetup", { "-setsocksfirewallproxystate", service, "on" });
365 QProcess::execute("/usr/sbin/networksetup", { "-setsocksfirewallproxy", service, address, QSTRN(socksPort) });
366 }
367 }
368
369#endif
370 //
371 // Trigger plugin events
372 QMap<Events::SystemProxy::SystemProxyType, int> portSettings;
373 if (hasHTTP)
374 portSettings.insert(Events::SystemProxy::SystemProxyType::SystemProxy_HTTP, httpPort);
375 if (hasSOCKS)
376 portSettings.insert(Events::SystemProxy::SystemProxyType::SystemProxy_SOCKS, socksPort);
377 PluginHost->Send_SystemProxyEvent({ portSettings, Events::SystemProxy::SystemProxyStateType::SetProxy });
378 }

380 void ClearSystemProxy()
381 {
382 LOG(MODULE_PROXY, "Clearing System Proxy")
383
384#ifdef Q_OS_WIN
385 LOG(MODULE_PROXY, "Cleaning system proxy settings.")
386 INTERNET_PER_CONN_OPTION_LIST list;
387 BOOL bReturn;
388 DWORD dwBufSize = sizeof(list);
389 // Fill out list struct.
390 list.dwSize = sizeof(list);
391 // nullptr == LAN, otherwise connectoid name.
392 list.pszConnection = nullptr;
393 // Set three options.
394 list.dwOptionCount = 1;
395 list.pOptions = new INTERNET_PER_CONN_OPTION[list.dwOptionCount];
396
397 // Make sure the memory was allocated.
398 if (nullptr == list.pOptions)
399 {
400 // Return FALSE if the memory wasn't allocated.
401 LOG(MODULE_PROXY, "Failed to allocat memory in DisableConnectionProxy()")
402 }
403
404 // Set flags.
405 list.pOptions[0].dwOption = INTERNET_PER_CONN_FLAGS;
406 list.pOptions[0].Value.dwValue = PROXY_TYPE_DIRECT;
407 //
408 // Set the options on the connection.
409 bReturn = InternetSetOption(nullptr, INTERNET_OPTION_PER_CONNECTION_OPTION, &list, dwBufSize);
410 delete[] list.pOptions;
411 InternetSetOption(nullptr, INTERNET_OPTION_SETTINGS_CHANGED, nullptr, 0);
412 InternetSetOption(nullptr, INTERNET_OPTION_REFRESH, nullptr, 0);
413#elif defined(Q_OS_LINUX)
414 QList<ProcessArgument> actions;
415 const bool isKDE = qEnvironmentVariable("XDG_SESSION_DESKTOP") == "KDE" || qEnvironmentVariable("XDG_SESSION_DESKTOP") == "plasma";
416 const auto configRoot = QStandardPaths::writableLocation(QStandardPaths::ConfigLocation);
417
418 // Setting System Proxy Mode to: None
419 {
420 // for GNOME:
421 {
422 actions << ProcessArgument{ "gsettings", { "set", "org.gnome.system.proxy", "mode", "none" } };
423 }
424
425 // for KDE:
426 if (isKDE)
427 {
428 actions << ProcessArgument{ "kwriteconfig5",
429 { "--file", configRoot + "/kioslaverc", //
430 "--group", "Proxy Settings", //
431 "--key", "ProxyType", "0" } };
432 }
433 }
434
435 // Notify kioslaves to reload system proxy configuration.
436 if (isKDE)
437 {
438 actions << ProcessArgument{ "dbus-send",
439 { "--type=signal", "/KIO/Scheduler", //
440 "org.kde.KIO.Scheduler.reparseSlaveConfiguration", //
441 "string:''" } };
442 }
443
444 // Execute the Actions
445 for (const auto &action : actions)
446 {
447 // execute and get the code
448 const auto returnCode = QProcess::execute(action.first, action.second);
449 // print out the commands and result codes
450 DEBUG(MODULE_PROXY, QString("[%1] Program: %2, Args: %3").arg(returnCode).arg(action.first).arg(action.second.join(";")))
451 }
452
453#else
454 for (const auto &service : macOSgetNetworkServices())
455 {
456 LOG(MODULE_PROXY, "Clearing proxy for interface: " + service)
457 QProcess::execute("/usr/sbin/networksetup", { "-setautoproxystate", service, "off" });
458 QProcess::execute("/usr/sbin/networksetup", { "-setwebproxystate", service, "off" });
459 QProcess::execute("/usr/sbin/networksetup", { "-setsecurewebproxystate", service, "off" });
460 QProcess::execute("/usr/sbin/networksetup", { "-setsocksfirewallproxystate", service, "off" });
461 }
462
463#endif
464 //
465 // Trigger plugin events
466 PluginHost->Send_SystemProxyEvent(Events::SystemProxy::EventObject{ {}, Events::SystemProxy::SystemProxyStateType::ClearProxy });
467 }
