28 int QvPluginHost::RefreshPluginList()
29 {
30 ClearPlugins();
31 LOG(MODULE_PLUGINHOST, "Reloading plugin list")
32 for (const auto &pluginDirPath : Qv2rayAssetsPaths("plugins"))
33 {
34 const QStringList entries = QDir(pluginDirPath).entryList(QDir::Files);
35 for (const auto &fileName : entries)
36 {
37 DEBUG(MODULE_PLUGINHOST, "Loading plugin: " + fileName + " from: " + pluginDirPath)
38 //
39 QvPluginInfo info;
40 auto pluginFullPath = QDir(pluginDirPath).absoluteFilePath(fileName);
41 info.libraryPath = pluginFullPath;
42 info.pluginLoader = new QPluginLoader(pluginFullPath, this);
43 // auto meta = pluginLoader.metaData();
44 // You should not call "delete" on this object, it's handled by the QPluginLoader
45 QObject *plugin = info.pluginLoader->instance();
46 if (plugin == nullptr)
47 {
48 LOG(MODULE_PLUGINHOST, info.pluginLoader->errorString());
49 continue;
50 }
51 info.pluginInterface = qobject_cast<Qv2rayInterface *>(plugin);
52 if (info.pluginInterface == nullptr)
53 {
54 LOG(MODULE_PLUGINHOST, "Failed to cast from QObject to Qv2rayPluginInterface")
55 info.pluginLoader->unload();
56 continue;
57 }
58
59 if (info.pluginInterface->QvPluginInterfaceVersion != QV2RAY_PLUGIN_INTERFACE_VERSION)
60 {
61 // The plugin was built for a not-compactable version of Qv2ray. Don't load the plugin by default.
62 LOG(MODULE_PLUGINHOST, info.libraryPath + " is built with an older Interface, ignoring")
63 QvMessageBoxWarn(nullptr, tr("Cannot load plugin"),
64 tr("The plugin located here cannot be loaded: ") + NEWLINE + info.libraryPath + NEWLINE NEWLINE +
65 tr("This plugin was built against an older/newer version of the Plugin Interface.") + NEWLINE +
66 tr("Please contact the plugin provider or report the issue to Qv2ray Workgroup."));
67 continue;
68 }
69 info.metadata = info.pluginInterface->GetMetadata();
70 if (plugins.contains(info.metadata.InternalName))
71 {
72 LOG(MODULE_PLUGINHOST, "Found another plugin with the same internal name: " + info.metadata.InternalName + ". Skipped")
73 continue;
74 }
75 connect(plugin, SIGNAL(PluginLog(const QString &)), this, SLOT(QvPluginLog(const QString &)));
76 connect(plugin, SIGNAL(PluginErrorMessageBox(const QString &)), this, SLOT(QvPluginMessageBox(const QString &)));
77 LOG(MODULE_PLUGINHOST, "Loaded plugin: \"" + info.metadata.Name + "\" made by: \"" + info.metadata.Author + "\"")
78 plugins.insert(info.metadata.InternalName, info);
79 }
80 }
81 return plugins.count();
82 }

147 bool QvPluginHost::InitializePlugin(const QString &internalName)
148 {
149 const auto &plugin = plugins[internalName];
150 if (plugin.isLoaded)
151 {
152 LOG(MODULE_PLUGINHOST, "The plugin: \"" + internalName + "\" has already been loaded.")
153 return true;
154 }
155 if (!GlobalConfig.pluginConfig.pluginStates.contains(internalName))
156 {
157 // If not contained, default to enabled.
158 GlobalConfig.pluginConfig.pluginStates[internalName] = true;
159 }
160 // If the plugin is disabled
161 if (!GlobalConfig.pluginConfig.pluginStates[internalName])
162 {
163 LOG(MODULE_PLUGINHOST, "Cannot load a plugin that's been disabled.")
164 return false;
165 }
166
167 auto conf = JsonFromString(StringFromFile(QV2RAY_PLUGIN_SETTINGS_DIR + internalName + ".conf"));
168 plugins[internalName].pluginInterface->Initialize(QV2RAY_PLUGIN_SETTINGS_DIR + internalName + "/", conf);
169 plugins[internalName].isLoaded = true;
170 return true;
171 }

250 const QList<std::tuple<QString, QString, QJsonObject>> QvPluginHost::TryDeserializeShareLink(const QString &sharelink, //
251 QString *aliasPrefix, //
252 QString *errMessage, //
253 QString *newGroupName, //
254 bool *ok) const
255 {
256 Q_UNUSED(newGroupName)
257 QList<std::tuple<QString, QString, QJsonObject>> data;
258 *ok = false;
259 for (const auto &plugin : plugins)
260 {
261 if (plugin.isLoaded && plugin.metadata.SpecialPluginType.contains(SPECIAL_TYPE_SERIALIZOR))
262 {
263 auto serializer = plugin.pluginInterface->GetSerializer();
264 bool thisPluginCanHandle = false;
265 for (const auto &prefix : serializer->ShareLinkPrefixes())
266 {
267 thisPluginCanHandle = thisPluginCanHandle || sharelink.startsWith(prefix);
268 }
269 if (thisPluginCanHandle)
270 {
271 const auto &[protocol, outboundSettings] = serializer->DeserializeOutbound(sharelink, aliasPrefix, errMessage);
272 if (errMessage->isEmpty())
273 {
274 data << std::tuple{ *aliasPrefix, protocol, outboundSettings };
275 *ok = true;
276 }
277 break;
278 }
279 }
280 }
281 return data;
282 }

354 const QString GetPluginTypeString(const SPECIAL_TYPE_FLAGS &types)
355 {
356 QStringList typesList;
357 if (types.isEmpty())
358 {
359 typesList << QObject::tr("Normal Plugin");
360 }
361 for (auto type : types)
362 {
363 switch (type)
364 {
365 case SPECIAL_TYPE_KERNEL: typesList << QObject::tr("Kernel"); break;
366 case SPECIAL_TYPE_SERIALIZOR: typesList << QObject::tr("Share Link Parser"); break;
367 default: typesList << QObject::tr("Unknown type."); break;
368 }
369 }
370 return typesList.join(NEWLINE);
371 }

373 const QString GetPluginCapabilityString(const CAPABILITY_FLAGS &caps)
374 {
375 QStringList capsString;
376 if (caps.isEmpty())
377 {
378 capsString << QObject::tr("No Capability");
379 }
380 for (auto cap : caps)
381 {
382 switch (cap)
383 {
384 case CAPABILITY_CONNECTIVITY: capsString << QObject::tr("Connection State Change"); break;
385 case CAPABILITY_CONNECTION_ENTRY: capsString << QObject::tr("Connection Change"); break;
386 case CAPABILITY_STATS: capsString << QObject::tr("Statistics Event"); break;
387 case CAPABILITY_SYSTEM_PROXY: capsString << QObject::tr("System Proxy Event"); break;
388 default: capsString << QObject::tr("Unknown"); break;
389 }
390 }
391 return capsString.join(NEWLINE);
