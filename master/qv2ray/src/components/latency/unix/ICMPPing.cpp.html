23 /// 1s complementary checksum
24 uint16_t ping_checksum(const char *buf, size_t size)
25 {
26 size_t i;
27 uint64_t sum = 0;
28
29 for (i = 0; i < size; i += 2)
30 {
31 sum += *(uint16_t *) buf;
32 buf += 2;
33 }
34 if (size - i > 0)
35 {
36 sum += *(uint8_t *) buf;
37 }
38
39 while ((sum >> 16) != 0)
40 {
41 sum = (sum & 0xffff) + (sum >> 16);
42 }
43
44 return (uint16_t) ~sum;
45 }

55 void ICMPPing::start(int ttl)
56 {
57 if (((socketId = socket(PF_INET, SOCK_DGRAM, IPPROTO_ICMP)) < 0))
58 {
59 data.errorMessage = "EPING_SOCK: " + QObject::tr("Socket creation failed");
60 data.avg = LATENCY_TEST_VALUE_ERROR;
61 testHost->OnLatencyTestCompleted(req.id, data);
62 return;
63 }
64 // set TTL
65 if (setsockopt(socketId, SOL_IP, IP_TTL, &ttl, sizeof(ttl)) != 0)
66 {
67 data.errorMessage = "EPING_TTL: " + QObject::tr("Failed to setup TTL value");
68 data.avg = LATENCY_TEST_VALUE_ERROR;
69 testHost->OnLatencyTestCompleted(req.id, data);
70 return;
71 }
72 data.totalCount = req.totalCount;
73 data.failedCount = 0;
74 data.worst = 0;
75 data.avg = 0;
76 if (isAddr() == -1)
77 {
78 getAddrHandle = loop->resource<uvw::GetAddrInfoReq>();
79 sprintf(digitBuffer, "%d", req.port);
80 }
81 async_DNS_lookup(0, 0);
82 }
83
84 bool ICMPPing::notifyTestHost()
85 {
86 if (data.failedCount + successCount == data.totalCount)
87 {
88 if (data.failedCount == data.totalCount)
89 data.avg = LATENCY_TEST_VALUE_ERROR;
90 else
91 data.errorMessage.clear(), data.avg = data.avg / successCount;
92 testHost->OnLatencyTestCompleted(req.id, data);
93 if (timeoutTimer)
94 {
95 timeoutTimer->stop();
96 timeoutTimer->clear();
97 timeoutTimer->close();
98 }
99 if (pollHandle)
100 {
101 if (!pollHandle->closing())
102 pollHandle->stop();
103 pollHandle->clear();
104 pollHandle->close();
105 }
106 return true;
107 }
108 return false;
109 }

111 void ICMPPing::ping()
112 {
113 timeoutTimer = loop->resource<uvw::TimerHandle>();
114 uvw::OSSocketHandle osSocketHandle{ socketId };
115 pollHandle = loop->resource<uvw::PollHandle>(osSocketHandle);
116 timeoutTimer->once<uvw::TimerEvent>([this, ptr = std::weak_ptr<ICMPPing>{ shared_from_this() }](auto &, uvw::TimerHandle &h) {
117 if (ptr.expired())
118 return;
119 else
120 {
121 auto p = ptr.lock();
122 pollHandle->clear();
123 if (!pollHandle->closing())
124 pollHandle->stop();
125 pollHandle->close();
126 successCount = 0;
127 data.failedCount = data.totalCount = req.totalCount;
128 notifyTestHost();
129 }
130 });
131 timeoutTimer->start(uvw::TimerHandle::Time{ 10000 }, uvw::TimerHandle::Time{ 0 });
132 auto pollEvent = uvw::Flags<uvw::PollHandle::Event>::from<uvw::PollHandle::Event::READABLE>();
133 pollHandle->on<uvw::PollEvent>([this, ptr = shared_from_this()](uvw::PollEvent &, uvw::PollHandle &h) {
134 timeval end;
135 sockaddr_in addr;
136 socklen_t slen = sizeof(sockaddr_in);
137 int rlen = 0;
138 char buf[1024];
139 do
140 {
141 do
142 {
143 rlen = recvfrom(socketId, buf, 1024, 0, (struct sockaddr *) &addr, &slen);
144 } while (rlen == -1 && errno == EINTR);
145
146 // skip malformed
147 #ifdef Q_OS_MAC
148 if (rlen < sizeof(icmp) + 20)
149 #else
150 if (rlen < sizeof(icmp))
151 #endif
152 continue;
153
154 #ifdef Q_OS_MAC
155 auto &resp = *reinterpret_cast<icmp *>(buf + 20);
156 #else
157 auto &resp = *reinterpret_cast<icmp *>(buf);
158 #endif
159 // skip the ones we didn't send
160 auto cur_seq = resp.icmp_hun.ih_idseq.icd_seq;
161 if (cur_seq >= seq)
162 continue;
163
164 switch (resp.icmp_type)
165 {
166 case ICMP_ECHOREPLY:
167 gettimeofday(&end, nullptr);
168 data.avg +=
169 1000 * (end.tv_sec - startTimevals[cur_seq - 1].tv_sec) + (end.tv_usec - startTimevals[cur_seq - 1].tv_usec) / 1000;
170 successCount++;
171 notifyTestHost();
172 continue;
173 case ICMP_UNREACH:
174 data.errorMessage = "EPING_DST: " + QObject::tr("Destination unreachable");
175 data.failedCount++;
176 if (notifyTestHost())
177 {
178 h.clear();
179 h.close();
180 return;
181 }
182 continue;
183 case ICMP_TIMXCEED:
184 data.errorMessage = "EPING_TIME: " + QObject::tr("Timeout");
185 data.failedCount++;
186 if (notifyTestHost())
187 {
188 h.clear();
189 h.close();
190 return;
191 }
192 continue;
193 default:
194 data.errorMessage = "EPING_UNK: " + QObject::tr("Unknown error");
195 data.failedCount++;
196 if (notifyTestHost())
197 {
198 h.clear();
199 h.close();
200 return;
201 }
202 continue;
203 }
204 } while (rlen > 0);
205 });
206 pollHandle->start(pollEvent);
207 for (int i = 0; i < req.totalCount; ++i)
208 {
209 // prepare echo request packet
210 icmp _icmp_request;
211 memset(&_icmp_request, 0, sizeof(_icmp_request));
212 _icmp_request.icmp_type = ICMP_ECHO;
213 _icmp_request.icmp_hun.ih_idseq.icd_id = 0; // SOCK_DGRAM & 0 => id will be set by kernel
214 _icmp_request.icmp_hun.ih_idseq.icd_seq = seq++;
215 _icmp_request.icmp_cksum = ping_checksum(reinterpret_cast<char *>(&_icmp_request), sizeof(_icmp_request));
216 int n;
217 timeval start;
218 gettimeofday(&start, nullptr);
219 startTimevals.push_back(start);
220 do
221 {
222 n = ::sendto(socketId, &_icmp_request, sizeof(icmp), 0, (struct sockaddr *) &storage, sizeof(struct sockaddr));
223 } while (n < 0 && errno == EINTR);
224 }
225 }
