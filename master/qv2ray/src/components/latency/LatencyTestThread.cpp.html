29 void LatencyTestThread::run()
30 {
31 loop = uvw::Loop::create();
32 stopTimer = loop->resource<uvw::TimerHandle>();
33 stopTimer->on<uvw::TimerEvent>([this](auto &, auto &handle) {
34 if (isStop)
35 {
36 if (!requests.empty())
37 requests.clear();
38 int timer_count = 0;
39 uv_walk(
40 loop->raw(),
41 [](uv_handle_t *handle, void *arg) {
42 int &counter = *static_cast<int *>(arg);
43 if (uv_is_closing(handle) == 0)
44 counter++;
45 },
46 &timer_count);
47 if (timer_count == 1) // only current timer
48 {
49 handle.stop();
50 handle.close();
51 loop->clear();
52 loop->close();
53 loop->stop();
54 }
55 }
56 else
57 {
58 if (requests.empty())
59 return;
60 std::unique_lock<std::mutex> lockGuard{ m };
61 auto parent = qobject_cast<LatencyTestHost *>(this->parent());
62 for (auto &req : requests)
63 {
64 switch (req.method)
65 {
66 case ICMPING:
67 {
68 auto ptr = std::make_shared<icmping::ICMPPing>(loop, req, parent);
69 ptr->start();
70 }
71 break;
72 case TCPING:
73 default:
74 {
75 auto ptr = std::make_shared<tcping::TCPing>(loop, req, parent);
76 ptr->start();
77 break;
78 }
79 }
80 }
81 requests.clear();
82 }
83 });
84 stopTimer->start(uvw::TimerHandle::Time{ 500 }, uvw::TimerHandle::Time{ 500 });
85 loop->run();
86 }
