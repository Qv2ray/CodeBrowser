13 struct DNSServerObject
14 {
15 bool QV2RAY_DNS_IS_COMPLEX_DNS;
16 QString address;
17 int port;
18 QList<QString> domains;
19 QList<QString> expectIPs;
20 DNSServerObject() : QV2RAY_DNS_IS_COMPLEX_DNS(false), port(53){};
21 DNSServerObject(const QString &_address) : DNSServerObject()
22 {
23 address = _address;
24 };
25
26 friend bool operator==(const DNSServerObject &left, const DNSServerObject &right)
27 {
28 return left.QV2RAY_DNS_IS_COMPLEX_DNS == right.QV2RAY_DNS_IS_COMPLEX_DNS && //
29 left.address == right.address && //
30 left.port == right.port && //
31 left.domains == right.domains && //
32 left.expectIPs == right.expectIPs;
33 }
34 JSONSTRUCT_REGISTER(DNSServerObject, F(QV2RAY_DNS_IS_COMPLEX_DNS, address, port, domains, expectIPs))
35 };
36 QMap<QString, QString> hosts;
37 QList<DNSServerObject> servers;
38 QString clientIp;
39 QString tag;
40 friend bool operator==(const DNSObject &left, const DNSObject &right)
41 {
42 return left.hosts == right.hosts && left.servers == right.servers && left.clientIp == right.clientIp && left.tag == right.tag;
43 }
44 JSONSTRUCT_REGISTER(DNSObject, F(hosts, servers, clientIp, tag))
45 };
