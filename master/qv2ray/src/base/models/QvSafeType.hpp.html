50 template<typename enumKey, typename TValue, typename = typename std::enable_if<std::is_enum<enumKey>::value>::type>
51 struct QvEnumMap : QMap<enumKey, TValue>
52 {
53 static constexpr auto ENUM_JSON_KEY_PREFIX = "$";
54 void loadJson(const QJsonValue &json_object)
55 {
56 QMap<QString, TValue> data;
57 JsonStructHelper::___json_struct_load_data(data, json_object);
58 this->clear();
59 for (QString k_str : data.keys())
60 {
61 auto k = (enumKey) k_str.remove(ENUM_JSON_KEY_PREFIX).toInt();
62 this->insert(k, data[ENUM_JSON_KEY_PREFIX + k_str]);
63 }
64 }
65 [[nodiscard]] static auto fromJson(const QJsonValue &json)
66 {
67 QvEnumMap t;
68 t.loadJson(json);
69 return t;
70 }
71 [[nodiscard]] const QJsonObject toJson() const
72 {
73 QMap<QString, TValue> data;
74 for (const auto &k : this->keys())
75 {
76 data[ENUM_JSON_KEY_PREFIX + QString::number(k)] = this->value(k);
77 }
78 return JsonStructHelper::___json_struct_store_data(data).toObject();
79 }
80 };
