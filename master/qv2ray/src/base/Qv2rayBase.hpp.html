124 inline QStringList Qv2rayAssetsPaths(const QString &dirName)
125 {
126 // Configuration Path
127 QStringList list;
128 list << QV2RAY_CONFIG_DIR + dirName;
129 list << ":/" + dirName;
130 //
131#ifdef Q_OS_LINUX
132 // Linux platform directories.
133 list << QString("/lib/qv2ray/" + dirName);
134 list << QString("/usr/lib/qv2ray/" + dirName);
135 list << QString("/usr/local/lib/qv2ray/" + dirName);
136 //
137 list << QString("/usr/share/qv2ray/" + dirName);
138 list << QString("/usr/local/share/qv2ray/" + dirName);
139 // For AppImage?
140 list << QString(QDir(QCoreApplication::applicationDirPath() + "/../share/qv2ray/" + dirName).absolutePath());
141 // For Snap
142 if (qEnvironmentVariableIsSet("SNAP"))
143 {
144 list << QString(qEnvironmentVariable("SNAP") + "/usr/share/qv2ray/" + dirName);
145 }
146#elif defined(Q_OS_MAC)
147 // macOS platform directories.
148 list << QDir(QCoreApplication::applicationDirPath() + "/../Resources/" + dirName).absolutePath();
149#endif
150 list << QStandardPaths::locateAll(QStandardPaths::AppDataLocation, dirName, QStandardPaths::LocateDirectory);
151 list << QStandardPaths::locateAll(QStandardPaths::AppConfigLocation, dirName, QStandardPaths::LocateDirectory);
152 // This is the default behavior on Windows
153 list << QCoreApplication::applicationDirPath() + "/" + dirName;
154 list.removeDuplicates();
155 return list;
156 }
