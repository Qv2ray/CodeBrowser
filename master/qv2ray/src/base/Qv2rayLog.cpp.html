23 void __QV2RAY_LOG_FUNC__(int type, const std::string &func, int line, const QString &module, const QString &log)
24 {
25 auto logString = QString("[" % module % "]: " % log);
26 auto funcPrepend = QString::fromStdString(func + ":" + std::to_string(line) + " ");
27
28#ifdef QT_DEBUG
29 // Debug build version, we only print info for DEBUG logs and print
30 // ALL info when debugLog presents,
31 if (type == QV2RAY_LOG_DEBUG || StartupOption.debugLog)
32 {
33 logString.prepend(funcPrepend);
34 }
35#else
36 // We only process DEBUG log in Release mode
37 if (type == QV2RAY_LOG_DEBUG)
38 {
39 if (StartupOption.debugLog)
40 {
41 logString.prepend(funcPrepend);
42 }
43 else
44 {
45 // Discard debug log in non-debug Qv2ray version with
46 // no-debugLog mode.
47 return;
48 }
49 }
50#endif
51#ifdef Q_OS_ANDROID
52 __android_log_write(ANDROID_LOG_INFO, "Qv2ray", logString.toStdString().c_str());
53#else
54 std::cout << logString.toStdString() << std::endl;
55#endif
56 {
57 QMutexLocker _(&__loggerMutex);
58 __loggerBuffer->append(logString + NEWLINE);
59 }
60 } // namespace Qv2ray::base
61
62 QString readLastLog()
63 {
64 QMutexLocker _(&__purgerMutex);
65 {
66 QMutexLocker locker(&__loggerMutex);
67 __loggerBuffer.swap(__purgerBuffer);
68 }
69 auto result = __purgerBuffer->join("");
70 __purgerBuffer->clear();
71 return result;
72 }
