111 void Qv2rayApplication::onMessageReceived(quint32 clientId, QByteArray _msg)
112 {
113 // Sometimes SingleApplication will send message with clientId == 0, ignore them.
114 if (clientId == instanceId())
115 return;
116 const auto msg = Qv2rayProcessArguments::fromJson(JsonFromString(_msg));
117 LOG(MODULE_INIT, "Client ID: " + QSTRN(clientId) + ", message received, version: " + msg.version)
118 DEBUG(MODULE_INIT, _msg)
119 //
120 const auto currentVersion = semver::version::from_string(QV2RAY_VERSION_STRING);
121 const auto newVersionString = msg.version.isEmpty() ? "0.0.0" : msg.version.toStdString();
122 const auto newVersion = semver::version::from_string(newVersionString);
123 //
124 if (newVersion > currentVersion)
125 {
126 const auto newPath = msg.fullArgs.first();
127 QString message;
128 message += tr("A new version of Qv2ray is attemping to start:") + NEWLINE;
129 message += NEWLINE;
130 message += tr("New version information: ") + NEWLINE;
131 message += tr("Qv2ray version: %1").arg(msg.version) + NEWLINE;
132 message += tr("Qv2ray path: %1").arg(newPath) + NEWLINE;
133 message += NEWLINE;
134 message += tr("Do you want to exit and launch that new version?");
135
136 const auto result = QvMessageBoxAsk(nullptr, tr("New version detected"), message);
137 if (result == QMessageBox::Yes)
138 {
139 Qv2rayProcessArgument._qvNewVersionPath = newPath;
140 QuitApplication(QVEXIT_NEW_VERSION);
141 }
142 }
143
144 for (const auto &argument : msg.arguments)
145 {
146 switch (argument)
147 {
148 case Qv2rayProcessArguments::EXIT:
149 {
150 QuitApplication();
151 break;
152 }
153 case Qv2rayProcessArguments::NORMAL:
154 {
155 mainWindow->show();
156 mainWindow->raise();
157 mainWindow->activateWindow();
158 break;
159 }
160 case Qv2rayProcessArguments::RECONNECT:
161 {
162 ConnectionManager->RestartConnection();
163 break;
164 }
165 case Qv2rayProcessArguments::DISCONNECT:
166 {
167 ConnectionManager->StopConnection();
168 break;
169 }
170 case Qv2rayProcessArguments::QV2RAY_LINK:
171 {
172 for (const auto &link : msg.links)
173 {
174 const auto url = QUrl::fromUserInput(link);
175 const auto command = url.host();
176 auto subcommands = url.path().split("/");
177 subcommands.removeAll("");
178 QMap<QString, QString> args;
179 for (const auto &kvp : QUrlQuery(url).queryItems())
180 {
181 args.insert(kvp.first, kvp.second);
182 }
183 if (command == "open")
184 {
185 emit mainWindow->ProcessCommand(command, subcommands, args);
186 }
187 }
188 break;
189 }
190 }
191 }
192 }

195 Qv2rayExitCode Qv2rayApplication::RunQv2ray()
196 {
197 // Show MainWindow
198 mainWindow = new MainWindow();
199 if (Qv2rayProcessArgument.arguments.contains(Qv2rayProcessArguments::QV2RAY_LINK))
200 {
201 for (const auto &link : Qv2rayProcessArgument.links)
202 {
203 const auto url = QUrl::fromUserInput(link);
204 const auto command = url.host();
205 auto subcommands = url.path().split("/");
206 subcommands.removeAll("");
207 QMap<QString, QString> args;
208 for (const auto &kvp : QUrlQuery(url).queryItems())
209 {
210 args.insert(kvp.first, kvp.second);
211 }
212 if (command == "open")
213 {
214 emit mainWindow->ProcessCommand(command, subcommands, args);
215 }
216 }
217 }
218#ifdef Q_OS_MACOS
219 connect(this, &QApplication::applicationStateChanged, [this](Qt::ApplicationState state) {
220 switch (state)
221 {
222 case Qt::ApplicationActive:
223 {
224 mainWindow->show();
225 mainWindow->raise();
226 mainWindow->activateWindow();
227 break;
228 }
229 case Qt::ApplicationHidden:
230 case Qt::ApplicationInactive:
231 case Qt::ApplicationSuspended: break;
232 }
233 });
234#endif
235 return Qv2rayExitCode(exec());
236 }

238 bool Qv2rayApplication::FindAndCreateInitialConfiguration()
239 {
240 if (initialized)
241 {
242 LOG(MODULE_INIT, "Qv2ray has already been initialized!")
243 return false;
244 }
245 LOG(MODULE_INIT, "Application exec path: " + applicationDirPath())
246 // Non-standard paths needs special handing for "_debug"
247 const auto currentPathConfig = applicationDirPath() + "/config" QV2RAY_CONFIG_DIR_SUFFIX;
248 const auto homeQv2ray = QDir::homePath() + "/.qv2ray" QV2RAY_CONFIG_DIR_SUFFIX;
249 //
250 // Standard paths already handles the "_debug" suffix for us.
251 const auto configQv2ray = QStandardPaths::writableLocation(QStandardPaths::AppConfigLocation);
252 //
253 //
254 // Some built-in search paths for Qv2ray to find configs. (load the first one if possible).
255 QStringList configFilePaths;
256 const auto useManualConfigPath = qEnvironmentVariableIsSet(QV2RAY_CONFIG_PATH_ENV_NAME);
257 const auto manualConfigPath = qEnvironmentVariable(QV2RAY_CONFIG_PATH_ENV_NAME);
258 if (useManualConfigPath)
259 {
260 LOG(MODULE_INIT, "Using config path from env: " + manualConfigPath)
261 configFilePaths << manualConfigPath;
262 }
263 else
264 {
265 configFilePaths << currentPathConfig;
266 configFilePaths << configQv2ray;
267 configFilePaths << homeQv2ray;
268 }
269 QString configPath = "";
270 bool hasExistingConfig = false;
271
272 for (const auto &path : configFilePaths)
273 {
274 // Verify the config path, check if the config file exists and in the
275 // correct JSON format. True means we check for config existence as
276 // well. ----------------------------------------------|HERE|
277 bool isValidConfigPath = CheckSettingsPathAvailability(path, true);
278
279 // If we already found a valid config file. just simply load it...
280 if (hasExistingConfig)
281 break;
282
283 if (isValidConfigPath)
284 {
285 DEBUG(MODULE_INIT, "Path: " + path + " is valid.")
286 configPath = path;
287 hasExistingConfig = true;
288 }
289 else
290 {
291 LOG(MODULE_INIT, "Path: " + path + " does not contain a valid config file.")
292 }
293 }
294
295 if (hasExistingConfig)
296 {
297 // Use the config path found by the checks above
298 SetConfigDirPath(configPath);
299 LOG(MODULE_INIT, "Using " + QV2RAY_CONFIG_DIR + " as the config path.")
300 }
301 else
302 {
303 // If there's no existing config.
304 //
305 // Create new config at these dirs, these are default values for each platform.
306 if (useManualConfigPath)
307 {
308 configPath = manualConfigPath;
309 }
310 else
311 {
312#if defined(Q_OS_WIN) && !defined(QV2RAY_NO_ASIDECONFIG)
313 configPath = currentPathConfig;
314#else
315 configPath = configQv2ray;
316#endif
317 }
318
319 bool hasPossibleNewLocation = QDir().mkpath(configPath) && CheckSettingsPathAvailability(configPath, false);
320 // Check if the dirs are write-able
321 if (!hasPossibleNewLocation)
322 {
323 // None of the path above can be used as a dir for storing config.
324 // Even the last folder failed to pass the check.
325 LOG(MODULE_INIT, "FATAL")
326 LOG(MODULE_INIT, " ---> CANNOT find a proper place to store Qv2ray config files.")
327 QvMessageBoxWarn(nullptr, tr("Cannot Start Qv2ray"),
328 tr("Cannot find a place to store config files.") + NEWLINE + //
329 tr("Qv2ray has searched these paths below:") + NEWLINE + NEWLINE + //
330 configFilePaths.join(NEWLINE) + NEWLINE + //
331 tr("It usually means you don't have the write permission to all of those locations.") + NEWLINE + //
332 tr("Qv2ray will now exit.")); //
333 return false;
334 }
335
336 // Found a valid config dir, with write permission, but assume no config is located in it.
337 LOG(MODULE_INIT, "Set " + configPath + " as the config path.")
338 SetConfigDirPath(configPath);
339
340 if (QFile::exists(QV2RAY_CONFIG_FILE))
341 {
342 // As we already tried to load config from every possible dir.
343 //
344 // This condition branch (!hasExistingConfig check) holds the fact that current config dir,
345 // should NOT contain any valid file (at least in the same name)
346 //
347 // It usually means that QV2RAY_CONFIG_FILE here has a corrupted JSON format.
348 //
349 // Otherwise Qv2ray would have loaded this config already instead of notifying to create a new config in this folder.
350 //
351 LOG(MODULE_INIT, "This should not occur: Qv2ray config exists but failed to load.")
352 QvMessageBoxWarn(nullptr, tr("Failed to initialise Qv2ray"),
353 tr("Failed to determine the location of config file:") + NEWLINE + //
354 tr("Qv2ray has found a config file, but it failed to be loaded due to some errors.") + NEWLINE + //
355 tr("A workaround is to remove the this file and restart Qv2ray:") + NEWLINE + //
356 QV2RAY_CONFIG_FILE + NEWLINE + //
357 tr("Qv2ray will now exit.") + NEWLINE + //
358 tr("Please report if you think it's a bug.")); //
359 return false;
360 }
361
362 Qv2rayConfigObject conf;
363 conf.kernelConfig.KernelPath(QString(QV2RAY_DEFAULT_VCORE_PATH));
364 conf.kernelConfig.AssetsPath(QString(QV2RAY_DEFAULT_VASSETS_PATH));
365 conf.logLevel = 3;
366 conf.uiConfig.language = QLocale::system().name();
367 conf.defaultRouteConfig.dnsConfig.servers << QvConfig_DNS::DNSServerObject{ "1.1.1.1" } //
368 << QvConfig_DNS::DNSServerObject{ "8.8.8.8" } //
369 << QvConfig_DNS::DNSServerObject{ "8.8.4.4" };
370
371 // Save initial config.
372 SaveGlobalSettings(conf);
373 LOG(MODULE_INIT, "Created initial config file.")
374 }
375
376 if (!QDir(QV2RAY_GENERATED_DIR).exists())
377 {
378 // The dir used to generate final config file, for V2Ray interaction.
379 QDir().mkdir(QV2RAY_GENERATED_DIR);
380 LOG(MODULE_INIT, "Created config generation dir at: " + QV2RAY_GENERATED_DIR)
381 }
382 return true;
383 }

385 bool Qv2rayApplication::LoadConfiguration()
386 {
387 // Load the config for upgrade, but do not parse it to the struct.
388 auto conf = JsonFromString(StringFromFile(QV2RAY_CONFIG_FILE));
389 const auto configVersion = conf["config_version"].toInt();
390
391 if (configVersion > QV2RAY_CONFIG_VERSION)
392 {
393 // Config version is larger than the current version...
394 // This is rare but it may happen....
395 QvMessageBoxWarn(nullptr, tr("Qv2ray Cannot Continue"), //
396 tr("You are running a lower version of Qv2ray compared to the current config file.") + NEWLINE + //
397 tr("Please check if there's an issue explaining about it.") + NEWLINE + //
398 tr("Or submit a new issue if you think this is an error.") + NEWLINE + NEWLINE + //
399 tr("Qv2ray will now exit."));
400 return false;
401 }
402 else if (configVersion < QV2RAY_CONFIG_VERSION)
403 {
404 // That is the config file needs to be upgraded.
405 conf = Qv2ray::UpgradeSettingsVersion(configVersion, QV2RAY_CONFIG_VERSION, conf);
406 }
407
408 // Load config object from upgraded config QJsonObject
409 auto confObject = Qv2rayConfigObject::fromJson(conf);
410
411 const auto allTranslations = Qv2rayTranslator->GetAvailableLanguages();
412 const auto osLanguage = QLocale::system().name();
413
414 if (!allTranslations.contains(confObject.uiConfig.language))
415 {
416 // If we need to reset the language.
417 if (allTranslations.contains(osLanguage))
418 {
419 confObject.uiConfig.language = osLanguage;
420 }
421 else if (!allTranslations.isEmpty())
422 {
423 confObject.uiConfig.language = allTranslations.first();
424 }
425 // If configured language is not found.
426 LOG(MODULE_UI, "Fall back language setting to: " + osLanguage)
427 }
428
429 if (!Qv2rayTranslator->InstallTranslation(confObject.uiConfig.language))
430 {
431 QvMessageBoxWarn(nullptr, "Translation Failed",
432 "Cannot load translation for " + confObject.uiConfig.language + NEWLINE + //
433 "English is now used." + NEWLINE + NEWLINE + //
434 "Please go to Preferences Window to change language or open an Issue");
435 confObject.uiConfig.language = "en_US";
436 }
437
438 // Let's save the config.
439 SaveGlobalSettings(confObject);
440 return true;
441 }

452 Qv2rayPreInitResult Qv2rayApplication::PreInitialize(int argc, char **argv)
453 {
454 QString errorMessage;
455 Qv2rayPreInitResult result;
456 {
457 QCoreApplication coreApp(argc, argv);
458 const auto &args = coreApp.arguments();
459 Qv2rayProcessArgument.version = QV2RAY_VERSION_STRING;
460 Qv2rayProcessArgument.fullArgs = args;
461 result = ParseCommandLine(&errorMessage, args);
462 LOG(MODULE_INIT, "Qv2ray PreInitialization: " + errorMessage)
463 if (result != PRE_INIT_RESULT_CONTINUE)
464 return result;
465#ifdef Q_OS_WIN
466 const auto appPath = QDir::toNativeSeparators(coreApp.applicationFilePath());
467 const auto regPath = "HKEY_CURRENT_USER\\Software\\Classes\\" + QV2RAY_URL_SCHEME;
468
469 QSettings reg(regPath, QSettings::NativeFormat);
470
471 reg.setValue("Default", "Qv2ray");
472 reg.setValue("URL Protocol", "");
473
474 reg.beginGroup("DefaultIcon");
475 reg.setValue("Default", QString("%1,1").arg(appPath));
476 reg.endGroup();
477
478 reg.beginGroup("shell");
479 reg.beginGroup("open");
480 reg.beginGroup("command");
481 reg.setValue("Default", appPath + " %1");
482#endif
483 }
484
485 // noScaleFactors = disable HiDPI
486 if (StartupOption.noScaleFactor)
487 {
488 LOG(MODULE_INIT, "Force set QT_SCALE_FACTOR to 1.")
489 DEBUG(MODULE_UI, "Original QT_SCALE_FACTOR was: " + qEnvironmentVariable("QT_SCALE_FACTOR"))
490 qputenv("QT_SCALE_FACTOR", "1");
491 }
492 else
493 {
494 DEBUG(MODULE_INIT, "High DPI scaling is enabled.")
495 QCoreApplication::setAttribute(Qt::AA_EnableHighDpiScaling);
496#if QT_VERSION >= QT_VERSION_CHECK(5, 14, 0)
497 QGuiApplication::setHighDpiScaleFactorRoundingPolicy(Qt::HighDpiScaleFactorRoundingPolicy::PassThrough);
498#endif
499 }
500 return result;
501 }

503 Qv2rayPreInitResult Qv2rayApplication::ParseCommandLine(QString *errorMessage, const QStringList &_argx_)
504 {
505 QStringList filteredArgs;
506 for (const auto &arg : _argx_)
507 {
508#ifdef Q_OS_MACOS
509 if (arg.contains("-psn"))
510 continue;
511#endif
512 filteredArgs << arg;
513 }
514 QCommandLineParser parser;
515 //
516 QCommandLineOption noAPIOption("noAPI", tr("Disable gRPC API subsystem"));
517 QCommandLineOption noPluginsOption("noPlugin", tr("Disable plugins feature"));
518 QCommandLineOption noScaleFactorOption("noScaleFactor", tr("Disable Qt UI scale factor"));
519 QCommandLineOption debugOption("debug", tr("Enable debug output"));
520 QCommandLineOption noAutoConnectionOption("noAutoConnection", tr("Do not automatically connect"));
521 QCommandLineOption disconnectOption("disconnect", tr("Stop current connection"));
522 QCommandLineOption reconnectOption("reconnect", tr("Reconnect last connection"));
523 QCommandLineOption exitOption("exit", tr("Exit Qv2ray"));
524 //
525 parser.setApplicationDescription(tr("Qv2ray - A cross-platform Qt frontend for V2Ray."));
526 parser.setSingleDashWordOptionMode(QCommandLineParser::ParseAsLongOptions);
527 //
528 parser.addOption(noAPIOption);
529 parser.addOption(noPluginsOption);
530 parser.addOption(noScaleFactorOption);
531 parser.addOption(debugOption);
532 parser.addOption(noAutoConnectionOption);
533 parser.addOption(disconnectOption);
534 parser.addOption(reconnectOption);
535 parser.addOption(exitOption);
536 //
537 auto helpOption = parser.addHelpOption();
538 auto versionOption = parser.addVersionOption();
539
540 if (!parser.parse(filteredArgs))
541 {
542 *errorMessage = parser.errorText();
543 return PRE_INIT_RESULT_CONTINUE;
544 }
545
546 if (parser.isSet(versionOption))
547 {
548 parser.showVersion();
549 return PRE_INIT_RESULT_QUIT;
550 }
551
552 if (parser.isSet(helpOption))
553 {
554 parser.showHelp();
555 return PRE_INIT_RESULT_QUIT;
556 }
557
558 for (const auto &arg : parser.positionalArguments())
559 {
560 if (arg.startsWith(QV2RAY_URL_SCHEME + "://"))
561 {
562 Qv2rayProcessArgument.arguments << Qv2rayProcessArguments::QV2RAY_LINK;
563 Qv2rayProcessArgument.links << arg;
564 }
565 }
566
567 if (parser.isSet(exitOption))
568 {
569 DEBUG(MODULE_INIT, "disconnectOption is set.")
570 Qv2rayProcessArgument.arguments << Qv2rayProcessArguments::EXIT;
571 }
572
573 if (parser.isSet(disconnectOption))
574 {
575 DEBUG(MODULE_INIT, "disconnectOption is set.")
576 Qv2rayProcessArgument.arguments << Qv2rayProcessArguments::DISCONNECT;
577 }
578
579 if (parser.isSet(reconnectOption))
580 {
581 DEBUG(MODULE_INIT, "reconnectOption is set.")
582 Qv2rayProcessArgument.arguments << Qv2rayProcessArguments::RECONNECT;
583 }
584
585 if (parser.isSet(noAPIOption))
586 {
587 DEBUG(MODULE_INIT, "noAPIOption is set.")
588 StartupOption.noAPI = true;
589 }
590
591 if (parser.isSet(debugOption))
592 {
593 DEBUG(MODULE_INIT, "debugOption is set.")
594 StartupOption.debugLog = true;
595 }
596
597 if (parser.isSet(noScaleFactorOption))
598 {
599 DEBUG(MODULE_INIT, "noScaleFactorOption is set.")
600 StartupOption.noScaleFactor = true;
601 }
602
603 if (parser.isSet(noAutoConnectionOption))
604 {
605 DEBUG(MODULE_INIT, "noAutoConnectOption is set.")
606 StartupOption.noAutoConnection = true;
607 }
608
609 if (parser.isSet(noPluginsOption))
610 {
611 DEBUG(MODULE_INIT, "noPluginOption is set.")
612 StartupOption.noPlugins = true;
613 }
614 *errorMessage = "OK";
615 return PRE_INIT_RESULT_CONTINUE;
616 }
