16const QString SayLastWords() noexcept
17{
18 QStringList msg;
19 msg << "------- BEGIN QV2RAY CRASH REPORT -------";
20#ifdef Q_OS_WIN
21 void *stack[1024];
22 HANDLE process = GetCurrentProcess();
23 SymInitialize(process, NULL, TRUE);
24 SymSetOptions(SYMOPT_LOAD_ANYTHING);
25 WORD numberOfFrames = CaptureStackBackTrace(0, 1024, stack, NULL);
26 SYMBOL_INFO *symbol = (SYMBOL_INFO *) malloc(sizeof(SYMBOL_INFO) + (512 - 1) * sizeof(TCHAR));
27 symbol->MaxNameLen = 512;
28 symbol->SizeOfStruct = sizeof(SYMBOL_INFO);
29 DWORD displacement;
30 IMAGEHLP_LINE64 *line = (IMAGEHLP_LINE64 *) malloc(sizeof(IMAGEHLP_LINE64));
31 line->SizeOfStruct = sizeof(IMAGEHLP_LINE64);
32 //
33 for (int i = 0; i < numberOfFrames; i++)
34 {
35 const auto address = (DWORD64) stack[i];
36 SymFromAddr(process, address, NULL, symbol);
37 if (SymGetLineFromAddr64(process, address, &displacement, line))
38 {
39 msg << QString("[%1]: %2 (%3:%4)").arg(symbol->Address).arg(symbol->Name).arg(line->FileName).arg(line->LineNumber);
40 }
41 else
42 {
43 msg << QString("[%1]: %2 SymGetLineFromAddr64[%3]").arg(symbol->Address).arg(symbol->Name).arg(GetLastError());
44 }
45 }
46#endif
47
48 if (KernelInstance)
49 {
50 msg << "Active Kernel Instances:";
51 const auto kernels = KernelInstance->GetActiveKernelProtocols();
52 msg << JsonToString(JsonStructHelper::___json_struct_store_data(static_cast<QList<QString>>(kernels)).toArray(), QJsonDocument::Compact);
53 msg << "Current Connection:";
54 //
55 const auto currentConnection = KernelInstance->CurrentConnection();
56 msg << JsonToString(currentConnection.toJson(), QJsonDocument::Compact);
57 msg << NEWLINE;
58 //
59 if (ConnectionManager && !currentConnection.isEmpty())
60 {
61 msg << "Active Connection Settings:";
62 const auto connection = ConnectionManager->GetConnectionMetaObject(currentConnection.connectionId);
63 auto group = ConnectionManager->GetGroupMetaObject(currentConnection.groupId);
64 //
65 // Do not collect private data.
66 // msg << NEWLINE;
67 // msg << JsonToString(ConnectionManager->GetConnectionRoot(currentConnection.connectionId));
68 group.subscriptionOption.address = "HIDDEN";
69 //
70 msg << JsonToString(connection.toJson(), QJsonDocument::Compact);
71 msg << NEWLINE;
72 msg << "Group:";
73 msg << JsonToString(group.toJson(), QJsonDocument::Compact);
74 msg << NEWLINE;
75 }
76 }
77 if (PluginHost)
78 {
79 msg << "Plugins:";
80 const auto plugins = PluginHost->AvailablePlugins();
81 for (const auto &plugin : plugins)
82 {
83 const auto data = PluginHost->GetPluginMetadata(plugin);
84 QList<QString> dataList;
85 dataList << data.Name;
86 dataList << data.Author;
87 dataList << data.InternalName;
88 dataList << data.Description;
89 msg << JsonToString(JsonStructHelper::___json_struct_store_data(dataList).toArray(), QJsonDocument::Compact);
90 }
91 msg << NEWLINE;
92 }
93
94 msg << "GlobalConfig:";
95 msg << JsonToString(GlobalConfig.toJson(), QJsonDocument::Compact);
96 msg << "------- END OF QV2RAY CRASH REPORT -------";
97 return msg.join(NEWLINE);
98}

100void signalHandler(int signum)
101{
102 std::cout << "Qv2ray: Interrupt signal (" << signum << ") received." << std::endl;
103
104 if (signum == SIGTERM)
105 {
106 if (qApp)
107 qApp->exit();
108 return;
109 }
110 std::cout << "Collecting StackTrace" << std::endl;
111 const auto msg = SayLastWords();
112 const auto filePath = QV2RAY_CONFIG_DIR + "bugreport/QvBugReport_" + QSTRN(system_clock::to_time_t(system_clock::now())) + ".stacktrace";
113 {
114 std::cout << msg.toStdString() << std::endl;
115 QDir().mkpath(QV2RAY_CONFIG_DIR + "bugreport/");
116 StringToFile(msg, filePath);
117 std::cout << "Backtrace saved in: " + filePath.toStdString() << std::endl;
118 }
119 if (qvApp)
120 {
121 qApp->clipboard()->setText(filePath);
122 QString message = QObject::tr("Qv2ray has encountered an uncaught exception: ") + NEWLINE + //
123 QObject::tr("Please report a bug via Github with the file located here: ") + NEWLINE NEWLINE + //
124 filePath;
125 QvMessageBoxWarn(nullptr, "UNCAUGHT EXCEPTION", message);
126 }
127 exit(-99);
128}

130QPair<Qv2rayExitCode, std::optional<QString>> RunQv2rayApplicationScoped(int argc, char *argv[])
131{
132 Qv2rayApplication app(argc, argv);
133
134 const auto setupStatus = app.SetupQv2ray();
135 switch (setupStatus)
136 {
137 case Qv2rayApplication::NORMAL: break;
138 case Qv2rayApplication::SINGLE_APPLICATION: return { QVEXIT_SECONDARY_INSTANCE, std::nullopt };
139 case Qv2rayApplication::FAILED: return { QVEXIT_EARLY_SETUP_FAIL, app.tr("Qv2ray early initialization failed.") };
140 }
141
142 LOG("LICENCE", NEWLINE //
143 "This program comes with ABSOLUTELY NO WARRANTY." NEWLINE //
144 "This is free software, and you are welcome to redistribute it" NEWLINE //
145 "under certain conditions." NEWLINE NEWLINE //
146 "Copyright (c) 2019-2020 Qv2ray Development Group." NEWLINE //
147 "Third-party libraries that have been used in Qv2ray can be found in the About page." NEWLINE)
148
149#ifdef QT_DEBUG
150 std::cerr << "WARNING: ================ This is a debug build, many features are not stable enough. ================" << std::endl;
151#endif
152 //
153 // Qv2ray Initialize, find possible config paths and verify them.
154 if (!app.FindAndCreateInitialConfiguration())
155 {
156 LOG(MODULE_INIT, "Cannot find or create initial configuration file.")
157 return { QVEXIT_CONFIG_PATH_FAIL, app.tr("Cannot create initial config file.") };
158 }
159 if (!app.LoadConfiguration())
160 {
161 LOG(MODULE_INIT, "Cannot load existing configuration file.")
162 return { QVEXIT_CONFIG_FILE_FAIL, "Configuration file currupted" };
163 }
164
165 // Check OpenSSL version for auto-update and subscriptions
166 auto osslReqVersion = QSslSocket::sslLibraryBuildVersionString();
167 auto osslCurVersion = QSslSocket::sslLibraryVersionString();
168 LOG(MODULE_NETWORK, "Current OpenSSL version: " + osslCurVersion)
169
170 if (!QSslSocket::supportsSsl())
171 {
172 LOG(MODULE_NETWORK, "Required OpenSSL version: " + osslReqVersion)
173 LOG(MODULE_NETWORK, "OpenSSL library MISSING, Quitting.")
174 QvMessageBoxWarn(nullptr, QObject::tr("Dependency Missing"),
175 QObject::tr("Cannot find openssl libs") + NEWLINE +
176 QObject::tr("This could be caused by a missing of `openssl` package in your system.") + NEWLINE +
177 QObject::tr("If you are using an AppImage from Github Action, please report a bug.") + NEWLINE + //
178 NEWLINE + QObject::tr("Technical Details") + NEWLINE + //
179 "OSsl.Rq.V=" + osslReqVersion + NEWLINE + //
180 "OSsl.Cr.V=" + osslCurVersion);
181 return { QVEXIT_SSL_FAIL, app.tr("Cannot start Qv2ray without OpenSSL") };
182 }
183
184 app.InitializeGlobalVariables();
185
186#ifndef Q_OS_WIN
187 signal(SIGUSR1, [](int) { ConnectionManager->RestartConnection(); });
188 signal(SIGUSR2, [](int) { ConnectionManager->StopConnection(); });
189#endif
190 return { app.RunQv2ray(), std::nullopt };
191}

193int main(int argc, char *argv[])
194{
195 // Register signal handlers.
196 signal(SIGABRT, signalHandler);
197 signal(SIGSEGV, signalHandler);
198 signal(SIGTERM, signalHandler);
199#ifndef Q_OS_WIN
200 signal(SIGHUP, signalHandler);
201 signal(SIGKILL, signalHandler);
202#endif
203 //
204 // This line must be called before any other ones, since we are using these
205 // values to identify instances.
206 QApplication::setApplicationVersion(QV2RAY_VERSION_STRING);
207 //
208#ifdef QT_DEBUG
209 QApplication::setApplicationName("qv2ray_debug");
210 QApplication::setApplicationDisplayName("Qv2ray - " + QObject::tr("Debug version"));
211#else
212 QApplication::setApplicationName("qv2ray");
213 QApplication::setApplicationDisplayName("Qv2ray");
214#endif
215 //
216 // parse the command line before starting as a Qt application
217 switch (Qv2rayApplication::PreInitialize(argc, argv))
218 {
219 case PRE_INIT_RESULT_QUIT: return QVEXIT_NORMAL;
220 case PRE_INIT_RESULT_CONTINUE: break;
221 case PRE_INIT_RESULT_ERROR:
222 {
223 QApplication errorApplication{ argc, argv };
224 QvMessageBoxWarn(nullptr, "Cannot Start Qv2ray!", "Early initialization failed!");
225 return QVEXIT_PRE_INITIALIZE_FAIL;
226 }
227 default: Q_UNREACHABLE();
228 }
229 const auto &[rcode, str] = RunQv2rayApplicationScoped(argc, argv);
230 if (rcode == QVEXIT_NEW_VERSION)
231 {
232 LOG(MODULE_INIT, "Starting new version of Qv2ray: " + Qv2rayProcessArgument._qvNewVersionPath)
233 QProcess::startDetached(Qv2rayProcessArgument._qvNewVersionPath, {});
234 }
235 else if (str)
236 {
237 QApplication errorApplication{ argc, argv };
238 QvMessageBoxWarn(nullptr, errorApplication.tr("Cannot start Qv2ray"),
239 ACCESS_OPTIONAL_VALUE(str) + //
240 NEWLINE + //
241 NEWLINE + //
242 errorApplication.tr("Qv2ray will now exit!"));
243 }
244
245 return rcode;
246}
