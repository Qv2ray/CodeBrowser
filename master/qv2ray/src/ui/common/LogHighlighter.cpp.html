122 void SyntaxHighlighter::highlightBlock(const QString &text)
123 {
124 for (const HighlightingRule &rule : qAsConst(highlightingRules))
125 {
126 QRegularExpressionMatchIterator matchIterator = rule.pattern.globalMatch(text);
127
128 while (matchIterator.hasNext())
129 {
130 QRegularExpressionMatch match = matchIterator.next();
131 setFormat(match.capturedStart(), match.capturedLength(), rule.format);
132 }
133 }
134
135 setCurrentBlockState(0);
136 }
