13 QString DecodeQRCode(const QImage &img)
14 {
15 DecodeHints hints;
16 hints.setTryHarder(true);
17 hints.setTryRotate(true);
18 //
19 auto ImgFmtFromQImg = [](const QImage &img) {
20 switch (img.format())
21 {
22 case QImage::Format_ARGB32:
23 case QImage::Format_RGB32:
24#if Q_BYTE_ORDER == Q_LITTLE_ENDIAN
25 return ImageFormat::BGRX;
26#else
27 return ImageFormat::XRGB;
28#endif
29 case QImage::Format_RGB888: return ImageFormat::RGB;
30 case QImage::Format_RGBX8888:
31 case QImage::Format_RGBA8888: return ImageFormat::RGBX;
32 case QImage::Format_Grayscale8: return ImageFormat::Lum;
33 default: return ImageFormat::None;
34 }
35 };
36
37 auto exec = [&](const QImage &img) {
38 return QString::fromStdWString(ZXing::ReadBarcode({ img.bits(), img.width(), img.height(), ImgFmtFromQImg(img) }, hints).text());
39 };
40
41 return ImgFmtFromQImg(img) == ImageFormat::None ? exec(img.convertToFormat(QImage::Format_RGBX8888)) : exec(img);
42 }

44 QImage EncodeQRCode(const QString &content, const QSize &size)
45 {
46 int eccLevel = 1;
47 try
48 {
49 MultiFormatWriter writer(ZXing::BarcodeFormat::QR_CODE);
50 writer.setMargin(1);
51 writer.setEccLevel(eccLevel);
52
53 const auto bitmap = writer.encode(content.toStdWString(), size.width(), size.height());
54 const auto BM = bitmap.toByteMatrix();
55 //
56 const auto black = qRgba(0, 0, 0, 255);
57 const auto white = qRgba(255, 255, 255, 255);
58 //
59 auto image = QImage(bitmap.width(), bitmap.width(), QImage::Format_ARGB32);
60 image.fill(white);
61 //
62 for (auto i = 0; i < bitmap.width(); ++i)
63 {
64 for (auto j = 0; j < bitmap.height(); ++j)
65 {
66 // 0 means black?
67 if (bitmap.get(j, i))
68 {
69 image.setPixel(i, j, black);
70 }
71 }
72 }
73 return image;
74 }
75 catch (const std::exception &e)
76 {
77 return {};
78 }
