86 void vCoreConfigJsonHighlighter::SetRule(const QString &kind, const QString &pattern, QTextCharFormat format)
87 {
88 JsonHighlightingRule rule;
89 rule.name = kind;
90 rule.pattern = QRegularExpression(pattern);
91 rule.format = format;
92 highlightingRules.append(rule);
93 }

95 void vCoreConfigJsonHighlighter::highlightBlock(const QString &text)
96 {
97 for (const auto &rule : std::as_const(highlightingRules))
98 {
99 QRegularExpressionMatchIterator matchIterator = rule.pattern.globalMatch(text);
100
101 while (matchIterator.hasNext())
102 {
103 QRegularExpressionMatch match = matchIterator.next();
104 setFormat(match.capturedStart(), match.capturedLength(), rule.format);
105 }
106 }
107
108 setCurrentBlockState(0);
109 }
