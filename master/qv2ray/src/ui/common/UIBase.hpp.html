13 inline QPixmap ApplyEffectToImage(QPixmap src, QGraphicsEffect *effect, int extent = 0)
14 {
15 if (src.isNull() || !effect)
16 return src;
17 QGraphicsScene scene;
18 QGraphicsPixmapItem item;
19 item.setPixmap(src);
20 item.setGraphicsEffect(effect);
21 scene.addItem(&item);
22 QImage res(src.size() + QSize(extent * 2, extent * 2), QImage::Format_ARGB32);
23 res.fill(Qt::transparent);
24 QPainter ptr(&res);
25 scene.render(&ptr, QRectF(), QRectF(-extent, -extent, src.width() + extent * 2, src.height() + extent * 2));
26 return QPixmap::fromImage(res);
27 }

29 inline QPixmap BlurImage(const QPixmap &pixmap, const double rad)
30 {
31 auto pBlur = new QGraphicsBlurEffect();
32 pBlur->setBlurRadius(rad);
33 return ApplyEffectToImage(pixmap, pBlur, 0);
34 }

36 inline QPixmap ColorizeImage(const QPixmap &pixmap, const QColor &color, const qreal factor)
37 {
38 auto pColor = new QGraphicsColorizeEffect();
39 pColor->setColor(color);
40 pColor->setStrength(factor);
41 return ApplyEffectToImage(pixmap, pColor, 0);
