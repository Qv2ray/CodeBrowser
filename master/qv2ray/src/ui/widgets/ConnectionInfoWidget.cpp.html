40ConnectionInfoWidget::ConnectionInfoWidget(QWidget *parent) : QWidget(parent)
41{

61void ConnectionInfoWidget::ShowDetails(const ConnectionGroupPair &_identifier)
62{
63 this->groupId = _identifier.groupId;
64 this->connectionId = _identifier.connectionId;
65 bool isConnection = connectionId != NullConnectionId;
66 //
67 editBtn->setEnabled(isConnection);
68 editJsonBtn->setEnabled(isConnection);
69 connectBtn->setEnabled(isConnection);
70 stackedWidget->setCurrentIndex(isConnection ? INDEX_CONNECTION : INDEX_GROUP);
71 if (isConnection)
72 {
73 auto shareLink = ConvertConfigToString(_identifier);
74 //
75 shareLinkTxt->setText(shareLink);
76 protocolLabel->setText(GetConnectionProtocolString(connectionId));
77 //
78 groupLabel->setText(GetDisplayName(groupId, 175));
79 auto [protocol, host, port] = GetConnectionInfo(connectionId);
80 Q_UNUSED(protocol)
81 addressLabel->setText(host);
82 portLabel->setNum(port);
83 //
84 shareLinkTxt->setCursorPosition(0);
85 auto isDarkTheme = GlobalConfig.uiConfig.useDarkTheme;
86 qrPixmap = QPixmap::fromImage(EncodeQRCode(shareLink, qrLabel->size() * devicePixelRatio()));
87 //
88 qrPixmapBlured = BlurImage(ColorizeImage(qrPixmap, isDarkTheme ? QColor(Qt::black) : QColor(Qt::white), 0.7), 35);
89 //
90 isRealPixmapShown = false;
91 qrLabel->setPixmap(IsComplexConfig(connectionId) ? QPixmap(":/assets/icons/qv2ray.png") : qrPixmapBlured);
92 qrLabel->setScaledContents(true);
93 //
94 connectBtn->setIcon(KernelInstance->CurrentConnection().connectionId == connectionId ? QICON_R("stop.png") : QICON_R("connect.png"));
95 }
96 else
97 {
98 connectBtn->setIcon(QICON_R("connect.png"));
99 groupNameLabel->setText(GetDisplayName(groupId));
100 QStringList shareLinks;
101 for (const auto &connection : ConnectionManager->Connections(groupId))
102 {
103 shareLinks << ConvertConfigToString({ connection, groupId }, !GlobalConfig.uiConfig.useOldShareLinkFormat);
104 }
105 //
106 auto complexCount = shareLinks.removeAll(QV2RAY_SERIALIZATION_COMPLEX_CONFIG_PLACEHOLDER);
107 complexCount += shareLinks.removeAll("");
108 if (complexCount > 0)
109 {
110 shareLinks << "# " + tr("(Ignored %n complex config(s))", "", complexCount);
111 }
112 //
113 groupShareTxt->setPlainText(shareLinks.join(NEWLINE));
114 const auto &groupMetaData = ConnectionManager->GetGroupMetaObject(groupId);
115 groupSubsLinkTxt->setText(groupMetaData.isSubscription ? groupMetaData.subscriptionOption.address : tr("Not a subscription"));
116 }
117}

181bool ConnectionInfoWidget::eventFilter(QObject *object, QEvent *event)
182{
183 if (shareLinkTxt->underMouse() && event->type() == QEvent::MouseButtonRelease)
184 {
185 if (!shareLinkTxt->hasSelectedText())
186 shareLinkTxt->selectAll();
187 }
188 else if (groupSubsLinkTxt->underMouse() && event->type() == QEvent::MouseButtonRelease)
189 {
190 if (!groupSubsLinkTxt->hasSelectedText())
191 groupSubsLinkTxt->selectAll();
192 }
193 else if (qrLabel->underMouse() && event->type() == QEvent::MouseButtonRelease)
194 {
195 qrLabel->setPixmap(IsComplexConfig(connectionId) ? QPixmap(":/assets/icons/qv2ray.png") :
196 (isRealPixmapShown ? qrPixmapBlured : qrPixmap));
197 isRealPixmapShown = !isRealPixmapShown;
198 }
199
200 return QWidget::eventFilter(object, event);
201}
