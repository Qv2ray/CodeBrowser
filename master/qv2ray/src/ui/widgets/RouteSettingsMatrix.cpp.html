73QvConfig_Route RouteSettingsMatrixWidget::GetRouteConfig() const
74{
75 QvConfig_Route conf;
76 conf.domainStrategy = this->domainStrategyCombo->currentText();
77 conf.domains.block = SplitLines(blockDomainTxt->toPlainText().replace(" ", ""));
78 conf.domains.direct = SplitLines(directDomainTxt->toPlainText().replace(" ", ""));
79 conf.domains.proxy = SplitLines(proxyDomainTxt->toPlainText().replace(" ", ""));
80 //
81 conf.ips.block = SplitLines(blockIPTxt->toPlainText().replace(" ", ""));
82 conf.ips.direct = SplitLines(directIPTxt->toPlainText().replace(" ", ""));
83 conf.ips.proxy = SplitLines(proxyIPTxt->toPlainText().replace(" ", ""));
84 return conf;
85}

91/**
92 * @brief RouteSettingsMatrixWidget::on_importSchemeBtn_clicked
93 * @author DuckSoft <realducksoft@gmail.com>
94 * @todo add some debug output
95 */
96void RouteSettingsMatrixWidget::on_importSchemeBtn_clicked()
97{
98 try
99 {
100 // open up the file dialog and choose a file.
101 auto filePath = this->openFileDialog();
102 if (!filePath)
103 return;
104
105 // read the file and parse back to struct.
106 // if error occurred on parsing, an exception will be thrown.
107 auto content = StringFromFile(ACCESS_OPTIONAL_VALUE(filePath));
108 auto scheme = Qv2rayRouteScheme::fromJson(JsonFromString(content));
109
110 // show the information of this scheme to user,
111 // and ask user if he/she wants to import and apply this.
112 auto strPrompt = tr("Import scheme '%1' made by '%2'? \r\n Description: %3").arg(scheme.name, scheme.author, scheme.description);
113 auto decision = QvMessageBoxAsk(this, tr("Importing Scheme"), strPrompt);
114
115 // if user don't want to import, just leave.
116 if (decision != QMessageBox::Yes)
117 return;
118
119 // write the scheme onto the window
120 this->SetRouteConfig(static_cast<QvConfig_Route>(scheme));
121
122 // done
123 LOG(MODULE_SETTINGS, "Imported route config: " + scheme.name + " by: " + scheme.author)
124 }
125 catch (std::exception &e)
126 {
127 LOG(MODULE_UI, "Exception: " + QString(e.what()))
128 // TODO: Give some error as Notification
129 }
130}

132/**
133 * @brief RouteSettingsMatrixWidget::on_exportSchemeBtn_clicked
134 * @author DuckSoft <realducksoft@gmail.com>
135 */
136void RouteSettingsMatrixWidget::on_exportSchemeBtn_clicked()
137{
138 try
139 {
140 // parse the config back from the window components
141 auto config = this->GetRouteConfig();
142
143 // init some constants
144 const auto dialogTitle = tr("Exporting Scheme");
145
146 // scheme name?
147 bool ok = false;
148 auto schemeName = QInputDialog::getText(this, dialogTitle, tr("Scheme name:"), QLineEdit::Normal, tr("Unnamed Scheme"), &ok);
149 if (!ok)
150 return;
151
152 // scheme author?
153 auto schemeAuthor = QInputDialog::getText(this, dialogTitle, tr("Author:"), QLineEdit::Normal, "Anonymous <mystery@example.com>", &ok);
154 if (!ok)
155 return;
156
157 // scheme description?
158 auto schemeDescription =
159 QInputDialog::getText(this, dialogTitle, tr("Description:"), QLineEdit::Normal, tr("The author is too lazy to leave a comment"));
160 if (!ok)
161 return;
162
163 // where to save?
164 auto savePath = this->saveFileDialog();
165 if (!savePath)
166 return;
167
168 // construct the data structure
169 Qv2rayRouteScheme scheme;
170 scheme.name = schemeName;
171 scheme.author = schemeAuthor;
172 scheme.description = schemeDescription;
173 scheme.domainStrategy = config.domainStrategy;
174 scheme.ips = config.ips;
175 scheme.domains = config.domains;
176
177 // serialize and write out
178 auto content = JsonToString(scheme.toJson());
179 StringToFile(content, ACCESS_OPTIONAL_VALUE(savePath));
180
181 // done
182 // TODO: Give some success as Notification
183 QvMessageBoxInfo(this, dialogTitle, tr("Your route scheme has been successfully exported!"));
184 }
185 catch (...)
186 {
187 // TODO: Give some error as Notification
188 }
189}

196std::optional<QString> RouteSettingsMatrixWidget::saveFileDialog()
197{
198 QFileDialog dialog;
199 dialog.setFileMode(QFileDialog::AnyFile);
200 dialog.setOption(QFileDialog::Option::DontConfirmOverwrite, false);
201 dialog.setNameFilter(tr("QvRoute Schemes(*.json)"));
202 dialog.setAcceptMode(QFileDialog::AcceptMode::AcceptSave);
203 if (!dialog.exec() || dialog.selectedFiles().length() != 1)
204 {
205 return std::nullopt;
206 }
207 return dialog.selectedFiles().first();
208}
