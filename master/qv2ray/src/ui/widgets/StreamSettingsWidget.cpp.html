28void StreamSettingsWidget::SetStreamObject(const StreamSettingsObject &sso)
29{
30 stream = sso;
31 //
32 transportCombo->setCurrentText(stream.network);
33 // TLS
34 tlsCB->setChecked(stream.security == "tls");
35 serverNameTxt->setText(stream.tlsSettings.serverName);
36 allowInsecureCB->setChecked(stream.tlsSettings.allowInsecure);
37 allowInsecureCiphersCB->setChecked(stream.tlsSettings.allowInsecureCiphers);
38 disableSessionResumptionCB->setChecked(stream.tlsSettings.disableSessionResumption);
39 alpnTxt->setPlainText(stream.tlsSettings.alpn.join(NEWLINE));
40 // TCP
41 tcpHeaderTypeCB->setCurrentText(stream.tcpSettings.header.type);
42 tcpRequestTxt->setPlainText(JsonToString(stream.tcpSettings.header.request.toJson()));
43 tcpRespTxt->setPlainText(JsonToString(stream.tcpSettings.header.response.toJson()));
44 // HTTP
45 httpHostTxt->setPlainText(stream.httpSettings.host.join(NEWLINE));
46 httpPathTxt->setText(stream.httpSettings.path);
47 // WS
48 wsPathTxt->setText(stream.wsSettings.path);
49 QString wsHeaders;
50 for (auto i = 0; i < stream.wsSettings.headers.count(); i++)
51 {
52 wsHeaders = wsHeaders % stream.wsSettings.headers.keys().at(i) % "|" % stream.wsSettings.headers.values().at(i) % NEWLINE;
53 }
54
55 wsHeadersTxt->setPlainText(wsHeaders);
56 // mKCP
57 kcpMTU->setValue(stream.kcpSettings.mtu);
58 kcpTTI->setValue(stream.kcpSettings.tti);
59 kcpHeaderType->setCurrentText(stream.kcpSettings.header.type);
60 kcpCongestionCB->setChecked(stream.kcpSettings.congestion);
61 kcpReadBufferSB->setValue(stream.kcpSettings.readBufferSize);
62 kcpUploadCapacSB->setValue(stream.kcpSettings.uplinkCapacity);
63 kcpDownCapacitySB->setValue(stream.kcpSettings.downlinkCapacity);
64 kcpWriteBufferSB->setValue(stream.kcpSettings.writeBufferSize);
65 kcpSeedTxt->setText(stream.kcpSettings.seed);
66 // DS
67 dsPathTxt->setText(stream.dsSettings.path);
68 // QUIC
69 quicKeyTxt->setText(stream.quicSettings.key);
70 quicSecurityCB->setCurrentText(stream.quicSettings.security);
71 quicHeaderTypeCB->setCurrentText(stream.quicSettings.header.type);
72 // SOCKOPT
73 tProxyCB->setCurrentText(stream.sockopt.tproxy);
74 tcpFastOpenCB->setChecked(stream.sockopt.tcpFastOpen);
75 soMarkSpinBox->setValue(stream.sockopt.mark);
76}

88void StreamSettingsWidget::on_httpHostTxt_textChanged()
89{
90 try
91 {
92 QStringList hosts = httpHostTxt->toPlainText().replace("\r", "").split("\n");
93 stream.httpSettings.host.clear();
94
95 for (auto host : hosts)
96 {
97 if (!host.trimmed().isEmpty())
98 {
99 stream.httpSettings.host.push_back(host.trimmed());
100 }
101 }
102
103 BLACK(httpHostTxt)
104 }
105 catch (...)
106 {
107 RED(httpHostTxt)
108 }
109}

111void StreamSettingsWidget::on_wsHeadersTxt_textChanged()
112{
113 try
114 {
115 QStringList headers = SplitLines(wsHeadersTxt->toPlainText());
116 stream.wsSettings.headers.clear();
117
118 for (auto header : headers)
119 {
120 if (header.isEmpty())
121 continue;
122
123 auto index = header.indexOf("|");
124
125 if (index < 0)
126 throw "fast fail to set RED color";
127
128 auto key = header.left(index);
129 auto value = header.right(header.length() - index - 1);
130 stream.wsSettings.headers[key] = value;
131 }
132
133 BLACK(wsHeadersTxt)
134 }
135 catch (...)
136 {
137 RED(wsHeadersTxt)
138 }
139}
