27DnsSettingsWidget::DnsSettingsWidget(QWidget *parent) : QWidget(parent)
28{
29 setupUi(this);
30 QvMessageBusConnect(DnsSettingsWidget);
31 //
32 auto sourceStringsDomain = ReadGeoSiteFromFile(GlobalConfig.kernelConfig.AssetsPath() + "/geosite.dat");
33 auto sourceStringsIP = ReadGeoSiteFromFile(GlobalConfig.kernelConfig.AssetsPath() + "/geoip.dat");
34 //
35 domainListTxt = new AutoCompleteTextEdit("geosite", sourceStringsDomain, this);
36 ipListTxt = new AutoCompleteTextEdit("geoip", sourceStringsIP, this);
37 connect(domainListTxt, &AutoCompleteTextEdit::textChanged,
38 [&]() { this->dns.servers[currentServerIndex].domains = SplitLines(domainListTxt->toPlainText()); });
39 connect(ipListTxt, &AutoCompleteTextEdit::textChanged,
40 [&]() { this->dns.servers[currentServerIndex].expectIPs = SplitLines(ipListTxt->toPlainText()); });
41
42 //
43 domainsLayout->addWidget(domainListTxt);
44 expectedIPsLayout->addWidget(ipListTxt);
45 detailsSettingsGB->setCheckable(true);
46 detailsSettingsGB->setChecked(false);
47 UPDATEUI
48}

61void DnsSettingsWidget::SetDNSObject(const DNSObject &_dns)
62{
63 this->dns = _dns;
64 //
65 dnsClientIPTxt->setText(dns.clientIp);
66 dnsTagTxt->setText(dns.tag);
67 serversListbox->clear();
68 for (const auto &server : dns.servers)
69 {
70 serversListbox->addItem(server.address);
71 }
72 if (serversListbox->count() > 0)
73 {
74 serversListbox->setCurrentRow(0);
75 ShowCurrentDnsServerDetails();
76 }
77 staticResolvedDomainsTable->clearContents();
78 for (const auto &[host, ip] : dns.hosts.toStdMap())
79 {
80 const auto rowId = staticResolvedDomainsTable->rowCount();
81 staticResolvedDomainsTable->insertRow(rowId);
82 staticResolvedDomainsTable->setItem(rowId, 0, new QTableWidgetItem(host));
83 staticResolvedDomainsTable->setItem(rowId, 1, new QTableWidgetItem(ip));
84 }
85 staticResolvedDomainsTable->resizeColumnsToContents();
86 UPDATEUI
87}

89bool DnsSettingsWidget::CheckIsValidDNS() const
90{
91 if (!dns.clientIp.isEmpty() && !IsValidIPAddress(dns.clientIp))
92 return false;
93 for (const auto &server : dns.servers)
94 {
95 if (!IsValidDNSServer(server.address))
96 return false;
97 }
98 return true;
99}

116void DnsSettingsWidget::ShowCurrentDnsServerDetails()
117{
118 serverAddressTxt->setText(dns.servers[currentServerIndex].address);
119 //
120 domainListTxt->setPlainText(dns.servers[currentServerIndex].domains.join(NEWLINE));
121 ipListTxt->setPlainText(dns.servers[currentServerIndex].expectIPs.join(NEWLINE));
122 //
123 serverPortSB->setValue(dns.servers[currentServerIndex].port);
124 detailsSettingsGB->setChecked(dns.servers[currentServerIndex].QV2RAY_DNS_IS_COMPLEX_DNS);
125 //
126 if (serverAddressTxt->text().isEmpty() || IsValidDNSServer(serverAddressTxt->text()))
127 {
128 BLACK(serverAddressTxt)
129 }
130 else
131 {
132 RED(serverAddressTxt)
133 }
134 ProcessDnsPortEnabledState();
135}

137DNSObject DnsSettingsWidget::GetDNSObject()
138{
139 dns.hosts.clear();
140 for (auto i = 0; i < staticResolvedDomainsTable->rowCount(); i++)
141 {
142 const auto &item1 = staticResolvedDomainsTable->item(i, 0);
143 const auto &item2 = staticResolvedDomainsTable->item(i, 1);
144 if (item1 && item2)
145 dns.hosts[item1->text()] = item2->text();
146 }
147 return dns;
148}

171void DnsSettingsWidget::on_removeServerBtn_clicked()
172{
173 dns.servers.removeAt(currentServerIndex);
174 // Block the signals
175 serversListbox->blockSignals(true);
176 auto item = serversListbox->item(currentServerIndex);
177 serversListbox->removeItemWidget(item);
178 delete item;
179 serversListbox->blockSignals(false);
180 UPDATEUI
181
182 if (serversListbox->count() > 0)
183 {
184 if (currentServerIndex < 0)
185 serversListbox->setCurrentRow(0);
186 ShowCurrentDnsServerDetails();
187 }
