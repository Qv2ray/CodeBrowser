13InboundEditor::InboundEditor(INBOUND root, QWidget *parent) : QDialog(parent), original(root)
14{
15 QvMessageBusConnect(InboundEditor);
16 setupUi(this);
17 this->root = root;
18 auto inboundType = root["protocol"].toString();
19 allocate = root["allocate"].toObject();
20 sniffing = root["sniffing"].toObject();
21
22 if (inboundType == "http")
23 {
24 httpSettings = root["settings"].toObject();
25 }
26 else if (inboundType == "socks")
27 {
28 socksSettings = root["settings"].toObject();
29 }
30 else if (inboundType == "dokodemo-door")
31 {
32 dokoSettings = root["settings"].toObject();
33 dokotproxy = QJsonIO ::GetValue(root, "streamSettings", "sockopt", "tproxy").toString();
34 if (dokotproxy.isEmpty())
35 dokotproxy = "off";
36 }
37 else if (inboundType == "mtproto")
38 {
39 mtSettings = root["settings"].toObject();
40 }
41 else
42 {
43 if (!root["protocol"].toString().isEmpty())
44 {
45 LOG(MODULE_UI, "Unsupported inbound type: " + inboundType)
46 QvMessageBoxWarn(this, tr("Inbound type not supported"),
47 tr("The inbound type is not supported by Qv2ray (yet). Please use JsonEditor to change the settings") + "\r\n" +
48 tr("Inbound: ") + inboundType);
49 }
50 else
51 {
52 LOG(MODULE_UI, "Creating new inbound config")
53 root["protocol"] = inboundType = "http";
54 }
55 }
56
57 LoadUIData();
58}

77INBOUND InboundEditor::GenerateNewRoot()
78{
79 INBOUND _newRoot = root;
80 auto inboundType = root["protocol"].toString();
81
82 if (inboundType.isNull() || inboundType.isEmpty())
83 {
84 inboundType = "http";
85 }
86
87 if (inboundType == "http")
88 {
89 // Remove useless, misleading 'accounts' array.
90 if (httpAccountListBox->count() == 0)
91 {
92 httpSettings.remove("accounts");
93 }
94
95 _newRoot["settings"] = httpSettings;
96 }
97 else if (inboundType == "socks")
98 {
99 // See above
100 if (socksAccountListBox->count() == 0)
101 {
102 socksSettings.remove("accounts");
103 }
104
105 _newRoot["settings"] = socksSettings;
106 }
107 else if (inboundType == "dokodemo-door")
108 {
109 _newRoot["settings"] = dokoSettings;
110 QJsonIO::SetValue(_newRoot, dokotproxy, "streamSettings", "sockopt", "tproxy");
111 }
112 else if (inboundType == "mtproto")
113 {
114 _newRoot["settings"] = mtSettings;
115 }
116
117 _newRoot["protocol"] = inboundType;
118 _newRoot["sniffing"] = sniffing;
119 _newRoot["allocate"] = allocate;
120 return _newRoot;
121}

123void InboundEditor::LoadUIData()
124{
125 isLoading = true;
126 auto x = allocate["strategy"].toString();
127 allocate["strategy"] = x.isEmpty() ? "always" : x;
128 strategyCombo->setCurrentText(x);
129 //
130 refreshNumberBox->setValue(allocate["refresh"].toInt());
131 concurrencyNumberBox->setValue(allocate["concurrency"].toInt());
132 enableSniffingCB->setChecked(sniffing["enabled"].toBool());
133 //
134 destOverrideList->setEnabled(sniffing["enabled"].toBool());
135
136 for (auto item : sniffing["destOverride"].toArray())
137 {
138 if (item.toString().toLower() == "http")
139 destOverrideList->item(0)->setCheckState(Qt::Checked);
140
141 if (item.toString().toLower() == "tls")
142 destOverrideList->item(1)->setCheckState(Qt::Checked);
143 }
144
145 inboundTagTxt->setText(root["tag"].toString());
146 inboundHostTxt->setText(root["listen"].toString());
147 inboundPortTxt->setText(root["port"].toVariant().toString());
148 inboundProtocolCombo->setCurrentText(root["protocol"].toString());
149 // HTTP
150 httpTimeoutSpinBox->setValue(httpSettings["timeout"].toInt());
151 httpTransparentCB->setChecked(httpSettings["allowTransparent"].toBool());
152 httpUserLevelSB->setValue(httpSettings["userLevel"].toInt());
153 httpAccountListBox->clear();
154
155 for (auto user : httpSettings["accounts"].toArray())
156 {
157 httpAccountListBox->addItem(user.toObject()["user"].toString() + ":" + user.toObject()["pass"].toString());
158 }
159
160 // SOCKS
161 socksAuthCombo->setCurrentText(socksSettings["auth"].toString());
162 socksUDPCB->setChecked(socksSettings["udp"].toBool());
163 socksUDPIPAddrTxt->setText(socksSettings["ip"].toString());
164 socksUserLevelSB->setValue(socksSettings["userLevel"].toInt());
165
166 for (auto user : socksSettings["accounts"].toArray())
167 {
168 socksAccountListBox->addItem(user.toObject()["user"].toString() + ":" + user.toObject()["pass"].toString());
169 }
170
171 // Dokodemo-Door
172 dokoFollowRedirectCB->setChecked(dokoSettings["followRedirect"].toBool());
173 dokoIPAddrTxt->setText(dokoSettings["address"].toString());
174 dokoPortSB->setValue(dokoSettings["port"].toInt());
175 dokoTimeoutSB->setValue(dokoSettings["timeout"].toInt());
176 dokoUserLevelSB->setValue(dokoSettings["userLevel"].toInt());
177 dokoTCPCB->setChecked(dokoSettings["network"].toString().contains("tcp"));
178 dokoUDPCB->setChecked(dokoSettings["network"].toString().contains("udp"));
179 dokotproxyCombo->setCurrentText(dokotproxy);
180 // MTProto
181 mtEMailTxt->setText(mtSettings["users"].toArray().first().toObject()["email"].toString());
182 mtUserLevelSB->setValue(mtSettings["users"].toArray().first().toObject()["level"].toInt());
183 mtSecretTxt->setText(mtSettings["users"].toArray().first().toObject()["secret"].toString());
184 isLoading = false;
185}

226void InboundEditor::on_httpRemoveUserBtn_clicked()
227{
228 CHECKLOADING
229
230 if (httpAccountListBox->currentRow() != -1)
231 {
232 auto item = httpAccountListBox->currentItem();
233 auto list = httpSettings["accounts"].toArray();
234
235 for (int i = 0; i < list.count(); i++)
236 {
237 auto user = list[i].toObject();
238 auto entry = user["user"].toString() + ":" + user["pass"].toString();
239
240 if (entry == item->text().trimmed())
241 {
242 list.removeAt(i);
243 httpSettings["accounts"] = list;
244 LOG(MODULE_UI, "Removed http inbound user " + entry)
245 httpAccountListBox->takeItem(httpAccountListBox->currentRow());
246 }
247 }
248
249 // QvMessageBox(this, tr("Removing a user"), tr("No user has been
250 // removed. Why?"));
251 }
252 else
253 {
254 QvMessageBoxWarn(this, tr("Removing a user"), tr("You haven't selected a user yet."));
255 }
256}

258void InboundEditor::on_httpAddUserBtn_clicked()
259{
260 CHECKLOADING
261 auto user = httpAddUserTxt->text();
262 auto pass = httpAddPasswordTxt->text();
263 //
264 auto list = httpSettings["accounts"].toArray();
265
266 for (int i = 0; i < list.count(); i++)
267 {
268 auto _user = list[i].toObject();
269
270 if (_user["user"].toString() == user)
271 {
272 QvMessageBoxWarn(this, tr("Add a user"), tr("This user exists already."));
273 return;
274 }
275 }
276
277 httpAddUserTxt->clear();
278 httpAddPasswordTxt->clear();
279 QJsonObject entry;
280 entry["user"] = user;
281 entry["pass"] = pass;
282 list.append(entry);
283 httpAccountListBox->addItem(user + ":" + pass);
284 httpSettings["accounts"] = list;
285}
