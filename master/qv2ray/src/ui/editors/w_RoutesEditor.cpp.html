149void RouteEditor::onNodeClicked(Node &n)
150{
151 LOADINGCHECK
152
153 if (isExiting)
154 return;
155
156 auto isOut = outboundNodes.values().contains(n.id());
157 auto isIn = inboundNodes.values().contains(n.id());
158 auto isRule = ruleNodes.values().contains(n.id());
159
160 if (isRule)
161 {
162 // It's a rule object
163 currentRuleTag = GetFirstNodeData(n.id(), RuleNode)->GetRuleTag();
164 DEBUG(MODULE_GRAPH, "Selecting rule: " + currentRuleTag)
165 ShowCurrentRuleDetail();
166 toolBox->setCurrentIndex(1);
167 }
168 else if (isOut || isIn)
169 {
170 // It's an inbound or an outbound.
171 QString alias;
172 QString host;
173 QString port;
174 QString protocol;
175
176 if (isOut)
177 {
178 alias = GetFirstNodeData(n.id(), OutboundNode)->GetOutbound();
179 QJsonObject _root = outbounds[alias].raw();
180 int _port;
181 GetOutboundInfo(OUTBOUND(_root), &host, &_port, &protocol);
182 port = QString::number(_port);
183 }
184 else
185 {
186 alias = GetFirstNodeData(n.id(), InboundNode)->GetInbound();
187 QJsonObject _root = inbounds[alias].raw();
188 host = _root["listen"].toString();
189 protocol = _root["protocol"].toString();
190 // Port could be a string, or an integer.
191 port = _root["port"].toVariant().toString();
192 }
193
194 tagLabel->setText(alias);
195 protocolLabel->setText(protocol);
196 portLabel->setText(port);
197 hostLabel->setText(host);
198 }
199 else
200 {
201 LOG(MODULE_GRAPH, "Selected an unknown node, RARE.")
202 }

205void RouteEditor::onConnectionCreated(QtNodes::Connection const &c)
206{
207 LOADINGCHECK
208
209 if (isExiting)
210 return;
211
212 // Connection Established
213 auto const &sourceNode = c.getNode(PortType::Out);
214 auto const &targetNode = c.getNode(PortType::In);
215
216 if (inboundNodes.values().contains(sourceNode->id()) && ruleNodes.values().contains(targetNode->id()))
217 {
218 // It's a inbound-rule connection
219 onNodeClicked(*sourceNode);
220 onNodeClicked(*targetNode);
221 LOG(MODULE_GRAPH, "Inbound-rule new connection.")
222 // Get all connected inbounds to this rule node.
223 // QStringList has an helper to let us remove duplicates, see below.
224 QSet<QString> _inbounds;
225 //
226 // Workaround for removing a connection within the loop.
227 QList<std::shared_ptr<QtNodes::Connection>> connectionsTobeRemoved;
228 for (auto &&[_, conn] : nodeScene->connections())
229 {
230 const auto &inNode = conn->getNode(PortType::In);
231 const auto &outNode = conn->getNode(PortType::Out);
232 // If a connection is not current Id, but with same IN/OUT nodes.
233 // It is a "duplicated" connection.
234 if (inNode->id() == targetNode->id() && outNode->id() == sourceNode->id() && conn->id() != c.id())
235 {
236 connectionsTobeRemoved << (conn);
237 }
238 // Append all inbounds
239 if (inNode->id() == targetNode->id())
240 {
241 _inbounds.insert(GetFirstNodeData(outNode->id(), InboundNode)->GetInbound());
242 }
243 }
244 for (const auto &connRemoved : connectionsTobeRemoved)
245 {
246 nodeScene->deleteConnection(*connRemoved);
247 }
248
249 CurrentRule.inboundTag = _inbounds.values();
250 }
251 else if (ruleNodes.values().contains(sourceNode->id()) && outboundNodes.values().contains(targetNode->id()))
252 {
253 // It's a rule-outbound connection
254 onNodeClicked(*sourceNode);
255 onNodeClicked(*targetNode);
256 CurrentRule.outboundTag = GetFirstNodeData(targetNode->id(), OutboundNode)->GetOutbound();
257 // Connecting to an outbound will disable the balancer feature.
258 CurrentRule.QV2RAY_RULE_USE_BALANCER = false;
259 // Update balancer settings.
260 ShowCurrentRuleDetail();
261 LOG(MODULE_GRAPH, "Updated outbound: " + CurrentRule.outboundTag)
262 }
263 else
264 {
265 // It's an impossible connection
266 LOG(MODULE_GRAPH, "Unrecognized connection, RARE.")
267 }

270void RouteEditor::onConnectionDeleted(QtNodes::Connection const &c)
271{
272 LOADINGCHECK
273
274 if (isExiting)
275 return;
276
277 // Connection Deleted
278 const auto &source = c.getNode(PortType::Out);
279 const auto &target = c.getNode(PortType::In);
280
281 if (inboundNodes.values().contains(source->id()) && ruleNodes.values().contains(target->id()))
282 {
283 // It's a inbound-rule connection
284 onNodeClicked(*source);
285 onNodeClicked(*target);
286 currentRuleTag = GetFirstNodeData(target->id(), RuleNode)->GetRuleTag();
287 auto _inboundTag = GetFirstNodeData(source->id(), InboundNode)->GetInbound();
288 LOG(MODULE_UI, "Removing inbound: " + _inboundTag + " from rule: " + currentRuleTag)
289 CurrentRule.inboundTag.removeAll(_inboundTag);
290 }
291 else if (ruleNodes.values().contains(source->id()) && outboundNodes.values().contains(target->id()))
292 {
293 // It's a rule-outbound connection
294 onNodeClicked(*source);
295 onNodeClicked(*target);
296 currentRuleTag = GetFirstNodeData(source->id(), RuleNode)->GetRuleTag();
297 auto _outboundTag = GetFirstNodeData(target->id(), OutboundNode)->GetOutbound();
298
299 if (!CurrentRule.QV2RAY_RULE_USE_BALANCER && CurrentRule.outboundTag == _outboundTag)
300 {
301 CurrentRule.outboundTag.clear();
302 }
303
304 LOG(MODULE_GRAPH, "Removing an outbound: " + _outboundTag)
305 }
306 else
307 {
308 // It's an impossible connection
309 LOG(MODULE_GRAPH, "Selected an unknown node, RARE.")
310 }

313CONFIGROOT RouteEditor::OpenEditor()
314{
315 auto result = this->exec();
316
317 if (rules.isEmpty())
318 {
319 // Prevent empty rule list causing mis-detection of config type to
320 // simple.
321 on_addRouteBtn_clicked();
322 }
323
324 // If clicking OK
325 if (result == QDialog::Accepted)
326 {
327 QJsonArray rulesArray;
328 QJsonArray _balancers;
329
330 // Append rules by order
331 for (auto i = 0; i < ruleListWidget->count(); i++)
332 {
333 auto _rule = rules[ruleListWidget->item(i)->text()];
334 auto ruleJsonObject = _rule.toJson();
335
336 // Process balancer for a rule
337 if (_rule.QV2RAY_RULE_USE_BALANCER)
338 {
339 // Do not use outbound tag.
340 ruleJsonObject.remove("outboundTag");
341
342 // Find balancer list
343 if (!balancers.contains(_rule.balancerTag))
344 {
345 LOG(MODULE_UI, "Cannot find a balancer for tag: " + _rule.balancerTag)
346 }
347 else
348 {
349 auto _balancerList = balancers[_rule.balancerTag];
350 QJsonObject balancerEntry;
351 balancerEntry["tag"] = _rule.balancerTag;
352 balancerEntry["selector"] = QJsonArray::fromStringList(_balancerList);
353 _balancers.append(balancerEntry);
354 }
355 }
356
357 // Remove some empty fields.
358 if (_rule.port.isEmpty())
359 {
360 ruleJsonObject.remove("port");
361 }
362
363 if (_rule.network.isEmpty())
364 {
365 ruleJsonObject.remove("network");
366 }
367
368 rulesArray.append(ruleJsonObject);
369 }
370
371 QJsonObject routing;
372 routing["domainStrategy"] = domainStrategy;
373 routing["rules"] = rulesArray;
374 routing["balancers"] = _balancers;
375 //
376 QJsonArray _inbounds;
377 QJsonArray _outbounds;
378
379 // Convert our internal data format to QJsonArray
380 for (auto x : inbounds)
381 {
382 if (x.isEmpty())
383 continue;
384
385 _inbounds.append(x.raw());
386 }
387
388 for (auto x : outbounds)
389 {
390 if (x.isEmpty())
391 continue;
392
393 if (getTag(x) == defaultOutbound)
394 {
395 LOG(MODULE_CONNECTION, "Pushing default outbound to the front.")
396 // Put the default outbound to the first.
397 _outbounds.push_front(x.raw());
398 }
399 else
400 {
401 _outbounds.push_back(x.raw());
402 }
403 }
404
405 root["inbounds"] = _inbounds;
406 root["outbounds"] = _outbounds;
407 root["routing"] = routing;
408 return root;
409 }
410 else
411 {
412 return original;
413 }
414}

428void RouteEditor::ShowCurrentRuleDetail()
429{
430 LOADINGCHECK
431

433 {
434 LOG(MODULE_UI, "WARNING, trying to access a non-exist rule entry. return.")
435 return;
436 }
437
438 if (!rules.contains(currentRuleTag))

440 QvMessageBoxWarn(this, tr("Show rule details"), tr("A rule cannot be found: ") + currentRuleTag);
441 LOG(MODULE_UI, "WARNING, trying to access a non-exist rule entry. return.")

445 // Switch to the detailed page.

477 netTCPRB->setChecked(network.contains("tcp"));

479 //
480 // Set protocol checkboxes.

482 routeProtocolHTTPCB->setChecked(protocol.contains("http"));
483 routeProtocolTLSCB->setChecked(protocol.contains("tls"));
484 routeProtocolBTCB->setChecked(protocol.contains("bittorrent"));

486 // Port

489 // Users

683void RouteEditor::on_addDefaultBtn_clicked()
684{
685 LOADINGCHECK
686 // Add default connection from GlobalConfig
687 //
688 auto _Inconfig = GlobalConfig.inboundConfig;
689 QJsonObject sniffingOff{ { "enabled", false } };
690 QJsonObject sniffingOn{ { "enabled", true }, { "destOverride", QJsonArray{ "http", "tls" } } };
691 //
692 if (_Inconfig.useHTTP)
693 {
694 INBOUND _in_HTTP;
695 _in_HTTP.insert("listen", _Inconfig.listenip);
696 _in_HTTP.insert("port", _Inconfig.httpSettings.port);
697 _in_HTTP.insert("protocol", "http");
698 _in_HTTP.insert("tag", "http_gConf");
699 if (!_Inconfig.httpSettings.sniffing)
700 {
701 _in_HTTP.insert("sniffing", sniffingOff);
702 }
703 else
704 {
705 _in_HTTP.insert("sniffing", sniffingOn);
706 }
707
708 if (_Inconfig.httpSettings.useAuth)
709 {
710 auto httpInSettings = GenerateHTTPIN(QList<AccountObject>() << _Inconfig.httpSettings.account);
711 _in_HTTP.insert("settings", httpInSettings);
712 }
713
714 AddInbound(_in_HTTP);
715 }
716 if (_Inconfig.useSocks)
717 {
718 auto _in_socksConf = GenerateSocksIN((_Inconfig.socksSettings.useAuth ? "password" : "noauth"), //
719 QList<AccountObject>() << _Inconfig.socksSettings.account, //
720 _Inconfig.socksSettings.enableUDP, //
721 _Inconfig.socksSettings.localIP);
722 auto _in_SOCKS = GenerateInboundEntry(_Inconfig.listenip, _Inconfig.socksSettings.port, "socks", _in_socksConf, "SOCKS_gConf");
723 if (!_Inconfig.socksSettings.sniffing)
724 {
725 _in_SOCKS.insert("sniffing", sniffingOff);
726 }
727 else
728 {
729 _in_SOCKS.insert("sniffing", sniffingOn);
730 }
731 AddInbound(_in_SOCKS);
732 }
733
734 if (_Inconfig.useTPROXY)
735 {
736 QList<QString> networks;
737#define _ts_ _Inconfig.tProxySettings
738 if (_ts_.hasTCP)
739 networks << "tcp";

819void RouteEditor::on_delBtn_clicked()
820{
821 if (nodeScene->selectedNodes().empty())
822 {
823 QvMessageBoxWarn(this, tr("Remove Items"), tr("Please select a node from the graph to continue."));
824 return;
825 }
826
827 auto firstNode = nodeScene->selectedNodes()[0];
828 auto isInbound = inboundNodes.values().contains(firstNode->id());
829 auto isOutbound = outboundNodes.values().contains(firstNode->id());
830 auto isRule = ruleNodes.values().contains(firstNode->id());
831
832 // Get the tag first, and call inbounds/outbounds/rules container variable
833 // remove() Remove the node last since some events may trigger. Then remove
834 // the node container.
835 if (isInbound)
836 {
837 currentInboundOutboundTag = GetFirstNodeData(firstNode->id(), InboundNode)->GetInbound();
838 nodeScene->removeNode(*nodeScene->node(inboundNodes[currentInboundOutboundTag]));
839 inboundNodes.remove(currentInboundOutboundTag);
840
841 // Remove corresponded inbound tags from the rules.
842 for (auto k : rules.keys())
843 {
844 auto v = rules[k];
845 v.inboundTag.removeAll(currentInboundOutboundTag);
846 rules[k] = v;
847 }
848
849 inbounds.remove(currentInboundOutboundTag);
850 }
851 else if (isOutbound)
852 {
853 currentInboundOutboundTag = GetFirstNodeData(firstNode->id(), OutboundNode)->GetOutbound();
854 outbounds.remove(currentInboundOutboundTag);
855 ResolveDefaultOutboundTag(currentInboundOutboundTag, "");
856
857 // Remove corresponded outbound tags from the rules.
858 for (auto k : rules.keys())
859 {
860 auto v = rules[k];
861
862 if (v.outboundTag == currentInboundOutboundTag)
863 v.outboundTag.clear();
864
865 rules[k] = v;
866 }
867
868 nodeScene->removeNode(*nodeScene->node(outboundNodes[currentInboundOutboundTag]));
869 outboundNodes.remove(currentInboundOutboundTag);
870 }
871 else if (isRule)
872 {
873 ruleEnableCB->setEnabled(false);
874 ruleTagLineEdit->setEnabled(false);
875 ruleRenameBtn->setEnabled(false);
876 auto RuleTag = GetFirstNodeData(firstNode->id(), RuleNode)->GetRuleTag();
877 currentRuleTag.clear();
878 routeRuleGroupBox->setEnabled(false);
879 routeEditGroupBox->setEnabled(false);
880 rules.remove(RuleTag);
881 nodeScene->removeNode(*nodeScene->node(ruleNodes[RuleTag]));
882 ruleNodes.remove(RuleTag);
883 //
884 // Remove item from the rule order list widget.
885 ruleListWidget->takeItem(ruleListWidget->row(ruleListWidget->findItems(RuleTag, Qt::MatchExactly).first()));
886 CHECKEMPTYRULES
887 // currentRuleTag = rules.firstKey();
888 // ShowCurrentRuleDetail();
889 }
890 else
891 {
892 LOG(MODULE_UI, "Unknown node selected.")
893 QvMessageBoxWarn(this, tr("Error"), tr("Qv2ray entered an unknown state."));
894 }

896void RouteEditor::on_editBtn_clicked()
897{
898 if (nodeScene->selectedNodes().empty())
899 {
900 QvMessageBoxWarn(this, tr("Edit Inbound/Outbound"), tr("Please select a node from the graph to continue."));
901 return;
902 }
903
904 const auto firstNode = nodeScene->selectedNodes().at(0);
905 const auto &isInbound = inboundNodes.values().contains(firstNode->id());
906 const auto &isOutbound = outboundNodes.values().contains(firstNode->id());
907
908 if (isInbound)
909 {
910 currentInboundOutboundTag = GetFirstNodeData(firstNode->id(), InboundNode)->GetInbound();
911
912 if (!inbounds.contains(currentInboundOutboundTag))
913 {
914 QvMessageBoxWarn(this, tr("Edit Inbound"), tr("No inbound tag found: ") + currentInboundOutboundTag);
915 return;
916 }
917
918 auto _in = inbounds[currentInboundOutboundTag];
919 INBOUND _result;
920 auto protocol = _in["protocol"].toString();
921 int _code;
922
923 if (protocol != "http" && protocol != "mtproto" && protocol != "socks" && protocol != "dokodemo-door")
924 {
925 QvMessageBoxWarn(this, tr("Cannot Edit"),
926 tr("Currently, this type of outbound is not supported by the editor.") + "\r\n" +
927 tr("We will launch Json Editor instead."));
928 statusLabel->setText(tr("Opening JSON editor"));
929 JsonEditor w(_in, this);
930 _result = INBOUND(w.OpenEditor());
931 _code = w.result();
932 }
933 else
934 {
935 InboundEditor w(_in, this);
936 statusLabel->setText(tr("Opening default inbound editor"));
937 _result = w.OpenEditor();
938 _code = w.result();
939 }
940
941 statusLabel->setText(tr("OK"));
942
943 if (_code == QDialog::Accepted)
944 {
945 bool isTagChanged = getTag(_in) != getTag(_result);
946
947 if (isTagChanged)
948 {
949 auto newTag = getTag(_result);
950 RenameItemTag(RENAME_INBOUND, getTag(_in), &newTag);
951 }
952
953 DEBUG(MODULE_UI, "Removed old tag: " + getTag(_in))
954 inbounds.remove(getTag(_in));
955 DEBUG(MODULE_UI, "Adding new tag: " + getTag(_result))
956 inbounds.insert(getTag(_result), _result);
957 }
958 }
959 else if (isOutbound)
960 {
961 currentInboundOutboundTag = GetFirstNodeData(firstNode->id(), OutboundNode)->GetOutbound();
962
963 if (!outbounds.contains(currentInboundOutboundTag))
964 {
965 QvMessageBoxWarn(this, tr("Edit Inbound"), tr("No inbound tag found: ") + currentInboundOutboundTag);
966 return;
967 }
968
969 OUTBOUND _result;
970 auto _out = outbounds.value(currentInboundOutboundTag);
971 auto protocol = _out["protocol"].toString().toLower();
972 int _code;
973
974 bool guisupport = true;
975 if (protocol != "vmess" && protocol != "shadowsocks" && protocol != "socks" && protocol != "http")
976 {
977 guisupport = false;
978 auto pluginEditorWidgetsInfo = PluginHost->GetOutboundEditorWidgets();
979 for (const auto &plugin : pluginEditorWidgetsInfo)
980 {
981 for (const auto &_d : plugin->OutboundCapabilities())
982 {
983 guisupport = guisupport || protocol == _d.protocol;
984 }
985 }
986 }
987
988 if (!guisupport)
989 {
990 QvMessageBoxWarn(this, tr("Unsupported Outbound Type"),
991 tr("This outbound entry is not supported by the GUI editor.") + NEWLINE +
992 tr("We will launch Json Editor instead."));
993 JsonEditor w(_out, this);
994 statusLabel->setText(tr("Opening JSON editor"));
995 _result = OUTBOUND(w.OpenEditor());
996 _code = w.result();
997 }
998 else
999 {
1000 OutboundEditor w(_out, this);
1001 statusLabel->setText(tr("Opening default outbound editor."));
1002 _result = w.OpenEditor();
1003 _code = w.result();
1004 }
1005
1006 if (_code == QDialog::Accepted)
1007 {
1008 bool isTagChanged = getTag(_out) != getTag(_result);
1009
1010 if (isTagChanged)
1011 {
1012 auto newTag = getTag(_result);
1013 DEBUG(MODULE_UI, "Outbound tag is changed: " + newTag)
1014 RenameItemTag(RENAME_OUTBOUND, getTag(_out), &newTag);
1015 }
1016
1017 DEBUG(MODULE_UI, "Adding new tag: " + getTag(_result))
1018 outbounds.insert(getTag(_result), _result);
1019 statusLabel->setText(tr("OK"));
1020 }
1021 }
1022 else
1023 {
1024 LOG(MODULE_UI, "Cannot apply 'edit' operation to non-inbound and non-outbound")
1025 }
1026}
