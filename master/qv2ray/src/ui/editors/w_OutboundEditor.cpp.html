30OutboundEditor::OutboundEditor(QWidget *parent) : QDialog(parent), tag(OUTBOUND_TAG_PROXY)
31{
32 QvMessageBusConnect(OutboundEditor);

34 //

36 outboundType = "vmess";
37 ALLOCATE_USERS;

40 streamSettingsWidget->SetStreamObject({});
41 transportFrame->addWidget(streamSettingsWidget);
42 //
43 auto pluginEditorWidgetsInfo = PluginHost->GetOutboundEditorWidgets();
44 for (const auto &plugin : pluginEditorWidgetsInfo)
45 {
46 for (const auto &_d : plugin->OutboundCapabilities())
47 {
48 outBoundTypeCombo->addItem(_d.displayName, _d.protocol);
49 auto index = outboundTypeStackView->addWidget(plugin);
50 pluginWidgets.insert(index, { _d, plugin });
51 }
52 }
53 //
54 outboundType = "vmess";
55 useForwardProxy = false;
56}

85QString OutboundEditor::GetFriendlyName()
86{
87 auto host = ipLineEdit->text().replace(":", "-").replace("/", "_").replace("\\", "_");
88 auto port = portLineEdit->text().replace(":", "-").replace("/", "_").replace("\\", "_");
89 auto type = outboundType;
90 QString name = tag.isEmpty() ? host + "-[" + port + "]-" + type : tag;
91 return name;

94OUTBOUND OutboundEditor::GenerateConnectionJson()
95{
96 OUTBOUNDSETTING settings;
97 auto streaming = streamSettingsWidget->GetStreamSettings().toJson();
98 ALLOCATE_USERS;
99 switch (outboundTypeStackView->currentIndex())
100 {
101 case STACK_PAGE_VMESS:
102 {
103 // VMess is only a ServerObject, and we need an array { "vnext": [] }
104 QJsonArray vnext;
105 vmess.address = serverAddress;
106 vmess.port = serverPort;
107 vnext.append(vmess.toJson());
108 settings.insert("vnext", vnext);
109 break;
110 }
111 case STACK_PAGE_VLESS:
112 {
113 QJsonArray vnext;
114 vless.address = serverAddress;
115 vless.port = serverPort;
116 vnext.append(vless.toJson());
117 settings.insert("vnext", vnext);
118 break;
119 }
120
121 case STACK_PAGE_SHADOWSOCKS:
122 {
123 // streaming = QJsonObject();
124 // LOG(MODULE_CONNECTION, "Shadowsocks outbound does not need StreamSettings.")
125 QJsonArray servers;
126 shadowsocks.address = serverAddress;
127 shadowsocks.port = serverPort;
128 servers.append(shadowsocks.toJson());
129 settings["servers"] = servers;
130 break;
131 }
132 case STACK_PAGE_SOCKS:
133 {
134 if (!socks.users.isEmpty() && socks.users.first().user.isEmpty() && socks.users.first().pass.isEmpty())
135 {
136 LOG(MODULE_UI, "Removed empty user form SOCKS settings")
137 socks.users.clear();
138 }
139 socks.address = serverAddress;
140 socks.port = serverPort;
141 // streaming = QJsonObject();
142 // LOG(MODULE_CONNECTION, "Socks outbound does not need StreamSettings.")
143 QJsonArray servers;
144 servers.append(socks.toJson());
145 settings["servers"] = servers;
146 break;
147 }
148 case STACK_PAGE_HTTP:
149 {
150 if (!http.users.isEmpty() && http.users.first().user.isEmpty() && http.users.first().pass.isEmpty())
151 {
152 LOG(MODULE_UI, "Removed empty user form HTTP settings")
153 http.users.clear();
154 }
155 http.address = serverAddress;
156 http.port = serverPort;
157 // streaming = QJsonObject();
158 // LOG(MODULE_CONNECTION, "Http outbound does not need StreamSettings.")
159 QJsonArray servers;
160 servers.append(http.toJson());
161 settings["servers"] = servers;
162 break;
163 }
164 default:
165 {
166 streaming = QJsonObject();
167 bool processed = false;
168 for (const auto &plugin : pluginWidgets)
169 {
170 if (plugin.first.protocol == outboundType)
171 {
172 plugin.second->SetHostInfo(serverAddress, serverPort);
173 settings = OUTBOUNDSETTING(plugin.second->GetContent());
174 processed = true;
175 break;
176 }
177 }
178 if (!processed)
179 {
180 QvMessageBoxWarn(this, tr("Unknown outbound type."),
181 tr("The specified outbound type is not supported, this may happen due to a plugin failure."));
182 }
183 break;
184 }
185 }
186
187 auto root = GenerateOutboundEntry(outboundType, settings, streaming, muxConfig, "0.0.0.0", tag);
188 root[QV2RAY_USE_FPROXY_KEY] = useForwardProxy;
189 return root;
190}

192void OutboundEditor::ReloadGUI()
193{
194 tag = originalConfig["tag"].toString();
195 tagTxt->setText(tag);
196 outboundType = originalConfig["protocol"].toString("vmess");
197 muxConfig = originalConfig["mux"].toObject();
198 useForwardProxy = originalConfig[QV2RAY_USE_FPROXY_KEY].toBool(false);
199 streamSettingsWidget->SetStreamObject(StreamSettingsObject::fromJson(originalConfig["streamSettings"].toObject()));
200 //
201 useFPCB->setChecked(useForwardProxy);
202 muxEnabledCB->setChecked(muxConfig["enabled"].toBool());
203 muxConcurrencyTxt->setValue(muxConfig["concurrency"].toInt());
204 //
205 ALLOCATE_USERS;
206 //
207 const auto &settings = originalConfig["settings"].toObject();
208 //
209 if (outboundType == "vmess")
210 {
211 outBoundTypeCombo->setCurrentIndex(STACK_PAGE_VMESS);
212 vmess = VMessServerObject::fromJson(settings["vnext"].toArray().first().toObject());
213 serverAddress = vmess.address;
214 serverPort = vmess.port;
215 if (vmess.users.empty())
216 vmess.users.push_back({});
217 const auto &user = vmess.users.front();
218 idLineEdit->setText(user.id);
219 alterLineEdit->setValue(user.alterId);
220 securityCombo->setCurrentText(user.security);
221 testsEnabledCombo->setCurrentText(user.testsEnabled);
222 }
223 else if (outboundType == "vless")
224 {
225 outBoundTypeCombo->setCurrentIndex(STACK_PAGE_VLESS);
226 vless = VLessServerObject::fromJson(settings["vnext"].toArray().first().toObject());
227 serverAddress = vless.address;
228 serverPort = vless.port;
229 if (vless.users.isEmpty())
230 vless.users.push_back({});
231 const auto &user = vless.users.front();
232 vLessIDTxt->setText(user.id);
233 vLessSecurityCombo->setCurrentText(user.encryption);
234 }
235 else if (outboundType == "shadowsocks")
236 {
237 outBoundTypeCombo->setCurrentIndex(STACK_PAGE_SHADOWSOCKS);
238 shadowsocks = ShadowSocksServerObject::fromJson(settings["servers"].toArray().first().toObject());
239 serverAddress = shadowsocks.address;
240 serverPort = shadowsocks.port;
241 // ShadowSocks Configs
242 ss_emailTxt->setText(shadowsocks.email);
243 ss_otaCheckBox->setChecked(shadowsocks.ota);
244 ss_passwordTxt->setText(shadowsocks.password);
245 ss_encryptionMethod->setCurrentText(shadowsocks.method);
246 }
247 else if (outboundType == "socks")
248 {
249 outBoundTypeCombo->setCurrentIndex(STACK_PAGE_SOCKS);
250 socks = SocksServerObject::fromJson(settings["servers"].toArray().first().toObject());
251 serverAddress = socks.address;
252 serverPort = socks.port;
253 if (socks.users.empty())
254 socks.users.push_back({});
255 socks_PasswordTxt->setText(socks.users.front().pass);
256 socks_UserNameTxt->setText(socks.users.front().user);
257 }
258 else if (outboundType == "http")
259 {
260 outBoundTypeCombo->setCurrentIndex(STACK_PAGE_HTTP);
261 http = HttpServerObject::fromJson(settings["servers"].toArray().first().toObject());
262 serverAddress = http.address;
263 serverPort = http.port;
264 if (http.users.empty())
265 http.users.push_back({});
266 http_PasswordTxt->setText(http.users.front().pass);
267 http_UserNameTxt->setText(http.users.front().user);
268 }
269 else
270 {
271 bool processed = false;
272 for (const auto &index : pluginWidgets.keys())
273 {
274 const auto &plugin = pluginWidgets.value(index);
275 if (plugin.first.protocol == outboundType)
276 {
277 useFPCB->setEnabled(false);
278 useFPCB->setToolTip(tr("Forward proxy has been disabled when using plugin outbound"));
279 plugin.second->SetContent(settings);
280 outBoundTypeCombo->setCurrentIndex(index);
281 const auto &[_address, _port] = plugin.second->GetHostInfo();
282 serverAddress = _address;
283 serverPort = _port;
284 processed = true;
285 break;
286 }
287 }
288 if (!processed)
289 {
290 LOG(MODULE_UI, "Outbound type: " + outboundType + " is not supported.")
291 QvMessageBoxWarn(this, tr("Unknown outbound."),
292 tr("The specified outbound type is invalid, this may be caused by a plugin failure.") + NEWLINE +
293 tr("Please use the JsonEditor or reload the plugin."));
294 reject();
295 }
296 }
297 //
298 ipLineEdit->setText(serverAddress);
299 portLineEdit->setText(QSTRN(serverPort));

317void OutboundEditor::on_idLineEdit_textEdited(const QString &arg1)
318{
319 const static QRegularExpression regExpUUID("^[0-9a-f]{8}(-[0-9a-f]{4}){3}-[0-9a-f]{12}$",
320 QRegularExpression::PatternOption::CaseInsensitiveOption);
321
322 if (!regExpUUID.match(arg1).hasMatch())
323 {
324 RED(idLineEdit);
325 }
326 else
327 {
328 BLACK(idLineEdit);
329 }

365void OutboundEditor::on_outBoundTypeCombo_currentIndexChanged(int index)
366{
367 // 0, 1, 2, 3, 4 as built-in vmess, vless, ss, socks, http
368 outboundTypeStackView->setCurrentIndex(index);
369 if (index < builtInOutboundTypes)
370 {
371 outboundType = outBoundTypeCombo->currentText().toLower();
372 useFPCB->setEnabled(true);
373 useFPCB->setToolTip("");
374 streamSettingsWidget->setEnabled(!useFPCB->isChecked());
375 }
376 else
377 {
378 outboundType = pluginWidgets.value(index).first.protocol;
379 useFPCB->setChecked(false);
380 useFPCB->setEnabled(false);
381 useFPCB->setToolTip(tr("Forward proxy has been disabled when using plugin outbound"));
382 streamSettingsWidget->setEnabled(false);
383 }
384}
