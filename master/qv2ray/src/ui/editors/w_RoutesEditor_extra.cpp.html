53QString RouteEditor::AddNewRule()
54{
55 // Add Route
56 RuleObject rule;
57 //
58 rule.QV2RAY_RULE_ENABLED = true;
59 rule.QV2RAY_RULE_USE_BALANCER = false;
60 // Default balancer tag, it's a random string.
61 auto bTag = GenerateRandomString();
62 rule.QV2RAY_RULE_TAG = rules.isEmpty() ? tr("Default rule") : (tr("rule") + "-" + GenerateRandomString(6));
63 rule.balancerTag = bTag;
64 balancers[bTag] = QStringList();
65 AddRule(rule);
66 return rule.QV2RAY_RULE_TAG;
67}

69void RouteEditor::AddRule(RuleObject rule)
70{
71 // Prevent duplicate
72 if (ruleNodes.contains(rule.QV2RAY_RULE_TAG))
73 {
74 rule.QV2RAY_RULE_TAG += "-" + GenerateRandomString(5);
75 }
76
77 rules.insert(rule.QV2RAY_RULE_TAG, rule);
78 auto pos = nodeGraphWidget->pos();
79 pos.setX(pos.x() + 350 + GRAPH_GLOBAL_OFFSET_X);
80 pos.setY(pos.y() + ruleNodes.count() * 120 + GRAPH_GLOBAL_OFFSET_Y);
81 auto _nodeData = std::make_unique<QvRuleNodeModel>(std::make_shared<RuleNodeData>(rule.QV2RAY_RULE_TAG));
82 auto &node = nodeScene->createNode(std::move(_nodeData));
83 nodeScene->setNodePosition(node, pos);
84
85 for (auto inTag : rule.inboundTag)
86 {
87 if (!inboundNodes.contains(inTag))
88 {
89 LOG(MODULE_UI, "No inbound tag found for rule: " + rule.QV2RAY_RULE_TAG + ", inbound tag: " + inTag)
90 QvMessageBoxWarn(this, tr("No Inbound"), tr("No inbound item found: ") + inTag);
91 rule.inboundTag.removeAll(inTag);
92 }
93 else
94 {
95 auto inboundNode = inboundNodes.value(inTag);
96 auto conn = nodeScene->createConnection(node, 0, *nodeScene->node(inboundNode), 0);
97 connect(conn.get(), &QtNodes::Connection::connectionCompleted, this, &RouteEditor::onConnectionCreated);
98 }
99 }
100
101 // If not using balancers (use outbound tag)
102 if (!rule.QV2RAY_RULE_USE_BALANCER)
103 {
104 if (outboundNodes.contains(rule.outboundTag))
105 {
106 DEBUG(MODULE_GRAPH, "Found outbound tag: " + rule.outboundTag + ", for rule: " + rule.QV2RAY_RULE_TAG)
107 auto conn = nodeScene->createConnection(*nodeScene->node(outboundNodes.value(rule.outboundTag)), 0, node, 0);
108 connect(conn.get(), &QtNodes::Connection::connectionCompleted, this, &RouteEditor::onConnectionCreated);
109 }
110 else
111 {
112 LOG(MODULE_GRAPH, "Outbound tag not found: " + rule.outboundTag + ", for: " + rule.QV2RAY_RULE_TAG)
113 // QvMessageBoxWarn(this, tr("No outbound tag"), tr("Please connect
114 // the rule with an outbound."));
115 }
116 }
117
118 this->ruleNodes.insert(rule.QV2RAY_RULE_TAG, node.id());
119 ruleListWidget->addItem(rule.QV2RAY_RULE_TAG);
120}

123void RouteEditor::RenameItemTag(ROUTE_EDIT_MODE mode, const QString originalTag, QString *newTag)
124{
125 switch (mode)
126 {
127 case RENAME_RULE:
128 if (rules.contains(originalTag) && ruleNodes.contains(originalTag))
129 {
130 if (rules.contains(*newTag) && ruleNodes.contains(*newTag))
131 {
132 QvMessageBoxWarn(this, tr("Rename tags"), tr("The new tag has been used, we appended a postfix."));
133 *newTag += "_" + GenerateRandomString(5);
134 }
135
136 const auto &nodeDataModel = (QvRuleNodeModel *) nodeScene->node(ruleNodes.value(originalTag))->nodeDataModel();
137
138 if (nodeDataModel == nullptr)
139 {
140 LOG(MODULE_GRAPH, "EMPTY NODE WARN")
141 }
142
143 nodeDataModel->setData(*newTag);
144 //
145 auto rule = rules.take(originalTag);
146 rule.QV2RAY_RULE_TAG = *newTag;
147 rules.insert(*newTag, rule);
148 ruleNodes.insert(*newTag, ruleNodes.take(originalTag));
149 //
150 // No other operation needed, but need to rename the one in the
151 // ruleOrder list widget.
152 auto items = ruleListWidget->findItems(originalTag, Qt::MatchExactly);
153
154 if (items.isEmpty())
155 {
156 LOG(MODULE_UI, "Cannot find a node: " + originalTag)
157 }
158 else
159 {
160 items.first()->setText(*newTag);
161 }
162
163 if (currentRuleTag == originalTag)
164 {
165 currentRuleTag = *newTag;
166 }
167 }
168 else
169 {
170 LOG(MODULE_UI, "There's nothing match " + originalTag + " in the containers.")
171 }
172
173 break;
174
175 case RENAME_OUTBOUND:
176 if (outbounds.contains(originalTag) && outboundNodes.contains(originalTag))
177 {
178 if (outbounds.contains(*newTag) && outboundNodes.contains(*newTag))
179 {
180 QvMessageBoxWarn(this, tr("Rename tags"), tr("The new tag has been used, we appended a random string to the tag."));
181 *newTag += "_" + GenerateRandomString(5);
182 }
183
184 auto out = outbounds.take(originalTag);
185 out["tag"] = *newTag;
186 outbounds.insert(*newTag, out);
187 outboundNodes.insert(*newTag, outboundNodes.take(originalTag));
188 const auto &node = (QvOutboundNodeModel *) nodeScene->node(outboundNodes.value(*newTag))->nodeDataModel();
189
190 if (node == nullptr)
191 {
192 LOG(MODULE_GRAPH, "EMPTY NODE WARN")
193 }
194
195 node->setData(*newTag);
196
197 // Change outbound tag in rules accordingly.
198 for (const auto &k : rules.keys())
199 {
200 auto v = rules.value(k);
201
202 if (v.outboundTag == originalTag)
203 {
204 v.outboundTag = *newTag;
205 // Put this inside the if block since no need an extra
206 // operation if the condition is false.
207 rules.insert(k, v);
208 }
209 }
210
211 // Resolve default outbound.
212 ResolveDefaultOutboundTag(originalTag, *newTag);
213 }
214 else
215 {
216 LOG(MODULE_UI, "Failed to rename an outbound --> Item not found.")
217 }
218
219 break;
220
221 case RENAME_INBOUND:
222 if (inbounds.contains(originalTag) && inboundNodes.contains(originalTag))
223 {
224 if (inbounds.contains(*newTag) && inboundNodes.contains(*newTag))
225 {
226 QvMessageBoxWarn(this, tr("Rename tags"), tr("The new tag has been used, we appended a postfix."));
227 *newTag += "_" + GenerateRandomString(5);
228 }
229 auto in = inbounds.take(originalTag);
230 in["tag"] = *newTag;
231 inbounds.insert(*newTag, in);
232 inboundNodes.insert(*newTag, inboundNodes.take(originalTag));
233 const auto &node = (QvInboundNodeModel *) nodeScene->node(inboundNodes.value(*newTag))->nodeDataModel();
234
235 if (node == nullptr)
236 {
237 LOG(MODULE_GRAPH, "EMPTY NODE WARN")
238 }
239
240 node->setData(*newTag);
241
242 // Change inbound tag in rules accordingly.
243 // k -> rule tag
244 // v -> rule object
245 for (const auto &k : rules.keys())
246 {
247 auto v = rules.value(k);
248
249 if (v.inboundTag.contains(originalTag))
250 {
251 v.inboundTag.append(*newTag);
252 v.inboundTag.removeAll(originalTag);
253 // Put this inside the if block since no need an extra
254 // operation if the condition is false.
255 rules.insert(k, v);
256 }
257 }
258 }
259 else
260 {
261 LOG(MODULE_UI, "Failed to rename an outbound --> Item not found.")
262 }
263
264 break;
265 }
266}

269void RouteEditor::ResolveDefaultOutboundTag(const QString original, const QString newTag)
270{
271 LOG(MODULE_UI, "Resolving default outbound settings: default=" + defaultOutbound + " original=" + original + " new=" + newTag)
272 auto isDefaultChanged = original == defaultOutbound;
273 //
274 isLoading = true;
275 defaultOutboundCombo->clear();
276 defaultOutboundCombo->addItems(outbounds.keys());
277 isLoading = false;
278 //
279 if (!isDefaultChanged)
280 {
281 LOG(MODULE_UI, "Default outbound is not changed: retaining: " + defaultOutbound)
282 // Just simply restore the default one.
283 defaultOutboundCombo->setCurrentText(defaultOutbound);
284 }
285 else if (newTag.isEmpty())
286 {
287 LOG(MODULE_UI, "Default outbound is removed, using first key from the outbounds as the default one.")
288
289 // Removed the default one, so set the first one as the default.
290 if (outbounds.isEmpty())
291 {
292 defaultOutbound.clear();
293 }
294 else
295 {
296 defaultOutbound = getTag(outbounds.first());
297 defaultOutboundCombo->addItem(outbounds.firstKey());
298 }
299 }
300 else
301 {
302 LOG(MODULE_UI, "Default outbound is renamed, new tag is " + newTag)
303 defaultOutboundCombo->setCurrentText(newTag);
304 defaultOutbound = newTag;
305 }
