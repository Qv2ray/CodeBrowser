94void GroupManager::onRCMExportConnectionTriggered()
95{
96 const auto &list = GET_SELECTED_CONNECTION_IDS(SELECTED_ROWS_INDEX);
97 QFileDialog d;
98 switch (list.count())
99 {
100 case 0: return;
101 case 1:
102 {
103 const auto id = ConnectionId(list.first());
104 auto filePath = d.getSaveFileName(this, GetDisplayName(id));
105 if (filePath.isEmpty())
106 return;
107 auto root = RouteManager->GenerateFinalConfig({ id, currentGroupId }, false);
108 //
109 // Apply export filter
110 ExportConnectionFilter(root);
111 //
112 if (filePath.endsWith(".json"))
113 {
114 filePath += ".json";
115 }
116 //
117 StringToFile(JsonToString(root), filePath);
118 QDesktopServices::openUrl(QUrl::fromLocalFile(QFileInfo(filePath).absoluteDir().absolutePath()));
119 break;
120 }
121 default:
122 {
123 const auto path = d.getExistingDirectory();
124 if (path.isEmpty())
125 return;
126 for (const auto &connId : list)
127 {
128 ConnectionId id(connId);
129 auto root = RouteManager->GenerateFinalConfig({ id, currentGroupId });
130 //
131 // Apply export filter
132 ExportConnectionFilter(root);
133 //
134 const auto fileName = RemoveInvalidFileName(GetDisplayName(id)) + ".json";
135 StringToFile(JsonToString(root), path + "/" + fileName);
136 }
137 QDesktopServices::openUrl(QUrl::fromLocalFile(path));
138 break;
139 }
140 }

168void GroupManager::reloadConnectionsList(const GroupId &group)
169{
170 if (group == NullGroupId)
171 return;
172 connectionsTable->clearContents();
173 connectionsTable->model()->removeRows(0, connectionsTable->rowCount());
174 const auto &connections = ConnectionManager->Connections(group);
175 for (auto i = 0; i < connections.count(); i++)
176 {
177 const auto &conn = connections.at(i);
178 connectionsTable->insertRow(i);
179 auto displayNameItem = new QTableWidgetItem(GetDisplayName(conn));
180 displayNameItem->setData(Qt::UserRole, conn.toString());
181 auto typeItem = new QTableWidgetItem(GetConnectionProtocolString(conn));
182 const auto [type, host, port] = GetConnectionInfo(conn);
183 auto hostPortItem = new QTableWidgetItem(host + ":" + QSTRN(port));
184 //
185 QStringList groupsNamesString;
186 for (const auto &group : ConnectionManager->GetGroupId(conn))
187 {
188 groupsNamesString.append(GetDisplayName(group));
189 }
190 auto groupsItem = new QTableWidgetItem(groupsNamesString.join(";"));
191 connectionsTable->setItem(i, 0, displayNameItem);
192 connectionsTable->setItem(i, 1, typeItem);
193 connectionsTable->setItem(i, 2, hostPortItem);
194 connectionsTable->setItem(i, 3, groupsItem);
195 }
196 connectionsTable->resizeColumnsToContents();
197}

277void GroupManager::on_updateButton_clicked()
278{
279 if (subAddrTxt->text().trimmed().isEmpty())

286 QvMessageBoxWarn(this, tr("Update Subscription"), tr("The subscription link is invalid."));

289 if (QvMessageBoxAsk(this, tr("Update Subscription"), tr("Would you like to update the subscription?")) == QMessageBox::Yes)

298void GroupManager::on_removeGroupButton_clicked()
299{
300 if (QvMessageBoxAsk(this, tr("Deleting a group"), tr("All connections will be moved to default group, do you want to continue?")) ==

302 {
303 ConnectionManager->DeleteGroup(currentGroupId);
304 auto item = groupList->currentItem();
305 int index = groupList->row(item);
306 groupList->removeItemWidget(item);
307 delete item;
308 if (groupList->count() > 0)
309 {
310 index = std::max(index, 0);
311 index = std::min(index, groupList->count() - 1);
312 groupList->setCurrentItem(groupList->item(index));
313 on_groupList_itemClicked(groupList->item(index));
314 }
315 else
316 {
317 groupInfoGroupBox->setEnabled(false);

339void GroupManager::on_groupList_itemClicked(QListWidgetItem *item)
340{
341 if (item == nullptr)
342 {
343 return;
344 }
345 groupInfoGroupBox->setEnabled(true);
346 currentGroupId = GroupId(item->data(Qt::UserRole).toString());
347 //
348 groupNameTxt->setText(GetDisplayName(currentGroupId));
349 const auto &groupMetaObject = ConnectionManager->GetGroupMetaObject(currentGroupId);
350 groupIsSubscriptionGroup->setChecked(groupMetaObject.isSubscription);
351 subAddrTxt->setText(groupMetaObject.subscriptionOption.address);
352 lastUpdatedLabel->setText(timeToString(groupMetaObject.lastUpdatedDate));
353 createdAtLabel->setText(timeToString(groupMetaObject.creationDate));
354 updateIntervalSB->setValue(groupMetaObject.subscriptionOption.updateInterval);
355 IncludeKeywords->clear();
356 for (const auto &key : groupMetaObject.subscriptionOption.IncludeKeywords)
357 {
358 auto str = key.trimmed();
359 if (!str.isEmpty())
360 {
361 IncludeKeywords->appendPlainText(str);
362 }
363 }
364 ExcludeKeywords->clear();
365 for (const auto &key : groupMetaObject.subscriptionOption.ExcludeKeywords)
366 {
367 auto str = key.trimmed();
368 if (!str.isEmpty())
369 {
370 ExcludeKeywords->appendPlainText(str);
371 }
372 }
373 IncludeRelation->setCurrentIndex(groupMetaObject.subscriptionOption.IncludeRelation);
374 ExcludeRelation->setCurrentIndex(groupMetaObject.subscriptionOption.ExcludeRelation);
375 //
376 // Load DNS / Route config
377 const auto routeId = ConnectionManager->GetGroupRoutingId(currentGroupId);
378 {
379 const auto &[overrideDns, dns] = RouteManager->GetDNSSettings(routeId);
380 dnsSettingsWidget->SetDNSObject(dns);
381 dnsSettingsGB->setChecked(overrideDns);
382 }
383 {
384 const auto &[overrideRoute, route] = RouteManager->GetAdvancedRoutingSettings(routeId);
385 routeSettingsWidget->SetRouteConfig(route);
386 routeSettingsGB->setChecked(overrideRoute);
387 }
388 reloadConnectionsList(currentGroupId);
389}

471void GroupManager::ExportConnectionFilter(CONFIGROOT &root)
472{
473 root.remove("api");
474 root.remove("stats");
475 QJsonArray inbounds = root["inbounds"].toArray();
476 for (int i = root["inbounds"].toArray().count() - 1; i >= 0; i--)
477 {
478 auto obj = root["inbounds"].toArray().at(i).toObject();
479 if (obj["tag"] == API_TAG_INBOUND)
480 {
481 inbounds.removeAt(i);
482 }
483 }
484 root["inbounds"] = inbounds;
485}
