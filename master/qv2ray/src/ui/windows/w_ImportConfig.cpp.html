57QMultiMap<QString, CONFIGROOT> ImportConfigWindow::SelectConnection(bool outboundsOnly)
58{
59 // partial import means only import as an outbound, will set outboundsOnly to
60 // false and disable the checkbox
61 keepImportedInboundCheckBox->setEnabled(!outboundsOnly);
62 routeEditBtn->setEnabled(!outboundsOnly);
63 groupCombo->setEnabled(false);
64 this->exec();
65 QMultiMap<QString, CONFIGROOT> conn;
66 for (const auto &connEntry : connectionsToNewGroup.values())
67 {
68 conn += connEntry;
69 }
70 for (const auto &connEntry : connectionsToExistingGroup.values())
71 {
72 conn += connEntry;
73 }
74 return result() == Accepted ? conn : QMultiMap<QString, CONFIGROOT>{};

77int ImportConfigWindow::PerformImportConnection()
78{
79 this->exec();
80 int count = 0;
81 for (const auto &groupObject : connectionsToNewGroup)
82 {
83 const auto groupName = connectionsToNewGroup.key(groupObject);
84 GroupId groupId = ConnectionManager->CreateGroup(groupName, false);
85 for (const auto &connConf : groupObject)
86 {
87 auto connName = groupObject.key(connConf);
88
89 auto [protocol, host, port] = GetConnectionInfo(connConf);
90 if (connName.isEmpty())
91 {
92 connName = protocol + "/" + host + ":" + QSTRN(port) + "-" + GenerateRandomString(5);
93 }
94 ConnectionManager->CreateConnection(connConf, connName, groupId, true);
95 }
96 }
97
98 for (const auto &groupObject : connectionsToExistingGroup)
99 {
100 const auto groupId = connectionsToExistingGroup.key(groupObject);
101 for (const auto &connConf : groupObject)
102 {
103 auto connName = groupObject.key(connConf);
104 auto [protocol, host, port] = GetConnectionInfo(connConf);
105 if (connName.isEmpty())
106 {
107 connName = protocol + "/" + host + ":" + QSTRN(port) + "-" + GenerateRandomString(5);
108 }
109 ConnectionManager->CreateConnection(connConf, connName, groupId, true);
110 }
111 }
112
113 return count;
114}

122void ImportConfigWindow::on_qrFromScreenBtn_clicked()
123{
124 bool hideQv2ray = hideQv2rayCB->isChecked();
125
126 if (hideQv2ray)
127 {
128 UIMessageBus.EmitGlobalSignal(QvMBMessage::HIDE_WINDOWS);
129 }
130
131 QApplication::processEvents();
132 QThread::msleep(doubleSpinBox->value() * 1000UL);

135 if (hideQv2ray)

139
140 if (w.result() == QDialog::Accepted)
141 {
142 auto str = DecodeQRCode(pix);
143 if (str.trimmed().isEmpty())

145 LOG(MODULE_UI, "Cannot decode QR Code from an image, size: h=" + QSTRN(pix.width()) + ", v=" + QSTRN(pix.height()))

147 }
148 else
149 {
150 qrCodeLinkTxt->setText(str.trimmed());
151 }
152 }

155void ImportConfigWindow::on_beginImportBtn_clicked()
156{
157 QString aliasPrefix = nameTxt->text();
158
159 switch (tabWidget->currentIndex())
160 {
161 case LINK_PAGE:
162 {
163 QStringList linkList = SplitLines(vmessConnectionStringTxt->toPlainText());
164 //
165 // Clear UI and error lists
166 linkErrors.clear();
167 vmessConnectionStringTxt->clear();
168 errorsList->clear();
169 //
170 LOG(MODULE_IMPORT, QSTRN(linkList.count()) + " string(s) found in vmess box.")
171
172 while (!linkList.isEmpty())
173 {
174 aliasPrefix = nameTxt->text();
175 const auto link = linkList.takeFirst().trimmed();
176 if (link.isEmpty() || link.startsWith("#") || link.startsWith("//"))
177 continue;
178
179 // warn if someone tries to import a https:// link
180 if (link.startsWith("https://"))
181 {
182 errorsList->addItem(tr("WARNING: You may have mistaken 'subscription link' with 'share link'"));
183 }
184
185 QString errMessage;
186 QString newGroupName;
187 const auto config = ConvertConfigFromString(link, &aliasPrefix, &errMessage, &newGroupName);
188
189 // If the config is empty or we have any err messages.
190 if (config.isEmpty() || !errMessage.isEmpty())
191 {
192 // To prevent duplicated values.
193 linkErrors[link] = QSTRN(linkErrors.count() + 1) + ": " + errMessage;
194 continue;
195 }
196 else if (newGroupName.isEmpty())
197 {
198 for (const auto &conf : config)
199 {
200 connectionsToExistingGroup[GroupId{ groupCombo->currentData().toString() }].insert(conf.first, conf.second);
201 }
202 }
203 else
204 {
205 for (const auto &conf : config)
206 {
207 connectionsToNewGroup[newGroupName].insert(conf.first, conf.second);
208 }
209 }
210 }
211
212 if (!linkErrors.isEmpty())
213 {
214 for (const auto &item : linkErrors)
215 {
216 vmessConnectionStringTxt->appendPlainText(linkErrors.key(item));
217 errorsList->addItem(item);
218 }
219
220 vmessConnectionStringTxt->setLineWidth(errorsList->lineWidth());
221 errorsList->sortItems();
222 return;
223 }
224
225 break;
226 }
227 case QRCODE_PAGE:
228 {
229 QString errorMsg;
230 const auto root = ConvertConfigFromString(qrCodeLinkTxt->text(), &aliasPrefix, &errorMsg);
231 if (!errorMsg.isEmpty())
232 {
233 QvMessageBoxWarn(this, tr("Failed to import connection"), errorMsg);
234 break;
235 }
236 for (const auto &conf : root)
237 {
238 connectionsToExistingGroup[GroupId{ groupCombo->currentData().toString() }].insert(conf.first, conf.second);
239 }
240 break;
241 }
242 case MANUAL_PAGE:
243 {
244 break;
245 }
246 case ADVANCED_PAGE:
247 {
248 // From File...
249 bool ImportAsComplex = keepImportedInboundCheckBox->isChecked();
250 QString path = fileLineTxt->text();
251
252 if (!V2RayKernelInstance::ValidateConfig(path))
253 {
254 QvMessageBoxWarn(this, tr("Import config file"), tr("Failed to check the validity of the config file."));
255 return;
256 }
257
258 aliasPrefix += "_" + QFileInfo(path).fileName();
259 CONFIGROOT config = ConvertConfigFromFile(path, ImportAsComplex);
260 connectionsToExistingGroup[GroupId{ groupCombo->currentData().toString() }].insert(aliasPrefix, config);
261 break;
262 }
263 }
264
265 accept();
266}

288void ImportConfigWindow::on_errorsList_currentItemChanged(QListWidgetItem *current, QListWidgetItem *previous)
289{
290 Q_UNUSED(previous)
291
292 if (current == nullptr)
293 {
294 return;
295 }
296
297 auto currentErrorText = current->text();
298 auto vmessEntry = linkErrors.key(currentErrorText);
299 //
300 auto startPos = vmessConnectionStringTxt->toPlainText().indexOf(vmessEntry);
301 auto endPos = startPos + vmessEntry.length();
302
303 if (startPos < 0)
304 {
305 return;
306 }
307
308 // Select vmess string that is invalid.
309 QTextCursor c = vmessConnectionStringTxt->textCursor();
310 c.setPosition(startPos);
311 c.setPosition(endPos, QTextCursor::KeepAnchor);
312 vmessConnectionStringTxt->setTextCursor(c);
313}
