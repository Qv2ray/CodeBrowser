9PluginManageWindow::PluginManageWindow(QWidget *parent) : QvDialog(parent)
10{
11 setupUi(this);
12 for (auto &plugin : PluginHost->AvailablePlugins())
13 {
14 const auto &info = PluginHost->GetPluginMetadata(plugin);
15 auto item = new QListWidgetItem(pluginListWidget);
16 item->setCheckState(PluginHost->GetPluginEnableState(info.InternalName) ? Qt::Checked : Qt::Unchecked);
17 item->setData(Qt::UserRole, info.InternalName);
18 item->setText(info.Name + " (" + (PluginHost->GetPluginLoadState(info.InternalName) ? tr("Loaded") : tr("Not loaded")) + ")");
19 pluginListWidget->addItem(item);
20 }
21 isLoading = false;
22}

29void PluginManageWindow::on_pluginListWidget_currentItemChanged(QListWidgetItem *current, QListWidgetItem *previous)
30{
31 Q_UNUSED(previous)
32 auto &info = PluginHost->GetPluginMetadata(current->data(Qt::UserRole).toString());
33
34 pluginIconLabel->setPixmap(info.Icon.pixmap(pluginIconLabel->size() * devicePixelRatio()));
35 //
36 pluginNameLabel->setText(info.Name);
37 pluginAuthorLabel->setText(info.Author);
38 pluginDescriptionLabel->setText(info.Description);
39 pluginLibPathLabel->setText(PluginHost->GetPluginLibraryPath(info.InternalName));
40 pluginStateLabel->setText(PluginHost->GetPluginLoadState(info.InternalName) ? tr("Loaded") : tr("Not loaded"));
41 pluginTypeLabel->setText(GetPluginTypeString(info.SpecialPluginType));
42 pluginHookTypeLabel->setText(GetPluginCapabilityString(info.Capabilities));
43 //
44 if (!current)
45 {
46 return;
47 }
48 if (settingsWidget || settingsWidget.get())
49 {
50 pluginSettingsLayout->removeWidget(settingsWidget.get());
51 settingsWidget.reset();
52 }
53 if (!PluginHost->GetPluginLoadState(info.InternalName))
54 {
55 pluginUnloadLabel->setVisible(true);
56 pluginUnloadLabel->setText(tr("Plugin Not Loaded"));
57 return;
58 }
59 settingsWidget = PluginHost->GetPluginSettingsWidget(info.InternalName);
60 if (settingsWidget)
61 {
62 pluginUnloadLabel->setVisible(false);
63 settingsWidget.get()->setParent(this);
64 pluginSettingsLayout->addWidget(settingsWidget.get());
65 }
66 else
67 {
68 pluginUnloadLabel->setVisible(true);
69 pluginUnloadLabel->setText(tr("Plugin does not have settings widget."));
70 }
71}

79void PluginManageWindow::on_pluginListWidget_itemChanged(QListWidgetItem *item)
80{
81 if (isLoading)
82 return;
83 bool isEnabled = item->checkState() == Qt::Checked;
84 auto pluginInternalName = item->data(Qt::UserRole).toString();
85 PluginHost->SetPluginEnableState(pluginInternalName, isEnabled);
86 auto &info = PluginHost->GetPluginMetadata(pluginInternalName);
87 item->setText(info.Name + " (" + (PluginHost->GetPluginLoadState(info.InternalName) ? tr("Loaded") : tr("Not loaded")) + ")");
88 //
89 if (!isEnabled)
90 {
91 QvMessageBoxInfo(this, tr("Disabling a plugin"), tr("This plugin will keep loaded until the next time Qv2ray starts."));
92 }
93}

95void PluginManageWindow::on_pluginEditSettingsJsonBtn_clicked()
96{
97 if (const auto &current = pluginListWidget->currentItem(); current != nullptr)
98 {
99 const auto &info = PluginHost->GetPluginMetadata(current->data(Qt::UserRole).toString());
100 if (!PluginHost->GetPluginLoadState(info.InternalName))
101 {
102 QvMessageBoxWarn(this, tr("Plugin not loaded"), tr("This plugin is not loaded, please enable or reload the plugin to continue."));
103 return;
104 }
105 JsonEditor w(PluginHost->GetPluginSettings(info.InternalName));
106 auto newConf = w.OpenEditor();
107 if (w.result() == QDialog::Accepted)
108 {
109 PluginHost->SetPluginSettings(info.InternalName, newConf);
110 }
111 }
112}

120void PluginManageWindow::on_openPluginFolder_clicked()
121{
122 QDir pluginPath(QV2RAY_CONFIG_DIR + "plugins/");
123 if (!pluginPath.exists())
124 {
125 pluginPath.mkpath(QV2RAY_CONFIG_DIR + "plugins/");
126 }
127#ifdef FALL_BACK_TO_XDG_OPEN
128 QProcess::execute("xdg-open", { pluginPath.absolutePath() });
129#else
130 QDesktopServices::openUrl(QUrl::fromLocalFile(pluginPath.absolutePath()));
131#endif
132}
