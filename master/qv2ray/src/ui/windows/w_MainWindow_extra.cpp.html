6void MainWindow::MWSetSystemProxy()
7{
8 const auto inboundInfo = KernelInstance->GetInboundInfo();
9
10 InboundInfoObject httpInboundInfo;
11 InboundInfoObject socksInboundInfo;
12
13 for (const auto &info : inboundInfo)
14 {
15 if (info.protocol == "http")
16 httpInboundInfo = info;
17 if (info.protocol == "socks")
18 socksInboundInfo = info;
19 }
20
21 const bool httpEnabled = httpInboundInfo.port > 0;
22 const bool socksEnabled = socksInboundInfo.port > 0;
23 QString proxyAddress;
24
25 if (httpEnabled || socksEnabled)
26 {
27 proxyAddress = httpEnabled ? httpInboundInfo.listenIp : socksInboundInfo.listenIp;
28 if (proxyAddress == "0.0.0.0")
29 proxyAddress = "127.0.0.1";
30
31 if (proxyAddress == "::")
32 proxyAddress = "::1";
33
34 LOG(MODULE_UI, "ProxyAddress: " + proxyAddress);
35 SetSystemProxy(proxyAddress, httpInboundInfo.port, socksInboundInfo.port);
36 qvAppTrayIcon->setIcon(Q_TRAYICON("tray-systemproxy.png"));
37 if (!GlobalConfig.uiConfig.quietMode)
38 {
39 qvApp->showMessage(tr("System proxy configured."));
40 }
41 }
42 else
43 {
44 LOG(MODULE_PROXY, "Neither of HTTP nor SOCKS is enabled, cannot set system proxy.")
45 QvMessageBoxWarn(this, tr("Cannot set system proxy"), tr("Both HTTP and SOCKS inbounds are not enabled"));
46 }
47}

72void MainWindow::CheckSubscriptionsUpdate()
73{
74 QList<QPair<QString, GroupId>> updateList;
75 QStringList updateNamesList;
76
77 for (const auto &entry : ConnectionManager->Subscriptions())
78 {
79 const auto info = ConnectionManager->GetGroupMetaObject(entry);
80 //
81 // The update is ignored.
82 if (info.subscriptionOption.updateInterval == 0)
83 continue;
84 //
85 const auto lastRenewDate = QDateTime::fromTime_t(info.lastUpdatedDate);
86 const auto renewTime = lastRenewDate.addSecs(info.subscriptionOption.updateInterval * 86400);
87
88 if (renewTime <= QDateTime::currentDateTime())
89 {
90 updateList << QPair{ info.displayName, entry };
91 updateNamesList << info.displayName;
92 LOG(MODULE_SUBSCRIPTION, QString("Subscription update \"%1\": L=%2 R=%3 I=%4")
93 .arg(info.displayName)
94 .arg(lastRenewDate.toString())
95 .arg(QSTRN(info.subscriptionOption.updateInterval))
96 .arg(renewTime.toString()))
97 }
98 }
99
100 if (!updateList.isEmpty())
101 {
102 const auto result = GlobalConfig.uiConfig.quietMode ? QMessageBox::Yes :
103 QvMessageBoxAsk(this, tr("Update Subscriptions"), //
104 tr("Do you want to update these subscriptions?") + NEWLINE + //
105 updateNamesList.join(NEWLINE), //
106 QMessageBox::Ignore);
107
108 for (const auto &[name, id] : updateList)
109 {
110 if (result == QMessageBox::Yes)
111 {
112 LOG(MODULE_UI, "Updating subscription: " + name)
113 ConnectionManager->UpdateSubscriptionAsync(id);
114 }
115 else if (result == QMessageBox::Ignore)
116 {
117 LOG(MODULE_UI, "Ignored subscription update: " + name)
118 ConnectionManager->IgnoreSubscriptionUpdate(id);
119 }
120 }
121 }
