274void PreferencesWindow::on_buttonBox_accepted()
275{
276 // Note:
277 // A signal-slot connection from buttonbox_accpted to QDialog::accepted()
278 // has been removed. To prevent closing this Dialog.
279 QSet<int> ports;
280 auto size = 0;
281
282 if (CurrentConfig.inboundConfig.useHTTP)
283 {
284 size++;
285 ports << CurrentConfig.inboundConfig.httpSettings.port;
286 }
287
288 if (CurrentConfig.inboundConfig.useSocks)
289 {
290 size++;
291 ports << CurrentConfig.inboundConfig.socksSettings.port;
292 }
293
294 if (CurrentConfig.inboundConfig.useTPROXY)
295 {
296 size++;
297 ports << CurrentConfig.inboundConfig.tProxySettings.port;
298 }
299
300 if (!StartupOption.noAPI)
301 {
302 size++;
303 ports << CurrentConfig.kernelConfig.statsPort;
304 }
305
306 if (ports.size() != size)
307 {
308 // Duplicates detected.
309 QvMessageBoxWarn(this, tr("Preferences"), tr("Duplicated port numbers detected, please check the port number settings."));
310 }
311 else if (!IsValidIPAddress(CurrentConfig.inboundConfig.listenip))
312 {
313 QvMessageBoxWarn(this, tr("Preferences"), tr("Invalid inbound listening address."));
314 }
315 else if (!IsIPv4Address(CurrentConfig.inboundConfig.tProxySettings.tProxyIP))
316 {
317 QvMessageBoxWarn(this, tr("Preferences"), tr("Invalid tproxy listening ivp4 address."));
318 }
319 else if (CurrentConfig.inboundConfig.tProxySettings.tProxyV6IP != "" &&
320 !IsIPv6Address(CurrentConfig.inboundConfig.tProxySettings.tProxyV6IP))
321 {
322 QvMessageBoxWarn(this, tr("Preferences"), tr("Invalid tproxy listening ipv6 address."));
323 }
324 else if (!dnsSettingsWidget->CheckIsValidDNS())
325 {
326 QvMessageBoxWarn(this, tr("Preferences"), tr("Invalid DNS settings."));
327 }
328 else
329 {
330 if (CurrentConfig.uiConfig.language != GlobalConfig.uiConfig.language)
331 {
332 // Install translator
333 if (Qv2rayTranslator->InstallTranslation(CurrentConfig.uiConfig.language))
334 {
335 UIMessageBus.EmitGlobalSignal(QvMBMessage::RETRANSLATE);
336 QApplication::processEvents();
337 }
338 else
339 {
340 LOG(MODULE_UI, "Failed to translate UI to: " + CurrentConfig.uiConfig.language)
341 }
342 }
343 CurrentConfig.defaultRouteConfig.routeConfig = routeSettingsWidget->GetRouteConfig();
344 if (!(CurrentConfig.defaultRouteConfig.routeConfig == GlobalConfig.defaultRouteConfig.routeConfig))
345 {
346 NEEDRESTART
347 }
348 CurrentConfig.defaultRouteConfig.dnsConfig = dnsSettingsWidget->GetDNSObject();
349 if (!(CurrentConfig.defaultRouteConfig.dnsConfig == GlobalConfig.defaultRouteConfig.dnsConfig))
350 {
351 NEEDRESTART
352 }
353 //
354 //
355 if (CurrentConfig.uiConfig.theme != GlobalConfig.uiConfig.theme)
356 {
357 StyleManager->ApplyStyle(CurrentConfig.uiConfig.theme);
358 }
359 SaveGlobalSettings(CurrentConfig);
360 UIMessageBus.EmitGlobalSignal(QvMBMessage::UPDATE_COLORSCHEME);
361 if (NeedRestart && !KernelInstance->CurrentConnection().isEmpty())
362 {
363 this->setEnabled(false);
364 qApp->processEvents();
365 ConnectionManager->RestartConnection();
366 }
367 emit accept();
368 }
369}

501void PreferencesWindow::on_tProxyCheckBox_stateChanged(int arg1)
502{
503 LOADINGCHECK
504#ifdef Q_OS_LINUX
505 // Setting up tProxy for linux
506 // Steps:
507 // --> 1. Copy V2Ray core files to the QV2RAY_TPROXY_VCORE_PATH and QV2RAY_TPROXY_VCTL_PATH dir.
508 // --> 2. Change GlobalConfig.v2CorePath.
509 // --> 3. Call `pkexec setcap
510 // CAP_NET_ADMIN,CAP_NET_RAW,CAP_NET_BIND_SERVICE=eip` on the V2Ray core.
511 auto const kernelPath = CurrentConfig.kernelConfig.KernelPath();
512 if (arg1 == Qt::Checked)
513 {
514 // We enable it!
515 if (QvMessageBoxAsk(this, tr("Enable tProxy Support"),
516 tr("This will append capabilities to the V2Ray executable.") + NEWLINE + NEWLINE +
517 tr("Qv2ray will copy your V2Ray core to this path: ") + NEWLINE + QV2RAY_TPROXY_VCORE_PATH + NEWLINE + NEWLINE +
518 tr("If anything goes wrong after enabling this, please check issue #57 or the link below:") + NEWLINE +
519 " https://github.com/Qv2ray/Qv2ray/wiki/FAQ ") != QMessageBox::Yes)
520 {
521 tProxyCheckBox->setChecked(false);
522 LOG(MODULE_UI, "Canceled enabling tProxy feature.")
523 }
524 else
525 {
526
527 LOG(MODULE_VCORE, "ENABLING tProxy Support")
528 LOG(MODULE_FILEIO, " --> Origin V2Ray core file is at: " + kernelPath)
529 auto v2ctlPath = QFileInfo(kernelPath).absolutePath() + "/v2ctl";
530 auto newPath = QFileInfo(QV2RAY_TPROXY_VCORE_PATH).absolutePath();
531 QString mkPathResult = QDir().mkpath(newPath) ? "OK" : "FAILED";
532 LOG(MODULE_FILEIO, " --> mkPath result: " + mkPathResult)
533 //
534 LOG(MODULE_FILEIO, " --> Origin v2ctl file is at: " + v2ctlPath)
535 LOG(MODULE_FILEIO, " --> New V2Ray files will be placed in: " + newPath)
536 //
537 LOG(MODULE_FILEIO, " --> Copying files....")
538
539 if (QFileInfo(kernelPath).absoluteFilePath() != QFileInfo(QV2RAY_TPROXY_VCORE_PATH).absoluteFilePath())
540 {
541 // Only trying to remove file when they are not in the default
542 // dir. (In other words...) Keep using the current files.
543 // <Because we don't know where else we can copy the file
544 // from...>
545 //
546 if (QFile(QV2RAY_TPROXY_VCORE_PATH).exists())
547 {
548 LOG(MODULE_FILEIO, QString(QV2RAY_TPROXY_VCORE_PATH) + ": File already exists.")
549 LOG(MODULE_FILEIO, QString(QV2RAY_TPROXY_VCORE_PATH) + ": Deleting file.")
550 QFile(QV2RAY_TPROXY_VCORE_PATH).remove();
551 }
552
553 if (QFile(QV2RAY_TPROXY_VCTL_PATH).exists())
554 {
555 LOG(MODULE_FILEIO, QV2RAY_TPROXY_VCTL_PATH + ": File already exists.")
556 LOG(MODULE_FILEIO, QV2RAY_TPROXY_VCTL_PATH + ": Deleting file.")
557 QFile(QV2RAY_TPROXY_VCTL_PATH).remove();
558 }
559
560 QString vCoreresult = QFile(kernelPath).copy(QV2RAY_TPROXY_VCORE_PATH) ? "OK" : "FAILED";
561 LOG(MODULE_FILEIO, " --> V2Ray Core: " + vCoreresult)
562 //
563 QString vCtlresult = QFile(v2ctlPath).copy(QV2RAY_TPROXY_VCTL_PATH) ? "OK" : "FAILED";
564 LOG(MODULE_FILEIO, " --> V2Ray Ctl: " + vCtlresult)
565 //
566
567 if (vCoreresult == "OK" && vCtlresult == "OK")
568 {
569 LOG(MODULE_VCORE, " --> Done copying files.")
570 on_vCorePathTxt_textEdited(QV2RAY_TPROXY_VCORE_PATH);
571 }
572 else
573 {
574 LOG(MODULE_VCORE, "FAILED to copy V2Ray files. Aborting.")
575 QvMessageBoxWarn(this, tr("Enable tProxy Support"),
576 tr("Qv2ray cannot copy one or both V2Ray files from: ") + NEWLINE + NEWLINE + kernelPath + NEWLINE +
577 v2ctlPath + NEWLINE + NEWLINE + tr("to this path: ") + NEWLINE + newPath);
578 return;
579 }
580 }
581 else
582 {
583 LOG(MODULE_VCORE, "Skipped removing files since the current V2Ray core is in the default path.")
584 LOG(MODULE_VCORE, " --> Actually because we don't know where else to obtain the files.")
585 }
586
587 LOG(MODULE_UI, "Calling pkexec and setcap...")
588 int ret = QProcess::execute("pkexec", { "/usr/sbin/setcap CAP_NET_ADMIN,CAP_NET_RAW,CAP_NET_BIND_SERVICE=eip " + kernelPath });
589 if (ret != 0)
590 {
591 LOG(MODULE_UI, "WARN: setcap exits with code: " + QSTRN(ret))
592 QvMessageBoxWarn(this, tr("Preferences"), tr("Failed to setcap onto V2Ray executable. You may need to run `setcap` manually."));
593 }
594
595 CurrentConfig.tProxySupport = true;
596 NEEDRESTART
597 }
598 }
599 else
600 {
601 int ret = QProcess::execute("pkexec", { "/usr/sbin/setcap -r " + kernelPath });
602
603 if (ret != 0)
604 {
605 LOG(MODULE_UI, "WARN: setcap exits with code: " + QSTRN(ret))
606 QvMessageBoxWarn(this, tr("Preferences"), tr("Failed to setcap onto V2Ray executable. You may need to run `setcap` manually."));
607 }
608
609 CurrentConfig.tProxySupport = false;
610 NEEDRESTART
611 }
612
613#else
614 Q_UNUSED(arg1)
615 // No such tProxy thing on Windows and macOS
616 QvMessageBoxWarn(this, tr("Preferences"), tr("tProxy is not supported on macOS and Windows"));
617 CurrentConfig.tProxySupport = false;
618 tProxyCheckBox->setChecked(false);
619#endif
620}

810void PreferencesWindow::on_checkVCoreSettings_clicked()
811{
812 auto vcorePath = vCorePathTxt->text();
813 auto vAssetsPath = vCoreAssetsPathTxt->text();
814 QString result;
815
816 // prevent some bullshit situations.
817 if (const auto vCorePathSmallCased = vcorePath.toLower();
818 vCorePathSmallCased.endsWith("qv2ray") || vCorePathSmallCased.endsWith("qv2ray.exe"))
819 {
820 const auto strWarnTitle = tr("Watch Out!");
821 const auto strWarnContent = //
822 tr("You may be about to set V2Ray core incorrectly to Qv2ray itself, which is absolutely not correct.\r\n"
823 "This won't trigger a fork bomb, however, since Qv2ray works in singleton mode.\r\n"
824 "If your V2Ray core filename happened to be 'qv2ray'-something, you are totally free to ignore this warning.");
825 const auto answer = QMessageBox::warning(this, strWarnTitle, strWarnContent, //
826 QMessageBox::StandardButton::Abort | QMessageBox::StandardButton::Ignore, //
827 QMessageBox::StandardButton::Abort);
828 if (answer == QMessageBox::StandardButton::Abort)
829 return;
830 }
831 else if (vCorePathSmallCased.endsWith("v2ctl") || vCorePathSmallCased.endsWith("v2ctl.exe"))
832 {
833 const auto strWarnTitle = tr("Watch Out!");
834 const auto strWarnContent = //
835 tr("You may be about to set V2Ray core incorrectly to V2Ray Control executable, which is absolutely not correct.\r\n"
836 "The filename of V2Ray core is usually 'v2ray' or 'v2ray.exe'. Make sure to choose it wisely.\r\n"
837 "If you insist to proceed, we're not providing with any support.");
838 const auto answer = QMessageBox::warning(this, strWarnTitle, strWarnContent, //
839 QMessageBox::StandardButton::Abort | QMessageBox::StandardButton::Ignore, //
840 QMessageBox::StandardButton::Abort);
841 if (answer == QMessageBox::StandardButton::Abort)
842 return;
843 }
844
845 if (!V2RayKernelInstance::ValidateKernel(vcorePath, vAssetsPath, &result))
846 {
847 QvMessageBoxWarn(this, tr("V2Ray Core Settings"), result);
848 }
849 else if (!result.toLower().contains("v2ray"))
850 {
851 const auto strWarnContent = //
852 tr("This does not seem like an output from V2Ray Core.\r\n"
853 "If you've been looking for plugin cores, you should change this in plugin settings rather than here.\r\n"
854 "Output: \r\n\r\n") +
855 result;
856 QvMessageBoxWarn(this, tr("'V2Ray Core' Settings"), strWarnContent);
857 }
858 else
859 {
860 QvMessageBoxInfo(this, tr("V2Ray Core Settings"),
861 tr("V2Ray path configuration check passed.") + NEWLINE + NEWLINE + tr("Current version of V2Ray is: ") + NEWLINE +
862 result);
863 }
864}

1109void PreferencesWindow::on_pushButton_clicked()
1110{
1111 const auto ntpTitle = tr("NTP Checker");
1112 const auto ntpHint = tr("Check date and time from server:");
1113 const static QStringList ntpServerList = { "cn.pool.ntp.org", //
1114 "cn.ntp.org.cn", //
1115 "edu.ntp.org.cn", //
1116 "time.pool.aliyun.com", //
1117 "time1.cloud.tencent.com", //
1118 "ntp.neu.edu.cn" };
1119 bool ok = false;
1120 QString ntpServer = QInputDialog::getItem(this, ntpTitle, ntpHint, ntpServerList, 0, true, &ok).trimmed();
1121 if (!ok)
1122 return;
1123
1124 auto client = new ntp::NtpClient(this);
1125 connect(client, &ntp::NtpClient::replyReceived, [&](const QHostAddress &, quint16, const ntp::NtpReply &reply) {
1126 const int offsetSecTotal = reply.localClockOffset() / 1000;
1127 if (offsetSecTotal >= 90 || offsetSecTotal <= -90)
1128 {
1129 const auto inaccurateWarning = tr("Your time offset is %1 seconds, which is too high.") + NEWLINE + //
1130 tr("Please synchronize your system to use V2Ray.");
1131 QvMessageBoxWarn(this, tr("Time Inaccurate"), inaccurateWarning.arg(offsetSecTotal));
1132 }
1133 else if (offsetSecTotal > 15 || offsetSecTotal < -15)
1134 {
1135 const auto smallErrorWarning = tr("Your time offset is %1 seconds, which is a little high.") + NEWLINE + //
1136 tr("V2Ray may still work, but we suggest you synchronize your clock.");
1137 QvMessageBoxInfo(this, tr("Time Somewhat Inaccurate"), smallErrorWarning.arg(offsetSecTotal));
1138 }
1139 else
1140 {
1141 const auto accurateInfo = tr("Your time offset is %1 seconds, which looks good.") + NEWLINE + //
1142 tr("V2Ray may not suffer from time inaccuracy.");
1143 QvMessageBoxInfo(this, tr("Time Accurate"), accurateInfo.arg(offsetSecTotal));
1144 }
1145 });
1146
1147 const auto hostInfo = QHostInfo::fromName(ntpServer);
1148 if (hostInfo.error() == QHostInfo::NoError)
1149 {
1150 client->sendRequest(hostInfo.addresses().first(), 123);
1151 }
1152 else
1153 {
1154 QvMessageBoxWarn(this, ntpTitle, tr("Failed to lookup server: %1").arg(hostInfo.errorString()));
1155 }
1156}
