107void MainWindow::ReloadRecentConnectionList()
108{
109 QList<ConnectionGroupPair> newRecentConnections;
110 const auto iterateRange = std::min(GlobalConfig.uiConfig.maxJumpListCount, GlobalConfig.uiConfig.recentConnections.count());
111 for (auto i = 0; i < iterateRange; i++)
112 {
113 const auto &item = GlobalConfig.uiConfig.recentConnections.at(i);
114 if (newRecentConnections.contains(item) || item.isEmpty())
115 continue;
116 newRecentConnections << item;
117 }
118 GlobalConfig.uiConfig.recentConnections = newRecentConnections;
119}

353void MainWindow::ProcessCommand(QString command, QStringList commands, QMap<QString, QString> args)
354{
355 if (commands.isEmpty())
356 return;
357 if (command == "open")
358 {
359 const auto subcommand = commands.takeFirst();
360 QvDialog *w;
361 if (subcommand == "preference")
362 w = new PreferencesWindow();
363 else if (subcommand == "plugin")
364 w = new PluginManageWindow();
365 else if (subcommand == "group")
366 w = new GroupManager();
367 else if (subcommand == "import")
368 w = new ImportConfigWindow();
369 else
370 return;
371 w->processCommands(command, commands, args);
372 w->exec();
373 delete w;
374 }
375}

390void MainWindow::keyPressEvent(QKeyEvent *e)
391{
392 if (focusWidget() == connectionListWidget)
393 {
394 CheckCurrentWidget;
395 if (e->key() == Qt::Key_Enter || e->key() == Qt::Key_Return)
396 {
397 // If pressed enter or return on connectionListWidget. Try to connect to the selected connection.
398 if (widget->IsConnection())
399 {
400 widget->BeginConnection();
401 }
402 else
403 {
404 connectionListWidget->expandItem(connectionListWidget->currentItem());
405 }
406 }
407 else if (e->key() == Qt::Key_F2)
408 {
409 widget->BeginRename();
410 }
411 else if (e->key() == Qt::Key_Delete)
412 {
413 on_action_RCM_DeleteThese_triggered();
414 }
415 }
416
417 if (e->key() == Qt::Key_Escape)
418 {
419 auto widget = GetItemWidget(connectionListWidget->currentItem());
420 // Check if this key was accpted by the ConnectionItemWidget
421 if (widget && widget->IsRenaming())
422 {
423 widget->CancelRename();
424 return;
425 }
426 else if (this->isActiveWindow())
427 {
428 this->close();
429 }
430 }
431 else if (e->modifiers() & Qt::ControlModifier && e->key() == Qt::Key_Q)
432 {
433 if (QvMessageBoxAsk(this, tr("Quit Qv2ray"), tr("Are you sure to exit Qv2ray?")) == QMessageBox::Yes)
434 on_actionExit_triggered();
435 }
436}

561void MainWindow::on_action_RCM_DeleteThese_triggered()
562{
563 QList<ConnectionGroupPair> connlist;
564
565 for (const auto &item : connectionListWidget->selectedItems())
566 {
567 auto widget = GetItemWidget(item);
568 if (widget)
569 {
570 const auto identifier = widget->Identifier();
571 if (widget->IsConnection())
572 {
573 connlist.append(identifier);
574 }
575 else
576 {
577 for (const auto &conns : ConnectionManager->GetGroupMetaObject(identifier.groupId).connections)
578 {
579 ConnectionGroupPair i;
580 i.connectionId = conns;
581 i.groupId = identifier.groupId;
582 connlist.append(i);
583 }
584 }
585 }
586 }
587
588 LOG(MODULE_UI, "Selected " + QSTRN(connlist.count()) + " items")
589
590 if (connlist.isEmpty())
591 {
592 // Remove nothing means doing nothing.
593 return;
594 }
595
596 const auto strRemoveConnTitle = tr("Removing Connection(s)", "", connlist.count());
597 const auto strRemoveConnContent = tr("Are you sure to remove selected connection(s)?", "", connlist.count());
598 if (QvMessageBoxAsk(this, strRemoveConnTitle, strRemoveConnContent) != QMessageBox::Yes)
599 {
600 return;
601 }
602
603 for (const auto &conn : connlist)
604 {
605 if (ConnectionManager->IsConnected(conn))
606 ConnectionManager->StopConnection();
607 if (GlobalConfig.autoStartId == conn)

776void MainWindow::OnStatsAvailable(const ConnectionGroupPair &id, const QMap<StatisticsType, QvStatsSpeedData> &data)
777{
778 if (!ConnectionManager->IsConnected(id))
779 return;
780 // This may not be, or may not precisely be, speed per second if the backend
781 // has "any" latency. (Hope not...)
782 //
783 QMap<SpeedWidget::GraphID, long> pointData;
784 bool isOutbound = GlobalConfig.uiConfig.graphConfig.useOutboundStats;
785 bool hasDirect = isOutbound && GlobalConfig.uiConfig.graphConfig.hasDirectStats;
786 for (const auto &type : data.keys())
787 {
788 const auto upSpeed = data[type].first.first;
789 const auto downSpeed = data[type].first.second;
790 switch (type)
791 {
792 case API_INBOUND:
793 if (!isOutbound)
794 {
795 pointData[SpeedWidget::INBOUND_UP] = upSpeed;
796 pointData[SpeedWidget::INBOUND_DOWN] = downSpeed;
797 }
798 break;
799 case API_OUTBOUND_PROXY:
800 if (isOutbound)
801 {
802 pointData[SpeedWidget::OUTBOUND_PROXY_UP] = upSpeed;
803 pointData[SpeedWidget::OUTBOUND_PROXY_DOWN] = downSpeed;
804 }
805 break;
806 case API_OUTBOUND_DIRECT:
807 if (hasDirect)
808 {
809 pointData[SpeedWidget::OUTBOUND_DIRECT_UP] = upSpeed;
810 pointData[SpeedWidget::OUTBOUND_DIRECT_DOWN] = downSpeed;
811 }
812 break;
813 case API_OUTBOUND_BLACKHOLE: break;
814 }
815 }
816
817 speedChartWidget->AddPointData(pointData);
818 //
819 const auto upSpeed = data[CurrentStatAPIType].first.first;
820 const auto downSpeed = data[CurrentStatAPIType].first.second;
821 auto totalSpeedUp = FormatBytes(upSpeed) + "/s";
822 auto totalSpeedDown = FormatBytes(downSpeed) + "/s";
823 auto totalDataUp = FormatBytes(data[CurrentStatAPIType].second.first);
824 auto totalDataDown = FormatBytes(data[CurrentStatAPIType].second.second);
825 //
826 netspeedLabel->setText(totalSpeedUp + NEWLINE + totalSpeedDown);
827 dataamountLabel->setText(totalDataUp + NEWLINE + totalDataDown);
828 //
829 qvAppTrayIcon->setToolTip(TRAY_TOOLTIP_PREFIX NEWLINE + tr("Connected: ") + GetDisplayName(id.connectionId) + //
830 NEWLINE "Up: " + totalSpeedUp + " Down: " + totalSpeedDown);
831 //
832 // Set data accordingly
833 if (connectionNodes.contains(id))
834 {
835 connectionNodes.value(id)->setText(MW_ITEM_COL_DATA, NumericString(GetConnectionTotalData(id.connectionId)));
836 }
837}

839void MainWindow::OnVCoreLogAvailable(const ConnectionGroupPair &id, const QString &log)
840{
841 Q_UNUSED(id);
842 FastAppendTextDocument(log.trimmed(), vCoreLogDocument);
843 // vCoreLogDocument->setPlainText(vCoreLogDocument->toPlainText() + log);
844 // From https://gist.github.com/jemyzhang/7130092
845 auto maxLines = GlobalConfig.uiConfig.maximumLogLines;
846 auto block = vCoreLogDocument->begin();
847
848 while (block.isValid())
849 {
850 if (vCoreLogDocument->blockCount() > maxLines)
851 {
852 QTextCursor cursor(block);
853 block = block.next();
854 cursor.select(QTextCursor::BlockUnderCursor);
855 cursor.movePosition(QTextCursor::NextCharacter, QTextCursor::KeepAnchor);
856 cursor.removeSelectedText();
857 continue;
858 }
859
860 break;
861 }
862}

864void MainWindow::OnEditRequested(const ConnectionId &id)
865{
866 auto outBoundRoot = ConnectionManager->GetConnectionRoot(id);
867 CONFIGROOT root;

872 LOG(MODULE_UI, "INFO: Opening route editor.")

875 isChanged = routeWindow.result() == QDialog::Accepted;
