11constexpr auto QV2RAY_PLUGIN_INTERFACE_VERSION = 2;

13namespace Qv2rayPlugin
14{
15 class Qv2rayInterface
16 {
17 public:
18 const int QvPluginInterfaceVersion = QV2RAY_PLUGIN_INTERFACE_VERSION;
19 //
20 virtual ~Qv2rayInterface() = default;
21 //
22 /// Basic metainfo of this plugin
23 virtual const QvPluginMetadata GetMetadata() const = 0;
24 /// Initilise Plugin using plugin settings directory and a Json-contained plugin settings.
25 virtual bool Initialize(const QString &, const QJsonObject &) = 0;
26 /// Update Plugin Settings.
27 virtual bool UpdateSettings(const QJsonObject &) = 0;
28 /// Get Plugin Setting
29 virtual const QJsonObject GetSettngs() = 0;
30 /// \brief Get the settings widget of this plugin.
31 /// \warning The owner of this widget is the caller.
32 /// \returns A unique pointer to QWidget of settings widget, the widget will be deleted after use each time.
33 virtual std::unique_ptr<QWidget> GetSettingsWidget() = 0;
34 /// \brief Get editor widget of this plugin, if any, according to the UI_TYPE
35 /// \arg UI_TYPE type of the desired ui
36 /// \warning The owner of this widget is the caller, the widget will be deleted after use each time.
37 /// \returns A unique pointer to the QvPluginEditor instance.
38 virtual std::unique_ptr<QvPluginEditor> GetEditorWidget(UI_TYPE) = 0;
39 /// \brief Get Kernel Instance if the QvPluginMetadata contains SPECIAL_TYPE_KERNEL
40 /// \warning The owner of QvPluginKernel instance will be transfered to its caller.
41 virtual std::unique_ptr<QvPluginKernel> CreateKernel() = 0;
42 /// \brief Get Serializer Instance if the QvPluginMetadata contains SPECIAL_TYPE_SERIALIZOR
43 /// \warning The owner of this QvPluginSerializer instance is the plugin instance itself.
44 virtual std::shared_ptr<QvPluginSerializer> GetSerializer() = 0;
45 /// \brief Get QvPluginEventHandler Instance. This should not return a nullptr
46 /// \warning The owner of this QvPluginEventHandler instance is the plugin instance itself.
47 virtual std::shared_ptr<QvPluginEventHandler> GetEventHandler() = 0;
48 //
49 /// \brief Signal to emit when a plugin need to write log messages.
50 virtual void PluginLog(const QString &) const = 0;
51 /// \brief Signal to emit when a plugin need to popup an error message box.
52 virtual void PluginErrorMessageBox(const QString &) const = 0;
53 };
54} // namespace Qv2rayPlugin
