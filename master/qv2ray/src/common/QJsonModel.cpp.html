98QJsonTreeItem *QJsonTreeItem::load(const QJsonValue &value, QJsonTreeItem *parent)
99{
100 QJsonTreeItem *rootItem = new QJsonTreeItem(parent);
101 rootItem->setKey("root");
102
103 if (value.isObject())
104 {
105 // Get all QJsonValue childs
106 for (QString key : value.toObject().keys())
107 {
108 QJsonValue v = value.toObject().value(key);
109 QJsonTreeItem *child = load(v, rootItem);
110 child->setKey(key);
111 child->setType(v.type());
112 rootItem->appendChild(child);
113 }
114 }
115 else if (value.isArray())
116 {
117 // Get all QJsonValue childs
118 int index = 0;
119
120 for (QJsonValue v : value.toArray())
121 {
122 QJsonTreeItem *child = load(v, rootItem);
123 child->setKey(QString::number(index));
124 child->setType(v.type());
125 rootItem->appendChild(child);
126 ++index;
127 }
128 }
129 else
130 {
131 rootItem->setValue(value.toVariant().toString());
132 rootItem->setType(value.type());
133 }
134
135 return rootItem;
136}

172bool QJsonModel::load(const QString &fileName)
173{
174 QFile file(fileName);
175 bool success = false;
176
177 if (file.open(QIODevice::ReadOnly))
178 {
179 success = load(&file);
180 file.close();
181 }
182 else
183 success = false;
184
185 return success;
186}

193bool QJsonModel::loadJson(const QByteArray &json)
194{
195 auto const &jdoc = QJsonDocument::fromJson(json);
196
197 if (!jdoc.isNull())
198 {
199 beginResetModel();
200 delete mRootItem;
201
202 if (jdoc.isArray())
203 {
204 mRootItem = QJsonTreeItem::load(QJsonValue(jdoc.array()));
205 mRootItem->setType(QJsonValue::Array);
206 }
207 else
208 {
209 mRootItem = QJsonTreeItem::load(QJsonValue(jdoc.object()));
210 mRootItem->setType(QJsonValue::Object);
211 }
212
213 endResetModel();
214 return true;
215 }
216
217 qDebug() << Q_FUNC_INFO << "cannot load json";
218 return false;
219}

221QVariant QJsonModel::data(const QModelIndex &index, int role) const
222{
223 if (!index.isValid())
224 return QVariant();
225
226 QJsonTreeItem *item = static_cast<QJsonTreeItem *>(index.internalPointer());
227
228 if (role == Qt::DisplayRole)
229 {
230 if (index.column() == 0)
231 return QString("%1").arg(item->key());
232
233 if (index.column() == 1)
234 return QString("%1").arg(item->value());
235 }
236 else if (Qt::EditRole == role)
237 {
238 if (index.column() == 1)
239 {
240 return QString("%1").arg(item->value());
241 }
242 }
243
244 return QVariant();
245}

247bool QJsonModel::setData(const QModelIndex &index, const QVariant &value, int role)
248{
249 int col = index.column();
250
251 if (Qt::EditRole == role)
252 {
253 if (col == 1)
254 {
255 QJsonTreeItem *item = static_cast<QJsonTreeItem *>(index.internalPointer());
256 item->setValue(value.toString());
257 emit dataChanged(index, index, { Qt::EditRole });
258 return true;
259 }
260 }
261
262 return false;

278QModelIndex QJsonModel::index(int row, int column, const QModelIndex &parent) const
279{
280 if (!hasIndex(row, column, parent))
281 return QModelIndex();
282
283 QJsonTreeItem *parentItem;
284
285 if (!parent.isValid())
286 parentItem = mRootItem;
287 else
288 parentItem = static_cast<QJsonTreeItem *>(parent.internalPointer());
289
290 QJsonTreeItem *childItem = parentItem->child(row);
291
292 if (childItem)
293 return createIndex(row, column, childItem);
294 else
295 return QModelIndex();
296}

298QModelIndex QJsonModel::parent(const QModelIndex &index) const
299{
300 if (!index.isValid())
301 return QModelIndex();
302
303 QJsonTreeItem *childItem = static_cast<QJsonTreeItem *>(index.internalPointer());
304 QJsonTreeItem *parentItem = childItem->parent();
305
306 if (parentItem == mRootItem)
307 return QModelIndex();
308
309 return createIndex(parentItem->row(), 0, parentItem);
310}

312int QJsonModel::rowCount(const QModelIndex &parent) const
313{
314 QJsonTreeItem *parentItem;
315
316 if (parent.column() > 0)
317 return 0;
318
319 if (!parent.isValid())
320 parentItem = mRootItem;
321 else
322 parentItem = static_cast<QJsonTreeItem *>(parent.internalPointer());
323
324 return parentItem->childCount();
325}

333Qt::ItemFlags QJsonModel::flags(const QModelIndex &index) const
334{
335 int col = index.column();
336 auto item = static_cast<QJsonTreeItem *>(index.internalPointer());
337 auto isArray = QJsonValue::Array == item->type();
338 auto isObject = QJsonValue::Object == item->type();
339
340 if ((col == 1) && !(isArray || isObject))
341 {
342 return Qt::ItemIsEditable | QAbstractItemModel::flags(index);
343 }
344 else
345 {
346 return QAbstractItemModel::flags(index);
347 }
348}

351{

367QJsonValue QJsonModel::genJson(QJsonTreeItem *item) const
368{
369 auto type = item->type();
370 int nchild = item->childCount();
371
372 if (QJsonValue::Object == type)
373 {
374 QJsonObject jo;
375
376 for (int i = 0; i < nchild; ++i)
377 {
378 auto ch = item->child(i);
379 auto key = ch->key();
380 jo.insert(key, genJson(ch));
381 }
382
383 return jo;
384 }
385 else if (QJsonValue::Array == type)
386 {
387 QJsonArray arr;
388
389 for (int i = 0; i < nchild; ++i)
390 {
391 auto ch = item->child(i);
392 arr.append(genJson(ch));
393 }
394
395 return arr;
396 }
397 else
398 {
399 QJsonValue va(item->value());
400 return va;
401 }
