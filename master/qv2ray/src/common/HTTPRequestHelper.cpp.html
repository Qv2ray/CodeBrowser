16 void NetworkRequestHelper::setAccessManagerAttributes(QNetworkRequest &request, QNetworkAccessManager &accessManager)
17 {
18 switch (GlobalConfig.networkConfig.proxyType)
19 {
20 case Qv2rayConfig_Network::QVPROXY_NONE:
21 {
22 DEBUG(MODULE_NETWORK, "Get without proxy.")
23 accessManager.setProxy(QNetworkProxy(QNetworkProxy::ProxyType::NoProxy));
24 break;
25 }
26 case Qv2rayConfig_Network::QVPROXY_SYSTEM:
27 {
28 accessManager.setProxy(QNetworkProxyFactory::systemProxyForQuery().first());
29 break;
30 }
31 case Qv2rayConfig_Network::QVPROXY_CUSTOM:
32 {
33 QNetworkProxy p{
34 GlobalConfig.networkConfig.type == "http" ? QNetworkProxy::HttpProxy : QNetworkProxy::Socks5Proxy, //
35 GlobalConfig.networkConfig.address, //
36 quint16(GlobalConfig.networkConfig.port) //
37 };
38 accessManager.setProxy(p);
39 break;
40 }
41 default: Q_UNREACHABLE();
42 }
43
44 if (accessManager.proxy().type() == QNetworkProxy::Socks5Proxy)
45 {
46 DEBUG(MODULE_NETWORK, "Adding HostNameLookupCapability to proxy.")
47 accessManager.proxy().setCapabilities(accessManager.proxy().capabilities() | QNetworkProxy::HostNameLookupCapability);
48 }
49
50 request.setAttribute(QNetworkRequest::RedirectPolicyAttribute, QNetworkRequest::NoLessSafeRedirectPolicy);
51
52#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
53 // request.setAttribute(QNetworkRequest::Http2AllowedAttribute, true);
54#else
55 // request.setAttribute(QNetworkRequest::HTTP2AllowedAttribute, true);
56#endif
57
58 auto ua = GlobalConfig.networkConfig.userAgent;
59 ua.replace("$VERSION", QV2RAY_VERSION_STRING);
60 request.setHeader(QNetworkRequest::KnownHeaders::UserAgentHeader, ua);
61 }

63 QByteArray NetworkRequestHelper::HttpGet(const QUrl &url)
64 {
65 QNetworkRequest request;
66 QNetworkAccessManager accessManager;
67 request.setUrl(url);
68 setAccessManagerAttributes(request, accessManager);
69 auto _reply = accessManager.get(request);
70 //
71 {
72 QEventLoop loop;
73 QObject::connect(&accessManager, &QNetworkAccessManager::finished, &loop, &QEventLoop::quit);
74 loop.exec();
75 }
76 //
77 // Data or timeout?
78 LOG(MODULE_NETWORK, _reply->errorString());
79 auto data = _reply->readAll();
80 return data;
81 }

83 void NetworkRequestHelper::AsyncHttpGet(const QString &url, std::function<void(const QByteArray &)> funcPtr)
84 {
85 QNetworkRequest request;
86 request.setUrl(url);
87 setAccessManagerAttributes(request, accessManager);
88 auto reply = accessManager.get(request);
89 QObject::connect(reply, &QNetworkReply::finished, [=]() {
90 {
91#if QT_VERSION >= QT_VERSION_CHECK(5, 15, 0)
92 bool h2Used = reply->attribute(QNetworkRequest::Http2WasUsedAttribute).toBool();
93#else
94 bool h2Used = reply->attribute(QNetworkRequest::HTTP2WasUsedAttribute).toBool();
95#endif
96 if (h2Used)
97 DEBUG(MODULE_NETWORK, "HTTP/2 was used.")
98
99 if (reply->error() != QNetworkReply::NoError)
100 LOG(MODULE_NETWORK, "Network error: " + QString(QMetaEnum::fromType<QNetworkReply::NetworkError>().key(reply->error())))
101
102 funcPtr(reply->readAll());
103 }
104 });
105 }
