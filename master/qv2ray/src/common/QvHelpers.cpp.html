8 const QString GenerateRandomString(int len)
9 {
10 const QString possibleCharacters("abcdefghijklmnopqrstuvwxyz");
11 QString randomString;
12
13 for (int i = 0; i < len; ++i)
14 {
15 uint rand = QRandomGenerator::system()->generate();
16 uint max = static_cast<uint>(possibleCharacters.length());
17 QChar nextChar = possibleCharacters[rand % max];
18 randomString.append(nextChar);
19 }
20
21 return randomString;
22 }

30 QString StringFromFile(QFile &source)
31 {
32 bool wasOpened = source.isOpen();
33 if (!wasOpened)
34 source.open(QFile::ReadOnly);
35 auto byteArray = source.readAll();
36 if (!wasOpened)
37 source.close();
38 //
39 QTextCodec *codec = QTextCodec::codecForName("UTF-8");
40 QTextCodec::ConverterState state;
41 const QString text = codec->toUnicode(byteArray.constData(), byteArray.size(), &state);
42 if (state.invalidChars > 0)
43 {
44 LOG(MODULE_FILEIO, "Not a valid UTF-8 sequence: " + source.fileName())
45 }
46 return state.invalidChars > 0 ? byteArray : text;
47 }

49 bool StringToFile(const QString &text, const QString &targetpath)
50 {
51 bool override = false;
52 {
53 QFileInfo info(targetpath);
54 override = info.exists();
55 if (!override && !info.dir().exists())
56 info.dir().mkpath(info.dir().path());
57 }
58 QSaveFile f{ targetpath };
59 f.open(QIODevice::WriteOnly);
60 f.write(text.toUtf8());
61 f.commit();
62 return override;
63 }

113 // backported from QvPlugin-SSR.
114 QString SafeBase64Encode(const QString &string, bool trim)
115 {
116 QString base64 = string.toUtf8().toBase64();
117 if (trim)
118 {
119 auto tmp = base64.replace(QChar('+'), QChar('-')).replace(QChar('/'), QChar('_'));
120 auto crbedin = tmp.crbegin();
121 auto idx = tmp.length();
122 while (crbedin != tmp.crend() && (*crbedin) == '=') idx -= 1, crbedin++;
123 return idx != tmp.length() ? tmp.remove(idx, tmp.length() - idx) : tmp;
124 }
125 else
126 {
127 return base64.replace(QChar('+'), QChar('-')).replace(QChar('/'), QChar('_'));
128 }
129 }

162 QString FormatBytes(const int64_t b)
163 {
164 auto _bytes = b;
165 char str[64];
166 const char *sizes[5] = { "B", "KB", "MB", "GB", "TB" };
167 int i;
168 double dblByte = _bytes;
169
170 for (i = 0; i < 5 && _bytes >= 1000; i++, _bytes /= 1000) dblByte = _bytes / 1000.0;
171
172 sprintf(str, "%.2f", dblByte);
173 return QString(str) + " " + QString(sizes[i]);
174 }

190 /// This returns a file name without extensions.
191 void DeducePossibleFileName(const QString &baseDir, QString *fileName, const QString &extension)
192 {
193 int i = 1;
194
195 if (!QDir(baseDir).exists())
196 {
197 QDir(baseDir).mkpath(baseDir);
198 LOG(MODULE_FILEIO, "Making path: " + baseDir)
199 }
200
201 while (true)
202 {
203 if (!QFile(baseDir + "/" + fileName + "_" + QSTRN(i) + extension).exists())
204 {
205 *fileName = *fileName + "_" + QSTRN(i);
206 return;
207 }
208 else
209 {
210 DEBUG(MODULE_FILEIO, "File with name: " + *fileName + "_" + QSTRN(i) + extension + " already exists")
211 }
212
213 i++;
214 }
