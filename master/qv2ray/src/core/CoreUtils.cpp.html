13 bool IsComplexConfig(const CONFIGROOT &root)
14 {
15 bool cRouting = root.contains("routing");
16 bool cRule = cRouting && root["routing"].toObject().contains("rules");
17 bool cRules = cRule && root["routing"].toObject()["rules"].toArray().count() > 0;
18 //
19 bool cInbounds = root.contains("inbounds");
20 bool cInboundCount = cInbounds && root["inbounds"].toArray().count() > 0;
21 //
22 bool cOutbounds = root.contains("outbounds");
23 bool cOutboundCount = cOutbounds && root["outbounds"].toArray().count() > 1;
24 return cRules || cInboundCount || cOutboundCount;
25 }

27 bool GetOutboundInfo(const OUTBOUND &out, QString *host, int *port, QString *protocol)
28 {
29 // Set initial values.
30 *host = QObject::tr("N/A");
31 *port = 0;
32 *protocol = out["protocol"].toString(QObject::tr("N/A")).toLower();
33
34 if (*protocol == "vmess" || *protocol == "vless")
35 {
36 auto Server = VMessServerObject::fromJson(QJsonIO::GetValue(out, "settings", "vnext", 0).toObject());
37 *host = Server.address;
38 *port = Server.port;
39 return true;
40 }
41 else if (*protocol == "shadowsocks")
42 {
43 auto Server = ShadowSocksServerObject::fromJson(QJsonIO::GetValue(out, "settings", "servers", 0).toObject());
44 *host = Server.address;
45 *port = Server.port;
46 return true;
47 }
48 else if (*protocol == "socks")
49 {
50 auto Server = SocksServerObject::fromJson(QJsonIO::GetValue(out, "settings", "servers", 0).toObject());
51 *host = Server.address;
52 *port = Server.port;
53 return true;
54 }
55 else if (*protocol == "http")
56 {
57 auto Server = HttpServerObject::fromJson(QJsonIO::GetValue(out, "settings", "servers", 0).toObject());
58 *host = Server.address;
59 *port = Server.port;
60 return true;
61 }
62 else
63 {
64 bool status;
65 auto info = PluginHost->TryGetOutboundInfo(*protocol, out["settings"].toObject(), &status);
66 *host = info.hostName;
67 *port = info.port;
68 return status;
69 }
70 }

83 {
84 if (status != nullptr)
85 *status = false;
86 //
87 //

90 OUTBOUND outBoundRoot = OUTBOUND(item.toObject());

92 int port;
93 QString outboundType = "";
94
95 if (GetOutboundInfo(outBoundRoot, &host, &port, &outboundType))

97 if (status != nullptr)
98 *status = true;

133 const QString GetConnectionProtocolString(const ConnectionId &id)
134 {
135 // Don't bother with the complex connection configs.
136 if (IsComplexConfig(id))
137 {
138 return QV2RAY_SERIALIZATION_COMPLEX_CONFIG_PLACEHOLDER;
139 }
140 CONFIGROOT root = ConnectionManager->GetConnectionRoot(id);
141 QString result;
142 QStringList streamProtocols;
143 auto outbound = root["outbounds"].toArray().first().toObject();
144 result.append(outbound["protocol"].toString());
145
146 if (outbound.contains("streamSettings") && outbound["streamSettings"].toObject().contains("network"))
147 {
148 result.append(" / " + outbound["streamSettings"].toObject()["network"].toString());
149 if (outbound["streamSettings"].toObject().contains("tls"))
150 {
151 result.append(outbound["streamSettings"].toObject()["tls"].toBool() ? " / tls" : "");
152 }
153 }
154
155 return result;
156 }
