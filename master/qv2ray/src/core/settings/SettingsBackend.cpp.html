29 bool CheckSettingsPathAvailability(const QString &_path, bool checkExistingConfig)
30 {
31 auto path = _path;
32
33 if (!path.endsWith("/"))
34 path.append("/");
35
36 // Does not exist.
37 if (!QDir(path).exists())
38 return false;
39
40 {
41 // A temp file used to test file permissions in that folder.
42 QFile testFile(path + ".qv2ray_test_file" + QSTRN(QTime::currentTime().msecsSinceStartOfDay()));
43
44 if (!testFile.open(QFile::OpenModeFlag::ReadWrite))
45 {
46 LOG(MODULE_SETTINGS, "Directory at: " + path + " cannot be used as a valid config file path.")
47 LOG(MODULE_SETTINGS, "---> Cannot create a new file or open a file for writing.")
48 return false;
49 }
50
51 testFile.write("Qv2ray test file, feel free to remove.");
52 testFile.flush();
53 testFile.close();
54
55 if (!testFile.remove())
56 {
57 // This is rare, as we can create a file but failed to remove it.
58 LOG(MODULE_SETTINGS, "Directory at: " + path + " cannot be used as a valid config file path.")
59 LOG(MODULE_SETTINGS, "---> Cannot remove a file.")
60 return false;
61 }
62 }
63
64 if (!checkExistingConfig)
65 {
66 // Just pass the test
67 return true;
68 }
69
70 // Check if an existing config is found.
71 QFile configFile(path + "Qv2ray.conf");
72
73 // No such config file.
74 if (!configFile.exists())
75 return false;
76
77 if (!configFile.open(QIODevice::ReadWrite))
78 {
79 LOG(MODULE_SETTINGS, "File: " + configFile.fileName() + " cannot be opened!")
80 return false;
81 }
82
83 const auto err = VerifyJsonString(StringFromFile(configFile));
84 if (!err.isEmpty())
85 {
86 LOG(MODULE_INIT, "Json parse returns: " + err)
87 return false;
88 }
89
90 // If the file format is valid.
91 const auto conf = JsonFromString(StringFromFile(configFile));
92 LOG(MODULE_SETTINGS, "Found a config file, v=" + conf["config_version"].toString() + " path=" + path)
93 configFile.close();
94 return true;
95 }
