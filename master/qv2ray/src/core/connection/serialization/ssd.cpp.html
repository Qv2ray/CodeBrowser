18 // These below are super strict checking schemes, but necessary.
19#define MUST_EXIST(fieldName) \
20 if (!obj.contains((fieldName)) || obj[(fieldName)].isUndefined() || obj[(fieldName)].isNull()) \
21 { \
22 *logList << QObject::tr("invalid ssd link: json: field %1 must exist").arg(fieldName); \
23 return {}; \
24 }
25#define MUST_PORT(fieldName) \
26 MUST_EXIST(fieldName); \
27 if (int value = obj[(fieldName)].toInt(-1); value < 0 || value > 65535) \
28 { \
29 *logList << QObject::tr("invalid ssd link: json: field %1 must be valid port number"); \
30 return {}; \
31 }
32#define MUST_STRING(fieldName) \
33 MUST_EXIST(fieldName); \
34 if (!obj[(fieldName)].isString()) \
35 { \
36 *logList << QObject::tr("invalid ssd link: json: field %1 must be of type 'string'").arg(fieldName); \
37 return {}; \
38 }
39#define MUST_ARRAY(fieldName) \
40 MUST_EXIST(fieldName); \
41 if (!obj[(fieldName)].isArray()) \
42 { \
43 *logList << QObject::tr("invalid ssd link: json: field %1 must be an array").arg(fieldName); \
44 return {}; \
45 }
46
47#define SERVER_SHOULD_BE_OBJECT(server) \
48 if (!server.isObject()) \
49 { \
50 *logList << QObject::tr("skipping invalid ssd server: server must be an object"); \
51 continue; \
52 }
53#define SHOULD_EXIST(fieldName) \
54 if (serverObject[(fieldName)].isUndefined()) \
55 { \
56 *logList << QObject::tr("skipping invalid ssd server: missing required field %1").arg(fieldName); \
57 continue; \
58 }
59#define SHOULD_STRING(fieldName) \
60 SHOULD_EXIST(fieldName); \
61 if (!serverObject[(fieldName)].isString()) \
62 { \
63 *logList << QObject::tr("skipping invalid ssd server: field %1 should be of type 'string'").arg(fieldName); \
64 continue; \
65 }

67 // std::pair<std::optional<std::pair<QString, QList<std::pair<QString, ShadowSocksServerObject>>>>, QStringList>
68 // AKA
69 //
70 // pair<optional<std::pair<QString, QList<pair<QString, ShadowSocksServerObject>>>>, QStringList>
71 // That is....
72 //
73 // A pair of an error string list, together with some optionally existed pair, which contains a QString for airport name and a List of
74 // pairs that contains a QString for connection name and finally, our ShadowSocksServerObject
75 //
76 QList<QPair<QString, CONFIGROOT>> Deserialize(const QString &uri, QString *groupName, QStringList *logList)
77 {
78 // ssd links should begin with "ssd://"
79 if (!uri.startsWith("ssd://"))
80 {
81 *logList << QObject::tr("Invalid ssd link: should begin with ssd://");
82 return {};
83 }
84
85 // decode base64
86 const auto ssdURIBody = QStringRef(&uri, 6, uri.length() - 6);
87 const auto decodedJSON = SafeBase64Decode(ssdURIBody.toString()).toUtf8();
88
89 if (decodedJSON.length() == 0)
90 {
91 *logList << QObject::tr("Invalid ssd link: base64 parse failed");
92 return {};
93 }
94
95 // parse json
96 QJsonParseError err;
97 QJsonDocument document = QJsonDocument::fromJson(decodedJSON, &err);
98
99 if (document.isNull())
100 {
101 *logList << QObject::tr("Invalid ssd link: json parse failed: ") % err.errorString();
102 return {};
103 }
104
105 // json should be an object
106 if (!document.isObject())
107 {
108 *logList << QObject::tr("Invalid ssd link: found non-object json, aborting");
109 return {};
110 }
111
112 // casting to object
113 QJsonObject obj = document.object();
114 //
115 // obj.airport
116 MUST_STRING("airport");
117 *groupName = obj["airport"].toString();
118 // obj.port
119 MUST_PORT("port");
120 const int port = obj["port"].toInt();
121 // obj.encryption
122 MUST_STRING("encryption");
123 const QString encryption = obj["encryption"].toString();
124
125 // check: rc4-md5 is not supported by v2ray-core
126 // TODO: more checks, including all algorithms
127 if (encryption.toLower() == "rc4-md5")
128 {
129 *logList << QObject::tr("Invalid ssd link: rc4-md5 encryption is not supported by v2ray-core");
130 return {};
131 }
132
133 // obj.password
134 MUST_STRING("password");
135 const QString password = obj["password"].toString();
136 // obj.servers
137 MUST_ARRAY("servers");
138 //
139 QList<QPair<QString, CONFIGROOT>> serverList;
140 //
141
142 // iterate through the servers
143 for (QJsonValueRef server : obj["servers"].toArray())
144 {
145 SERVER_SHOULD_BE_OBJECT(server);
146 QJsonObject serverObject = server.toObject();
147 ShadowSocksServerObject ssObject;
148 // encryption
149 ssObject.method = encryption;
150 // password
151 ssObject.password = password;
152 // address :-> "server"
153 SHOULD_STRING("server");
154 const QString serverAddress = serverObject["server"].toString();
155 ssObject.address = serverAddress;
156
157 // port selection:
158 // normal: use global settings
159 // overriding: use current config
160 if (serverObject["port"].isUndefined())
161 {
162 ssObject.port = port;
163 }
164 else if (auto currPort = serverObject["port"].toInt(-1); (currPort >= 0 && currPort <= 65535))
165 {
166 ssObject.port = currPort;
167 }
168 else
169 {
170 ssObject.port = port;
171 }
172
173 // name decision:
174 // untitled: using server:port as name
175 // entitled: using given name
176 QString nodeName;
177
178 if (serverObject["remarks"].isUndefined())
179 {
180 nodeName = QString("%1:%2").arg(ssObject.address).arg(ssObject.port);
181 }
182 else if (serverObject["remarks"].isString())
183 {
184 nodeName = serverObject["remarks"].toString();
185 }
186 else
187 {
188 nodeName = QString("%1:%2").arg(ssObject.address).arg(ssObject.port);
189 }
190
191 // ratio decision:
192 // unspecified: ratio = 1
193 // specified: use given value
194 double ratio = 1.0;
195
196 if (auto currRatio = serverObject["ratio"].toDouble(-1.0); currRatio != -1.0)
197 {
198 ratio = currRatio;
199 }
200 else if (!serverObject["ratio"].isUndefined())
201 {
202 //*logList << QObject::tr("Invalid ratio encountered. using fallback value.");
203 }
204
205 // format the total name of the node.
206 const QString totalName = QV2RAY_SSD_DEFAULT_NAME_PATTERN.arg(*groupName, nodeName).arg(ratio);
207 // appending to the total list
208 CONFIGROOT root;
209 OUTBOUNDS outbounds;
210 outbounds.append(GenerateOutboundEntry("shadowsocks", GenerateShadowSocksOUT({ ssObject }), {}));
211 JADD(outbounds)
212 serverList.append({ totalName, root });
213 }
214
215 // returns the current result
216 return serverList;
217 }

219#undef MUST_PORT
220#undef MUST_ARRAY

222#undef SERVER_SHOULD_BE_OBJECT
223#undef SHOULD_EXIST
224#undef SHOULD_STRING
