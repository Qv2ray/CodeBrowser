11 // From https://github.com/2dust/v2rayN/wiki/分享链接格式说明(ver-2)
12 const QString Serialize(const StreamSettingsObject &transfer, const VMessServerObject &server, const QString &alias)
13 {
14 QJsonObject vmessUriRoot;
15 // Constant
16 vmessUriRoot["v"] = 2;
17 vmessUriRoot["ps"] = alias;
18 vmessUriRoot["add"] = server.address;
19 vmessUriRoot["port"] = server.port;
20 vmessUriRoot["id"] = server.users.front().id;
21 vmessUriRoot["aid"] = server.users.front().alterId;
22 vmessUriRoot["net"] = transfer.network == "http" ? "h2" : transfer.network;
23 vmessUriRoot["tls"] = transfer.security;
24
25 if (transfer.network == "tcp")
26 {
27 vmessUriRoot["type"] = transfer.tcpSettings.header.type;
28 }
29 else if (transfer.network == "kcp")
30 {
31 vmessUriRoot["type"] = transfer.kcpSettings.header.type;
32 }
33 else if (transfer.network == "quic")
34 {
35 vmessUriRoot["type"] = transfer.quicSettings.header.type;
36 vmessUriRoot["host"] = transfer.quicSettings.security;
37 vmessUriRoot["path"] = transfer.quicSettings.key;
38 }
39 else if (transfer.network == "ws")
40 {
41 auto x = transfer.wsSettings.headers;
42 auto host = x.contains("host");
43 auto CapHost = x.contains("Host");
44 auto realHost = host ? x["host"] : (CapHost ? x["Host"] : "");
45 //
46 vmessUriRoot["host"] = realHost;
47 vmessUriRoot["path"] = transfer.wsSettings.path;
48 }
49 else if (transfer.network == "h2" || transfer.network == "http")
50 {
51 vmessUriRoot["host"] = transfer.httpSettings.host.join(",");
52 vmessUriRoot["path"] = transfer.httpSettings.path;
53 }
54
55 if (!vmessUriRoot.contains("type") || vmessUriRoot["type"].toString().isEmpty())
56 {
57 vmessUriRoot["type"] = "none";
58 }
59
60 //
61 auto vmessPart = Base64Encode(JsonToString(vmessUriRoot, QJsonDocument::JsonFormat::Compact));
62 return "vmess://" + vmessPart;
63 }

65 // This generates global config containing only one outbound....
66 CONFIGROOT Deserialize(const QString &vmessStr, QString *alias, QString *errMessage)
67 {
68#define default CONFIGROOT()
69 QString vmess = vmessStr;
70
71 if (vmess.trimmed() != vmess)
72 {
73 LOG(MODULE_SETTINGS, "VMess string has some prefix/postfix spaces, trimming.")
74 vmess = vmessStr.trimmed();
75 }
76
77 // Reset errMessage
78 *errMessage = "";
79
80 if (!vmess.toLower().startsWith("vmess://"))
81 {
82 *errMessage = QObject::tr("VMess string should start with 'vmess://'");
83 return default;
84 }
85
86 QStringRef vmessJsonB64(&vmess, 8, vmess.length() - 8);
87 auto b64Str = vmessJsonB64.toString();
88
89 if (b64Str.isEmpty())
90 {
91 *errMessage = QObject::tr("VMess string should be a valid base64 string");
92 return default;
93 }
94
95 auto vmessString = SafeBase64Decode(b64Str);
96 auto jsonErr = VerifyJsonString(vmessString);
97
98 if (!jsonErr.isEmpty())
99 {
100 *errMessage = jsonErr;
101 return default;
102 }
103
104 auto vmessConf = JsonFromString(vmessString);
105
106 if (vmessConf.isEmpty())
107 {
108 *errMessage = QObject::tr("JSON should not be empty");
109 return default;
110 }
111
112 // --------------------------------------------------------------------------------------
113 CONFIGROOT root;
114 QString ps, add, id, net, type, host, path, tls;
115 int port, aid;
116 //
117 // __vmess_checker__func(key, values)
118 //
119 // - Key = Key in JSON and the variable name.
120 // - Values = Candidate variable list, if not match, the first one is used as default.
121 //
122 // - [[val.size() <= 1]] is used when only the default value exists.
123 //
124 // - It can be empty, if so, if the key is not in the JSON, or the value is empty, report an error.
125 // - Else if it contains one thing. if the key is not in the JSON, or the value is empty, use that one.
126 // - Else if it contains many things, when the key IS in the JSON but not within the THINGS, use the first in the THINGS
127 // - Else, use the JSON value
128 //
129#define __vmess_checker__func(key, values) \
130 { \
131 auto val = QStringList() values; \
132 if (vmessConf.contains(#key) && !vmessConf[#key].toVariant().toString().trimmed().isEmpty() && \
133 (val.size() <= 1 || val.contains(vmessConf[#key].toVariant().toString()))) \
134 { \
135 key = vmessConf[#key].toVariant().toString(); \
136 } \
137 else if (!val.isEmpty()) \
138 { \
139 key = val.first(); \
140 DEBUG(MODULE_IMPORT, "Using key \"" #key "\" from the first candidate list: " + key) \
141 } \
142 else \
143 { \
144 *errMessage = QObject::tr(#key " does not exist."); \
145 LOG(MODULE_IMPORT, "Cannot process \"" #key "\" since it's not included in the json object.") \
146 LOG(MODULE_IMPORT, " --> values: " + val.join(";")) \
147 LOG(MODULE_IMPORT, " --> PS: " + ps) \
148 } \
149 }
150
151 // vmess v1 upgrader
152 if (!vmessConf.contains("v"))
153 {
154 LOG(MODULE_IMPORT, "Detected deprecated vmess v1. Trying to upgrade...")
155 if (const auto network = vmessConf["net"].toString(); network == "ws" || network == "h2")
156 {
157 const QStringList hostComponents = vmessConf["host"].toString().replace(" ", "").split(";");
158 if (const auto nParts = hostComponents.length(); nParts == 1)
159 vmessConf["path"] = hostComponents[0], vmessConf["host"] = "";
160 else if (nParts == 2)
161 vmessConf["path"] = hostComponents[0], vmessConf["host"] = hostComponents[1];
162 else
163 vmessConf["path"] = "/", vmessConf["host"] = "";
164 }
165 }
166
167 // Strict check of VMess protocol, to check if the specified value
168 // is in the correct range.
169 //
170 // Get Alias (AKA ps) from address and port.
171 {
172 // Some idiot vmess:// links are using alterId...
173 aid = vmessConf.contains("aid") ? vmessConf.value("aid").toInt(VMESS_USER_ALTERID_DEFAULT) :
174 vmessConf.value("alterId").toInt(VMESS_USER_ALTERID_DEFAULT);
175 //
176 //
177 __vmess_checker__func(ps, << vmessConf["add"].toVariant().toString() + ":" + vmessConf["port"].toVariant().toString()); //
178 __vmess_checker__func(add, nothing); //
179 __vmess_checker__func(id, nothing); //
180 __vmess_checker__func(type, << "none" //
181 << "http" //
182 << "srtp" //
183 << "utp" //
184 << "wechat-video"); //
185 //
186 __vmess_checker__func(net, << "tcp" //
187 << "http" //
188 << "h2" //
189 << "ws" //
190 << "kcp" //
191 << "domainsocket" //
192 << "quic"); //
193 //
194 __vmess_checker__func(tls, << "none" //
195 << "tls"); //
196 path = vmessConf.contains("path") ? vmessConf["path"].toVariant().toString() : (net == "quic" ? "" : "/");
197 host = vmessConf.contains("host") ? vmessConf["host"].toVariant().toString() : (net == "quic" ? "none" : "");
198 }
199
200 // Repect connection type rather than obfs type
201 if (QStringList{ "srtp", "utp", "wechat-video" }.contains(type)) //
202 { //
203 if (net != "quic" && net != "kcp") //
204 { //
205 LOG(MODULE_CONNECTION, "Reset obfs settings from " + type + " to none") //
206 type = "none"; //
207 } //
208 }
209
210 port = vmessConf["port"].toVariant().toInt();
211 aid = vmessConf["aid"].toVariant().toInt();
212 //
213 // Apply the settings.
214 // User
215 VMessServerObject::UserObject user;
216 user.id = id;
217 user.alterId = aid;
218 //
219 // Server
220 VMessServerObject serv;
221 serv.port = port;
222 serv.address = add;
223 serv.users.push_back(user);
224 //
225 //
226 // Stream Settings
227 StreamSettingsObject streaming;
228
229 if (net == "tcp")
230 {
231 streaming.tcpSettings.header.type = type;
232 }
233 else if (net == "http" || net == "h2")
234 {
235 // Fill hosts for HTTP
236 for (const auto &_host : host.split(','))
237 {
238 if (!_host.isEmpty())
239 {
240 streaming.httpSettings.host << _host.trimmed();
241 }
242 }
243
244 streaming.httpSettings.path = path;
245 }
246 else if (net == "ws")
247 {
248 if (!host.isEmpty())
249 streaming.wsSettings.headers["Host"] = host;
250 streaming.wsSettings.path = path;
251 }
252 else if (net == "kcp")
253 {
254 streaming.kcpSettings.header.type = type;
255 }
256 else if (net == "domainsocket")
257 {
258 streaming.dsSettings.path = path;
259 }
260 else if (net == "quic")
261 {
262 streaming.quicSettings.security = host;
263 streaming.quicSettings.header.type = type;
264 streaming.quicSettings.key = path;
265 }
266
267 // FIXME: makeshift patch for #290.
268 // to be rewritten after refactoring.
269 if (tls == "tls" && host != "" && (net == "tcp" || net == "ws"))
270 {
271 streaming.tlsSettings.serverName = host;
272 streaming.tlsSettings.allowInsecure = false;
273 }
274
275 streaming.security = tls;
276 //
277 // Network type
278 // NOTE(DuckSoft): Damn vmess:// just don't write 'http' properly
279 if (net == "h2")
280 net = "http";
281 streaming.network = net;
282 //
283 // VMess root config
284 OUTBOUNDSETTING vConf;
285 QJsonArray vnextArray;
286 vnextArray.append(serv.toJson());
287 vConf["vnext"] = vnextArray;
288 auto outbound = GenerateOutboundEntry("vmess", vConf, streaming.toJson(), {}, "0.0.0.0", OUTBOUND_TAG_PROXY);
289 //
290 root["outbounds"] = QJsonArray{ outbound };
291 // If previous alias is empty, just the PS is needed, else, append a "_"
292 *alias = alias->trimmed().isEmpty() ? ps : *alias + "_" + ps;
293 return root;
294#undef default
295 }
