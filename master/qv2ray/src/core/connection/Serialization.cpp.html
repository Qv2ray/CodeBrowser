13 QList<QPair<QString, CONFIGROOT>> ConvertConfigFromString(const QString &link, QString *aliasPrefix, QString *errMessage,
14 QString *newGroupName)
15 {
16 const auto mkAllowInsecure = [](QJsonObject &conf) {
17 auto allowI = GlobalConfig.advancedConfig.setAllowInsecure;
18 auto allowSR = GlobalConfig.advancedConfig.setSessionResumption;
19 if (allowI || allowSR)
20 {
21 QJsonIO::SetValue(conf, allowI, "outbounds", 0, "streamSettings", "tlsSettings", "allowInsecure");
22 QJsonIO::SetValue(conf, !allowSR, "outbounds", 0, "streamSettings", "tlsSettings", "disableSessionResumption");
23 }
24 };
25
26 QList<QPair<QString, CONFIGROOT>> connectionConf;
27 if (link.startsWith("vmess://") && link.contains("@"))
28 {
29 auto conf = vmess_new::Deserialize(link, aliasPrefix, errMessage);
30 mkAllowInsecure(conf);
31 connectionConf << QPair{ *aliasPrefix, conf };
32 }
33 else if (link.startsWith("vmess://"))
34 {
35 auto conf = vmess::Deserialize(link, aliasPrefix, errMessage);
36 mkAllowInsecure(conf);
37 connectionConf << QPair{ *aliasPrefix, conf };
38 }
39 else if (link.startsWith("ss://") && !link.contains("plugin="))
40 {
41 auto conf = ss::Deserialize(link, aliasPrefix, errMessage);
42 connectionConf << QPair{ *aliasPrefix, conf };
43 }
44 else if (link.startsWith("ssd://"))
45 {
46 QStringList errMessageList;
47 connectionConf << ssd::Deserialize(link, newGroupName, &errMessageList);
48 *errMessage = errMessageList.join(NEWLINE);
49 }
50 else
51 {
52 bool ok = false;
53 const auto configs = PluginHost->TryDeserializeShareLink(link, aliasPrefix, errMessage, newGroupName, &ok);
54 if (ok)
55 {
56 errMessage->clear();
57 for (const auto &[_alias, _protocol, _outbound] : configs)
58 {
59 CONFIGROOT root;
60 auto outbound = GenerateOutboundEntry(_protocol, OUTBOUNDSETTING(_outbound), {});
61 QJsonIO::SetValue(root, outbound, "outbounds", 0);
62 connectionConf << QPair{ _alias, root };
63 }
64 }
65 else if (errMessage->isEmpty())
66 {
67 *errMessage = QObject::tr("Unsupported share link format.");
68 }
69 }
70
71 return connectionConf;
72 }

74 const QString ConvertConfigToString(const ConnectionGroupPair &identifier, bool isSip002)
75 {
76 auto alias = GetDisplayName(identifier.connectionId);
77 if (IsComplexConfig(identifier.connectionId))
78 {
79 DEBUG(MODULE_CONNECTION, "Ignored an complex config: " + alias)
80 return QV2RAY_SERIALIZATION_COMPLEX_CONFIG_PLACEHOLDER;
81 }
82 auto server = ConnectionManager->GetConnectionRoot(identifier.connectionId);
83 return ConvertConfigToString(alias, GetDisplayName(identifier.groupId), server, isSip002);
84 }
85
86 const QString ConvertConfigToString(const QString &alias, const QString &groupName, const CONFIGROOT &server, bool isSip002)
87 {
88 const auto outbound = OUTBOUND(server["outbounds"].toArray().first().toObject());
89 const auto type = outbound["protocol"].toString();
90 const auto &settings = outbound["settings"].toObject();
91 QString sharelink = "";
92 if (type == "vmess")
93 {
94 auto vmessServer = VMessServerObject::fromJson(settings["vnext"].toArray().first().toObject());
95 auto transport = StreamSettingsObject::fromJson(outbound["streamSettings"].toObject());
96 if (GlobalConfig.uiConfig.useOldShareLinkFormat)
97 {
98 sharelink = vmess::Serialize(transport, vmessServer, alias);
99 }
100 else
101 {
102 sharelink = vmess_new::Serialize(transport, vmessServer, alias);
103 }
104 }
105 else if (type == "shadowsocks")
106 {
107 auto ssServer = ShadowSocksServerObject::fromJson(settings["servers"].toArray().first().toObject());
108 sharelink = ss::Serialize(ssServer, alias, isSip002);
109 }
110 else
111 {
112 if (type.isEmpty())
113 {
114 DEBUG(MODULE_CONNECTION, "WARNING: Empty outbound type.")
115 }
116 else
117 {
118 bool ok = false;
119 sharelink = PluginHost->TrySerializeShareLink(type, settings, alias, groupName, &ok);
120 Q_UNUSED(ok)
121 }
122 }
123
124 return sharelink;
125 }
