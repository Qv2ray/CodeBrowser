8 CONFIGROOT ConvertConfigFromFile(const QString &sourceFilePath, bool importComplex)
9 {
10 auto root = CONFIGROOT(JsonFromString(StringFromFile(sourceFilePath)));
11
12 if (!importComplex)
13 {
14 root.remove("inbounds");
15 root.remove("routing");
16 root.remove("dns");
17 }
18
19 root.remove("log");
20 root.remove("api");
21 root.remove("stats");
22 return root;
23 }

25 QList<QPair<QString, CONFIGROOT>> GetConnectionConfigFromSubscription(const QByteArray &arr, const QString &groupName)
26 {
27 QList<QPair<QString, CONFIGROOT>> subscriptionContent;
28 auto subscriptionLines = SplitLines(TryDecodeSubscriptionString(arr));
29 for (const auto &line : subscriptionLines)
30 {
31 QString __alias;
32 QString __errMessage;
33 // Assign a group name, to pass the name check.
34 QString __groupName = groupName;
35 const auto connectionConfigMap = ConvertConfigFromString(line.trimmed(), &__alias, &__errMessage, &__groupName);
36 if (!__errMessage.isEmpty())
37 LOG(MODULE_SUBSCRIPTION, "Error: " + __errMessage)
38 subscriptionContent << connectionConfigMap;
39 }
40 return subscriptionContent;
41 }
