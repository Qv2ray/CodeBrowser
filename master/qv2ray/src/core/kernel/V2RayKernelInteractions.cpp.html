11 std::pair<bool, std::optional<QString>> V2RayKernelInstance::CheckAndSetCoreExecutableState(const QString &vCorePath)
12 {
13#ifdef Q_OS_UNIX
14 // For Linux/macOS users: if they cannot execute the core,
15 // then we shall grant the permission to execute it.
16
17 QFile coreFile(vCorePath);
18
19 if (!coreFile.permissions().testFlag(QFileDevice::ExeUser))
20 {
21 DEBUG(MODULE_VCORE, "Core file not executable. Trying to enable.")
22 const auto result = coreFile.setPermissions(coreFile.permissions().setFlag(QFileDevice::ExeUser));
23 if (!result)
24 {
25 DEBUG(MODULE_VCORE, "Failed to enable executable permission.")
26 const auto message = tr("Core file is lacking executable permission for the current user.") % //
27 tr("Qv2ray tried to set, but failed because permission denied.");
28 return { false, message };
29 }
30 else
31 {
32 DEBUG(MODULE_VCORE, "Core executable permission set.")
33 }
34 }
35 else
36 {
37 DEBUG(MODULE_VCORE, "Core file is executable.")
38 }
39
40 // Also do the same thing for v2ctl.
41 // TODO: Simplify This / Extract This Creepy Thing
42 const auto coreControlFilePath =
43 QDir::cleanPath(QFileInfo(coreFile).absoluteDir().path() + QDir::separator() + "v2ctl" QV2RAY_EXECUTABLE_FILENAME_SUFFIX);
44
45 QFile coreControlFile(coreControlFilePath);
46 if (!coreControlFile.permissions().testFlag(QFileDevice::ExeUser))
47 {
48 DEBUG(MODULE_VCORE, "Core control file not executable. Trying to enable.")
49 const auto result = coreControlFile.setPermissions(coreFile.permissions().setFlag(QFileDevice::ExeUser));
50
51 if (!result)
52 {
53 DEBUG(MODULE_VCORE, "Failed to enable executable permission for core control.")
54 const auto message = tr("Core control file is lacking executable permission for the current user.") + //
55 tr("Qv2ray tried to set, but failed because permission denied.");
56 return { false, message };
57 }
58 else
59 {
60 DEBUG(MODULE_VCORE, "Core control executable permission set.")
61 }
62 }
63 else
64 {
65 DEBUG(MODULE_VCORE, "Core control file is executable.")
66 }
67
68 return { true, std::nullopt };
69#endif
70
71 // For Windows and other users: just skip this check.
72 DEBUG(MODULE_VCORE, "Skipped check and set core executable state.")
73 return { true, tr("Check is skipped") };
74 }

76 bool V2RayKernelInstance::ValidateKernel(const QString &vCorePath, const QString &vAssetsPath, QString *message)
77 {
78 QFile coreFile(vCorePath);
79
80 if (!coreFile.exists())
81 {
82 DEBUG(MODULE_VCORE, "V2Ray core file cannot be found.")
83 *message = tr("V2Ray core executable not found.");
84 return false;
85 }
86
87 // Use open() here to prevent `executing` a folder, which may have the
88 // same name as the V2Ray core.
89 if (!coreFile.open(QFile::ReadOnly))
90 {
91 DEBUG(MODULE_VCORE, "V2Ray core file cannot be opened, possibly be a folder?")
92 *message = tr("V2Ray core file cannot be opened, please ensure there's a file instead of a folder.");
93 return false;
94 }
95
96 coreFile.close();
97
98 // Get Core ABI.
99 const auto [abi, err] = kernel::abi::deduceKernelABI(vCorePath);
100 if (err)
101 {
102 LOG(MODULE_VCORE, "Core ABI deduction failed: " + ACCESS_OPTIONAL_VALUE(err))
103 *message = ACCESS_OPTIONAL_VALUE(err);
104 return false;
105 }
106 LOG(MODULE_VCORE, "Core ABI: " + kernel::abi::abiToString(ACCESS_OPTIONAL_VALUE(abi)))
107
108 // Get Compiled ABI
109 auto compiledABI = kernel::abi::COMPILED_ABI_TYPE;
110 LOG(MODULE_VCORE, "Host ABI: " + kernel::abi::abiToString(compiledABI))
111
112 // Check ABI Compatibility.
113 switch (kernel::abi::checkCompatibility(compiledABI, ACCESS_OPTIONAL_VALUE(abi)))
114 {
115 case kernel::abi::ABI_NOPE:
116 {
117 LOG(MODULE_VCORE, "Host is incompatible with core")
118 *message = tr("V2Ray core is incompatible with your platform.\r\n" //
119 "Expected core ABI is %1, but got actual %2.\r\n" //
120 "Maybe you have downloaded the wrong core?")
121 .arg(kernel::abi::abiToString(compiledABI), kernel::abi::abiToString(ACCESS_OPTIONAL_VALUE(abi)));
122 return false;
123 }
124 case kernel::abi::ABI_MAYBE:
125 {
126 LOG(MODULE_VCORE, "WARNING: Host maybe incompatible with core");
127 break;
128 }
129 case kernel::abi::ABI_PERFECT:
130 {
131 LOG(MODULE_VCORE, "Host is compatible with core");
132 break;
133 }
134 }
135
136 // Check executable permissions.
137 const auto [isExecutableOk, strExecutableErr] = CheckAndSetCoreExecutableState(vCorePath);
138 if (!isExecutableOk)
139 {
140 *message = strExecutableErr.value_or("");
141 return false;
142 }
143
144 //
145 // Check file existance.
146 // From: https://www.v2fly.org/chapter_02/env.html#asset-location
147 //
148 bool hasGeoIP = FileExistsIn(QDir(vAssetsPath), "geoip.dat");
149 bool hasGeoSite = FileExistsIn(QDir(vAssetsPath), "geosite.dat");
150
151 if (!hasGeoIP && !hasGeoSite)
152 {
153 DEBUG(MODULE_VCORE, "V2Ray assets path contains none of those two files.")
154 *message = tr("V2Ray assets path is not valid.");
155 return false;
156 }
157
158 if (!hasGeoIP)
159 {
160 DEBUG(MODULE_VCORE, "No geoip.dat in assets path, aborting.")
161 *message = tr("No geoip.dat in assets path.");
162 return false;
163 }
164
165 if (!hasGeoSite)
166 {
167 DEBUG(MODULE_VCORE, "No geosite.dat in assets path, aborting.")
168 *message = tr("No geosite.dat in assets path.");
169 return false;
170 }
171
172 // Check if V2Ray core returns a version number correctly.
173 QProcess proc;
174#ifdef Q_OS_WIN32
175 // nativeArguments are required for Windows platform, without a
176 // reason...
177 proc.setProcessChannelMode(QProcess::MergedChannels);
178 proc.setProgram(vCorePath);
179 proc.setNativeArguments("--version");
180 proc.start();
181#else
182 proc.start(vCorePath, { "--version" });
183#endif
184 proc.waitForStarted();
185 proc.waitForFinished();
186 auto exitCode = proc.exitCode();
187
188 if (exitCode != 0)
189 {
190 DEBUG(MODULE_VCORE, "VCore failed with an exit code: " + QSTRN(exitCode))
191 *message = tr("V2Ray core failed with an exit code: ") + QSTRN(exitCode);
192 return false;
193 }
194
195 QString output = proc.readAllStandardOutput();
196 LOG(MODULE_VCORE, "V2Ray output: " + SplitLines(output).join(";"))
197
198 if (SplitLines(output).isEmpty())
199 {
200 *message = tr("V2Ray core returns empty string.");
201 return false;
202 }
203
204 *message = SplitLines(output).first();
205 return true;
206 }

208 bool V2RayKernelInstance::ValidateConfig(const QString &path)
209 {
210 QString V2RayCheckResult;
211 auto kernelPath = GlobalConfig.kernelConfig.KernelPath();
212 auto assetsPath = GlobalConfig.kernelConfig.AssetsPath();
213 if (ValidateKernel(kernelPath, assetsPath, &V2RayCheckResult))
214 {
215 DEBUG(MODULE_VCORE, "V2Ray version: " + V2RayCheckResult)
216 // Append assets location env.
217 QProcessEnvironment env = QProcessEnvironment::systemEnvironment();
218 env.insert("V2RAY_LOCATION_ASSET", assetsPath);
219 //
220 QProcess process;
221 process.setProcessEnvironment(env);
222 DEBUG(MODULE_VCORE, "Starting V2Ray core with test options")
223 process.start(kernelPath, QStringList{ "-test", "-config", path }, QIODevice::ReadWrite | QIODevice::Text);
224 process.waitForFinished();
225
226 if (process.exitCode() != 0)
227 {
228 QString output = QString(process.readAllStandardOutput());
229 QvMessageBoxWarn(nullptr, tr("Configuration Error"), output.mid(output.indexOf("anti-censorship.") + 17));
230 return false;
231 }
232 else
233 {
234 DEBUG(MODULE_VCORE, "Config file check passed.")
235 return true;
236 }
237 }
238 else
239 {
240 QvMessageBoxWarn(nullptr, tr("Cannot start V2Ray"), //
241 tr("V2Ray core settings is incorrect.") + NEWLINE + NEWLINE + //
242 tr("The error is: ") + NEWLINE + V2RayCheckResult);
243 return false;
244 }

247 V2RayKernelInstance::V2RayKernelInstance(QObject *parent) : QObject(parent)
248 {
249 vProcess = new QProcess();
250 connect(vProcess, &QProcess::readyReadStandardOutput, this,
251 [&]() { emit OnProcessOutputReadyRead(vProcess->readAllStandardOutput().trimmed()); });
252 connect(vProcess, &QProcess::stateChanged, [&](QProcess::ProcessState state) {
253 DEBUG(MODULE_VCORE, "V2Ray kernel process status changed: " + QVariant::fromValue(state).toString())
254
255 // If V2Ray crashed AFTER we start it.
256 if (KernelStarted && state == QProcess::NotRunning)
257 {
258 LOG(MODULE_VCORE, "V2Ray kernel crashed.")
259 StopConnection();
260 emit OnProcessErrored("V2Ray kernel crashed.");
261 }
262 });
263 apiWorker = new APIWorker();
264 qRegisterMetaType<StatisticsType>();
265 qRegisterMetaType<QMap<StatisticsType, QvStatsSpeed>>();
266 connect(apiWorker, &APIWorker::onAPIDataReady, this, &V2RayKernelInstance::OnNewStatsDataArrived);
267 KernelStarted = false;
268 }

270 std::optional<QString> V2RayKernelInstance::StartConnection(const CONFIGROOT &root)
271 {
272 if (KernelStarted)
273 {
274 LOG(MODULE_VCORE, "Status is invalid, expect STOPPED when calling StartConnection")
275 return tr("Invalid V2Ray Instance Status.");
276 }
277
278 // Write the final configuration to the disk.
279 QString json = JsonToString(root);
280 StringToFile(json, QV2RAY_GENERATED_FILE_PATH);
281 //
282 auto filePath = QV2RAY_GENERATED_FILE_PATH;
283
284 if (ValidateConfig(filePath))
285 {
286 QProcessEnvironment env = QProcessEnvironment::systemEnvironment();
287 env.insert("V2RAY_LOCATION_ASSET", GlobalConfig.kernelConfig.AssetsPath());
288 vProcess->setProcessEnvironment(env);
289 vProcess->start(GlobalConfig.kernelConfig.KernelPath(), { "-config", filePath }, QIODevice::ReadWrite | QIODevice::Text);
290 vProcess->waitForStarted();
291 DEBUG(MODULE_VCORE, "V2Ray core started.")
292 KernelStarted = true;
293
294 QMap<bool, QMap<QString, QString>> tagProtocolMap;
295 for (const auto isOutbound : { GlobalConfig.uiConfig.graphConfig.useOutboundStats, false })
296 {
297 for (const auto &item : root[isOutbound ? "outbounds" : "inbounds"].toArray())
298 {
299 const auto tag = item.toObject()["tag"].toString("");
300 if (tag == API_TAG_INBOUND)
301 continue;
302 if (tag.isEmpty())
303 {
304 LOG(MODULE_VCORE, "Ignored inbound with empty tag.")
305 continue;
306 }
307 tagProtocolMap[isOutbound][tag] = item.toObject()["protocol"].toString();
308 }
309 }
310
311 apiEnabled = false;
312 if (StartupOption.noAPI)
313 {
314 LOG(MODULE_VCORE, "API has been disabled by the command line arguments")
315 }
316 else if (!GlobalConfig.kernelConfig.enableAPI)
317 {
318 LOG(MODULE_VCORE, "API has been disabled by the global config option")
319 }
320 else if (tagProtocolMap.isEmpty())
321 {
322 LOG(MODULE_VCORE, "RARE: API is disabled since no inbound tags configured. This is usually caused by a bad complex config.")
323 }
324 else
325 {
326 DEBUG(MODULE_VCORE, "Starting API")
327 apiWorker->StartAPI(tagProtocolMap);
328 apiEnabled = true;
329 }
330
331 return std::nullopt;
332 }
333 else
334 {
335 KernelStarted = false;
336 return tr("V2Ray kernel failed to start.");
337 }

340 void V2RayKernelInstance::StopConnection()
341 {
342 if (apiEnabled)
343 {
344 apiWorker->StopAPI();
345 apiEnabled = false;
346 }
347
348 // Set this to false BEFORE close the Process, since we need this flag
349 // to capture the real kernel CRASH
350 KernelStarted = false;
351 vProcess->close();
352 // Block until V2Ray core exits
353 // Should we use -1 instead of waiting for 30secs?
354 vProcess->waitForFinished();
355 }

357 V2RayKernelInstance::~V2RayKernelInstance()
358 {
359 if (KernelStarted)
360 {
361 StopConnection();
362 }
363
364 delete apiWorker;
365 delete vProcess;
366 }
