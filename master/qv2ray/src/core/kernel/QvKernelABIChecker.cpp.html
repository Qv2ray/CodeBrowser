7 QvKernelABICompatibility checkCompatibility(QvKernelABIType hostType, QvKernelABIType targetType)
8 {
9#ifndef QV2RAY_TRUSTED_ABI
10 switch (hostType)
11 {
12 case ABI_WIN32:
13 case ABI_MACH_O:
14 case ABI_ELF_AARCH64:
15 case ABI_ELF_ARM:
16 case ABI_ELF_X86: return targetType == hostType ? ABI_PERFECT : ABI_NOPE;
17 case ABI_ELF_X86_64: return targetType == hostType ? ABI_PERFECT : targetType == ABI_ELF_X86 ? ABI_MAYBE : ABI_NOPE;
18 case ABI_ELF_OTHER: return targetType == hostType ? ABI_PERFECT : ABI_MAYBE;
19 case ABI_TRUSTED: return ABI_PERFECT;
20 default: return ABI_MAYBE;
21 }
22#else
23 return ABI_PERFECT;
24#endif
25 }

27 std::pair<std::optional<QvKernelABIType>, std::optional<QString>> deduceKernelABI(const QString &pathCoreExecutable)
28 {
29#ifdef QV2RAY_TRUSTED_ABI
30 return { QvKernelABIType::ABI_TRUSTED, std::nullopt };
31#else
32 QFile file(pathCoreExecutable);
33 if (!file.exists())
34 return { std::nullopt, QObject::tr("core executable file %1 does not exist").arg(pathCoreExecutable) };
35
36 if (!file.open(QIODevice::ReadOnly))
37 return { std::nullopt, QObject::tr("cannot open core executable file %1 in read-only mode").arg(pathCoreExecutable) };
38
39 if (file.atEnd())
40 return { std::nullopt, QObject::tr("core executable file %1 is an empty file").arg(pathCoreExecutable) };
41
42 const QByteArray arr = file.read(0x100);
43 if (arr.length() < 0x100)
44 return { std::nullopt, QObject::tr("core executable file %1 is too short to be executed").arg(pathCoreExecutable) };
45
46 if (quint32 elfMagicMaybe; QDataStream(arr) >> elfMagicMaybe, 0x7F454C46u == elfMagicMaybe)
47 {
48 quint16 elfInstruction;
49 if (QDataStream stream(arr); stream.skipRawData(0x12), stream >> elfInstruction, elfInstruction == 0x3E00u)
50 return { QvKernelABIType::ABI_ELF_X86_64, std::nullopt };
51 else if (elfInstruction == 0x0300u)
52 return { QvKernelABIType::ABI_ELF_X86, std::nullopt };
53 else if (elfInstruction == 0xB700u)
54 return { QvKernelABIType::ABI_ELF_AARCH64, std::nullopt };
55 else if (elfInstruction == 0x2800u)
56 return { QvKernelABIType::ABI_ELF_ARM, std::nullopt };
57 else
58 return { QvKernelABIType::ABI_ELF_OTHER, std::nullopt };
59 }
60 else if (quint16 dosMagicMaybe; QDataStream(arr) >> dosMagicMaybe, dosMagicMaybe == 0x4D5Au)
61 return { QvKernelABIType::ABI_WIN32, std::nullopt };
62 else if (quint32 machOMagicMaybe; QDataStream(arr) >> machOMagicMaybe, machOMagicMaybe == 0xCFFAEDFEu)
63 return { QvKernelABIType::ABI_MACH_O, std::nullopt };
64 else
65 return { std::nullopt, QObject::tr("cannot deduce the type of core executable file %1").arg(pathCoreExecutable) };
66#endif
67 }

69 QString abiToString(QvKernelABIType abi)
70 {
71 switch (abi)
72 {
73 case ABI_WIN32: return QObject::tr("Windows PE executable");
74 case ABI_MACH_O: return QObject::tr("macOS Mach-O executable");
75 case ABI_ELF_X86: return QObject::tr("ELF x86 executable");
76 case ABI_ELF_X86_64: return QObject::tr("ELF amd64 executable");
77 case ABI_ELF_AARCH64: return QObject::tr("ELF arm64 executable");
78 case ABI_ELF_ARM: return QObject::tr("ELF arm executable");
79 case ABI_ELF_OTHER: return QObject::tr("other ELF executable");
80 case ABI_TRUSTED: return QObject::tr("trusted abi");
81 default: return QObject::tr("unknown abi");
82 }
