12 QvConfigHandler::QvConfigHandler(QObject *parent) : QObject(parent)
13 {
14 asyncRequestHelper = new NetworkRequestHelper(this);
15 DEBUG(MODULE_CORE_HANDLER, "ConnectionHandler Constructor.")
16 const auto connectionJson = JsonFromString(StringFromFile(QV2RAY_CONFIG_DIR + "connections.json"));
17 const auto groupJson = JsonFromString(StringFromFile(QV2RAY_CONFIG_DIR + "groups.json"));
18 //
19 for (const auto &connectionId : connectionJson.keys())
20 {
21 connections.insert(ConnectionId{ connectionId }, ConnectionObject::fromJson(connectionJson.value(connectionId).toObject()));
22 }
23 //
24 for (const auto &groupId : groupJson.keys())
25 {
26 auto groupObject = GroupObject::fromJson(groupJson.value(groupId).toObject());
27 if (groupObject.displayName.isEmpty())
28 {
29 groupObject.displayName = tr("Group: %1").arg(GenerateRandomString(5));
30 }
31 groups.insert(GroupId{ groupId }, groupObject);
32 for (const auto &connId : groupObject.connections)
33 {
34 connections[connId].__qvConnectionRefCount++;
35 }
36 }
37 //
38 for (const auto &id : connections.keys())
39 {
40 auto const &connectionObject = connections.value(id);
41 if (connectionObject.__qvConnectionRefCount == 0)
42 {
43 QFile connectionFile(QV2RAY_CONNECTIONS_DIR + id.toString() + QV2RAY_CONFIG_FILE_EXTENSION);
44 if (connectionFile.exists())
45 {
46 if (!connectionFile.remove())
47 LOG(MODULE_CONNECTION, "Failed to remove connection config file")
48 }
49 connections.remove(id);
50 LOG(MODULE_CORE_HANDLER, "Dropped connection id: " + id.toString() + " since it's not in a group")
51 }
52 else
53 {
54 const auto connectionFilePath = QV2RAY_CONNECTIONS_DIR + id.toString() + QV2RAY_CONFIG_FILE_EXTENSION;
55 connectionRootCache[id] = CONFIGROOT(JsonFromString(StringFromFile(connectionFilePath)));
56 DEBUG(MODULE_CORE_HANDLER, "Loaded connection id: " + id.toString() + " into cache.")
57 }
58 }
59
60 // Force default group name.
61 if (!groups.contains(DefaultGroupId))
62 {
63 groups.insert(DefaultGroupId, {});
64 groups[DefaultGroupId].displayName = tr("Default Group");
65 groups[DefaultGroupId].isSubscription = false;
66 }
67 //
68 kernelHandler = new KernelInstanceHandler(this);
69 connect(kernelHandler, &KernelInstanceHandler::OnCrashed, this, &QvConfigHandler::OnKernelCrashed_p);
70 connect(kernelHandler, &KernelInstanceHandler::OnStatsDataAvailable, this, &QvConfigHandler::OnStatsDataArrived_p);
71 connect(kernelHandler, &KernelInstanceHandler::OnKernelLogAvailable, this, &QvConfigHandler::OnKernelLogAvailable);
72 connect(kernelHandler, &KernelInstanceHandler::OnConnected, this, &QvConfigHandler::OnConnected);
73 connect(kernelHandler, &KernelInstanceHandler::OnDisconnected, this, &QvConfigHandler::OnDisconnected);
74 //
75 tcpingHelper = new LatencyTestHost(5, this);
76 connect(tcpingHelper, &LatencyTestHost::OnLatencyTestCompleted, this, &QvConfigHandler::OnLatencyDataArrived_p);
77 //
78 // Save per 1 hour.
79 saveTimerId = startTimer(1 * 60 * 60 * 1000);
80 // Do not ping all...
81 pingConnectionTimerId = startTimer(60 * 1000);
82 }

203 bool QvConfigHandler::RemoveConnectionFromGroup(const ConnectionId &id, const GroupId &gid)
204 {
205 CheckValidId(id, false);
206 LOG(MODULE_CONNECTION, "Removing connection : " + id.toString())
207 if (groups[gid].connections.contains(id))
208 {
209 auto removedEntries = groups[gid].connections.removeAll(id);
210 if (removedEntries > 1)
211 {
212 LOG(MODULE_CONNECTION, "Found same connection occured multiple times in a group.")
213 }
214 // Decrease reference count.
215 connections[id].__qvConnectionRefCount -= removedEntries;
216 }
217
218 if (GlobalConfig.autoStartId == ConnectionGroupPair{ id, gid })
219 {
220 GlobalConfig.autoStartId.clear();
221 }
222 //
223 // Emit everything first then clear the connection map.
224 PluginHost->Send_ConnectionEvent({ Events::ConnectionEntry::RemovedFromGroup, GetDisplayName(id), "" });
225
226 emit OnConnectionRemovedFromGroup({ id, gid });
227
228 //
229 if (connections[id].__qvConnectionRefCount <= 0)
230 {
231 LOG(MODULE_CONNECTION, "Fully removing a connection from cache.")
232 connectionRootCache.remove(id);
233 //
234 QFile connectionFile(QV2RAY_CONNECTIONS_DIR + id.toString() + QV2RAY_CONFIG_FILE_EXTENSION);
235 if (connectionFile.exists())
236 {
237 if (!connectionFile.remove())
238 LOG(MODULE_CONNECTION, "Failed to remove connection config file")
239 }
240 connections.remove(id);
241 }
242 return true;

260 bool QvConfigHandler::MoveConnectionFromToGroup(const ConnectionId &id, const GroupId &sourceGid, const GroupId &targetGid)
261 {
262 CheckValidId(id, false);
263 CheckValidId(targetGid, false);
264 CheckValidId(sourceGid, false);
265 //
266 if (!groups[sourceGid].connections.contains(id))
267 {
268 LOG(MODULE_CONNECTION, "Trying to move a connection away from a group it does not belong to.")
269 return false;
270 }
271 if (groups[targetGid].connections.contains(id))
272 {
273 LOG(MODULE_CONNECTION, "The connection: " + id.toString() + " has already been in the target group: " + targetGid.toString())
274 auto removedCount = groups[sourceGid].connections.removeAll(id);
275 connections[id].__qvConnectionRefCount -= removedCount;
276 }
277 else
278 {
279 // If the target group does not contain this connection.
280 auto removedCount = groups[sourceGid].connections.removeAll(id);
281 connections[id].__qvConnectionRefCount -= removedCount;
282 //
283 groups[targetGid].connections.append(id);
284 connections[id].__qvConnectionRefCount++;
285 }
286
287 emit OnConnectionRemovedFromGroup({ id, sourceGid });
288 emit OnConnectionLinkedWithGroup({ id, targetGid });
289
290 return true;
291 }

293 const std::optional<QString> QvConfigHandler::DeleteGroup(const GroupId &id)
294 {
295 CheckValidId(id, {});
296 if (!groups.contains(id) || id == NullGroupId)
297 {
298 return tr("Group does not exist");
299 }
300
301 // Copy construct
302 auto list = groups[id].connections;
303 for (const auto &conn : list)
304 {
305 MoveConnectionFromToGroup(conn, id, DefaultGroupId);
306 }
307 //
308 PluginHost->Send_ConnectionEvent({ Events::ConnectionEntry::FullyRemoved, groups[id].displayName, "" });
309 //
310 groups.remove(id);
311 SaveConnectionConfig();
312 emit OnGroupDeleted(id, list);
313 if (id == DefaultGroupId)
314 {
315 groups[id].displayName = tr("Default Group");
316 }
317 return {};

514 bool QvConfigHandler::CHUpdateSubscription_p(const GroupId &id, const QByteArray &data)
515 {
516 CheckValidId(id, false);
517 if (!groups.contains(id))
518 {
519 return false;
520 }
521 //
522 // ====================================================================================== Begin reading subscription
523
524 auto _newConnections = GetConnectionConfigFromSubscription(data, GetDisplayName(id));
525 if (_newConnections.count() < 5)
526 {
527 LOG(MODULE_SUBSCRIPTION, "Found a subscription with less than 5 connections.")
528 if (QvMessageBoxAsk(nullptr, tr("Update Subscription"),
529 tr("%n entrie(s) have been found from the subscription source, do you want to continue?", "",
530 _newConnections.count())) != QMessageBox::Yes)
531
532 return false;
533 }
534 //
535 // ====================================================================================== Begin Connection Data Storage
536 // Anyway, we try our best to preserve the connection id.
537 QMultiMap<QString, ConnectionId> nameMap;
538 QMultiMap<std::tuple<QString, QString, int>, ConnectionId> typeMap;
539 {
540 // Store connection type metadata into map.
541 for (const auto &conn : groups[id].connections)
542 {
543 nameMap.insert(GetDisplayName(conn), conn);
544 const auto &&[protocol, host, port] = GetConnectionInfo(conn);
545 if (port != 0)
546 {
547 typeMap.insert({ protocol, host, port }, conn);
548 }
549 }
550 }
551 // ====================================================================================== End Connection Data Storage
552 //
553 bool hasErrorOccured = false;
554 // Copy construct here.
555 auto originalConnectionIdList = groups[id].connections;
556 groups[id].connections.clear();
557 //
558 decltype(_newConnections) filteredConnections;
559 //
560 for (const auto &config : _newConnections)
561 {
562 // filter connections
563 const bool isIncludeOperationAND = groups[id].subscriptionOption.IncludeRelation == RELATION_AND;
564 const bool isExcludeOperationOR = groups[id].subscriptionOption.ExcludeRelation == RELATION_OR;
565 //
566 // Initial includeConfig value
567 bool includeconfig = isIncludeOperationAND;
568 {
569 bool hasIncludeItemMatched = false;
570 for (const auto &key : groups[id].subscriptionOption.IncludeKeywords)
571 {
572 if (!key.trimmed().isEmpty())
573 {
574 hasIncludeItemMatched = true;
575 // WARN: MAGIC, DO NOT TOUCH
576 if (!isIncludeOperationAND == config.first.contains(key.trimmed()))
577 {
578 includeconfig = !isIncludeOperationAND;
579 break;
580 }
581 }
582 }
583 // If includekeywords is empty then include all configs.
584 if (!hasIncludeItemMatched)
585 includeconfig = true;
586 }
587 if (includeconfig)
588 {
589 bool hasExcludeItemMatched = false;
590 includeconfig = isExcludeOperationOR;
591 for (const auto &key : groups[id].subscriptionOption.ExcludeKeywords)
592 {
593 if (!key.trimmed().isEmpty())
594 {
595 hasExcludeItemMatched = true;
596 // WARN: MAGIC, DO NOT TOUCH
597 if (isExcludeOperationOR == config.first.contains(key.trimmed()))
598 {
599 includeconfig = !isExcludeOperationOR;
600 break;
601 }
602 }
603 }
604 // If excludekeywords is empty then don't exclude any configs.
605 if (!hasExcludeItemMatched)
606 includeconfig = true;
607 }
608
609 if (includeconfig)
610 {
611 filteredConnections << config;
612 }
613 }
614
615 LOG(MODULE_SUBSCRIPTION, "Filtered out less than 5 connections.")
616 const auto useFilteredConnections =
617 filteredConnections.count() > 5 ||
618 QvMessageBoxAsk(nullptr, tr("Update Subscription"),
619 tr("%1 out of %n entrie(s) have been filtered out, do you want to continue?", "", _newConnections.count())
620 .arg(filteredConnections.count()) +
621 NEWLINE + GetDisplayName(id)) == QMessageBox::Yes;
622
623 for (const auto &config : useFilteredConnections ? filteredConnections : _newConnections)
624 {
625 const auto &_alias = config.first;
626 // Should not have complex connection we assume.
627 bool canGetOutboundData = false;
628 auto outboundData = GetConnectionInfo(config.second, &canGetOutboundData);
629 //
630 // ====================================================================================== Begin guessing new ConnectionId
631 if (nameMap.contains(_alias))
632 {
633 // Just go and save the connection...
634 LOG(MODULE_CORE_HANDLER, "Reused connection id from name: " + _alias)
635 const auto _conn = nameMap.take(_alias);
636 groups[id].connections << _conn;
637 UpdateConnection(_conn, config.second, true);
638 // Remove Connection Id from the list.
639 originalConnectionIdList.removeAll(_conn);
640 typeMap.remove(typeMap.key(_conn));
641 }
642 else if (canGetOutboundData && typeMap.contains(outboundData))
643 {
644 LOG(MODULE_CORE_HANDLER, "Reused connection id from protocol/host/port pair for connection: " + _alias)
645 const auto _conn = typeMap.take(outboundData);
646 groups[id].connections << _conn;
647 // Update Connection Properties
648 UpdateConnection(_conn, config.second, true);
649 RenameConnection(_conn, _alias);
650 // Remove Connection Id from the list.
651 originalConnectionIdList.removeAll(_conn);
652 nameMap.remove(nameMap.key(_conn));
653 }
654 else
655 {
656 // New connection id is required since nothing matched found...
657 LOG(MODULE_CORE_HANDLER, "Generated new connection id for connection: " + _alias)
658 CreateConnection(config.second, _alias, id, true);
659 }
660 // ====================================================================================== End guessing new ConnectionId
661 }
662
663 // Check if anything left behind (not being updated or changed significantly)
664 if (!originalConnectionIdList.isEmpty())
665 {
666 bool needContinue =
667 QvMessageBoxAsk(nullptr, //
668 tr("Update Subscription"),
669 tr("There're %n connection(s) in the group that do not belong the current subscription (any more).", "",
670 originalConnectionIdList.count()) +
671 NEWLINE + GetDisplayName(id) + NEWLINE + tr("Would you like to remove them?")) == QMessageBox::Yes;
672 if (needContinue)
673 {
674 LOG(MODULE_CORE_HANDLER, "Removed old connections not have been matched.")
675 for (const auto &conn : originalConnectionIdList)
676 {
677 LOG(MODULE_CORE_HANDLER, "Removing connections not in the new subscription: " + conn.toString())
678 RemoveConnectionFromGroup(conn, id);
679 }
680 }
681 }
682
683 // Update the time
684 groups[id].lastUpdatedDate = system_clock::to_time_t(system_clock::now());
685 return hasErrorOccured;
686 }

688 void QvConfigHandler::OnStatsDataArrived_p(const ConnectionGroupPair &id, const QMap<StatisticsType, QvStatsSpeed> &data)
689 {
690 if (id.isEmpty())
691 return;
692
693 const auto &cid = id.connectionId;
694 QMap<StatisticsType, QvStatsSpeedData> result;
695 for (const auto t : data.keys())
696 {
697 const auto &stat = data[t];
698 connections[cid].stats[t].upLinkData += stat.first;
699 connections[cid].stats[t].downLinkData += stat.second;
700 result[t] = { stat, connections[cid].stats[t].toData() };
701 }
702
703 emit OnStatsAvailable(id, result);
704 PluginHost->Send_ConnectionStatsEvent({ GetDisplayName(cid), //
705 result[CurrentStatAPIType].first.first, //
706 result[CurrentStatAPIType].first.second, //
707 result[CurrentStatAPIType].second.first, //
708 result[CurrentStatAPIType].second.second });
709 }
710 const ConnectionGroupPair QvConfigHandler::CreateConnection(const CONFIGROOT &root, const QString &displayName, const GroupId &groupId,
711 bool skipSaveConfig)
712 {
713 LOG(MODULE_CORE_HANDLER, "Creating new connection: " + displayName)
714 ConnectionId newId(GenerateUuid());
715 groups[groupId].connections << newId;
716 connections[newId].creationDate = system_clock::to_time_t(system_clock::now());
717 connections[newId].displayName = displayName;
718 connections[newId].__qvConnectionRefCount = 1;
719 emit OnConnectionCreated({ newId, groupId }, displayName);
720 PluginHost->Send_ConnectionEvent({ Events::ConnectionEntry::Created, displayName, "" });
721 UpdateConnection(newId, root);
722 if (!skipSaveConfig)
723 {
724 SaveConnectionConfig();
725 }
726 return { newId, groupId };
727 }
