47 ROUTING RouteHandler::GenerateRoutes(bool enableProxy, bool bypassCN, const QString &outTag, const QvConfig_Route &routeConfig) const
48 {
49 ROUTING root;
50 root.insert("domainStrategy", routeConfig.domainStrategy);
51 //
52 // For Rules list
53 ROUTERULELIST rulesList;
54
55 // Private IPs should always NOT TO PROXY!
56 rulesList.append(GenerateSingleRouteRule("geoip:private", false, OUTBOUND_TAG_DIRECT));
57 //
58 if (!enableProxy)
59 {
60 // This is added to disable all proxies, as a alternative influence of #64
61 rulesList.append(GenerateSingleRouteRule("regexp:.*", true, OUTBOUND_TAG_DIRECT));
62 rulesList.append(GenerateSingleRouteRule("0.0.0.0/0", false, OUTBOUND_TAG_DIRECT));
63 rulesList.append(GenerateSingleRouteRule("::/0", false, OUTBOUND_TAG_DIRECT));
64 }
65 else
66 {
67 //
68 // Blocked.
69 if (!routeConfig.ips.block.isEmpty())
70 {
71 rulesList.append(GenerateSingleRouteRule(routeConfig.ips.block, false, OUTBOUND_TAG_BLACKHOLE));
72 }
73 if (!routeConfig.domains.block.isEmpty())
74 {
75 rulesList.append(GenerateSingleRouteRule(routeConfig.domains.block, true, OUTBOUND_TAG_BLACKHOLE));
76 }
77 //
78 // Proxied
79 if (!routeConfig.ips.proxy.isEmpty())
80 {
81 rulesList.append(GenerateSingleRouteRule(routeConfig.ips.proxy, false, outTag));
82 }
83 if (!routeConfig.domains.proxy.isEmpty())
84 {
85 rulesList.append(GenerateSingleRouteRule(routeConfig.domains.proxy, true, outTag));
86 }
87 //
88 // Directed
89 if (!routeConfig.ips.direct.isEmpty())
90 {
91 rulesList.append(GenerateSingleRouteRule(routeConfig.ips.direct, false, OUTBOUND_TAG_DIRECT));
92 }
93 if (!routeConfig.domains.direct.isEmpty())
94 {
95 rulesList.append(GenerateSingleRouteRule(routeConfig.domains.direct, true, OUTBOUND_TAG_DIRECT));
96 }
97 //
98 // Check if CN needs proxy, or direct.
99 if (bypassCN)
100 {
101 // No proxy agains CN addresses.
102 rulesList.append(GenerateSingleRouteRule("geoip:cn", false, OUTBOUND_TAG_DIRECT));
103 rulesList.append(GenerateSingleRouteRule("geosite:cn", true, OUTBOUND_TAG_DIRECT));
104 }
105 }
106
107 root.insert("rules", rulesList);
108 return root;
109 }

118 CONFIGROOT RouteHandler::GenerateFinalConfig(CONFIGROOT root, const GroupRoutingId &routingId, bool hasAPI) const
119 {
120 const auto &config = configs.contains(routingId) ? configs[routingId] : GlobalConfig.defaultRouteConfig;
121 //
122 const auto &connConf = config.overrideConnectionConfig ? config.connectionConfig : GlobalConfig.defaultRouteConfig.connectionConfig;
123 const auto &dnsConf = config.overrideDNS ? config.dnsConfig : GlobalConfig.defaultRouteConfig.dnsConfig;
124 const auto &routeConf = config.overrideRoute ? config.routeConfig : GlobalConfig.defaultRouteConfig.routeConfig;
125 const auto &fpConf = config.overrideForwardProxyConfig ? config.forwardProxyConfig : GlobalConfig.defaultRouteConfig.forwardProxyConfig;
126 //
127 // See: https://github.com/Qv2ray/Qv2ray/issues/129
128 // routeCountLabel in Mainwindow makes here failed to ENOUGH-ly
129 // check the routing tables
130 //
131 // Check if is complex BEFORE adding anything.
132 bool isComplex = IsComplexConfig(root);
133 //
134 //
135 // logObject.insert("access", QV2RAY_CONFIG_PATH + QV2RAY_VCORE_LOG_DIRNAME + QV2RAY_VCORE_ACCESS_LOG_FILENAME);
136 // logObject.insert("error", QV2RAY_CONFIG_PATH + QV2RAY_VCORE_LOG_DIRNAME + QV2RAY_VCORE_ERROR_LOG_FILENAME);
137 QJsonIO::SetValue(root, V2RayLogLevel[GlobalConfig.logLevel], "log", "loglevel");
138 //
139 // Since Qv2ray does not support settings DNS manually for now.
140 // These settings are being added for both complex config AND simple config.
141 if (root.contains("dns") && !root.value("dns").toObject().isEmpty())
142 {
143 // We assume the users are using THEIR DNS settings.
144 LOG(MODULE_CONNECTION, "Found DNS settings specified manually, skipping inserting GlobalConfig")
145 }
146 else
147 {
148 root.insert("dns", GenerateDNS(connConf.withLocalDNS, dnsConf));
149 }
150 //
151 //
152 // If inbounds list is empty we append our global configured inbounds to the config.
153 // The setting applies to BOTH complex config AND simple config.
154 // Just to ensure there's AT LEAST 1 possible inbound is being configured.
155 if (!root.contains("inbounds") || root.value("inbounds").toArray().empty())
156 {
157#define INCONF GlobalConfig.inboundConfig
158 INBOUNDS inboundsList;
159 const QJsonObject sniffingOff{ { "enabled", false } };
160 const QJsonObject sniffingOn{ { "enabled", true }, { "destOverride", QJsonArray{ "http", "tls" } } };
161
162 // HTTP Inbound
163 if (GlobalConfig.inboundConfig.useHTTP)
164 {
165 const auto httpInBoundObject = //
166 GenerateInboundEntry(INCONF.listenip, //
167 INCONF.httpSettings.port, //
168 "http", //
169 INBOUNDSETTING{}, //
170 "http_IN", //
171 { INCONF.httpSettings.sniffing ? sniffingOn : sniffingOff });
172 if (INCONF.httpSettings.useAuth)
173 {
174 QJsonIO::SetValue(httpInBoundObject, GenerateHTTPIN({ INCONF.httpSettings.account }), "settings");
175 }
176
177 inboundsList.append(httpInBoundObject);
178 }
179
180 // SOCKS Inbound
181 if (INCONF.useSocks)
182 {
183 const auto socksInBoundObject = //
184 GenerateInboundEntry(INCONF.listenip, //
185 INCONF.socksSettings.port, //
186 "socks", //
187 GenerateSocksIN(INCONF.socksSettings.useAuth ? "password" : "noauth", //
188 { INCONF.socksSettings.account }, //
189 INCONF.socksSettings.enableUDP, //
190 INCONF.socksSettings.localIP), //
191 "socks_IN", //
192 { INCONF.socksSettings.sniffing ? sniffingOn : sniffingOff });
193 inboundsList.append(socksInBoundObject);
194 }
195
196 // TPROXY
197 if (INCONF.useTPROXY)
198 {
199 QList<QString> networks;
200 if (INCONF.tProxySettings.hasTCP)
201 networks << "tcp";
202 if (INCONF.tProxySettings.hasUDP)
203 networks << "udp";
204 const auto tproxy_network = networks.join(",");
205 // tProxy IPv4 Settings
206 {
207 LOG(MODULE_CONNECTION, "Processing tProxy IPv4 inbound")
208 INBOUND tProxyIn = GenerateInboundEntry(INCONF.tProxySettings.tProxyIP, //
209 INCONF.tProxySettings.port, //
210 "dokodemo-door", //
211 GenerateDokodemoIN("", 0, tproxy_network, 0, true, 0), //
212 "tproxy_IN", //
213 {
214 { "enabled", true }, //
215 { "destOverride", QJsonArray{ "http", "tls" } } //
216 });
217 tProxyIn.insert("streamSettings", QJsonObject{ { "sockopt", QJsonObject{ { "tproxy", INCONF.tProxySettings.mode } } } });
218 inboundsList.append(tProxyIn);
219 }
220 if (!INCONF.tProxySettings.tProxyV6IP.isEmpty())
221 {
222 LOG(MODULE_CONNECTION, "Processing tProxy IPv6 inbound")
223 INBOUND tProxyIn = GenerateInboundEntry(INCONF.tProxySettings.tProxyV6IP, //
224 INCONF.tProxySettings.port, //
225 "dokodemo-door", //
226 GenerateDokodemoIN("", 0, tproxy_network, 0, true, 0), //
227 "tproxy_IN_V6", //
228 {
229 { "enabled", true }, //
230 { "destOverride", QJsonArray{ "http", "tls" } } //
231 });
232 tProxyIn.insert("streamSettings", QJsonObject{ { "sockopt", QJsonObject{ { "tproxy", INCONF.tProxySettings.mode } } } });
233 inboundsList.append(tProxyIn);
234 }
235 }
236
237 root["inbounds"] = inboundsList;
238 DEBUG(MODULE_CONNECTION, "Added global config inbounds to the config")
239 }
240#undef INCONF
241
242 // Process every inbounds to make sure a tag is configured, fixed
243 // API 0 speed issue when no tag is configured.
244 FillupTagsFilter(root, "inbounds");
245 //
246 //
247 // Note: The part below always makes the whole functionality in
248 // trouble...... BE EXTREME CAREFUL when changing these code
249 // below...
250 if (isComplex)
251 {
252 // For some config files that has routing entries already.
253 // We DO NOT add extra routings.
254 //
255 // HOWEVER, we need to verify the QV2RAY_RULE_ENABLED entry.
256 // And what's more, process (by removing unused items) from a
257 // rule object.
258 ROUTING routing(root["routing"].toObject());
259 ROUTERULELIST rules;
260 LOG(MODULE_CONNECTION, "Processing an existing routing table.")
261
262 for (const auto &_rule : routing["rules"].toArray())
263 {
264 auto _b = _rule.toObject();
265
266 if (_b.contains("QV2RAY_RULE_USE_BALANCER"))
267 {
268 // We use balancer, or the normal outbound
269 _b.remove(_b["QV2RAY_RULE_USE_BALANCER"].toBool(false) ? "outboundTag" : "balancerTag");
270 }
271 else
272 {
273 LOG(MODULE_SETTINGS, "We found a rule without QV2RAY_RULE_USE_BALANCER, so didn't process it.")
274 }
275
276 // If this entry has been disabled.
277 if (_b.contains("QV2RAY_RULE_ENABLED") && _b["QV2RAY_RULE_ENABLED"].toBool() == false)
278 {
279 LOG(MODULE_SETTINGS, "Discarded a rule as it's been set DISABLED")
280 }
281 else
282 {
283 rules.append(_b);
284 }
285 }
286
287 routing["rules"] = rules;
288 root["routing"] = routing;
289 }
290 else
291 {
292 LOG(MODULE_CONNECTION, "Inserting default values to simple config")
293 if (root["outbounds"].toArray().count() != 1)
294 {
295 // There are no ROUTING but 2 or more outbounds.... This is rare, but possible.
296 LOG(MODULE_CONNECTION, "WARN: This message usually indicates the config file has logic errors:")
297 LOG(MODULE_CONNECTION, "WARN: --> The config file has NO routing section, however more than 1 outbounds are detected.")
298 }
299 //
300 auto tag = getTag(OUTBOUND(QJsonIO::GetValue(root, "outbounds", 0).toObject()));
301 if (tag.isEmpty())
302 {
303 LOG(MODULE_CONNECTION, "Applying workaround when an outbound tag is empty")
304 tag = GenerateRandomString(15);
305 QJsonIO::SetValue(root, tag, "outbounds", 0, "tag");
306 }
307 root["routing"] = GenerateRoutes(connConf.enableProxy, connConf.bypassCN, tag, routeConf);
308 //
309 // Process forward proxy
310 //
311 if (fpConf.enableForwardProxy)
312 {
313 auto outboundArray = root["outbounds"].toArray();
314 auto firstOutbound = outboundArray.first().toObject();
315 if (firstOutbound[QV2RAY_USE_FPROXY_KEY].toBool(false))
316 {
317 LOG(MODULE_CONNECTION, "Applying forward proxy to current connection.")
318 PROXYSETTING proxy;
319 proxy["tag"] = OUTBOUND_TAG_FORWARD_PROXY;
320 firstOutbound["proxySettings"] = proxy;
321
322 // FP Outbound.
323 if (fpConf.type.toLower() == "http" || fpConf.type.toLower() == "socks")
324 {
325 auto fpOutbound =
326 GenerateHTTPSOCKSOut(fpConf.serverAddress, fpConf.port, fpConf.useAuth, fpConf.username, fpConf.password);
327 outboundArray.push_back(
328 GenerateOutboundEntry(fpConf.type.toLower(), fpOutbound, {}, {}, "0.0.0.0", OUTBOUND_TAG_FORWARD_PROXY));
329 }
330 else if (!fpConf.type.isEmpty())
331 {
332 DEBUG(MODULE_CONNECTION, "WARNING: Unsupported outbound type: " + fpConf.type)
333 }
334 else
335 {
336 DEBUG(MODULE_CONNECTION, "WARNING: Empty outbound type.")
337 }
338 }
339 else
340 {
341 // Remove proxySettings from firstOutbound
342 firstOutbound.remove("proxySettings");
343 }
344
345 outboundArray.replace(0, firstOutbound);
346 root["outbounds"] = outboundArray;
347 }
348#undef fpConf
349 //
350 // Process FREEDOM and BLACKHOLE outbound
351 {
352 OUTBOUNDS outbounds(root["outbounds"].toArray());
353 const auto freeDS = (connConf.v2rayFreedomDNS) ? "UseIP" : "AsIs";
354 outbounds.append(GenerateOutboundEntry("freedom", GenerateFreedomOUT(freeDS, ":0", 0), {}, {}, "0.0.0.0", OUTBOUND_TAG_DIRECT));
355 outbounds.append(GenerateOutboundEntry("blackhole", GenerateBlackHoleOUT(false), {}, {}, "0.0.0.0", OUTBOUND_TAG_BLACKHOLE));
356 root["outbounds"] = outbounds;
357 }
358 //
359 // Connection Filters
360 if (GlobalConfig.defaultRouteConfig.connectionConfig.dnsIntercept)
361 {
362 const auto hasTProxy = GlobalConfig.inboundConfig.useTPROXY && GlobalConfig.inboundConfig.tProxySettings.hasUDP;
363 const auto hasIPv6 = hasTProxy && (!GlobalConfig.inboundConfig.tProxySettings.tProxyV6IP.isEmpty());
364 const bool hasSocksUDP = GlobalConfig.inboundConfig.useSocks && GlobalConfig.inboundConfig.socksSettings.enableUDP;
365 DNSInterceptFilter(root, hasTProxy, hasIPv6, hasSocksUDP);
366 }
367
368 if (GlobalConfig.inboundConfig.useTPROXY && GlobalConfig.outboundConfig.mark > 0)
369 {
370 OutboundMarkSettingFilter(root, GlobalConfig.outboundConfig.mark);
371 }
372
373 if (connConf.bypassBT)
374 {
375 BypassBTFilter(root);
376 }
377 // Process mKCP seed.
378 mKCPSeedFilter(root);
379
380 // Remove empty Mux object from settings.
381 RemoveEmptyMuxFilter(root);
382 }
383
384 // Let's process some api features.
385 if (hasAPI && GlobalConfig.kernelConfig.enableAPI)
386 {
387 //
388 // Stats
389 root.insert("stats", QJsonObject());
390 //
391 // Routes
392 QJsonObject routing = root["routing"].toObject();
393 QJsonArray routingRules = routing["rules"].toArray();
394 QJsonObject APIRouteRoot{ { "type", "field" }, //
395 { "outboundTag", API_TAG_DEFAULT }, //
396 { "inboundTag", QJsonArray{ API_TAG_INBOUND } } };
397 routingRules.push_front(APIRouteRoot);
398 routing["rules"] = routingRules;
399 root["routing"] = routing;
400 //
401 // Policy
402 QJsonIO::SetValue(root, true, "policy", "system", "statsInboundUplink");
403 QJsonIO::SetValue(root, true, "policy", "system", "statsInboundDownlink");
404 QJsonIO::SetValue(root, true, "policy", "system", "statsOutboundUplink");
405 QJsonIO::SetValue(root, true, "policy", "system", "statsOutboundDownlink");
406 //
407 // Inbounds
408 INBOUNDS inbounds(root["inbounds"].toArray());
409 QJsonObject fakeDocodemoDoor{ { "address", "127.0.0.1" } };
410 const auto apiInboundsRoot = GenerateInboundEntry("127.0.0.1", GlobalConfig.kernelConfig.statsPort, "dokodemo-door",
411 INBOUNDSETTING(fakeDocodemoDoor), API_TAG_INBOUND);
412 inbounds.push_front(apiInboundsRoot);
413 root["inbounds"] = inbounds;
414 //
415 // API
416 root["api"] = GenerateAPIEntry(API_TAG_DEFAULT);
417 }
418
419 return root;
420 }
