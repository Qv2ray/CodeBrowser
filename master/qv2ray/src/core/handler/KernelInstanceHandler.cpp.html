10 KernelInstanceHandler::KernelInstanceHandler(QObject *parent) : QObject(parent)

12 KernelInstance = this;

16 connect(vCoreInstance, &V2RayKernelInstance::OnProcessErrored, this, &KernelInstanceHandler::OnKernelCrashed_p);
17 //
18 auto kernelList = PluginHost->GetPluginKernels();
19 for (const auto &internalName : kernelList.keys())
20 {
21 auto kernel = kernelList.value(internalName);
22 for (const auto &protocol : kernel)
23 {
24 if (outboundKernelMap.contains(protocol))
25 {
26 LOG(MODULE_PLUGINHOST, "Found multiple kernel providers for a protocol: " + protocol)
27 continue;
28 }
29 outboundKernelMap.insert(protocol, internalName);
30 }
31 }

39 std::optional<QString> KernelInstanceHandler::CheckPort(const QMap<QString, InboundInfoObject> &info, int plugins)
40 {
41 //
42 // Check inbound port allocation issue.
43 QStringList portDetectionErrorMessage;
44 auto portDetectionMsg = tr("Another process is using the port required to start the connection:") + NEWLINE + NEWLINE;
45 for (const auto &key : info.keys())
46 {
47 auto result = components::port::CheckTCPPortStatus(info[key].listenIp, info[key].port);
48 if (!result)
49 {
50 portDetectionErrorMessage << tr("Port: %1 for listening IP: %2 for inbound tag: \"%3\"") //
51 .arg(info[key].port)
52 .arg(info[key].listenIp)
53 .arg(key);
54 }
55 }
56 if (GlobalConfig.pluginConfig.v2rayIntegration)
57 {
58 for (int i = 0; i < plugins; i++)
59 {
60 auto result = components::port::CheckTCPPortStatus("127.0.0.1", GlobalConfig.pluginConfig.portAllocationStart + i);
61 if (!result)
62 {
63 portDetectionErrorMessage << tr("Port: %1 for listening IP: 127.0.0.1 for plugin integration.")
64 .arg(GlobalConfig.pluginConfig.portAllocationStart + i);
65 }
66 }
67 }
68 if (!portDetectionErrorMessage.isEmpty())
69 {
70 portDetectionMsg += portDetectionErrorMessage.join(NEWLINE);
71 return portDetectionMsg;
72 }
73 else
74 {
75 return std::nullopt;
76 }

79 std::optional<QString> KernelInstanceHandler::StartConnection(const ConnectionGroupPair &id, CONFIGROOT fullConfig)
80 {
81 StopConnection();
82 inboundInfo = GetConfigInboundInfo(fullConfig);
83 //
84 const auto inboundPorts = GetInboundPorts();
85 const auto inboundHosts = GetInboundHosts();
86 PluginHost->Send_ConnectivityEvent({ GetDisplayName(id.connectionId), inboundPorts, Events::Connectivity::Connecting });
87 // QList<std::tuple<QString, int, QString>> inboundInfo;
88 // for (const auto &inbound_v : fullConfig["inbounds"].toArray())
89 //{
90 // const auto &inbound = inbound_v.toObject();
91 // inboundInfo.push_back({ inbound["protocol"].toString(), inbound["port"].toInt(), inbound["tag"].toString() });
92 //}
93 //
94 using _k = Qv2rayPlugin::QvPluginKernel;
95 if (GlobalConfig.pluginConfig.v2rayIntegration)
96 {
97 //
98 // Process outbounds.
99 OUTBOUNDS processedOutbounds;
100 auto pluginPort = GlobalConfig.pluginConfig.portAllocationStart;
101 //
102 for (auto i = 0; i < fullConfig["outbounds"].toArray().count(); i++)
103 {
104 auto outbound = QJsonIO::GetValue(fullConfig, "outbounds", i).toObject();
105 const auto outProtocol = outbound["protocol"].toString();
106 //
107 if (!outboundKernelMap.contains(outProtocol))
108 {
109 // Normal outbound, or the one without a plugin supported.
110 // Marked as processed.
111 processedOutbounds.push_back(outbound);
112 LOG(MODULE_CONNECTION, "Outbound protocol " + outProtocol + " is not a registered plugin outbound.")
113 continue;
114 }
115 LOG(MODULE_CONNECTION, "Creating kernel plugin instance for protocol" + outProtocol)
116 auto kernel = PluginHost->CreatePluginKernel(outboundKernelMap[outProtocol]);
117 // New object does not need disconnect?
118 // disconnect(kernel, &QvPluginKernel::OnKernelStatsAvailable, this, &KernelInstanceHandler::OnStatsDataArrived_p);
119 //
120 QMap<QvPluginKernel::KernelSetting, QVariant> _inboundSettings;
121
122 LOG(MODULE_VCORE, "V2RayIntegration: " + QSTRN(pluginPort) + " = " + outProtocol)
123 _inboundSettings[_k::KERNEL_HTTP_ENABLED] = false;
124 _inboundSettings[_k::KERNEL_SOCKS_ENABLED] = true;
125 _inboundSettings[_k::KERNEL_SOCKS_PORT] = pluginPort;
126 _inboundSettings[_k::KERNEL_SOCKS_UDP_ENABLED] = GlobalConfig.inboundConfig.socksSettings.enableUDP;
127 _inboundSettings[_k::KERNEL_SOCKS_LOCAL_ADDRESS] = GlobalConfig.inboundConfig.socksSettings.localIP;
128 _inboundSettings[_k::KERNEL_LISTEN_ADDRESS] = "127.0.0.1";
129 LOG(MODULE_CONNECTION, "Sending connection settings to kernel.")
130 kernel->SetConnectionSettings(_inboundSettings, outbound["settings"].toObject());
131 activeKernels.push_back({ outProtocol, std::move(kernel) });
132 //
133 const auto pluginOutSettings = GenerateHTTPSOCKSOut("127.0.0.1", pluginPort, false, "", "");
134 //
135 const auto pluginOut = GenerateOutboundEntry("socks", pluginOutSettings, {}, {}, "0.0.0.0", outbound["tag"].toString());
136 //
137 // Add the integration outbound to the list.
138 processedOutbounds.push_back(pluginOut);
139 pluginPort++;
140 }
141 LOG(MODULE_CONNECTION, "Applying new outbound settings.")
142 fullConfig["outbounds"] = processedOutbounds;
143 RemoveEmptyMuxFilter(fullConfig);
144 }
145
146 //
147 // ======================================================================= Start Kernels
148 //
149 {
150 const auto portResult = CheckPort(inboundInfo, activeKernels.size());
151 if (portResult)
152 {
153 LOG(MODULE_CONNECTION, ACCESS_OPTIONAL_VALUE(portResult))
154 return portResult;
155 }
156 auto firstOutbound = fullConfig["outbounds"].toArray().first().toObject();
157 const auto firstOutboundProtocol = firstOutbound["protocol"].toString();
158 if (GlobalConfig.pluginConfig.v2rayIntegration)
159 {
160 LOG(MODULE_VCORE, "Starting kernels with V2RayIntegration.")
161 bool hasAllKernelStarted = true;
162 for (auto &[outboundProtocol, kernelObject] : activeKernels)
163 {
164 LOG(MODULE_CONNECTION, "Starting kernel for protocol: " + outboundProtocol)
165 bool status = kernelObject->StartKernel();
166 connect(kernelObject.get(), &QvPluginKernel::OnKernelCrashed, this, &KernelInstanceHandler::OnKernelCrashed_p,
167 Qt::QueuedConnection);
168 connect(kernelObject.get(), &QvPluginKernel::OnKernelLogAvailable, this, &KernelInstanceHandler::OnKernelLog_p,
169 Qt::QueuedConnection);
170 hasAllKernelStarted = hasAllKernelStarted && status;
171 if (!status)
172 {
173 LOG(MODULE_CONNECTION, "Plugin Kernel: " + outboundProtocol + " failed to start.")
174 break;
175 }
176 }
177 if (!hasAllKernelStarted)
178 {
179 StopConnection();
180 return tr("A plugin kernel failed to start. Please check the outbound settings.");
181 }
182 currentId = id;
183 //
184 // Also start V2Ray-core.
185 auto result = vCoreInstance->StartConnection(fullConfig);
186 //
187 if (result.has_value())
188 {
189 StopConnection();
190 PluginHost->Send_ConnectivityEvent({ GetDisplayName(id.connectionId), inboundPorts, Events::Connectivity::Disconnected });
191 return result;
192 }
193 else
194 {
195 emit OnConnected(id);
196 PluginHost->Send_ConnectivityEvent({ GetDisplayName(id.connectionId), inboundPorts, Events::Connectivity::Connected });
197 }
198 }
199 else if (outboundKernelMap.contains(firstOutboundProtocol))
200 {
201 // Connections without V2Ray Integration will have and ONLY have ONE kernel.
202 LOG(MODULE_CONNECTION, "Starting kernel " + firstOutboundProtocol + " without V2Ray Integration")
203 {
204 auto kernel = PluginHost->CreatePluginKernel(outboundKernelMap[firstOutbound["protocol"].toString()]);
205 activeKernels.push_back({ firstOutboundProtocol, std::move(kernel) });
206 }
207 Q_ASSERT(activeKernels.size() == 1);
208#define theKernel (activeKernels.front().second.get())
209 connect(theKernel, &QvPluginKernel::OnKernelStatsAvailable, this, &KernelInstanceHandler::OnPluginStatsDataRcvd_p,
210 Qt::QueuedConnection);
211 connect(theKernel, &QvPluginKernel::OnKernelCrashed, this, &KernelInstanceHandler::OnKernelCrashed_p, Qt::QueuedConnection);
212 connect(theKernel, &QvPluginKernel::OnKernelLogAvailable, this, &KernelInstanceHandler::OnKernelLog_p, Qt::QueuedConnection);
213 currentId = id;
214 //
215 QMap<QvPluginKernel::KernelSetting, QVariant> pluginSettings;
216
217 for (const auto &v : inboundInfo)
218 {
219 if (v.protocol != "http" && v.protocol != "socks")
220 continue;
221 pluginSettings[_k::KERNEL_HTTP_ENABLED] = pluginSettings[_k::KERNEL_HTTP_ENABLED].toBool() || v.protocol == "http";
222 pluginSettings[_k::KERNEL_SOCKS_ENABLED] = pluginSettings[_k::KERNEL_SOCKS_ENABLED].toBool() || v.protocol == "socks";
223 pluginSettings.insert(v.protocol.toLower() == "http" ? _k::KERNEL_HTTP_PORT : _k::KERNEL_SOCKS_PORT, v.port);
224 }
225
226 pluginSettings[_k::KERNEL_SOCKS_UDP_ENABLED] = GlobalConfig.inboundConfig.socksSettings.enableUDP;
227 pluginSettings[_k::KERNEL_SOCKS_LOCAL_ADDRESS] = GlobalConfig.inboundConfig.socksSettings.localIP;
228 pluginSettings[_k::KERNEL_LISTEN_ADDRESS] = GlobalConfig.inboundConfig.listenip;
229 //
230 theKernel->SetConnectionSettings(pluginSettings, firstOutbound["settings"].toObject());
231 bool kernelStarted = theKernel->StartKernel();
232#undef theKernel
233 if (kernelStarted)
234 {
235 emit OnConnected(id);
236 PluginHost->Send_ConnectivityEvent({ GetDisplayName(id.connectionId), inboundPorts, Events::Connectivity::Connected });
237 }
238 else
239 {
240 return tr("A plugin kernel failed to start. Please check the outbound settings.");
241 StopConnection();
242 }
243 }
244 else
245 {
246 LOG(MODULE_CONNECTION, "Starting V2Ray without plugin.")
247 currentId = id;
248 auto result = vCoreInstance->StartConnection(fullConfig);
249 if (result.has_value())
250 {
251 PluginHost->Send_ConnectivityEvent({ GetDisplayName(id.connectionId), inboundPorts, Events::Connectivity::Disconnected });
252 StopConnection();
253 return result;
254 }
255 else
256 {
257 emit OnConnected(id);
258 PluginHost->Send_ConnectivityEvent({ GetDisplayName(id.connectionId), inboundPorts, Events::Connectivity::Connected });
259 }
260 }
261 }
262 // Return
263 return std::nullopt;
264 }

277 void KernelInstanceHandler::StopConnection()
278 {
279 if (isConnected)
280 {
281 const auto inboundPorts = GetInboundPorts();
282 PluginHost->Send_ConnectivityEvent({ GetDisplayName(currentId.connectionId), inboundPorts, Events::Connectivity::Disconnecting });
283 if (vCoreInstance->KernelStarted)
284 {
285 vCoreInstance->StopConnection();
286 }
287 for (const auto &[kernel, kernelObject] : activeKernels)
288 {
289 LOG(MODULE_CONNECTION, "Stopping plugin kernel: " + kernel)
290 kernelObject->StopKernel();
291 }
292 emit OnDisconnected(currentId);
293 PluginHost->Send_ConnectivityEvent({ GetDisplayName(currentId.connectionId), inboundPorts, Events::Connectivity::Disconnected });
294 }
295 else
296 {
297 LOG(MODULE_CORE_HANDLER, "Cannot disconnect when there's nothing connected.")
298 }
299 currentId.clear();
300 activeKernels.clear();
301 }
