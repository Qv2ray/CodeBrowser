7TEST_CASE("Test VLESS URL Parsing")
8{
9 QvTestApplication app;
10 QString alias;
11 QString errMessage;
12
13 SECTION("VLESSTCPXTLSSplice")
14 {
15 const static auto url = "vless://b0dd64e4-0fbd-4038-9139-d1f32a68a0dc@qv2ray.net:3279?security=xtls&flow=rprx-xtls-splice#VLESSTCPXTLSSplice";
16
17 const auto result = vless::Deserialize(url, &alias, &errMessage);
18
19 INFO("Parsed: " << QJsonDocument(result).toJson().toStdString());
20 REQUIRE(errMessage.isEmpty());
21 REQUIRE(alias.toStdString() == "VLESSTCPXTLSSplice");
22 }

24 SECTION("ALPN Parse Test")
25 {
26 const static auto url = "vless://24a613c1-de83-4c63-ba73-a9d08c88fec3@qv2ray.net:13432?security=xtls&alpn=h2%2Chttp%2F1.1";
27
28 const auto result = vless::Deserialize(url, &alias, &errMessage);
29
30 INFO("Parsed: " << QJsonDocument(result).toJson().toStdString());
31 REQUIRE(errMessage.isEmpty());
32
33 const auto alpnField = QJsonIO::GetValue(result, "outbounds", 0, "streamSettings", "xtlsSettings", "alpn");
34 REQUIRE(alpnField.isArray());
35
36 const auto alpnArray = alpnField.toArray();
37 REQUIRE(!alpnArray.empty());
38 REQUIRE(alpnArray.size() == 2);
39
40 const auto firstALPN = alpnArray.first();
41 REQUIRE(firstALPN.isString());
42
43 const auto firstALPNString = firstALPN.toString();
44 REQUIRE(firstALPNString.toStdString() == "h2");
45
46 const auto lastALPN = alpnArray.last();
47 REQUIRE(lastALPN.isString());
48
49 const auto lastALPNString = lastALPN.toString();
50 REQUIRE(lastALPNString.toStdString() == "http/1.1");
51 }
