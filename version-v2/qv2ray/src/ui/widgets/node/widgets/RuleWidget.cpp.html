42void QvNodeRuleWidget::setValue(std::shared_ptr<RuleObject> _ruleptr)
43{
44 this->ruleptr = _ruleptr;
45 // Switch to the detailed page.
46 ruleEnableCB->setEnabled(true);
47 ruleEnableCB->setChecked(rule.QV2RAY_RULE_ENABLED);
48 ruleTagLineEdit->setEnabled(true);
49 LOAD_FLAG_BEGIN
50 ruleTagLineEdit->setText(rule.QV2RAY_RULE_TAG);
51 isLoading = false;
52 // Networks
53 auto network = rule.network.toLower();
54 netUDPRB->setChecked(network.contains("udp"));
55 netTCPRB->setChecked(network.contains("tcp"));
56 //
57 // Set protocol checkboxes.
58 auto protocol = rule.protocol;
59 routeProtocolHTTPCB->setChecked(protocol.contains("http"));
60 routeProtocolTLSCB->setChecked(protocol.contains("tls"));
61 routeProtocolBTCB->setChecked(protocol.contains("bittorrent"));
62 //
63 // Port
64 routePortTxt->setText(rule.port);
65 //
66 // Users
67 const auto sourcePorts = rule.sourcePort;
68 //
69 // Incoming Sources
70 const auto sources = rule.source.join(NEWLINE);
71 sourceIPList->setPlainText(sources);
72 //
73 // Domains
74 QString domains = rule.domain.join(NEWLINE);
75 hostList->setPlainText(domains);
76 //
77 // Outcoming IPs
78 QString ips = rule.ip.join(NEWLINE);
79 ipList->setPlainText(ips);
80 LOAD_FLAG_END
81}
