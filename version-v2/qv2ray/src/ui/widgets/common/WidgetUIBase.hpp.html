38class QvStateObject
39{
40 using options_save_func_t = std::function<QJsonValue()>;
41 using options_restore_func_t = std::function<void(QJsonValue)>;
42 using options_storage_type = std::map<QString, std::pair<options_save_func_t, options_restore_func_t>>;
43
44 public:
45 explicit QvStateObject(const QString &name) : windowName(name){};
46
47 protected:
48 void addStateOptions(QString name, std::pair<options_save_func_t, options_restore_func_t> funcs)
49 {
50 state_options_list[name] = funcs;
51 }
52
53 private:
54 const QString windowName;
55 options_storage_type state_options_list;
56
57#if QV2RAY_FEATURE(ui_has_store_state)
58 public:
59 void SaveState()
60 {
61 QvWidgetApplication->UIStates[windowName] = saveStateImpl();
62 }
63 void RestoreState()
64 {
65 restoreStateImpl(QvWidgetApplication->UIStates[windowName].toObject());
66 }
67
68 private:
69 QJsonObject saveStateImpl()
70 {
71 QJsonObject o;
72 for (const auto &[name, pair] : state_options_list)
73 o[name] = pair.first();
74 return o;
75 }
76 void restoreStateImpl(const QJsonObject &o)
77 {
78 for (const auto &[name, pair] : state_options_list)
79 if (o.contains(name))
80 pair.second(o[name]);
81 }
82#endif
83};

114 inline QPixmap ApplyEffectToImage(QPixmap src, QGraphicsEffect *effect, int extent = 0)
115 {
116 if (src.isNull() || !effect)
117 return src;
118 QGraphicsScene scene;
119 QGraphicsPixmapItem item;
120 item.setPixmap(src);
121 item.setGraphicsEffect(effect);
122 scene.addItem(&item);
123 QImage res(src.size() + QSize(extent * 2, extent * 2), QImage::Format_ARGB32);
124 res.fill(Qt::transparent);
125 QPainter ptr(&res);
126 scene.render(&ptr, QRectF(), QRectF(-extent, -extent, src.width() + extent * 2, src.height() + extent * 2));
127 // Clean up
128 item.setGraphicsEffect(nullptr);
129 return QPixmap::fromImage(res);
130 }

132 inline QPixmap BlurImage(const QPixmap &pixmap, const double rad)
133 {
134 auto pBlur = new QGraphicsBlurEffect();
135 pBlur->setBlurRadius(rad);
136 return ApplyEffectToImage(pixmap, pBlur, 0);
137 }

139 inline QPixmap ColorizeImage(const QPixmap &pixmap, const QColor &color, const qreal factor)
140 {
141 auto pColor = new QGraphicsColorizeEffect();
142 pColor->setColor(color);
143 pColor->setStrength(factor);
144 return ApplyEffectToImage(pixmap, pColor, 0);
145 }
