33QImage ScreenShotWindow::DoScreenShot()
34{
35 // The msleep is the only solution which prevent capturing our windows again. It works on KDE,
36 // https://www.qtcentre.org/threads/55708-Get-Desktop-Screenshot-Without-Application-Window-Being-Shown?p=248993#post248993
37 QThread::msleep(100);
38 QApplication::processEvents();
39 //
40 desktopImage = qApp->screenAt(QCursor::pos())->grabWindow(0);
41 scale = qApp->screenAt(QCursor::pos())->devicePixelRatio();
42 //
43 int w = desktopImage.width();
44 int h = desktopImage.height();
45 //
46 int r, g, b;
47 const auto _xdesktopImg = desktopImage.toImage();
48
49 QImage bg_grey(w, h, QImage::Format_RGB32);
50 for (int i = 0; i < w; i++)
51 {
52 for (int j = 0; j < h; j++)
53 {
54 r = static_cast<int>(qRed(_xdesktopImg.pixel(i, j)) * QV2RAY_SCREENSHOT_DIM_RATIO);
55 g = static_cast<int>(qGreen(_xdesktopImg.pixel(i, j)) * QV2RAY_SCREENSHOT_DIM_RATIO);
56 b = static_cast<int>(qBlue(_xdesktopImg.pixel(i, j)) * QV2RAY_SCREENSHOT_DIM_RATIO);
57 bg_grey.setPixel(i, j, qRgb(r, g, b));
58 }
59 }
60 setStyleSheet("QDialog { background-color: transparent; }");
61 bg_grey = bg_grey.scaled(bg_grey.size() / scale, Qt::KeepAspectRatio, Qt::TransformationMode::SmoothTransformation);
62 auto p = this->palette();
63 p.setBrush(QPalette::Window, bg_grey);
64 setPalette(p);
65 setWindowState(Qt::WindowState::WindowFullScreen);
66 setMouseTracking(true);
67 setWindowFlags(Qt::FramelessWindowHint | Qt::WindowStaysOnTopHint);
68 this->showFullScreen();
69 this->exec();
70 return resultImage;
71}

73void ScreenShotWindow::pSize()
74{
75 imgW = abs(end.x() - origin.x());
76 imgH = abs(end.y() - origin.y());
77 imgX = origin.x() < end.x() ? origin.x() : end.x();
78 imgY = origin.y() < end.y() ? origin.y() : end.y();
79 rubber.setGeometry(imgX, imgY, imgW, imgH);
80 fg->setGeometry(rubber.geometry());
81 fg->setPixmap(desktopImage.copy(imgX * scale, imgY * scale, imgW * scale, imgH * scale));
82}

106void ScreenShotWindow::mouseMoveEvent(QMouseEvent *e)
107{
108 if (e->buttons() & Qt::LeftButton)
109 {
110 end = e->pos();
111 pSize();
112 //
113 label->setText(QString("%1x%2").arg(imgW).arg(imgH));
114 label->adjustSize();
115 //
116 QRect labelRect(label->contentsRect());
117 QRect btnRect(startBtn->contentsRect());
118
119 if (imgY > labelRect.height())
120 {
121 label->move(imgX, imgY - labelRect.height());
122 }
123 else
124 {
125 label->move(imgX, imgY);
126 }
127
128 if (height() - imgY - imgH > btnRect.height())
129 {
130 startBtn->move(imgX + imgW - btnRect.width(), imgY + imgH);
131 }
132 else
133 {
134 startBtn->move(imgX + imgW - btnRect.width(), imgY + imgH - btnRect.height());
135 }
136
137 label->show();
138 startBtn->show();
139 }
140}
