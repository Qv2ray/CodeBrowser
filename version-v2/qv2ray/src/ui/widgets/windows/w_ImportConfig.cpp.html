71QMultiMap<QString, CONFIGROOT> ImportConfigWindow::SelectConnection(bool outboundsOnly)
72{
73 // partial import means only import as an outbound, will set outboundsOnly to
74 // false and disable the checkbox
75 keepImportedInboundCheckBox->setEnabled(!outboundsOnly);
76 groupCombo->setEnabled(false);
77 this->exec();
78 QMultiMap<QString, CONFIGROOT> conn;
79 for (const auto &connEntry : connectionsToNewGroup.values())
80 {
81 conn += connEntry;
82 }
83 for (const auto &connEntry : connectionsToExistingGroup.values())
84 {
85 conn += connEntry;
86 }
87 return result() == Accepted ? conn : QMultiMap<QString, CONFIGROOT>{};

90int ImportConfigWindow::PerformImportConnection()
91{
92 this->exec();
93 int count = 0;
94 for (const auto &groupObject : connectionsToNewGroup)
95 {
96 const auto groupName = connectionsToNewGroup.key(groupObject);
97 GroupId groupId = ConnectionManager->CreateGroup(groupName, false);
98 for (const auto &connConf : groupObject)
99 {
100 auto connName = groupObject.key(connConf);
101
102 auto [protocol, host, port] = GetConnectionInfo(connConf);
103 if (connName.isEmpty())
104 {
105 connName = protocol + "/" + host + ":" + QSTRN(port) + "-" + GenerateRandomString(5);
106 }
107 ConnectionManager->CreateConnection(connConf, connName, groupId, true);
108 }
109 }
110
111 for (const auto &groupObject : connectionsToExistingGroup)
112 {
113 const auto groupId = connectionsToExistingGroup.key(groupObject);
114 for (const auto &connConf : groupObject)
115 {
116 auto connName = groupObject.key(connConf);
117 auto [protocol, host, port] = GetConnectionInfo(connConf);
118 if (connName.isEmpty())
119 {
120 connName = protocol + "/" + host + ":" + QSTRN(port) + "-" + GenerateRandomString(5);
121 }
122 ConnectionManager->CreateConnection(connConf, connName, groupId, true);
123 }
124 }
125
126 return count;
127}

136void ImportConfigWindow::on_qrFromScreenBtn_clicked()
137{
138 bool hideQv2ray = hideQv2rayCB->isChecked();
139
140 if (hideQv2ray)
141 {
142 UIMessageBus.EmitGlobalSignal(QvMBMessage::HIDE_WINDOWS);
143 }
144
145 QApplication::processEvents();
146 QThread::msleep(doubleSpinBox->value() * 1000UL);
147 ScreenShotWindow w;
148 auto pix = w.DoScreenShot();
149 if (hideQv2ray)
150 {
151 UIMessageBus.EmitGlobalSignal(QvMBMessage::SHOW_WINDOWS);
152 }
153
154 if (w.result() == QDialog::Accepted)
155 {
156 auto str = DecodeQRCode(pix);
157 if (str.trimmed().isEmpty())
158 {
159 LOG("Cannot decode QR Code from an image, size:", pix.width(), pix.height());
160 QvMessageBoxWarn(this, tr("Capture QRCode"), tr("Cannot find a valid QRCode from this region."));
161 }
162 else
163 {
164 qrCodeLinkTxt->setText(str.trimmed());
165 }
166 }

198void ImportConfigWindow::on_beginImportBtn_clicked()
199{
200 QString aliasPrefix = nameTxt->text();
201
202 switch (tabWidget->currentIndex())
203 {
204 case LINK_PAGE:
205 {
206 QStringList linkList = SplitLines(vmessConnectionStringTxt->toPlainText());
207 //
208 // Clear UI and error lists
209 linkErrors.clear();
210 vmessConnectionStringTxt->clear();
211 errorsList->clear();
212 //
213 LOG(linkList.count(), "entries found.");
214
215 while (!linkList.isEmpty())
216 {
217 aliasPrefix = nameTxt->text();
218 const auto link = linkList.takeFirst().trimmed();
219 if (link.isEmpty() || link.startsWith("#") || link.startsWith("//"))
220 continue;
221
222 // warn if someone tries to import a https:// link
223 if (link.startsWith("https://"))
224 {
225 errorsList->addItem(tr("WARNING: You may have mistaken 'subscription link' with 'share link'"));
226 }
227
228 QString errMessage;
229 QString newGroupName;
230 const auto config = ConvertConfigFromString(link, &aliasPrefix, &errMessage, &newGroupName);
231
232 // If the config is empty or we have any err messages.
233 if (config.isEmpty() || !errMessage.isEmpty())
234 {
235 // To prevent duplicated values.
236 linkErrors[link] = QSTRN(linkErrors.count() + 1) + ": " + errMessage;
237 continue;
238 }
239 else if (newGroupName.isEmpty())
240 {
241 for (const auto &conf : config)
242 {
243 connectionsToExistingGroup[GroupId{ groupCombo->currentData().toString() }].insert(conf.first, conf.second);
244 }
245 }
246 else
247 {
248 for (const auto &conf : config)
249 {
250 connectionsToNewGroup[newGroupName].insert(conf.first, conf.second);
251 }
252 }
253 }
254
255 if (!linkErrors.isEmpty())
256 {
257 for (const auto &item : qAsConst(linkErrors))
258 {
259 vmessConnectionStringTxt->appendPlainText(linkErrors.key(item));
260 errorsList->addItem(item);
261 }
262
263 vmessConnectionStringTxt->setLineWidth(errorsList->lineWidth());
264 errorsList->sortItems();
265 return;
266 }
267
268 break;
269 }
270#if QV2RAY_FEATURE(ui_has_import_qrcode)
271 case QRCODE_PAGE:
272 {
273 QString errorMsg;
274 const auto root = ConvertConfigFromString(qrCodeLinkTxt->text(), &aliasPrefix, &errorMsg);
275 if (!errorMsg.isEmpty())
276 {
277 QvMessageBoxWarn(this, tr("Failed to import connection"), errorMsg);
278 break;
279 }
280 for (const auto &conf : root)
281 {
282 connectionsToExistingGroup[GroupId{ groupCombo->currentData().toString() }].insert(conf.first, conf.second);
283 }
284 break;
285 }
286#endif
287 case ADVANCED_PAGE:
288 {
289 // From File...
290 bool ImportAsComplex = keepImportedInboundCheckBox->isChecked();
291 const auto path = fileLineTxt->text();
292
293 if (const auto &result = V2RayKernelInstance::ValidateConfig(path); result)
294 {
295 QvMessageBoxWarn(this, tr("Import config file"), *result);
296 return;
297 }
298
299 aliasPrefix += "_" + QFileInfo(path).fileName();
300 CONFIGROOT config = ConvertConfigFromFile(path, ImportAsComplex);
301 connectionsToExistingGroup[GroupId{ groupCombo->currentData().toString() }].insert(aliasPrefix, config);
302 break;
303 }
304 }
305
306 accept();
307}

309void ImportConfigWindow::on_errorsList_currentItemChanged(QListWidgetItem *current, QListWidgetItem *previous)
310{
311 Q_UNUSED(previous)
312
313 if (!current)
314 return;
315
316 const auto currentErrorText = current->text();
317 const auto vmessEntry = linkErrors.key(currentErrorText);
318 //
319 const auto startPos = vmessConnectionStringTxt->toPlainText().indexOf(vmessEntry);
320 const auto endPos = startPos + vmessEntry.length();
321
322 if (startPos < 0)
323 return;
324
325 // Select vmess string that is invalid.
326 auto c = vmessConnectionStringTxt->textCursor();
327 c.setPosition(startPos);
328 c.setPosition(endPos, QTextCursor::KeepAnchor);
329 vmessConnectionStringTxt->setTextCursor(c);
330}
