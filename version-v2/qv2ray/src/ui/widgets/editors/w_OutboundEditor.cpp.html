14OutboundEditor::OutboundEditor(QWidget *parent) : QDialog(parent), tag(OUTBOUND_TAG_PROXY)
15{
16 QvMessageBusConnect(OutboundEditor);
17 setupUi(this);
18 //
19 streamSettingsWidget = new StreamSettingsWidget(this);
20 streamSettingsWidget->SetStreamObject({});
21 transportFrame->addWidget(streamSettingsWidget);
22 //
23 for (const auto &name : PluginHost->UsablePlugins())
24 {
25 const auto &plugin = PluginHost->GetPlugin(name);
26 if (!plugin->hasComponent(COMPONENT_GUI))
27 continue;
28
29 const auto guiInterface = plugin->pluginInterface->GetGUIInterface();
30
31 if (!guiInterface)
32 LOG("Found a plugin with COMPONENT_GUI but returns an invalid GUI interface: " + plugin->metadata.Name);
33
34 if (!guiInterface->GetComponents().contains(GUI_COMPONENT_OUTBOUND_EDITOR))
35 continue;
36
37 const auto editors = guiInterface->GetOutboundEditors();
38 for (const auto &editorInfo : editors)
39 {
40 outBoundTypeCombo->addItem(editorInfo.first.displayName, editorInfo.first.protocol);
41 outboundTypeStackView->addWidget(editorInfo.second);
42 pluginWidgets.insert(editorInfo.first.protocol, editorInfo.second);
43 }
44 }
45 outBoundTypeCombo->model()->sort(0);
46 useForwardProxy = false;
47}

83OUTBOUND OutboundEditor::generateConnectionJson()
84{
85 OUTBOUNDSETTING settings;
86 auto streaming = streamSettingsWidget->GetStreamSettings().toJson();
87 bool processed = false;
88 for (const auto &[protocol, widget] : pluginWidgets.toStdMap())
89 {
90 if (protocol == outboundType)
91 {
92 widget->SetHostAddress(serverAddress, serverPort);
93 settings = OUTBOUNDSETTING(widget->GetContent());
94 const auto prop = widget->property("");
95 const auto hasStreamSettings = GetProperty(widget, "QV2RAY_INTERNAL_HAS_STREAMSETTINGS");
96 if (!hasStreamSettings)
97 streaming = {};
98 processed = true;
99 break;
100 }
101 }
102 if (!processed)
103 {
104 QvMessageBoxWarn(this, tr("Unknown outbound type."),
105 tr("The specified outbound type is not supported, this may happen due to a plugin failure."));
106 }
107 auto root = GenerateOutboundEntry(tag, outboundType, settings, streaming, muxConfig);
108 root[QV2RAY_USE_FPROXY_KEY] = useForwardProxy;
109 return root;
110}

112void OutboundEditor::reloadGUI()
113{
114 tag = originalConfig["tag"].toString();
115 tagTxt->setText(tag);
116 outboundType = originalConfig["protocol"].toString("vmess");
117 muxConfig = originalConfig.contains("mux") ? originalConfig["mux"].toObject() : QJsonObject{};
118 useForwardProxy = originalConfig[QV2RAY_USE_FPROXY_KEY].toBool(false);
119 streamSettingsWidget->SetStreamObject(StreamSettingsObject::fromJson(originalConfig["streamSettings"].toObject()));
120 //
121 useFPCB->setChecked(useForwardProxy);
122 muxEnabledCB->setChecked(muxConfig["enabled"].toBool());
123 muxConcurrencyTxt->setValue(muxConfig["concurrency"].toInt());
124 //
125 const auto &settings = originalConfig["settings"].toObject();
126 //
127 bool processed = false;
128 for (const auto &[protocol, widget] : pluginWidgets.toStdMap())
129 {
130 if (protocol == outboundType)
131 {
132 outBoundTypeCombo->setCurrentIndex(outBoundTypeCombo->findData(protocol));
133 widget->SetContent(settings);
134 const auto &[_address, _port] = widget->GetHostAddress();
135 serverAddress = _address;
136 serverPort = _port;
137 ipLineEdit->setText(_address);
138 portLineEdit->setText(QSTRN(_port));
139 processed = true;
140 break;
141 }
142 }
143 if (!processed)
144 {
145 LOG("Outbound type: ", outboundType, " is not supported.");
146 QvMessageBoxWarn(this, tr("Unknown outbound."),
147 tr("The specified outbound type is invalid, this may be caused by a plugin failure.") + NEWLINE +
148 tr("Please use the JsonEditor or reload the plugin."));
149 reject();
150 }
151}

189void OutboundEditor::on_outBoundTypeCombo_currentIndexChanged(int)
190{
191 outboundType = outBoundTypeCombo->currentData().toString();
192 auto newWidget = pluginWidgets[outboundType];
193 if (!newWidget)
194 return;
195 outboundTypeStackView->setCurrentWidget(newWidget);
196 const auto hasStreamSettings = GetProperty(newWidget, "QV2RAY_INTERNAL_HAS_STREAMSETTINGS");
197 const auto hasForwardProxy = GetProperty(newWidget, "QV2RAY_INTERNAL_HAS_FORWARD_PROXY");
198 streamSettingsWidget->setEnabled(hasStreamSettings);
199 useFPCB->setEnabled(hasForwardProxy);
200 if (!hasForwardProxy)
201 useFPCB->setToolTip(tr("Forward proxy has been disabled when using plugin outbound"));
202}
