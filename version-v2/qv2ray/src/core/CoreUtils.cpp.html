15 bool IsComplexConfig(const CONFIGROOT &root)
16 {
17 bool cRouting = root.contains("routing");
18 bool cRule = cRouting && root["routing"].toObject().contains("rules");
19 bool cRules = cRule && root["routing"].toObject()["rules"].toArray().count() > 0;
20 //
21 bool cInbounds = root.contains("inbounds");
22 bool cInboundCount = cInbounds && root["inbounds"].toArray().count() > 0;
23 //
24 bool cOutbounds = root.contains("outbounds");
25 bool cOutboundCount = cOutbounds && root["outbounds"].toArray().count() > 1;
26 return cRules || cInboundCount || cOutboundCount;
27 }

29 bool GetOutboundInfo(const OUTBOUND &out, QString *host, int *port, QString *protocol)
30 {
31 *protocol = out["protocol"].toString(QObject::tr("N/A")).toLower();
32 bool ok;
33 const auto info = PluginHost->GetOutboundInfo(*protocol, out["settings"].toObject(), ok);
34 if (ok)
35 {
36 *host = info[INFO_SERVER].toString();
37 *port = info[INFO_PORT].toInt();
38 }
39 else
40 {
41 *host = QObject::tr("N/A");
42 *port = 0;
43 }
44 return ok;
45 }

108 const QString GetConnectionProtocolString(const ConnectionId &id)
109 {
110 // Don't bother with the complex connection configs.
111 if (IsComplexConfig(id))
112 {
113 return QV2RAY_SERIALIZATION_COMPLEX_CONFIG_PLACEHOLDER;
114 }
115
116 const auto root = ConnectionManager->GetConnectionRoot(id);
117 const auto outbound = root["outbounds"].toArray().first().toObject();
118
119 QStringList result;
120 result << outbound["protocol"].toString();
121
122 const auto streamSettings = outbound["streamSettings"].toObject();
123
124 if (streamSettings.contains("network"))
125 result << streamSettings["network"].toString();
126
127 const auto security = streamSettings["security"].toString();
128 if (!security.isEmpty() && security != "none")
129 result << streamSettings["security"].toString();
130
131 return result.join("+");
132 }

152 const QMap<QString, ProtocolSettingsInfoObject> GetInboundInfo(const CONFIGROOT &root)
153 {
154 QMap<QString, ProtocolSettingsInfoObject> inboundPorts;
155 for (const auto &inboundVal : root["inbounds"].toArray())
156 {
157 INBOUND in{ inboundVal.toObject() };
158 QString host, protocol;
159 int port;
160 if (GetInboundInfo(in, &host, &port, &protocol))
161 inboundPorts[getTag(in)] = { protocol, host, port };
162 }
163 return inboundPorts;
164 }
