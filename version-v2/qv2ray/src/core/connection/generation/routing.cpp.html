4 QJsonObject GenerateDNS(const QvConfig_DNS &dnsServer)

8 for (const auto &serv : dnsServer.servers)

19 ROUTERULE GenerateSingleRouteRule(RuleType t, const QStringList &rules, const QString &outboundTag, const QString &type)
20 {
21 ROUTERULE root;
22 auto list = rules;
23 list.removeAll("");
24 switch (t)
25 {
26 case RULE_DOMAIN: root.insert("domain", QJsonArray::fromStringList(rules)); break;
27 case RULE_IP: root.insert("ip", QJsonArray::fromStringList(rules)); break;
28 default: Q_UNREACHABLE();
29 }
30 JADD(outboundTag, type)
31 return root;
32 }

35 ROUTING GenerateRoutes(bool enableProxy, bool bypassCN, bool bypassLAN, const QString &outTag, const QvConfig_Route &routeConfig)
36 {
37 ROUTING root;
38 root.insert("domainStrategy", routeConfig.domainStrategy);
39 root.insert("domainMatcher", routeConfig.domainMatcher);
40 //
41 // For Rules list
42 QJsonArray rulesList;
43 if (bypassLAN)
44 rulesList << GenerateSingleRouteRule(RULE_IP, "geoip:private", OUTBOUND_TAG_DIRECT);
45 //
46 if (!enableProxy)
47 {
48 // This is added to disable all proxies, as a alternative influence of #64
49 rulesList << GenerateSingleRouteRule(RULE_DOMAIN, "regexp:.*", OUTBOUND_TAG_DIRECT);
50 rulesList << GenerateSingleRouteRule(RULE_IP, "0.0.0.0/0", OUTBOUND_TAG_DIRECT);
51 rulesList << GenerateSingleRouteRule(RULE_IP, "::/0", OUTBOUND_TAG_DIRECT);
52 }
53 else
54 {
55 //
56 // Blocked.
57 if (!routeConfig.ips.block.isEmpty())
58 rulesList << GenerateSingleRouteRule(RULE_IP, routeConfig.ips.block, OUTBOUND_TAG_BLACKHOLE);
59 if (!routeConfig.domains.block.isEmpty())
60 rulesList << GenerateSingleRouteRule(RULE_DOMAIN, routeConfig.domains.block, OUTBOUND_TAG_BLACKHOLE);
61 //
62 // Proxied
63 if (!routeConfig.ips.proxy.isEmpty())
64 rulesList << GenerateSingleRouteRule(RULE_IP, routeConfig.ips.proxy, outTag);
65 if (!routeConfig.domains.proxy.isEmpty())
66 rulesList << GenerateSingleRouteRule(RULE_DOMAIN, routeConfig.domains.proxy, outTag);
67 //
68 // Directed
69 if (!routeConfig.ips.direct.isEmpty())
70 rulesList << GenerateSingleRouteRule(RULE_IP, routeConfig.ips.direct, OUTBOUND_TAG_DIRECT);
71 if (!routeConfig.domains.direct.isEmpty())
72 rulesList << GenerateSingleRouteRule(RULE_DOMAIN, routeConfig.domains.direct, OUTBOUND_TAG_DIRECT);
73 //
74 // Check if CN needs proxy, or direct.
75 if (bypassCN)
76 {
77 // No proxy agains CN addresses.
78 rulesList << GenerateSingleRouteRule(RULE_IP, "geoip:cn", OUTBOUND_TAG_DIRECT);
79 rulesList << GenerateSingleRouteRule(RULE_DOMAIN, "geosite:cn", OUTBOUND_TAG_DIRECT);
80 }
81 }
82
83 root.insert("rules", rulesList);
84 return root;
85 }
