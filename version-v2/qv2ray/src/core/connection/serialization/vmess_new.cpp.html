18 CONFIGROOT Deserialize(const QString &vmessStr, QString *alias, QString *errMessage)
19 {
20 QUrl url{ vmessStr };
21 QUrlQuery query{ url };
22 //
23#define default CONFIGROOT()
24 if (!url.isValid())
25 {
26 *errMessage = QObject::tr("vmess:// url is invalid");
27 return default;
28 }
29
30 // If previous alias is empty, just the PS is needed, else, append a "_"
31 const auto name = url.fragment(QUrl::FullyDecoded).trimmed();
32 *alias = alias->isEmpty() ? name : (*alias + "_" + name);
33
34 VMessServerObject server;
35 server.users << VMessServerObject::UserObject{};
36
37 StreamSettingsObject stream;
38 QString net;
39 bool tls = false;
40 // Check streamSettings
41 {
42 for (const auto &_protocol : url.userName().split("+"))
43 {
44 if (_protocol == "tls")
45 tls = true;
46 else
47 net = _protocol;
48 }
49 if (!NetworkType.contains(net))
50 {
51 *errMessage = QObject::tr("Invalid streamSettings protocol: ") + net;
52 return default;
53 }
54 // L("net: " << net.toStdString());
55 // L("tls: " << tls);
56 stream.network = net;
57 stream.security = tls ? "tls" : "";
58 }
59 // Host Port UUID AlterID
60 {
61 const auto host = url.host();
62 int port = url.port();
63 QString uuid;
64 int aid;
65 {
66 const auto pswd = url.password();
67 const auto index = pswd.lastIndexOf("-");
68 uuid = pswd.mid(0, index);
69 aid = pswd.right(pswd.length() - index - 1).toInt();
70 }
71 server.address = host;
72 server.port = port;
73 server.users.first().id = uuid;
74 server.users.first().alterId = aid;
75 server.users.first().security = "auto";
76 }
77
78 const static auto getQueryValue = [&query](const QString &key, const QString &defaultValue) {
79 if (query.hasQueryItem(key))
80 return query.queryItemValue(key, QUrl::FullyDecoded);
81 else
82 return defaultValue;
83 };
84
85 //
86 // Begin transport settings parser
87 {
88 if (net == "tcp")
89 {
90 stream.tcpSettings.header.type = getQueryValue("type", "none");
91 }
92 else if (net == "http")
93 {
94 stream.httpSettings.host.append(getQueryValue("host", ""));
95 stream.httpSettings.path = getQueryValue("path", "/");
96 }
97 else if (net == "ws")
98 {
99 stream.wsSettings.headers["Host"] = getQueryValue("host", "");
100 stream.wsSettings.path = getQueryValue("path", "/");
101 }
102 else if (net == "kcp")
103 {
104 stream.kcpSettings.seed = getQueryValue("seed", "");
105 stream.kcpSettings.header.type = getQueryValue("type", "none");
106 }
107 else if (net == "quic")
108 {
109 stream.quicSettings.security = getQueryValue("security", "none");
110 stream.quicSettings.key = getQueryValue("key", "");
111 stream.quicSettings.header.type = getQueryValue("type", "none");
112 }
113 else if (net == "grpc")
114 {
115 stream.grpcSettings.serviceName = getQueryValue("serviceName", "");
116 }
117 else
118 {
119 *errMessage = QObject::tr("Unknown transport method: ") + net;
120 return default;
121 }
122 }
123#undef default
124 if (tls)
125 {
126 stream.tlsSettings.allowInsecure = !FalseTypes.contains(getQueryValue("allowInsecure", "false"));
127 stream.tlsSettings.serverName = getQueryValue("tlsServerName", "");
128 }
129 CONFIGROOT root;
130 OUTBOUNDSETTING vConf;
131 QJsonArray vnextArray;
132 vnextArray.append(server.toJson());
133 vConf["vnext"] = vnextArray;
134 auto outbound = GenerateOutboundEntry(OUTBOUND_TAG_PROXY, "vmess", vConf, stream.toJson());
135 //
136 root["outbounds"] = QJsonArray{ outbound };
137 return root;
138 }

140 const QString Serialize(const StreamSettingsObject &stream, const VMessServerObject &server, const QString &alias)
141 {
142 if (server.users.empty())
143 return QObject::tr("(Empty Users)");
144 QUrl url;
145 QUrlQuery query;
146 url.setFragment(alias, QUrl::StrictMode);
147
148 if (stream.network == "tcp")
149 {
150 if (!stream.tcpSettings.header.type.isEmpty() && stream.tcpSettings.header.type != "none")
151 query.addQueryItem("type", stream.tcpSettings.header.type);
152 }
153 else if (stream.network == "http")
154 {
155 if (!stream.httpSettings.host.isEmpty())
156 query.addQueryItem("host", stream.httpSettings.host.first());
157 query.addQueryItem("path", stream.httpSettings.path.isEmpty() ? "/" : stream.httpSettings.path);
158 }
159 else if (stream.network == "ws")
160 {
161 if (stream.wsSettings.headers.contains("Host") && !stream.wsSettings.headers["Host"].isEmpty())
162 query.addQueryItem("host", stream.wsSettings.headers["Host"]);
163 if (!stream.wsSettings.path.isEmpty() && stream.wsSettings.path != "/")
164 query.addQueryItem("path", stream.wsSettings.path);
165 }
166 else if (stream.network == "kcp")
167 {
168 if (!stream.kcpSettings.seed.isEmpty() && stream.kcpSettings.seed != "")
169 query.addQueryItem("seed", stream.kcpSettings.seed);
170 if (!stream.kcpSettings.header.type.isEmpty() && stream.kcpSettings.header.type != "none")
171 query.addQueryItem("type", stream.kcpSettings.header.type);
172 }
173 else if (stream.network == "quic")
174 {
175 if (!stream.quicSettings.security.isEmpty() && stream.quicSettings.security != "none")
176 query.addQueryItem("security", stream.quicSettings.security);
177 if (!stream.quicSettings.key.isEmpty())
178 query.addQueryItem("key", stream.quicSettings.key);
179 if (!stream.quicSettings.header.type.isEmpty() && stream.quicSettings.header.type != "none")
180 query.addQueryItem("headers", stream.quicSettings.header.type);
181 }
182 else if (stream.network == "grpc")
183 {
184 if (!stream.grpcSettings.serviceName.isEmpty())
185 query.addQueryItem("serviceName", stream.grpcSettings.serviceName);
186 }
187 else
188 {
189 return {};
190 }
191 bool hasTLS = stream.security == "tls" || stream.security == "xtls";
192 auto protocol = stream.network;
193 if (hasTLS)
194 protocol += "+tls";
195 if (stream.security == "tls")
196 {
197 if (!stream.tlsSettings.serverName.isEmpty())
198 query.addQueryItem("tlsServerName", stream.tlsSettings.serverName);
199 }
200 else if (stream.security == "xtls")
201 {
202 if (!stream.xtlsSettings.serverName.isEmpty())
203 query.addQueryItem("tlsServerName", stream.xtlsSettings.serverName);
204 }
205 url.setPath("/");
206 url.setScheme("vmess");
207 url.setPassword(server.users.first().id + "-" + QSTRN(server.users.first().alterId));
208 url.setHost(server.address);
209 url.setPort(server.port);
210 url.setUserName(protocol);
211 url.setQuery(query);
212 return url.toString();
213 }
