2// This file handles some important migration
3// from old to newer versions of Qv2ray.
4//

9#define QV_MODULE_NAME "SettingsUpgrade"
10
11#define UPGRADELOG(msg) LOG("[" + QSTRN(fromVersion) + "-" + QSTRN(fromVersion + 1) + "] --> " + msg)
12
13namespace Qv2ray
14{
15 // Private member
16 QJsonObject UpgradeConfig_Inc(int fromVersion, const QJsonObject &original)
17 {
18 auto root = original;
19 switch (fromVersion)
20 {
21 // Qv2ray version 2, RC 4
22 case 6:
23 {
24 // Moved API Stats port from connectionConfig to apiConfig
25 QJsonObject apiConfig;
26 apiConfig["enableAPI"] = true;
27 apiConfig["statsPort"] = root["connectionConfig"].toObject()["statsPort"].toInt();
28 root["apiConfig"] = apiConfig;
29 break;
30 }
31
32 case 7:
33 {
34 auto lang = root["uiConfig"].toObject()["language"].toString().replace("-", "_");
35 auto uiConfig = root["uiConfig"].toObject();
36 uiConfig["language"] = lang;
37 root["uiConfig"] = uiConfig;
38 UPGRADELOG("Changed language: " + lang);
39 break;
40 }
41
42 // From version 8 to 9, we introduced a lot of new connection metainfo(s)
43 case 8:
44 {
45 // Generate a default group
46 QJsonObject defaultGroup;
47 QStringList defaultGroupConnectionId;
48 defaultGroup["displayName"] = QObject::tr("Default Group");
49 QString defaultGroupId = "000000000000";
50
51 if (!QDir(QV2RAY_CONNECTIONS_DIR + defaultGroupId).exists())
52 {
53 QDir().mkpath(QV2RAY_CONNECTIONS_DIR + defaultGroupId);
54 }
55
56 QString autoStartId;
57 UPGRADELOG("Upgrading connections...");
58 QJsonObject rootConnections;
59
60 for (auto config : root["configs"].toArray())
61 {
62 UPGRADELOG("Migrating: " + config.toString());
63 //
64 // MOVE FILES.
65 // OLD PATH is at QV2RAY_CONFIG_DIR
66 auto filePath = QV2RAY_CONFIG_DIR + config.toString() + QV2RAY_CONFIG_FILE_EXTENSION;
67 auto configFile = QFile(filePath);
68 auto newUuid = GenerateUuid();
69 DEBUG("Generated new UUID: " + newUuid);
70
71 // Check Autostart Id
72 if (root["autoStartConfig"].toObject()["subscriptionName"].toString().isEmpty())
73 {
74 if (root["autoStartConfig"].toObject()["connectionName"].toString() == config.toString())
75 {
76 autoStartId = newUuid;
77 }
78 }
79
80 if (configFile.exists())
81 {
82 auto newPath = QV2RAY_CONNECTIONS_DIR + defaultGroupId + "/" + newUuid + QV2RAY_CONFIG_FILE_EXTENSION;
83 configFile.rename(newPath);
84 UPGRADELOG("Moved: " + filePath + " to " + newPath);
85 }
86 else
87 {
88 UPGRADELOG("WARNING! This file is not found, possible loss of data!");
89 continue;
90 }
91
92 QJsonObject connectionObject;
93 connectionObject["displayName"] = config.toString();
94 defaultGroupConnectionId << newUuid;
95 DEBUG("Pushed uuid: " + newUuid + " to default group.");
96 rootConnections[newUuid] = connectionObject;
97 }
98
99 // Upgrading subscriptions.
100 QJsonObject rootSubscriptions = root.take("subscriptions").toObject();
101 QJsonObject newSubscriptions;
102
103 for (auto i = 0; i < rootSubscriptions.count(); i++)
104 {
105 auto key = rootSubscriptions.keys().at(i);
106 auto value = rootSubscriptions.value(key);
107 //
108 UPGRADELOG("Upgrading subscription: " + key);
109 QString subsUuid = GenerateUuid();
110 QJsonObject subs;
111 QStringList subsConnectionIds;
112 subs["address"] = value["address"].toString();
113 subs["lastUpdated"] = value["lastUpdated"];
114 subs["updateInterval"] = value["updateInterval"];
115 subs["displayName"] = key;
116 //
117 auto baseDirPath = QV2RAY_CONFIG_DIR + "/subscriptions/" + key;
118 auto newDirPath = QV2RAY_CONFIG_DIR + "/subscriptions/" + subsUuid;
119 QDir newDir(newDirPath);
120
121 if (!newDir.exists())
122 {
123 newDir.mkpath(newDirPath);
124 }
125
126 // With extensions
127 auto fileList = GetFileList(baseDirPath);
128
129 // Copy every file within a subscription.
130 for (const auto &fileName : fileList)
131 {
132 auto subsConnectionId = GenerateUuid();
133 auto baseFilePath = baseDirPath + "/" + fileName;
134 auto newFilePath = newDirPath + "/" + subsConnectionId + QV2RAY_CONFIG_FILE_EXTENSION;
135 //
136 QJsonObject subsConnection;
137 subsConnection["displayName"] = fileName.chopped(QString(QV2RAY_CONFIG_FILE_EXTENSION).count());
138 QFile(baseFilePath).rename(newFilePath);
139 UPGRADELOG("Moved subscription file from: " + baseFilePath + " to: " + newFilePath);
140 subsConnectionIds << subsConnectionId;
141 rootConnections[subsConnectionId] = subsConnection;
142
143 //
144
145 // Check Autostart Id
146 if (root["autoStartConfig"].toObject()["subscriptionName"].toString() == key)
147 {
148 if (root["autoStartConfig"].toObject()["connectionName"].toString() == subsConnection["displayName"].toString())
149 {
150 autoStartId = subsConnectionId;
151 }
152 }
153 }
154
155 subs["connections"] = QJsonArray::fromStringList(subsConnectionIds);
156 newSubscriptions[subsUuid] = subs;
157 QDir().rmdir(baseDirPath);
158 }
159
160 defaultGroup["connections"] = QJsonArray::fromStringList(defaultGroupConnectionId);
161 QJsonObject groups;
162 groups[defaultGroupId] = defaultGroup;
163 root["autoStartId"] = autoStartId;
164 root["groups"] = groups;
165 root["connections"] = rootConnections;
166 root["subscriptions"] = newSubscriptions;
167 UPGRADELOG("Finished upgrading config, version 8.");
168 break;
169 }
170
171 // Added cross-platform vCore and vAssets path support;
172 case 9:
173 {
174 QJsonObject kernelConfig;
175#ifdef Q_OS_LINUX
176#define _VARNAME_VCOREPATH_ kernelConfig["v2CorePath_linux"]
177#define _VARNAME_VASSETSPATH_ kernelConfig["v2AssetsPath_linux"]
178 UPGRADELOG("Update kernel and assets paths for linux");
179#elif defined(Q_OS_MACOS)
180#define _VARNAME_VCOREPATH_ kernelConfig["v2CorePath_macx"]
181#define _VARNAME_VASSETSPATH_ kernelConfig["v2AssetsPath_macx"]
182 UPGRADELOG("Update kernel and assets paths for macOS");
183#elif defined(Q_OS_WIN)
184#define _VARNAME_VCOREPATH_ kernelConfig["v2CorePath_win"]
185#define _VARNAME_VASSETSPATH_ kernelConfig["v2AssetsPath_win"]
186 UPGRADELOG("Update kernel and assets paths for Windows");
187#endif
188 _VARNAME_VCOREPATH_ = root["v2CorePath"].toString();
189 _VARNAME_VASSETSPATH_ = root["v2AssetsPath"].toString();
190 //
191 root["kernelConfig"] = kernelConfig;
192#undef _VARNAME_VCOREPATH_
193#undef _VARNAME_VASSETSPATH_
194 break;
195 }
196 case 10:
197 {
198 // It's worth that use a seperated config number to notify all users about the deprecation of PAC
199
200 if (root["inboundConfig"].toObject()["pacConfig"].toObject()["enablePAC"].toBool(false))
201 {
202 QvMessageBoxWarn(
203 nullptr, QObject::tr("Deprecated"),
204 QObject::tr("PAC is now deprecated and is not encouraged to be used anymore.") + NEWLINE +
205 QObject::tr("It will be removed or be provided as a plugin in the future.") + NEWLINE + NEWLINE +
206 QObject::tr("PAC will still work currently, but please switch to the V2Ray built-in routing as soon as possible."));
207 }
208 else
209 {
210 UPGRADELOG("It's a wise choice not to use PAC.");
211 }
212 break;
213 }
214
215 // Splitted Qv2ray.conf,
216 case 11:
217 {
218 // Process AutoStartSettings
219 ConnectionGroupPair autoStartIdPair{ ConnectionId{ root["autoStartId"].toString() }, NullGroupId };
220
221 // Process connection entries.
222 //
223 {
224 // Moved root["connections"] into separated file: $QV2RAY_CONFIG_PATH/connections.json
225 QDir connectionsDir(QV2RAY_CONNECTIONS_DIR);
226 if (!connectionsDir.exists())
227 {
228 connectionsDir.mkpath(QV2RAY_CONNECTIONS_DIR);
229 }
230 const auto connectionsArray = root["connections"].toObject().keys();
231 QJsonObject newConnectionsArray;
232 ///
233 /// Connection.json
234 /// {
235 /// "connections" : [
236 /// {ID1, connectionObject 1},
237 /// {ID2, connectionObject 2},
238 /// {ID3, connectionObject 3},
239 /// {ID4, connectionObject 4},
240 /// ]
241 /// }
242 ///
243 for (const auto &connectionVal : connectionsArray)
244 {
245 // Config file migrations:
246 // Connection Object:
247 // importDate --> creationDate
248 // lastUpdatedDate --> now
249 //
250 auto connection = root["connections"].toObject()[connectionVal].toObject();
251 connection["creationDate"] = connection.take("importDate");
252 connection["lastUpdatedDate"] = (qint64) system_clock::to_time_t(system_clock::now());
253 UPGRADELOG("Migrating connection: " + connectionVal + " -- " + connection["displayName"].toString());
254 newConnectionsArray[connectionVal] = connection;
255 }
256 QJsonObject ConnectionJsonObject;
257 root["connections"] = QJsonArray::fromStringList(connectionsArray);
258 //
259 // Store Connection.json
260 StringToFile(JsonToString(newConnectionsArray), QV2RAY_CONFIG_DIR + "connections.json");
261 }
262 // Merged groups and subscriptions. $QV2RAY_GROUPS_PATH + groupId.json
263 {
264 // Susbcription Object
265 // Doesn't exist anymore, convert into normal group Object.
266 //
267 QMap<QString, QJsonObject> ConnectionsCache;
268 QJsonObject allGroupsObject;
269 const auto subscriptionKeys = root["subscriptions"].toObject().keys();
270 for (const auto &key : subscriptionKeys)
271 {
272 auto aSubscription = root["subscriptions"].toObject()[key].toObject();
273 QJsonObject subscriptionSettings;
274 subscriptionSettings["address"] = aSubscription.take("address");
275 subscriptionSettings["updateInterval"] = aSubscription.take("updateInterval");
276 aSubscription["lastUpdatedDate"] = (qint64) system_clock::to_time_t(system_clock::now());
277 aSubscription["creationDate"] = (qint64) system_clock::to_time_t(system_clock::now());
278 aSubscription["subscriptionOption"] = subscriptionSettings;
279 UPGRADELOG("Migrating subscription: " + key + " -- " + aSubscription["displayName"].toString());
280 //
281 if (autoStartIdPair.groupId != NullGroupId &&
282 aSubscription["connections"].toArray().contains(autoStartIdPair.connectionId.toString()))
283 {
284 autoStartIdPair.groupId = GroupId{ key };
285 }
286 //
287 for (const auto &cid : aSubscription["connections"].toArray())
288 {
289 ConnectionsCache[cid.toString()] = JsonFromString(
290 StringFromFile(QV2RAY_CONFIG_DIR + "subscriptions/" + key + "/" + cid.toString() + QV2RAY_CONFIG_FILE_EXTENSION));
291 }
292 //
293 allGroupsObject[key] = aSubscription;
294 }
295 //
296 root.remove("subscriptions");
297 //
298 const auto groupKeys = root["groups"].toObject().keys();
299 for (const auto &key : groupKeys)
300 {
301 // Group Object
302 // connections ---> ConnectionID
303 // idSubscription ---> if the group is a subscription
304 // subscriptionSettings ---> Originally SubscriptionObject
305 // creationDate ---> Now
306 // lastUpdateDate ---> Now
307 auto aGroup = root["groups"].toObject()[key].toObject();
308 aGroup["lastUpdatedDate"] = (qint64) system_clock::to_time_t(system_clock::now());
309 aGroup["creationDate"] = (qint64) system_clock::to_time_t(system_clock::now());
310 UPGRADELOG("Migrating group: " + key + " -- " + aGroup["displayName"].toString());
311 //
312 if (autoStartIdPair.groupId != NullGroupId &&
313 aGroup["connections"].toArray().contains(autoStartIdPair.connectionId.toString()))
314 {
315 autoStartIdPair.groupId = GroupId{ key };
316 }
317 for (const auto &cid : aGroup["connections"].toArray())
318 {
319 ConnectionsCache[cid.toString()] = JsonFromString(
320 StringFromFile(QV2RAY_CONFIG_DIR + "connections/" + key + "/" + cid.toString() + QV2RAY_CONFIG_FILE_EXTENSION));
321 }
322 //
323 allGroupsObject[key] = aGroup;
324 }
325 //
326 StringToFile(JsonToString(allGroupsObject), QV2RAY_CONFIG_DIR + "groups.json");
327 //
328 root.remove("groups"); //
329 UPGRADELOG("Removing unused directory");
330 QDir(QV2RAY_CONFIG_DIR + "subscriptions/").removeRecursively();
331 QDir(QV2RAY_CONFIG_DIR + "connections/").removeRecursively();
332 //
333 QDir().mkpath(QV2RAY_CONFIG_DIR + "connections/");
334 //
335 //
336 // FileSystem Migrations
337 // Move all files in GROUPS / SUBSCRIPTION subfolders into CONNECTIONS.
338 // Only Store (connections.json in CONFIG_PATH) and ($groupID.json in GROUP_PATH)
339 for (const auto &cid : ConnectionsCache.keys())
340 {
341 StringToFile(JsonToString(ConnectionsCache[cid]), QV2RAY_CONFIG_DIR + "connections/" + cid + QV2RAY_CONFIG_FILE_EXTENSION);
342 }
343 //
344 }
345
346 //
347 // Main Object
348 // Drop recentConnections since it's ill-formed and not supported yet.
349 // convert autoStartId into ConnectionGroupPair / instead of QString
350 // Remove subscriptions item.
351 root.remove("recentConnections");
352 root["autoStartId"] = autoStartIdPair.toJson();
353 // 1 here means FIXED
354 root["autoStartBehavior"] = 1;
355
356 // Moved apiConfig into kernelConfig
357 auto kernelConfig = root["kernelConfig"].toObject();
358 kernelConfig["enableAPI"] = root["apiConfig"].toObject()["enableAPI"];
359 kernelConfig["statsPort"] = root["apiConfig"].toObject()["statsPort"];
360 root["kernelConfig"] = kernelConfig;
361 UPGRADELOG("Finished upgrading config file for Qv2ray Group Routing update.");
362 break;
363 }
364 case 12:
365 {
366 auto inboundConfig = root["inboundConfig"].toObject();
367 //
368 QJsonObject socksSettings;
369 QJsonObject httpSettings;
370 QJsonObject tProxySettings;
371 QJsonObject systemProxySettings;
372 systemProxySettings["setSystemProxy"] = inboundConfig["setSystemProxy"];
373 //
374 socksSettings["port"] = inboundConfig["socks_port"];
375 socksSettings["useAuth"] = inboundConfig["socks_useAuth"];
376 socksSettings["enableUDP"] = inboundConfig["socksUDP"];
377 socksSettings["localIP"] = inboundConfig["socksLocalIP"];
378 socksSettings["account"] = inboundConfig["socksAccount"];
379 socksSettings["sniffing"] = inboundConfig["socksSniffing"];
380 //
381 httpSettings["port"] = inboundConfig["http_port"];
382 httpSettings["useAuth"] = inboundConfig["http_useAuth"];
383 httpSettings["account"] = inboundConfig["httpAccount"];
384 httpSettings["sniffing"] = inboundConfig["httpSniffing"];
385 //
386 tProxySettings["tProxyIP"] = inboundConfig["tproxy_ip"];
387 tProxySettings["port"] = inboundConfig["tproxy_port"];
388 tProxySettings["hasTCP"] = inboundConfig["tproxy_use_tcp"];
389 tProxySettings["hasUDP"] = inboundConfig["tproxy_use_udp"];
390 tProxySettings["followRedirect"] = inboundConfig["tproxy_followRedirect"];
391 tProxySettings["mode"] = inboundConfig["tproxy_mode"];
392 tProxySettings["dnsIntercept"] = inboundConfig["dnsIntercept"];
393 //
394 inboundConfig["systemProxySettings"] = systemProxySettings;
395 inboundConfig["socksSettings"] = socksSettings;
396 inboundConfig["httpSettings"] = httpSettings;
397 inboundConfig["tProxySettings"] = tProxySettings;
398 //
399 root["inboundConfig"] = inboundConfig;
400 break;
401 }
402 case 13:
403 {
404 const auto dnsList = QJsonIO::GetValue(root, "connectionConfig", "dnsList").toArray();
405 auto connectionConfig = root["connectionConfig"].toObject();
406 QJsonObject defaultRouteConfig;
407 defaultRouteConfig["forwardProxyConfig"] = connectionConfig.take("forwardProxyConfig");
408 defaultRouteConfig["routeConfig"] = connectionConfig.take("routeConfig");
409
410 for (auto i = 0; i < dnsList.count(); i++)
411 {
412 QJsonIO::SetValue(defaultRouteConfig, dnsList[i], "dnsConfig", "servers", i, "address");
413 QJsonIO::SetValue(defaultRouteConfig, false, "dnsConfig", "servers", i, "QV2RAY_DNS_IS_COMPLEX_DNS");
414 }
415 root["defaultRouteConfig"] = defaultRouteConfig;
416 break;
417 }
418 default:
419 {
420 //
421 // Due to technical issue, we cannot maintain all of those upgrade processes anymore. Check
422 // https://github.com/Qv2ray/Qv2ray/issues/353#issuecomment-586117507
423 // for more information, see commit 2f716a9a443b71ddb96aaab081de73c0095cb637
424 //
425 QvMessageBoxWarn(nullptr, QObject::tr("Configuration Upgrade Failed"),
426 QObject::tr("Unsupported config version number: ") + QSTRN(fromVersion) + NEWLINE + NEWLINE +
427 QObject::tr("Please upgrade firstly up to Qv2ray v2.0/v2.1 and try again."));
428 LOG("The configuration version of your old Qv2ray installation is out-of-date and that"
429 " version is not supported anymore, please try to update to an intermediate version of Qv2ray first.");
430 qApp->exit(1);
431 }
432 }
433 root["config_version"] = root["config_version"].toInt() + 1;
434 return root;
435 }
436
437 // Exported function
438 QJsonObject UpgradeSettingsVersion(int fromVersion, int toVersion, const QJsonObject &original)
439 {
440 auto root = original;
441 LOG("Migrating config from version ", fromVersion, "to", toVersion);
442
443 for (int i = fromVersion; i < toVersion; i++)
444 {
445 root = UpgradeConfig_Inc(i, root);
446 }
447
448 return root;
449 }
