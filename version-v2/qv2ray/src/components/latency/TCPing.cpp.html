11 int getSocket(int af, int socktype, int proto)
12 {
13 uv_os_sock_t fd;
14#ifndef INVALID_SOCKET
15#define INVALID_SOCKET -1
16#endif
17 if ((fd = socket(af, socktype, proto)) == INVALID_SOCKET)
18 {
19 return 0;
20 }
21
22 // Set TCP connection timeout per-socket level.
23 // See [https://github.com/libuv/help/issues/54] for details.
24#if defined(_WIN32) && !defined(__SYMBIAN32__)
25#ifndef TCP_MAXRT
26#define TCP_MAXRT 5
27#endif
28 setsockopt(fd, IPPROTO_TCP, TCP_MAXRT, (char *) &conn_timeout_sec, sizeof(conn_timeout_sec));
29#elif defined(__APPLE__)
30 // (billhoo) MacOS uses TCP_CONNECTIONTIMEOUT to do so.
31 setsockopt(fd, IPPROTO_TCP, TCP_CONNECTIONTIMEOUT, (char *) &conn_timeout_sec, sizeof(conn_timeout_sec));
32#else // Linux like systems
33 uint32_t conn_timeout_ms = conn_timeout_sec * 1000;
34 setsockopt(fd, IPPROTO_TCP, TCP_USER_TIMEOUT, (char *) &conn_timeout_ms, sizeof(conn_timeout_ms));
35#endif
36 return (int) fd;
37 }

57 {

67 void TCPing::ping()
68 {
69 for (; data.totalCount < req.totalCount; ++data.totalCount)
70 {
71 auto tcpClient = loop->resource<uvw::TCPHandle>();
72 tcpClient->open(getSocket(af, SOCK_STREAM, IPPROTO_TCP));
73 tcpClient->once<uvw::ErrorEvent>([ptr = shared_from_this(), this](const uvw::ErrorEvent &e, uvw::TCPHandle &h) {
74 LOG("error connecting to host: " + req.host + ":" + QSTRN(req.port) + " " + e.what());
75 data.failedCount += 1;
76 data.errorMessage = e.what();
77 notifyTestHost();
78 h.clear();
79 h.close();
80 });
81 tcpClient->once<uvw::ConnectEvent>([ptr = shared_from_this(), start = system_clock::now(), this](auto &, auto &h) {
82 ++successCount;
83 system_clock::time_point end = system_clock::now();
84 auto milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
85 long ms = milliseconds.count();
86 data.avg += ms;
87 data.worst = std::max(data.worst, ms);
88 data.best = std::min(data.best, ms);
89 notifyTestHost();
90 h.clear();
91 h.close();
92 });
93 tcpClient->connect(reinterpret_cast<const sockaddr &>(storage));
94 }
