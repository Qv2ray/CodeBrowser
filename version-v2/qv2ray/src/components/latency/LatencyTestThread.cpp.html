30 void LatencyTestThread::run()
31 {
32 loop = uvw::Loop::create();
33 stopTimer = loop->resource<uvw::TimerHandle>();
34 stopTimer->on<uvw::TimerEvent>([this](auto &, auto &handle) {
35 if (isStop)
36 {
37 if (!requests.empty())
38 requests.clear();
39 int timer_count = 0;
40 uv_walk(
41 loop->raw(),
42 [](uv_handle_t *handle, void *arg) {
43 int &counter = *static_cast<int *>(arg);
44 if (uv_is_closing(handle) == 0)
45 counter++;
46 },
47 &timer_count);
48 if (timer_count == 1) // only current timer
49 {
50 handle.stop();
51 handle.close();
52 loop->clear();
53 loop->close();
54 loop->stop();
55 }
56 }
57 else
58 {
59 if (requests.empty())
60 return;
61 std::unique_lock<std::mutex> lockGuard{ m };
62 auto parent = qobject_cast<LatencyTestHost *>(this->parent());
63 for (auto &req : requests)
64 {
65 switch (req.method)
66 {
67 case ICMPING:
68 {
69 auto ptr = std::make_shared<icmping::ICMPPing>(loop, req, parent);
70 ptr->start();
71 }
72 break;
73 case TCPING:
74 default:
75 {
76 auto ptr = std::make_shared<tcping::TCPing>(loop, req, parent);
77 ptr->start();
78 break;
79 }
80 case REALPING:
81 {
82 auto ptr = std::make_shared<realping::RealPing>(loop, req, parent);
83 ptr->start();
84 break;
85 }
86 }
87 }
88 requests.clear();
89 }
90 });
91 stopTimer->start(uvw::TimerHandle::Time{ 500 }, uvw::TimerHandle::Time{ 500 });
92 loop->run();
93 }
