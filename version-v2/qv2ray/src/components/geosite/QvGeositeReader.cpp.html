11 QMap<QString, QStringList> GeositeEntries;
12 QStringList ReadGeoSiteFromFile(const QString &filepath)
13 {
14 if (GeositeEntries.contains(filepath))
15 {
16 return GeositeEntries.value(filepath);
17 }
18 else
19 {
20 QStringList list;
21#ifndef ANDROID
22 LOG("Reading geosites from: " + filepath);
23 //
24 GOOGLE_PROTOBUF_VERIFY_VERSION;
25 //
26 QFile f(filepath);
27 bool opened = f.open(QFile::OpenModeFlag::ReadOnly);
28
29 if (!opened)
30 {
31 LOG("File cannot be opened: " + filepath);
32 return list;
33 }
34
35 auto content = f.readAll();
36 f.close();
37 //
38 v2ray::core::app::router::GeoSiteList sites;
39 sites.ParseFromArray(content.data(), content.size());
40
41 for (const auto &e : sites.entry())
42 {
43 // We want to use lower string.
44 list << QString::fromStdString(e.country_code()).toLower();
45 }
46
47 LOG("Loaded " + QSTRN(list.count()) + " geosite entries from data file.");
48 // Optional: Delete all global objects allocated by libprotobuf.
49 google::protobuf::ShutdownProtobufLibrary();
50#endif
51 list.sort();
52 GeositeEntries[filepath] = list;
53 return list;
54 }
55 }
