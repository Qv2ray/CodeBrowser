26 void QvUpdateChecker::CheckUpdate()
27 {
28#ifndef DISABLE_AUTO_UPDATE
29 if (QFile(QV2RAY_CONFIG_DIR + "QV2RAY_FEATURE_DISABLE_AUTO_UPDATE").exists())
30 return;
31 const auto &updateChannel = GlobalConfig.updateConfig.updateChannel;
32 LOG("Start checking update for channel ID: " + QSTRN(updateChannel));
33 requestHelper->AsyncHttpGet(UpdateChannelLink[updateChannel], &QvUpdateChecker::VersionUpdate);
34#endif

37 void QvUpdateChecker::VersionUpdate(const QByteArray &data)
38 {
39 // Version update handler.
40 const auto doc = QJsonDocument::fromJson(data);
41 const auto root = doc.isArray() ? doc.array().first().toObject() : doc.object();
42 if (root.isEmpty())
43 return;
44
45 const auto newVersionStr = root["tag_name"].toString("v").mid(1);
46 const auto currentVersionStr = QString(QV2RAY_VERSION_STRING);
47 const auto ignoredVersionStr = GlobalConfig.updateConfig.ignoredVersion.isEmpty() ? "0.0.0" : GlobalConfig.updateConfig.ignoredVersion;
48 //
49 bool hasUpdate = false;
50 try
51 {
52 const auto newVersion = semver::version::from_string(newVersionStr.toStdString());
53 const auto currentVersion = semver::version::from_string(currentVersionStr.toStdString());
54 const auto ignoredVersion = semver::version::from_string(ignoredVersionStr.toStdString());
55 //
56 LOG(QString("Received update info:") + NEWLINE + //
57 " --> Latest: " + newVersionStr + NEWLINE + //
58 " --> Current: " + currentVersionStr + NEWLINE + //
59 " --> Ignored: " + ignoredVersionStr);
60 // If the version is newer than us.
61 // And new version is newer than the ignored version.
62 hasUpdate = (newVersion > currentVersion && newVersion > ignoredVersion);
63 }
64 catch (...)
65 {
66 LOG("Some strange exception occured, cannot check update.");
67 }
68 if (hasUpdate)
69 {
70 const auto name = root["name"].toString("");
71 if (name.contains("NO_RELEASE"))
72 {
73 LOG("Found the recent release title with NO_RELEASE tag. Ignoring");
74 return;
75 }
76 const auto link = root["html_url"].toString("");
77 const auto versionMessage =
78 QString("A new version of Qv2ray has been found:" NEWLINE "v%1" NEWLINE NEWLINE "%2" NEWLINE "------------" NEWLINE "%3")
79 .arg(newVersionStr)
80 .arg(name)
81 .arg(root["body"].toString());
82
83 const auto result = QvMessageBoxAsk(nullptr, tr("Qv2ray Update"), versionMessage, { Yes, No, Ignore });
84 if (result == Yes)
85 {
86 QvCoreApplication->OpenURL(link);
87 }
88 else if (result == Ignore)
89 {
90 // Set and save ingored version.
91 GlobalConfig.updateConfig.ignoredVersion = newVersionStr;
92 SaveGlobalSettings();
93 }
94 }
95 else
96 {
97 LOG("No suitable updates found on channel " + QSTRN(GlobalConfig.updateConfig.updateChannel));
98 }
99 }
