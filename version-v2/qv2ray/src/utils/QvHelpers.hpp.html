73 inline bool IsIPv6Address(const QString &addr)
74 {
75 QHostAddress address(addr);
76 return QAbstractSocket::IPv6Protocol == address.protocol();
77 }
78
79 inline bool IsValidIPAddress(const QString &addr)
80 {

82 }

84 inline bool IsValidDNSServer(const QString &addr)
85 {
86 return IsIPv4Address(addr) //
87 || IsIPv6Address(addr) //
88 || addr.startsWith("https://") //
89 || addr.startsWith("https+local://") //
90 || addr.startsWith("quic+local://") //
91 || addr == "localhost" //
92 || addr == "fakedns";
93 }
