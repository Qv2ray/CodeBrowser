37void SocksInboundEditor::on_socksRemoveUserBtn_clicked()
38{
39 PLUGIN_EDITOR_LOADING_GUARD
40 if (socksAccountListBox->currentRow() != -1)
41 {
42 auto item = socksAccountListBox->currentItem();
43 auto list = content["accounts"].toArray();
44
45 for (int i = 0; i < list.count(); i++)
46 {
47 auto user = list[i].toObject();
48 auto entry = user["user"].toString() + ":" + user["pass"].toString();
49
50 if (entry == item->text().trimmed())
51 {
52 list.removeAt(i);
53 content["accounts"] = list;
54 socksAccountListBox->takeItem(socksAccountListBox->currentRow());
55 return;
56 }
57 }
58 }
59 else
60 {
61 InternalProtocolSupportPluginInstance->PluginErrorMessageBox(tr("Removing a user"), tr("You haven't selected a user yet."));
62 }
63}

65void SocksInboundEditor::on_socksAddUserBtn_clicked()
66{
67 PLUGIN_EDITOR_LOADING_GUARD
68 auto user = socksAddUserTxt->text();
69 auto pass = socksAddPasswordTxt->text();
70 //
71 auto list = content["accounts"].toArray();
72
73 for (int i = 0; i < list.count(); i++)
74 {
75 auto _user = list[i].toObject();
76
77 if (_user["user"].toString() == user)
78 {
79 InternalProtocolSupportPluginInstance->PluginErrorMessageBox(tr("Add a user"), tr("This user exists already."));
80 return;
81 }
82 }
83
84 socksAddUserTxt->clear();
85 socksAddPasswordTxt->clear();
86 QJsonObject entry;
87 entry["user"] = user;
88 entry["pass"] = pass;
89 list.append(entry);
90 socksAccountListBox->addItem(user + ":" + pass);
91 content["accounts"] = list;
92}
