22void VmessOutboundEditor::SetContent(const QJsonObject &content)
23{
24 this->content = content;
25 PLUGIN_EDITOR_LOADING_SCOPE({
26 if (content["vnext"].toArray().isEmpty())
27 content["vnext"] = QJsonArray{ QJsonObject{} };
28 vmess = VMessServerObject::fromJson(content["vnext"].toArray().first().toObject());
29 if (vmess.users.empty())
30 vmess.users.push_back({});
31 const auto &user = vmess.users.front();
32 idLineEdit->setText(user.id);
33 alterLineEdit->setValue(user.alterId);
34
35 securityCombo->setCurrentText(user.security);
36 })
37
38 if (alterLineEdit->value() > 0)
39 {
40 const auto msg = tr("VMess MD5 with Non-zero AlterID has been deprecated, please use VMessAEAD.");
41 InternalProtocolSupportPluginInstance->PluginErrorMessageBox(tr("Non AEAD VMess detected"), msg);
42 }
43}

45void VmessOutboundEditor::on_idLineEdit_textEdited(const QString &arg1)
46{
47 const static QRegularExpression regExpUUID("^[0-9a-f]{8}(-[0-9a-f]{4}){3}-[0-9a-f]{12}$", QRegularExpression::CaseInsensitiveOption);
48
49 if (!regExpUUID.match(arg1).hasMatch())
50 {
51 RED(idLineEdit)
52 }
53 else
54 {
55 BLACK(idLineEdit)
56 }
57 if (vmess.users.isEmpty())
58 vmess.users << VMessServerObject::UserObject{};
59 vmess.users.front().id = arg1;
60}
