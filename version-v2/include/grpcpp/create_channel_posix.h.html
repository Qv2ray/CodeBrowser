28namespace grpc {
29
30#ifdef GPR_SUPPORT_CHANNELS_FROM_FD
31
32/// Create a new \a Channel communicating over the given file descriptor.
33///
34/// \param target The name of the target.
35/// \param fd The file descriptor representing a socket.
36std::shared_ptr<Channel> CreateInsecureChannelFromFd(const grpc::string& target,
37 int fd);
38
39/// Create a new \a Channel communicating over given file descriptor with custom
40/// channel arguments.
41///
42/// \param target The name of the target.
43/// \param fd The file descriptor representing a socket.
44/// \param args Options for channel creation.
45std::shared_ptr<Channel> CreateCustomInsecureChannelFromFd(
46 const grpc::string& target, int fd, const ChannelArguments& args);
47
48namespace experimental {
49
50/// Create a new \a Channel communicating over given file descriptor with custom
51/// channel arguments.
52///
53/// \param target The name of the target.
54/// \param fd The file descriptor representing a socket.
55/// \param args Options for channel creation.
56/// \param interceptor_creators Vector of interceptor factory objects.
57std::shared_ptr<Channel> CreateCustomInsecureChannelWithInterceptorsFromFd(
58 const grpc::string& target, int fd, const ChannelArguments& args,
59 std::unique_ptr<std::vector<
60 std::unique_ptr<experimental::ClientInterceptorFactoryInterface>>>
61 interceptor_creators);
62
63} // namespace experimental

65#endif // GPR_SUPPORT_CHANNELS_FROM_FD

67} // namespace grpc
