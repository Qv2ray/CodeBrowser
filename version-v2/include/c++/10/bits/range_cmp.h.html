43 // Define std::identity here so that <iterator> and <ranges>

46 /// [func.identity] The identity function.
47 struct identity
48 {
49 template<typename _Tp>
50 constexpr _Tp&&
51 operator()(_Tp&& __t) const noexcept
52 { return std::forward<_Tp>(__t); }
53
54 using is_transparent = __is_transparent;
55 };

61namespace ranges
62{
63 namespace __detail
64 {
65 // BUILTIN-PTR-CMP(T, ==, U)
66 template<typename _Tp, typename _Up>
67 concept __eq_builtin_ptr_cmp
68 = requires (_Tp&& __t, _Up&& __u) { { __t == __u } -> same_as<bool>; }
69 && convertible_to<_Tp, const volatile void*>
70 && convertible_to<_Up, const volatile void*>
71 && (! requires(_Tp&& __t, _Up&& __u)
72 { operator==(std::forward<_Tp>(__t), std::forward<_Up>(__u)); }
73 &&
74 ! requires(_Tp&& __t, _Up&& __u)
75 { std::forward<_Tp>(__t).operator==(std::forward<_Up>(__u)); });
76
77 // BUILTIN-PTR-CMP(T, <, U)
78 template<typename _Tp, typename _Up>
79 concept __less_builtin_ptr_cmp
80 = requires (_Tp&& __t, _Up&& __u) { { __t < __u } -> same_as<bool>; }
81 && convertible_to<_Tp, const volatile void*>
82 && convertible_to<_Up, const volatile void*>
83 && (! requires(_Tp&& __t, _Up&& __u)
84 { operator<(std::forward<_Tp>(__t), std::forward<_Up>(__u)); }
85 && ! requires(_Tp&& __t, _Up&& __u)
86 { std::forward<_Tp>(__t).operator<(std::forward<_Up>(__u)); });
87 } // namespace __detail

89 // [range.cmp] Concept-constrained comparisons
90
91 /// ranges::equal_to function object type.
92 struct equal_to
93 {
94 template<typename _Tp, typename _Up>
95 requires equality_comparable_with<_Tp, _Up>
96 || __detail::__eq_builtin_ptr_cmp<_Tp, _Up>
97 constexpr bool
98 operator()(_Tp&& __t, _Up&& __u) const
99 noexcept(noexcept(std::declval<_Tp>() == std::declval<_Up>()))
100 { return std::forward<_Tp>(__t) == std::forward<_Up>(__u); }
101
102 using is_transparent = __is_transparent;
103 };
104
105 /// ranges::not_equal_to function object type.
106 struct not_equal_to
107 {
108 template<typename _Tp, typename _Up>
109 requires equality_comparable_with<_Tp, _Up>
110 || __detail::__eq_builtin_ptr_cmp<_Tp, _Up>
111 constexpr bool
112 operator()(_Tp&& __t, _Up&& __u) const
113 noexcept(noexcept(std::declval<_Up>() == std::declval<_Tp>()))
114 { return !equal_to{}(std::forward<_Tp>(__t), std::forward<_Up>(__u)); }
115
116 using is_transparent = __is_transparent;
117 };
118
119 /// ranges::less function object type.
120 struct less
121 {
122 template<typename _Tp, typename _Up>
123 requires totally_ordered_with<_Tp, _Up>
124 || __detail::__less_builtin_ptr_cmp<_Tp, _Up>
125 constexpr bool
126 operator()(_Tp&& __t, _Up&& __u) const
127 noexcept(noexcept(std::declval<_Tp>() < std::declval<_Up>()))
128 {
129 if constexpr (__detail::__less_builtin_ptr_cmp<_Tp, _Up>)
130 {
131#ifdef __cpp_lib_is_constant_evaluated
132 if (std::is_constant_evaluated())
133 return __t < __u;
134#endif
135 auto __x = reinterpret_cast<__UINTPTR_TYPE__>(
136 static_cast<const volatile void*>(std::forward<_Tp>(__t)));
137 auto __y = reinterpret_cast<__UINTPTR_TYPE__>(
138 static_cast<const volatile void*>(std::forward<_Up>(__u)));
139 return __x < __y;
140 }
141 else
142 return std::forward<_Tp>(__t) < std::forward<_Up>(__u);
143 }
144
145 using is_transparent = __is_transparent;
146 };
147
148 /// ranges::greater function object type.
149 struct greater
150 {
151 template<typename _Tp, typename _Up>
152 requires totally_ordered_with<_Tp, _Up>
153 || __detail::__less_builtin_ptr_cmp<_Up, _Tp>
154 constexpr bool
155 operator()(_Tp&& __t, _Up&& __u) const
156 noexcept(noexcept(std::declval<_Up>() < std::declval<_Tp>()))
157 { return less{}(std::forward<_Up>(__u), std::forward<_Tp>(__t)); }
158
159 using is_transparent = __is_transparent;
160 };

168 constexpr bool
169 operator()(_Tp&& __t, _Up&& __u) const
170 noexcept(noexcept(std::declval<_Tp>() < std::declval<_Up>()))
171 { return !less{}(std::forward<_Tp>(__t), std::forward<_Up>(__u)); }
172
