35// This file contains classes which describe a type of protocol message.
36// You can use a message's descriptor to learn at runtime what fields
37// it contains and what the types of those fields are. The Message
38// interface also allows you to dynamically access and modify individual
39// fields by passing the FieldDescriptor of the field you are interested
40// in.
41//
42// Most users will not care about descriptors, because they will write
43// code specific to certain protocol types and will simply use the classes
44// generated by the protocol compiler directly. Advanced users who want
45// to operate on arbitrary types (not known at compile time) may want to
46// read descriptors in order to learn about the contents of a message.
47// A very small number of users will want to construct their own
48// Descriptors, either because they are implementing Message manually or
49// because they are writing something like the protocol compiler.
50//
51// For an example of how you might use descriptors, see the code example
52// at the top of message.h.

169// A class to handle the simplest cases of a lazily linked descriptor
170// for a message type that isn't built at the time of cross linking,
171// which is needed when a pool has lazily_build_dependencies_ set.
172// Must be instantiated as mutable in a descriptor.
173namespace internal {
174class LIBPROTOBUF_EXPORT LazyDescriptor {
175 public:
176 // Init function to be called at init time of a descriptor containing
177 // a LazyDescriptor.
178 void Init() {
179 descriptor_ = NULL;
180 name_ = NULL;
181 once_ = NULL;
182 file_ = NULL;
183 }
184
185 // Sets the value of the descriptor if it is known during the descriptor
186 // building process. Not thread safe, should only be called during the
187 // descriptor build process. Should not be called after SetLazy has been
188 // called.
189 void Set(const Descriptor* descriptor);
190
191 // Sets the information needed to lazily cross link the descriptor at a later
192 // time, SetLazy is not thread safe, should be called only once at descriptor
193 // build time if the symbol wasn't found and building of the file containing
194 // that type is delayed because lazily_build_dependencies_ is set on the pool.
195 // Should not be called after Set() has been called.
196 void SetLazy(const string& name, const FileDescriptor* file);
197
198 // Returns the current value of the descriptor, thread-safe. If SetLazy(...)
199 // has been called, will do a one-time cross link of the type specified,
200 // building the descriptor file that contains the type if necessary.
201 inline const Descriptor* Get() {
202 Once();
203 return descriptor_;
204 }
205
206 private:
207 static void OnceStatic(LazyDescriptor* lazy);
208 void OnceInternal();
209 void Once();
210
211 const Descriptor* descriptor_;
212 const string* name_;
213 GoogleOnceDynamic* once_;
214 const FileDescriptor* file_;
215};

490// Describes a single field of a message. To get the descriptor for a given
491// field, first get the Descriptor for the message in which it is defined,
492// then call Descriptor::FindFieldByName(). To get a FieldDescriptor for
493// an extension, do one of the following:
494// - Get the Descriptor or FileDescriptor for its containing scope, then
495// call Descriptor::FindExtensionByName() or
496// FileDescriptor::FindExtensionByName().
497// - Given a DescriptorPool, call DescriptorPool::FindExtensionByNumber().
498// - Given a Reflection for a message object, call
499// Reflection::FindKnownExtensionByName() or
500// Reflection::FindKnownExtensionByNumber().
501// Use DescriptorPool to construct your own descriptors.
502class LIBPROTOBUF_EXPORT FieldDescriptor {
503 public:
504 // Identifies a field type. 0 is reserved for errors. The order is weird
505 // for historical reasons. Types 12 and up are new in proto2.
506 enum Type {
507 TYPE_DOUBLE = 1, // double, exactly eight bytes on the wire.
508 TYPE_FLOAT = 2, // float, exactly four bytes on the wire.
509 TYPE_INT64 = 3, // int64, varint on the wire. Negative numbers
510 // take 10 bytes. Use TYPE_SINT64 if negative
511 // values are likely.
512 TYPE_UINT64 = 4, // uint64, varint on the wire.
513 TYPE_INT32 = 5, // int32, varint on the wire. Negative numbers
514 // take 10 bytes. Use TYPE_SINT32 if negative
515 // values are likely.
516 TYPE_FIXED64 = 6, // uint64, exactly eight bytes on the wire.
517 TYPE_FIXED32 = 7, // uint32, exactly four bytes on the wire.
518 TYPE_BOOL = 8, // bool, varint on the wire.
519 TYPE_STRING = 9, // UTF-8 text.
520 TYPE_GROUP = 10, // Tag-delimited message. Deprecated.
521 TYPE_MESSAGE = 11, // Length-delimited message.
522
523 TYPE_BYTES = 12, // Arbitrary byte array.
524 TYPE_UINT32 = 13, // uint32, varint on the wire
525 TYPE_ENUM = 14, // Enum, varint on the wire
526 TYPE_SFIXED32 = 15, // int32, exactly four bytes on the wire
527 TYPE_SFIXED64 = 16, // int64, exactly eight bytes on the wire
528 TYPE_SINT32 = 17, // int32, ZigZag-encoded varint on the wire
529 TYPE_SINT64 = 18, // int64, ZigZag-encoded varint on the wire
530
531 MAX_TYPE = 18, // Constant useful for defining lookup tables
532 // indexed by Type.
533 };
534
535 // Specifies the C++ data type used to represent the field. There is a
536 // fixed mapping from Type to CppType where each Type maps to exactly one
537 // CppType. 0 is reserved for errors.
538 enum CppType {
539 CPPTYPE_INT32 = 1, // TYPE_INT32, TYPE_SINT32, TYPE_SFIXED32
540 CPPTYPE_INT64 = 2, // TYPE_INT64, TYPE_SINT64, TYPE_SFIXED64
541 CPPTYPE_UINT32 = 3, // TYPE_UINT32, TYPE_FIXED32
542 CPPTYPE_UINT64 = 4, // TYPE_UINT64, TYPE_FIXED64
543 CPPTYPE_DOUBLE = 5, // TYPE_DOUBLE
544 CPPTYPE_FLOAT = 6, // TYPE_FLOAT
545 CPPTYPE_BOOL = 7, // TYPE_BOOL
546 CPPTYPE_ENUM = 8, // TYPE_ENUM
547 CPPTYPE_STRING = 9, // TYPE_STRING, TYPE_BYTES
548 CPPTYPE_MESSAGE = 10, // TYPE_MESSAGE, TYPE_GROUP
549
550 MAX_CPPTYPE = 10, // Constant useful for defining lookup tables
551 // indexed by CppType.
552 };
553
554 // Identifies whether the field is optional, required, or repeated. 0 is
555 // reserved for errors.
556 enum Label {
557 LABEL_OPTIONAL = 1, // optional
558 LABEL_REQUIRED = 2, // required
559 LABEL_REPEATED = 3, // repeated
560
561 MAX_LABEL = 3, // Constant useful for defining lookup tables
562 // indexed by Label.
563 };

575 const string& name() const; // Name of this field within the message.
576 const string& full_name() const; // Fully-qualified name of the field.
577 const string& json_name() const; // JSON name of this field.
578 const FileDescriptor* file() const;// File in which this field was defined.
579 bool is_extension() const; // Is this an extension field?
580 int number() const; // Declared tag number.
581
582 // Same as name() except converted to lower-case. This (and especially the
583 // FindFieldByLowercaseName() method) can be useful when parsing formats
584 // which prefer to use lowercase naming style. (Although, technically
585 // field names should be lowercased anyway according to the protobuf style
586 // guide, so this only makes a difference when dealing with old .proto files
587 // which do not follow the guide.)
588 const string& lowercase_name() const;
589
590 // Same as name() except converted to camel-case. In this conversion, any
591 // time an underscore appears in the name, it is removed and the next
592 // letter is capitalized. Furthermore, the first letter of the name is
593 // lower-cased. Examples:
594 // FooBar -> fooBar
595 // foo_bar -> fooBar
596 // fooBar -> fooBar
597 // This (and especially the FindFieldByCamelcaseName() method) can be useful
598 // when parsing formats which prefer to use camel-case naming style.
599 const string& camelcase_name() const;
600
601 Type type() const; // Declared type of this field.
602 const char* type_name() const; // Name of the declared type.
603 CppType cpp_type() const; // C++ type of this field.
604 const char* cpp_type_name() const; // Name of the C++ type.
605 Label label() const; // optional/required/repeated
606

608 bool is_optional() const; // shorthand for label() == LABEL_OPTIONAL
609 bool is_repeated() const; // shorthand for label() == LABEL_REPEATED
610 bool is_packable() const; // shorthand for is_repeated() &&
611 // IsTypePackable(type())

613 // options().packed()

615 // message_type()->options().map_entry()

669 // An extension may be declared within the scope of another message. If this

1474// Used to construct descriptors.
1475//
1476// Normally you won't want to build your own descriptors. Message classes
1477// constructed by the protocol compiler will provide them for you. However,
1478// if you are implementing Message on your own, or if you are writing a
1479// program which can operate on totally arbitrary types and needs to load
1480// them from some sort of database, you might need to.
1481//
1482// Since Descriptors are composed of a whole lot of cross-linked bits of
1483// data that would be a pain to put together manually, the
1484// DescriptorPool class is provided to make the process easier. It can
1485// take a FileDescriptorProto (defined in descriptor.proto), validate it,
1486// and convert it to a set of nicely cross-linked Descriptors.
1487//
1488// DescriptorPool also helps with memory management. Descriptors are
1489// composed of many objects containing static data and pointers to each
1490// other. In all likelihood, when it comes time to delete this data,
1491// you'll want to delete it all at once. In fact, it is not uncommon to
1492// have a whole pool of descriptors all cross-linked with each other which
1493// you wish to delete all at once. This class represents such a pool, and
1494// handles the memory management for you.
1495//
1496// You can also search for descriptors within a DescriptorPool by name, and
1497// extensions by number.

1500 // Create a normal, empty DescriptorPool.

1503 // Constructs a DescriptorPool that, when it can't find something among the
1504 // descriptors already in the pool, looks for it in the given
1505 // DescriptorDatabase.
1506 // Notes:
1507 // - If a DescriptorPool is constructed this way, its BuildFile*() methods
1508 // must not be called (they will assert-fail). The only way to populate
1509 // the pool with descriptors is to call the Find*By*() methods.
1510 // - The Find*By*() methods may block the calling thread if the
1511 // DescriptorDatabase blocks. This in turn means that parsing messages
1512 // may block if they need to look up extensions.
1513 // - The Find*By*() methods will use mutexes for thread-safety, thus making
1514 // them slower even when they don't have to fall back to the database.
1515 // In fact, even the Find*By*() methods of descriptor objects owned by
1516 // this pool will be slower, since they will have to obtain locks too.
1517 // - An ErrorCollector may optionally be given to collect validation errors
1518 // in files loaded from the database. If not given, errors will be printed
1519 // to GOOGLE_LOG(ERROR). Remember that files are built on-demand, so this
1520 // ErrorCollector may be called from any thread that calls one of the
1521 // Find*By*() methods.
1522 // - The DescriptorDatabase must not be mutated during the lifetime of
1523 // the DescriptorPool. Even if the client takes care to avoid data races,
1524 // changes to the content of the DescriptorDatabase may not be reflected
1525 // in subsequent lookups in the DescriptorPool.

1682 // underlay for a new DescriptorPool in which you add only the new file.
