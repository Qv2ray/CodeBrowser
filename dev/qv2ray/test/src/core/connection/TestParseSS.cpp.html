8SCENARIO("Test Parse Shadowsocks url", "[ParseSSUrl]")

16 WHEN("the url without padding")
17 {
18 auto c = ss::Deserialize("ss://YmYtY2ZiOnRlc3RAMTkyLjE2OC4xMDAuMTo4ODg4", &alias, &err);
19 s = ShadowSocksServerObject::fromJson(QJsonIO::GetValue(c, "outbounds", 0, "settings", "servers", 0));
