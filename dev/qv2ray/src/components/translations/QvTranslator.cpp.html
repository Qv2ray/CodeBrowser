11QStringList getLanguageSearchPaths()

13 // Configuration Path
14 QStringList list = QvCoreApplication->GetAssetsPaths("lang");

16 // If the translations have been embedded.

33 void QvTranslator::refreshTranslations()
34 {
35 searchPaths = getLanguageSearchPaths();
36 languages.clear();
37 for (const auto &path : searchPaths)
38 {
39 languages << QDir(path).entryList({ "*.qm" }, QDir::Hidden | QDir::Files);
40 }
41 std::transform(languages.begin(), languages.end(), languages.begin(), [](QString &fileName) { return fileName.replace(".qm", ""); });
42 languages.removeDuplicates();
43 DEBUG("Found translations: " + languages.join(" "));
44 }
45
46 bool QvTranslator::InstallTranslation(const QString &code)
47 {
48 for (const auto &path : searchPaths)
49 {
50 if (FileExistsIn(QDir(path), code + ".qm"))
51 {
52 DEBUG("Found " + code + " in folder: " + path);
53 QTranslator *translatorNew = new QTranslator();
54 bool success = translatorNew->load(code + ".qm", path);
55 if (!success)
56 {
57 LOG("Cannot load translation: " + code);
58 }
59 if (pTranslator)
60 {
61 LOG("Removed translations");
62 qApp->removeTranslator(pTranslator.get());
63 }
64 this->pTranslator.reset(translatorNew);
65 qApp->installTranslator(pTranslator.get());
66 LOG("Successfully installed a translator for", code);
67 return true;
68 }
69 }
70 return false;
71 }
