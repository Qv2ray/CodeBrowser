26static void check_multi_info(CURLM *curl_handle, int *success_num, LatencyTestResult *latencyTestResultPtr, RealPingGlobalInfo *info)
27{
28 CURLMsg *message;
29 int pending;
30 CURL *easy_handle;
31
32 while ((message = curl_multi_info_read(curl_handle, &pending)))
33 {
34 switch (message->msg)
35 {
36 case CURLMSG_DONE:
37 /* Do not use message data after calling curl_multi_remove_handle() and
38 curl_easy_cleanup(). As per curl_multi_info_read() docs:
39 "WARNING: The data the returned pointer points to will not survive
40 calling curl_multi_cleanup, curl_multi_remove_handle or
41 curl_easy_cleanup." */
42 easy_handle = message->easy_handle;
43 if (message->data.result == CURLE_OK)
44 {
45 success_num[0] += 1;
46 auto ms = info->_preserve_life_time->getHandleTime(easy_handle);
47 latencyTestResultPtr->avg += ms;
48 latencyTestResultPtr->best = std::min(latencyTestResultPtr->best, ms);
49 latencyTestResultPtr->worst = std::max(latencyTestResultPtr->worst, ms);
50 }
51 else
52 latencyTestResultPtr->failedCount += 1;
53 curl_multi_remove_handle(curl_handle, easy_handle);
54 curl_easy_cleanup(easy_handle);
55 break;
56 default: break;
57 }
58 }
59}

61static int start_timeout(CURLM *, long timeout_ms, void *userp)
62{
63 auto globalInfo = static_cast<RealPingGlobalInfo *>(userp);
64 if (timeout_ms < 0)
65 {
66 globalInfo->timer->stop();
67 }
68 else
69 {
70 if (timeout_ms == 0)
71 timeout_ms = 1; /* 0 means directly call socket_action, but we'll do it
72 in a bit */
73 globalInfo->timer->start(uvw::TimerHandle::Time{ timeout_ms }, uvw::TimerHandle::Time{ 0 });
74 }
75 return 0;
76}
77static int handle_socket(CURL *easy, curl_socket_t s, int action, void *userp, void *socketp)
78{
79 auto pRealPingGlobalInfo = static_cast<RealPingGlobalInfo *>(userp);
80 RealPingContext *curl_context;
81 uvw::Flags<uvw::PollHandle::Event> events{};
82
83 switch (action)
84 {
85 case CURL_POLL_IN:
86 case CURL_POLL_OUT:
87 curl_context = socketp ? (RealPingContext *) socketp : new RealPingContext{ s, pRealPingGlobalInfo->timer->loop() };
88
89 if (!socketp)
90 curl_multi_assign(pRealPingGlobalInfo->multiHandle, s, (void *) curl_context);
91
92 if (action == CURL_POLL_IN)
93 events = events | uvw::Flags<uvw::PollHandle::Event>::from<uvw::PollHandle::Event::READABLE>();
94 if (action == CURL_POLL_OUT)
95 events = events | uvw::Flags<uvw::PollHandle::Event>::from<uvw::PollHandle::Event::WRITABLE>();
96 curl_context->handle->on<uvw::ErrorEvent>([sockfd = curl_context->sockfd, pRealPingGlobalInfo](uvw::ErrorEvent &, uvw::PollHandle &) {
97 pRealPingGlobalInfo->timer->stop();
98 int running_handles;
99 auto flags = CURL_CSELECT_ERR;
100 curl_multi_socket_action(pRealPingGlobalInfo->multiHandle, sockfd, flags, &running_handles);
101 check_multi_info(pRealPingGlobalInfo->multiHandle, pRealPingGlobalInfo->successCountPtr, pRealPingGlobalInfo->latencyResultPtr,
102 pRealPingGlobalInfo);
103 pRealPingGlobalInfo->_preserve_life_time->notifyTestHost();
104 });
105 curl_context->handle->on<uvw::PollEvent>([sockfd = curl_context->sockfd, pRealPingGlobalInfo](uvw::PollEvent &e, uvw::PollHandle &) {
106 pRealPingGlobalInfo->timer->stop();
107 int running_handles;
108 int flags = 0;
109
110 if (e.flags & uvw::Flags<uvw::PollHandle::Event>{ uvw::PollHandle::Event::READABLE })
111 flags |= CURL_CSELECT_IN;
112 if (e.flags & uvw::Flags<uvw::PollHandle::Event>{ uvw::PollHandle::Event::WRITABLE })
113 flags |= CURL_CSELECT_OUT;
114
115 curl_multi_socket_action(pRealPingGlobalInfo->multiHandle, sockfd, flags, &running_handles);
116 check_multi_info(pRealPingGlobalInfo->multiHandle, pRealPingGlobalInfo->successCountPtr, pRealPingGlobalInfo->latencyResultPtr,
117 pRealPingGlobalInfo);
118 pRealPingGlobalInfo->_preserve_life_time->notifyTestHost();
119 });
120 pRealPingGlobalInfo->_preserve_life_time->recordHanleTime(easy);
121 curl_context->handle->start(events);
122 break;
123 case CURL_POLL_REMOVE:
124 if (socketp)
125 {
126 curl_context = static_cast<RealPingContext *>(socketp);
127 curl_context->handle->stop();
128 curl_context->handle->close();
129 delete curl_context;
130 curl_multi_assign(pRealPingGlobalInfo->multiHandle, s, nullptr);
131 }
132 break;
133 default: break;
134 }
135 return 0;
136}

158 std::string RealPing::getProxyAddress()
159 {
160 struct sockaddr_storage addr
161 {
162 };
163 const auto &ip = GlobalConfig.inboundConfig.listenip.toStdString();
164 bool is_addr_any = false;
165 bool is_ipv4 = (uv_ip4_addr(ip.c_str(), 0, reinterpret_cast<sockaddr_in *>(&addr)) == 0);
166 QString proxy_ip;
167 if (!is_ipv4)
168 {
169 uv_ip6_addr(ip.c_str(), 0, reinterpret_cast<sockaddr_in6 *>(&addr));
170 is_addr_any = isIPv6Any(reinterpret_cast<sockaddr_in6 &>(addr).sin6_addr);
171 if (!is_addr_any)
172 proxy_ip = "[" + GlobalConfig.inboundConfig.listenip + "]";
173 else
174 proxy_ip = "[::1]";
175 }
176 else
177 {
178 is_addr_any = isIPv4Any(reinterpret_cast<sockaddr_in &>(addr).sin_addr);
179 if (!is_addr_any)
180 proxy_ip = GlobalConfig.inboundConfig.listenip;
181 else
182 proxy_ip = "127.0.0.1";
183 }
184 QString protocolWithUsrPwd;
185 int port;
186 if (GlobalConfig.inboundConfig.useHTTP)
187 {
188 protocolWithUsrPwd = "http://";
189 if (GlobalConfig.inboundConfig.httpSettings.useAuth)
190 protocolWithUsrPwd +=
191 GlobalConfig.inboundConfig.httpSettings.account.user + ":" + GlobalConfig.inboundConfig.httpSettings.account.pass + "@";
192 port = GlobalConfig.inboundConfig.httpSettings.port;
193 }
194 else
195 {
196 protocolWithUsrPwd = "socks5://";
197 if (GlobalConfig.inboundConfig.socksSettings.useAuth)
198 protocolWithUsrPwd +=
199 GlobalConfig.inboundConfig.socksSettings.account.user + ":" + GlobalConfig.inboundConfig.socksSettings.account.pass + "@";
200 port = GlobalConfig.inboundConfig.socksSettings.port;
201 }
202 return (protocolWithUsrPwd + proxy_ip + ":" + QSTRN(port)).toStdString();

204 void RealPing::start()
205 {
206 if (!GlobalConfig.inboundConfig.useSocks && !GlobalConfig.inboundConfig.useHTTP)
207 {
208 data.avg = LATENCY_TEST_VALUE_ERROR;
209 testHost->OnLatencyTestCompleted(req.id, data);
210 return;
211 }
212 auto request_name = GlobalConfig.networkConfig.latencyRealPingTestURL.toStdString();
213 data.totalCount = 0;
214 data.failedCount = 0;
215 data.worst = 0;
216 data.avg = 0;
217 auto local_proxy_address = getProxyAddress();
218 auto curlMultiHandle = curl_multi_init();
219 auto globalInfo = std::make_shared<RealPingGlobalInfo>(
220 RealPingGlobalInfo{ shared_from_this(), curlMultiHandle, timeout->shared_from_this(), &successCount, &data });
221 timeout->once<uvw::CloseEvent>([globalInfo, curlMultiHandle](auto &, auto &h) {
222 curl_multi_cleanup(curlMultiHandle);
223 h.clear();
224 });
225 timeout->on<uvw::TimerEvent>([globalInfo, curlMultiHandle, suc_cnt_ptr = &successCount, fail_cnt_ptr = &data, this](auto &, auto &) {
226 int running_handles;
227 curl_multi_socket_action(curlMultiHandle, CURL_SOCKET_TIMEOUT, 0, &running_handles);
228 check_multi_info(curlMultiHandle, suc_cnt_ptr, fail_cnt_ptr, globalInfo.get());
229 notifyTestHost();
230 });
231 curl_multi_setopt(curlMultiHandle, CURLMOPT_SOCKETFUNCTION, handle_socket);
232 curl_multi_setopt(curlMultiHandle, CURLMOPT_SOCKETDATA, globalInfo.get());
233 curl_multi_setopt(curlMultiHandle, CURLMOPT_TIMERDATA, globalInfo.get());
234 curl_multi_setopt(curlMultiHandle, CURLMOPT_TIMERFUNCTION, start_timeout);
235 for (int i = 0; i < req.totalCount; ++i)
236 {
237 auto handle = curl_easy_init();
238 curl_easy_setopt(handle, CURLOPT_URL, request_name.c_str());
239 curl_easy_setopt(handle, CURLOPT_PROXY, local_proxy_address.c_str());
240 curl_easy_setopt(handle, CURLOPT_WRITEFUNCTION, noop_cb);
241 /* complete within 5 seconds */
242 curl_easy_setopt(handle, CURLOPT_CONNECTTIMEOUT, 5L);
243 curl_easy_setopt(handle, CURLOPT_TIMEOUT, 5L);
244 curl_multi_add_handle(curlMultiHandle, handle);
245 }
246 }
247 void RealPing::notifyTestHost()
248 {
249 if (data.failedCount + successCount == req.totalCount)
250 {
251 if (data.failedCount == req.totalCount)
252 data.avg = LATENCY_TEST_VALUE_ERROR;
253 else
254 data.errorMessage.clear(), data.avg = data.avg / successCount;
255 testHost->OnLatencyTestCompleted(req.id, data);
256 timeout->close();
257 }

263 long RealPing::getHandleTime(CURL *h)
264 {
265 if (reqStartTime.find(h) == reqStartTime.end())
266 return 0;
267 return std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now() - reqStartTime[h]).count();
268 }
269 void RealPing::recordHanleTime(CURL *h)
270 {
271 reqStartTime.emplace(h, std::chrono::system_clock::now());
272 }
