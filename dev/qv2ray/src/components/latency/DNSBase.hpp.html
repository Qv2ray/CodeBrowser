24 if (uv_ip4_addr(host.data(), req.port, reinterpret_cast<sockaddr_in *>(&storage)) == 0)

35 template<typename E, typename H>
36 void async_DNS_lookup(E &&e, H &&h)
37 {
38 co_enter(*this)
39 {
40 if (getAddrHandle)
41 {
42 getAddrHandle->once<uvw::ErrorEvent>(coro(async_DNS_lookup));
43 getAddrHandle->once<uvw::AddrInfoEvent>(coro(async_DNS_lookup));
44 co_yield return getAddrHandle->addrInfo(req.host.toStdString(), digitBuffer);
45 co_yield if constexpr (std::is_same_v<uvw::AddrInfoEvent, std::remove_reference_t<E>>)
46 {
47 if (getAddrInfoRes(e) != 0)
48 {
49 data.errorMessage = QObject::tr("DNS not resolved");
50 data.avg = LATENCY_TEST_VALUE_ERROR;
51 testHost->OnLatencyTestCompleted(req.id, data);
52 h.clear();
53 return;
54 }
55 }
56 else
57 {
58 if constexpr (std::is_same_v<uvw::ErrorEvent, std::remove_reference_t<E>>)
59 {
60 data.errorMessage = QObject::tr("DNS not resolved");
61 data.avg = LATENCY_TEST_VALUE_ERROR;
62 testHost->OnLatencyTestCompleted(req.id, data);
63 h.clear();

69 ping();

73 int getAddrInfoRes(uvw::AddrInfoEvent &e)
74 {
75 struct addrinfo *rp = nullptr;
76 for (rp = e.data.get(); rp != nullptr; rp = rp->ai_next)
77 if (rp->ai_family == AF_INET)
78 {
79 if (rp->ai_family == AF_INET)
80 {
81 af = AF_INET;
82 memcpy(&storage, rp->ai_addr, sizeof(struct sockaddr_in));
83 }
84 else if (rp->ai_family == AF_INET6)
85 {
86 af = AF_INET6;
87 memcpy(&storage, rp->ai_addr, sizeof(struct sockaddr_in6));
88 }
89 break;
90 }
91 if (rp == nullptr)
92 {
93 // fallback: if we can't find prefered AF, then we choose alternative.
94 for (rp = e.data.get(); rp != nullptr; rp = rp->ai_next)
95 {
96 if (rp->ai_family == AF_INET)
97 {
98 af = AF_INET;
99 memcpy(&storage, rp->ai_addr, sizeof(struct sockaddr_in));
100 }
101 else if (rp->ai_family == AF_INET6)
102 {
103 af = AF_INET6;
104 memcpy(&storage, rp->ai_addr, sizeof(struct sockaddr_in6));
105 }
106 break;
107 }
108 }
109 if (rp)
110 return 0;
111 return -1;
