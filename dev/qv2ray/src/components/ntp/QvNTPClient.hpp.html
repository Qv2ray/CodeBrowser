33 enum NtpLeapIndicator
34 {
35 NoWarning = 0, /**< No warning. */
36 LastMinute61Warning = 1, /**< Last minute has 61 seconds. */
37 LastMinute59Warning = 2, /**< Last minute has 59 seconds. */
38 UnsynchronizedWarning = 3, /**< Alarm condition (clock not synchronized). */
39 };
40
41 enum NtpMode
42 {
43 ReservedMode = 0, /**< Reserved. */
44 SymmetricActiveMode = 1, /**< Symmetric active. */
45 SymmetricPassiveMode = 2, /**< Symmetric passive. */
46 ClientMode = 3, /**< Client. */
47 ServerMode = 4, /**< Server. */
48 BroadcastMode = 5, /**< Broadcast. */
49 ControlMode = 6, /**< NTP control message. */
50 PrivateMode = 7, /**< Reserved for private use. */
51 };
52
53 enum NtpStratum
54 {
55 UnspecifiedStratum = 0, /**< Unspecified or unavailable. */
56 PrimaryStratum = 1, /**< Primary reference (e.g. radio-clock). */
57 SecondaryStratumFirst = 2, /**< Secondary reference (via NTP or SNTP). */
58 SecondaryStratumLast = 15,
59 UnsynchronizedStratum = 16, /**< Unsynchronized. */
60 /* 17-255 are reserved. */
61 };
62
63 struct NtpPacketFlags
64 {
65 unsigned char mode : 3;
66 unsigned char versionNumber : 3;
67 unsigned char leapIndicator : 2;
68 };

72 struct NtpTimestamp
73 {
74 quint32 seconds;
75 quint32 fraction;
76 static inline NtpTimestamp fromDateTime(const QDateTime &dateTime);
77 static inline QDateTime toDateTime(const NtpTimestamp &ntpTime);
78 };

80 struct NtpPacket
81 {
82 NtpPacketFlags flags;
83 quint8 stratum;
84 qint8 poll;
85 qint8 precision;
86 qint32 rootDelay;
87 qint32 rootDispersion;
88 qint8 referenceID[4];
89 NtpTimestamp referenceTimestamp;
90 NtpTimestamp originateTimestamp;
91 NtpTimestamp receiveTimestamp;
92 NtpTimestamp transmitTimestamp;
93 };
94
95 struct NtpAuthenticationInfo
96 {
97 quint32 keyId;
98 quint8 messageDigest[16];
99 };

101 struct NtpFullPacket
102 {
103 NtpPacket basic;
104 NtpAuthenticationInfo auth;
105 };

124 quint8 versionNumber() const;
125 NtpMode mode() const;

127 qreal pollInterval() const;
128 qreal precision() const;
129 QDateTime referenceTime() const;
130 QDateTime originTime() const;
131 QDateTime receiveTime() const;
132 QDateTime transmitTime() const;
133 QDateTime destinationTime() const;
134 qint64 roundTripDelay() const;
