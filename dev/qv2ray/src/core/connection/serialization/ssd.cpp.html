20 // These below are super strict checking schemes, but necessary.
21#define MUST_EXIST(fieldName) \
22 if (!obj.contains((fieldName)) || obj[(fieldName)].isUndefined() || obj[(fieldName)].isNull()) \
23 { \
24 *logList << QObject::tr("Invalid ssd link: json: field %1 must exist").arg(fieldName); \
25 return {}; \
26 }
27#define MUST_PORT(fieldName) \
28 MUST_EXIST(fieldName); \
29 if (int value = obj[(fieldName)].toInt(-1); value < 0 || value > 65535) \
30 { \
31 *logList << QObject::tr("Invalid ssd link: json: field %1 must be valid port number"); \
32 return {}; \
33 }
34#define MUST_STRING(fieldName) \
35 MUST_EXIST(fieldName); \
36 if (!obj[(fieldName)].isString()) \
37 { \
38 *logList << QObject::tr("Invalid ssd link: json: field %1 must be of type 'string'").arg(fieldName); \
39 return {}; \
40 }
41#define MUST_ARRAY(fieldName) \
42 MUST_EXIST(fieldName); \
43 if (!obj[(fieldName)].isArray()) \
44 { \
45 *logList << QObject::tr("Invalid ssd link: json: field %1 must be an array").arg(fieldName); \
46 return {}; \
47 }
48
49#define SERVER_SHOULD_BE_OBJECT(server) \
50 if (!server.isObject()) \
51 { \
52 *logList << QObject::tr("Skipping invalid ssd server: server must be an object"); \
53 continue; \
54 }
55#define SHOULD_EXIST(fieldName) \
56 if (serverObject[(fieldName)].isUndefined()) \
57 { \
58 *logList << QObject::tr("Skipping invalid ssd server: missing required field %1").arg(fieldName); \
59 continue; \
60 }
61#define SHOULD_STRING(fieldName) \
62 SHOULD_EXIST(fieldName); \
63 if (!serverObject[(fieldName)].isString()) \
64 { \
65 *logList << QObject::tr("Skipping invalid ssd server: field %1 should be of type 'string'").arg(fieldName); \
66 continue; \
67 }

69 QList<std::pair<QString, CONFIGROOT>> Deserialize(const QString &uri, QString *groupName, QStringList *logList)
70 {
71 LOG("SSD Link format is now deprecated.");
72 // ssd links should begin with "ssd://"
73 if (!uri.startsWith("ssd://"))
74 {
75 *logList << QObject::tr("Invalid ssd link: should begin with ssd://");
76 return {};
77 }
78
79 // decode base64
80 const auto ssdURIBody = uri.mid(6, uri.length() - 6); //(&uri, 6, uri.length() - 6);
81 const auto decodedJSON = SafeBase64Decode(ssdURIBody).toUtf8();
82
83 if (decodedJSON.length() == 0)
84 {
85 *logList << QObject::tr("Invalid ssd link: base64 parse failed");
86 return {};
87 }
88
89 const auto decodeError = VerifyJsonString(decodedJSON);
90 if (!decodeError.isEmpty())
91 {
92 *logList << QObject::tr("Invalid ssd link: json parse failed: ") % decodeError;
93 return {};
94 }
95
96 // casting to object
97 const auto obj = JsonFromString(decodedJSON);
98
99 // obj.airport
100 MUST_STRING("airport");
101 *groupName = obj["airport"].toString();
102
103 // obj.port
104 MUST_PORT("port");
105 const int port = obj["port"].toInt();
106
107 // obj.encryption
108 MUST_STRING("encryption");
109 const auto encryption = obj["encryption"].toString();
110
111 // check: rc4-md5 is not supported by v2ray-core
112 // TODO: more checks, including all algorithms
113 if (encryption.toLower() == "rc4-md5")
114 {
115 *logList << QObject::tr("Invalid ssd link: rc4-md5 encryption is not supported by v2ray-core");
116 return {};
117 }
118
119 // obj.password
120 MUST_STRING("password");
121 const auto password = obj["password"].toString();
122 // obj.servers
123 MUST_ARRAY("servers");
124 //
125 QList<std::pair<QString, CONFIGROOT>> serverList;
126 //
127
128 // iterate through the servers
129 for (const auto &server : obj["servers"].toArray())
130 {
131 SERVER_SHOULD_BE_OBJECT(server);
132 const auto serverObject = server.toObject();
133 ShadowSocksServerObject ssObject;
134
135 // encryption
136 ssObject.method = encryption;
137
138 // password
139 ssObject.password = password;
140
141 // address :-> "server"
142 SHOULD_STRING("server");
143 const auto serverAddress = serverObject["server"].toString();
144 ssObject.address = serverAddress;
145
146 // port selection:
147 // normal: use global settings
148 // overriding: use current config
149 if (serverObject["port"].isUndefined())
150 {
151 ssObject.port = port;
152 }
153 else if (auto currPort = serverObject["port"].toInt(-1); (currPort >= 0 && currPort <= 65535))
154 {
155 ssObject.port = currPort;
156 }
157 else
158 {
159 ssObject.port = port;
160 }
161
162 // name decision:
163 // untitled: using server:port as name
164 // entitled: using given name
165 QString nodeName;
166 if (serverObject["remarks"].isUndefined())
167 {
168 nodeName = QString("%1:%2").arg(ssObject.address).arg(ssObject.port);
169 }
170 else if (serverObject["remarks"].isString())
171 {
172 nodeName = serverObject["remarks"].toString();
173 }
174 else
175 {
176 nodeName = QString("%1:%2").arg(ssObject.address).arg(ssObject.port);
177 }
178
179 // ratio decision:
180 // unspecified: ratio = 1
181 // specified: use given value
182 double ratio = 1.0;
183 if (auto currRatio = serverObject["ratio"].toDouble(-1.0); currRatio != -1.0)
184 {
185 ratio = currRatio;
186 }
187 // else if (!serverObject["ratio"].isUndefined())
188 // {
189 // //*logList << QObject::tr("Invalid ratio encountered. using fallback value.");
190 // }
191
192 // format the total name of the node.
193 const auto finalName = QV2RAY_SSD_DEFAULT_NAME_PATTERN.arg(*groupName, nodeName).arg(ratio);
194 // appending to the total list
195 CONFIGROOT root;
196 OUTBOUNDS outbounds;
197 outbounds.append(GenerateOutboundEntry(OUTBOUND_TAG_PROXY, "shadowsocks", GenerateShadowSocksOUT({ ssObject }), {}));
198 root["outbounds"] = outbounds;
199 serverList.append({ finalName, root });
200 }
201
202 // returns the current result
203 return serverList;
204 }
