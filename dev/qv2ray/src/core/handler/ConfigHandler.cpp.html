14 QvConfigHandler::QvConfigHandler(QObject *parent) : QObject(parent)
15 {
16 DEBUG("ConnectionHandler Constructor.");
17 const auto connectionJson = JsonFromString(StringFromFile(QV2RAY_CONFIG_DIR + "connections.json"));
18 const auto groupJson = JsonFromString(StringFromFile(QV2RAY_CONFIG_DIR + "groups.json"));
19 //
20 for (const auto &connectionId : connectionJson.keys())
21 {
22 connections.insert(ConnectionId{ connectionId }, ConnectionObject::fromJson(connectionJson.value(connectionId).toObject()));
23 }
24 //
25 for (const auto &groupId : groupJson.keys())
26 {
27 auto groupObject = GroupObject::fromJson(groupJson.value(groupId).toObject());
28 if (groupObject.displayName.isEmpty())
29 {
30 groupObject.displayName = tr("Group: %1").arg(GenerateRandomString(5));
31 }
32 groups.insert(GroupId{ groupId }, groupObject);
33 for (const auto &connId : groupObject.connections)
34 {
35 connections[connId].__qvConnectionRefCount++;
36 }
37 }
38 //
39 for (const auto &id : connections.keys())
40 {
41 auto const &connectionObject = connections.value(id);
42 if (connectionObject.__qvConnectionRefCount == 0)
43 {
44 QFile connectionFile(QV2RAY_CONNECTIONS_DIR + id.toString() + QV2RAY_CONFIG_FILE_EXTENSION);
45 if (connectionFile.exists())
46 {
47 if (!connectionFile.remove())
48 LOG("Failed to remove connection config file");
49 }
50 connections.remove(id);
51 LOG("Dropped connection id: " + id.toString() + " since it's not in a group");
52 }
53 else
54 {
55 const auto connectionFilePath = QV2RAY_CONNECTIONS_DIR + id.toString() + QV2RAY_CONFIG_FILE_EXTENSION;
56 connectionRootCache[id] = CONFIGROOT(JsonFromString(StringFromFile(connectionFilePath)));
57 DEBUG("Loaded connection id: " + id.toString() + " into cache.");
58 }
59 }
60
61 // Force default group name.
62 if (!groups.contains(DefaultGroupId))
63 {
64 groups.insert(DefaultGroupId, {});
65 groups[DefaultGroupId].displayName = tr("Default Group");
66 groups[DefaultGroupId].isSubscription = false;
67 }
68 //
69 kernelHandler = new KernelInstanceHandler(this);
70 connect(kernelHandler, &KernelInstanceHandler::OnCrashed, this, &QvConfigHandler::p_OnKernelCrashed);
71 connect(kernelHandler, &KernelInstanceHandler::OnStatsDataAvailable, this, &QvConfigHandler::p_OnStatsDataArrived);
72 connect(kernelHandler, &KernelInstanceHandler::OnKernelLogAvailable, this, &QvConfigHandler::OnKernelLogAvailable);
73 connect(kernelHandler, &KernelInstanceHandler::OnConnected, this, &QvConfigHandler::OnConnected);
74 connect(kernelHandler, &KernelInstanceHandler::OnDisconnected, this, &QvConfigHandler::OnDisconnected);
75 //
76 pingHelper = new LatencyTestHost(5, this);
77 connect(pingHelper, &LatencyTestHost::OnLatencyTestCompleted, this, &QvConfigHandler::p_OnLatencyDataArrived);
78 //
79 // Save per 1 hour.
80 saveTimerId = startTimer(1 * 60 * 60 * 1000);
81 // Do not ping all...
82 pingConnectionTimerId = startTimer(60 * 1000);
83 }

199 bool QvConfigHandler::RemoveConnectionFromGroup(const ConnectionId &id, const GroupId &gid)
200 {
201 CheckValidId(id, false);
202 LOG("Removing connection : " + id.toString());
203 if (groups[gid].connections.contains(id))
204 {
205 auto removedEntries = groups[gid].connections.removeAll(id);
206 if (removedEntries > 1)
207 {
208 LOG("Found same connection occured multiple times in a group.");
209 }
210 // Decrease reference count.
211 connections[id].__qvConnectionRefCount -= removedEntries;
212 }
213
214 if (GlobalConfig.autoStartId == ConnectionGroupPair{ id, gid })
215 {
216 GlobalConfig.autoStartId.clear();
217 }
218
219 // Emit everything first then clear the connection map.
220 PluginHost->SendEvent({ Events::ConnectionEntry::RemovedFromGroup, GetDisplayName(id), "" });
221 emit OnConnectionRemovedFromGroup({ id, gid });
222
223 if (connections[id].__qvConnectionRefCount <= 0)
224 {
225 LOG("Fully removing a connection from cache.");
226 connectionRootCache.remove(id);
227 //
228 QFile connectionFile(QV2RAY_CONNECTIONS_DIR + id.toString() + QV2RAY_CONFIG_FILE_EXTENSION);
229 if (connectionFile.exists())
230 {
231 if (!connectionFile.remove())
232 LOG("Failed to remove connection config file");
233 }
234 connections.remove(id);
235 }
236 return true;

254 bool QvConfigHandler::MoveConnectionFromToGroup(const ConnectionId &id, const GroupId &sourceGid, const GroupId &targetGid)
255 {
256 CheckValidId(id, false);
257 CheckValidId(targetGid, false);
258 CheckValidId(sourceGid, false);
259 //
260 if (!groups[sourceGid].connections.contains(id))
261 {
262 LOG("Trying to move a connection away from a group it does not belong to.");
263 return false;
264 }
265 if (groups[targetGid].connections.contains(id))
266 {
267 LOG("The connection: " + id.toString() + " has already been in the target group: " + targetGid.toString());
268 const auto removedCount = groups[sourceGid].connections.removeAll(id);
269 connections[id].__qvConnectionRefCount -= removedCount;
270 }
271 else
272 {
273 // If the target group does not contain this connection.
274 const auto removedCount = groups[sourceGid].connections.removeAll(id);
275 connections[id].__qvConnectionRefCount -= removedCount;
276 //
277 groups[targetGid].connections.append(id);
278 connections[id].__qvConnectionRefCount++;
279 }
280
281 emit OnConnectionRemovedFromGroup({ id, sourceGid });
282 emit OnConnectionLinkedWithGroup({ id, targetGid });
283
284 return true;

502 bool QvConfigHandler::p_CHUpdateSubscription(const GroupId &id, const QByteArray &data)
503 {
504 CheckValidId(id, false);
505 //
506 // ====================================================================================== Begin reading subscription
507 std::shared_ptr<SubscriptionDecoder> decoder;
508
509 {
510 const auto type = groups[id].subscriptionOption.type;
511 for (const auto &plugin : PluginHost->UsablePlugins())
512 {
513 const auto pluginInfo = PluginHost->GetPlugin(plugin);
514 if (pluginInfo->hasComponent(COMPONENT_SUBSCRIPTION_ADAPTER))
515 {
516 const auto adapterInterface = pluginInfo->pluginInterface->GetSubscriptionAdapter();
517 for (const auto &[t, _] : adapterInterface->SupportedSubscriptionTypes())
518 {
519 if (t == type)
520 decoder = adapterInterface->GetSubscriptionDecoder(t);
521 }
522 }
523 }
524
525 if (decoder == nullptr)
526 {
527 QvMessageBoxWarn(nullptr, tr("Cannot Update Subscription"),
528 tr("Unknown subscription type: %1").arg(type) + NEWLINE + tr("A subscription plugin is missing?"));
529 return false;
530 }
531 }
532
533 const auto groupName = groups[id].displayName;
534 const auto result = decoder->DecodeData(data);
535 QList<std::pair<QString, CONFIGROOT>> _newConnections;
536
537 for (const auto &[name, json] : result.connections)
538 {
539 _newConnections.append({ name, CONFIGROOT(json) });
540 }
541 for (const auto &link : result.links)
542 {
543 // Assign a group name, to pass the name check.
544 QString _alias;
545 QString errMessage;
546 QString __groupName = groupName;
547 const auto connectionConfigMap = ConvertConfigFromString(link.trimmed(), &_alias, &errMessage, &__groupName);
548 if (!errMessage.isEmpty())
549 LOG("Error: ", errMessage);
550 _newConnections << connectionConfigMap;
551 }
552
553 if (_newConnections.count() < 5)
554 {
555 LOG("Found a subscription with less than 5 connections.");
556 if (QvMessageBoxAsk(
557 nullptr, tr("Update Subscription"),
558 tr("%n entrie(s) have been found from the subscription source, do you want to continue?", "", _newConnections.count())) != Yes)
559 return false;
560 }
561 //
562 // ====================================================================================== Begin Connection Data Storage
563 // Anyway, we try our best to preserve the connection id.
564 QMultiMap<QString, ConnectionId> nameMap;
565 QMultiMap<std::tuple<QString, QString, int>, ConnectionId> typeMap;
566 {
567 // Store connection type metadata into map.
568 for (const auto &conn : groups[id].connections)
569 {
570 nameMap.insert(GetDisplayName(conn), conn);
571 const auto &&[protocol, host, port] = GetConnectionInfo(conn);
572 if (port != 0)
573 {
574 typeMap.insert({ protocol, host, port }, conn);
575 }
576 }
577 }
578 // ====================================================================================== End Connection Data Storage
579 //
580 bool hasErrorOccured = false;
581 // Copy construct here.
582 auto originalConnectionIdList = groups[id].connections;
583 groups[id].connections.clear();
584 //
585 decltype(_newConnections) filteredConnections;
586 //
587 for (const auto &config : _newConnections)
588 {
589 // filter connections
590 const bool isIncludeOperationAND = groups[id].subscriptionOption.IncludeRelation == RELATION_AND;
591 const bool isExcludeOperationOR = groups[id].subscriptionOption.ExcludeRelation == RELATION_OR;
592 //
593 // Initial includeConfig value
594 bool includeconfig = isIncludeOperationAND;
595 {
596 bool hasIncludeItemMatched = false;
597 for (const auto &key : groups[id].subscriptionOption.IncludeKeywords)
598 {
599 if (!key.trimmed().isEmpty())
600 {
601 hasIncludeItemMatched = true;
602 // WARN: MAGIC, DO NOT TOUCH
603 if (!isIncludeOperationAND == config.first.contains(key.trimmed()))
604 {
605 includeconfig = !isIncludeOperationAND;
606 break;
607 }
608 }
609 }
610 // If includekeywords is empty then include all configs.
611 if (!hasIncludeItemMatched)
612 includeconfig = true;
613 }
614 if (includeconfig)
615 {
616 bool hasExcludeItemMatched = false;
617 includeconfig = isExcludeOperationOR;
618 for (const auto &key : groups[id].subscriptionOption.ExcludeKeywords)
619 {
620 if (!key.trimmed().isEmpty())
621 {
622 hasExcludeItemMatched = true;
623 // WARN: MAGIC, DO NOT TOUCH
624 if (isExcludeOperationOR == config.first.contains(key.trimmed()))
625 {
626 includeconfig = !isExcludeOperationOR;
627 break;
628 }
629 }
630 }
631 // If excludekeywords is empty then don't exclude any configs.
632 if (!hasExcludeItemMatched)
633 includeconfig = true;
634 }
635
636 if (includeconfig)
637 {
638 filteredConnections << config;
639 }
640 }
641
642 LOG("Filtered out less than 5 connections.");
643 const auto useFilteredConnections =
644 filteredConnections.count() > 5 ||
645 QvMessageBoxAsk(nullptr, tr("Update Subscription"),
646 tr("%1 out of %n entrie(s) have been filtered out, do you want to continue?", "", _newConnections.count())
647 .arg(filteredConnections.count()) +
648 NEWLINE + GetDisplayName(id)) == Yes;
649
650 for (const auto &config : useFilteredConnections ? filteredConnections : _newConnections)
651 {
652 const auto &_alias = config.first;
653 // Should not have complex connection we assume.
654 bool canGetOutboundData = false;
655 const auto &&[protocol, host, port] = GetConnectionInfo(config.second, &canGetOutboundData);
656 const auto outboundData = std::make_tuple(protocol, host, port);
657 //
658 // ====================================================================================== Begin guessing new ConnectionId
659 if (nameMap.contains(_alias))
660 {
661 // Just go and save the connection...
662 LOG("Reused connection id from name: " + _alias);
663 const auto _conn = nameMap.take(_alias);
664 groups[id].connections << _conn;
665 UpdateConnection(_conn, config.second, true);
666 // Remove Connection Id from the list.
667 originalConnectionIdList.removeAll(_conn);
668 typeMap.remove(typeMap.key(_conn));
669 }
670 else if (canGetOutboundData && typeMap.contains(outboundData))
671 {
672 LOG("Reused connection id from protocol/host/port pair for connection: " + _alias);
673 const auto _conn = typeMap.take(outboundData);
674 groups[id].connections << _conn;
675 // Update Connection Properties
676 UpdateConnection(_conn, config.second, true);
677 RenameConnection(_conn, _alias);
678 // Remove Connection Id from the list.
679 originalConnectionIdList.removeAll(_conn);
680 nameMap.remove(nameMap.key(_conn));
681 }
682 else
683 {
684 // New connection id is required since nothing matched found...
685 LOG("Generated new connection id for connection: " + _alias);
686 CreateConnection(config.second, _alias, id, true);
687 }
688 // ====================================================================================== End guessing new ConnectionId
689 }
690
691 // Check if anything left behind (not being updated or changed significantly)
692 if (!originalConnectionIdList.isEmpty())
693 {
694 bool needContinue = QvMessageBoxAsk(nullptr, //
695 tr("Update Subscription"),
696 tr("There're %n connection(s) in the group that do not belong the current subscription (any more).",
697 "", originalConnectionIdList.count()) +
698 NEWLINE + GetDisplayName(id) + NEWLINE + tr("Would you like to remove them?")) == Yes;
699 if (needContinue)
700 {
701 LOG("Removed old connections not have been matched.");
702 for (const auto &conn : originalConnectionIdList)
703 {
704 LOG("Removing connections not in the new subscription: " + conn.toString());
705 RemoveConnectionFromGroup(conn, id);
706 }
707 }
708 }
709
710 // Update the time
711 groups[id].lastUpdatedDate = system_clock::to_time_t(system_clock::now());
712 return hasErrorOccured;
713 }

737 const ConnectionGroupPair QvConfigHandler::CreateConnection(const CONFIGROOT &root, const QString &displayName, const GroupId &groupId,
738 bool skipSaveConfig)
739 {
740 LOG("Creating new connection: " + displayName);
741 ConnectionId newId(GenerateUuid());
