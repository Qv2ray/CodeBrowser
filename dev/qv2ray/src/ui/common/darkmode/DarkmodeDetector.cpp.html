19 bool isDarkMode()
20 {
21#if defined(Q_OS_WIN32)
22 QSettings settings(R"(HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Themes\Personalize)", QSettings::NativeFormat);
23 return settings.value("AppsUseLightTheme", 1).toInt() == 0;
24#elif defined(Q_OS_MAC)
25 bool isDark = false;
26
27 CFStringRef uiStyleKey = CFSTR("AppleInterfaceStyle");
28 CFStringRef uiStyle = nullptr;
29 CFStringRef darkUiStyle = CFSTR("Dark");
30
31 if (uiStyle = (CFStringRef) CFPreferencesCopyAppValue(uiStyleKey, kCFPreferencesCurrentApplication); uiStyle)
32 {
33 isDark = (kCFCompareEqualTo == CFStringCompare(uiStyle, darkUiStyle, 0));
34 CFRelease(uiStyle);
35 }
36
37 return isDark;
38#endif
39
40 if (!qApp || !qApp->style())
41 {
42 return false;
43 }
44 return qApp->style()->standardPalette().color(QPalette::Window).toHsl().lightness() < 110;
45 }
