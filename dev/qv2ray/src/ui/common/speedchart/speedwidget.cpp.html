46void SpeedWidget::AddPointData(QMap<SpeedWidget::GraphType, long> data)
47{
48 SpeedWidget::PointData point;
49 point.x = QDateTime::currentMSecsSinceEpoch() / 1000;
50 for (const auto &[id, data] : data.toStdMap())
51 {
52 if (m_properties.contains(id))
53 point.y[id] = data;
54 }
55
56 dataCollection.push_back(point);
57
58 while (dataCollection.length() > VIEWABLE)
59 {
60 dataCollection.removeFirst();
61 }
62 replot();
63}

65// use binary prefix standards from IEC 60027-2
66// see http://en.wikipedia.org/wiki/Kilobyte
67enum SizeUnit : int
68{
69 Byte, // 1000^0,
70 KByte, // 1000^1,
71 MByte, // 1000^2,
72 GByte, // 1000^3,
73 TByte, // 1000^4,
74 PByte, // 1000^5,
75 EByte // 1000^6
76};
77
78QString unitString(const SizeUnit unit, const bool isSpeed)
79{
80 const static QStringList units{ "B", "KB", "MB", "GB", "TB", "PB", "EB" };
81 auto unitString = units[unit];
82 if (isSpeed)
83 unitString += "/s";
84 return unitString;
85}
86
87int friendlyUnitPrecision(const SizeUnit unit)
88{
89 // friendlyUnit's number of digits after the decimal point
90 switch (unit)
91 {
92 case SizeUnit::Byte: return 0;
93 case SizeUnit::KByte:
94 case SizeUnit::MByte: return 1;
95 case SizeUnit::GByte: return 2;
96 default: return 3;
97 }
98}
99struct SplittedValue
100{
101 double arg;
102 SizeUnit unit;
103 qint64 sizeInBytes() const
104 {
105 auto size = arg;
106 for (int i = 0; i < static_cast<int>(unit); ++i)
107 {
108 size *= 1024;
109 }
110 return size;
111 }
112};
113
114SplittedValue getRoundedYScale(double value)
115{
116 if (value == 0.0)
117 return { 0, SizeUnit::Byte };
118
119 if (value <= 12.0)
120 return { 12, SizeUnit::Byte };
121
122 auto calculatedUnit = SizeUnit::Byte;
123
124 while (value > 1000)
125 {
126 value /= 1000;
127 calculatedUnit = static_cast<SizeUnit>(static_cast<int>(calculatedUnit) + 1);
128 }
129
130 if (value > 100.0)
131 {
132 int roundedValue = static_cast<int>(value / 40) * 40;
133 while (roundedValue < value)
134 roundedValue += 40;
135 return { static_cast<double>(roundedValue), calculatedUnit };
136 }
137
138 if (value > 10.0)
139 {
140 int roundedValue = static_cast<int>(value / 4) * 4;
141 while (roundedValue < value)
142 roundedValue += 4;
143 return { static_cast<double>(roundedValue), calculatedUnit };
144 }
145
146 for (const auto &roundedValue : roundingTable)
147 {
148 if (value <= roundedValue)
149 return { roundedValue, calculatedUnit };
150 }
151
152 return { 10.0, calculatedUnit };
153}
154
155QString formatLabel(const double argValue, const SizeUnit unit)
156{
157 // check is there need for digits after decimal separator
158 const int precision = (argValue < 10) ? friendlyUnitPrecision(unit) : 0;
159 return QLocale::system().toString(argValue, 'f', precision) + " " + unitString(unit, true);
160}

162void SpeedWidget::UpdateSpeedPlotSettings()
163{
164 auto &Graph = GlobalConfig.uiConfig.graphConfig;
165
166 const auto apply_penconfig = [&](StatisticsType x, QvPair<QvGraphPenConfig> y) {
167 if (!Graph.colorConfig.contains(x))
168 Graph.colorConfig[x] = y;
169 };
170
171 apply_penconfig(API_INBOUND, Qv2rayConfig_Graph::DefaultPen);
172 apply_penconfig(API_OUTBOUND_PROXY, Graph.colorConfig[API_INBOUND]);
173 apply_penconfig(API_OUTBOUND_DIRECT, Qv2rayConfig_Graph::DirectPen);
174
175 const auto getPen = [](const QvGraphPenConfig &conf) {
176 QPen p{ { conf.R, conf.G, conf.B } };
177 p.setStyle(conf.style);
178 p.setWidthF(conf.width);
179 return p;
180 };
181
182 m_properties.clear();
183 if (Graph.useOutboundStats)
184 {
185 m_properties[OUTBOUND_PROXY_UP] = { tr("Proxy") + " ↑", getPen(Graph.colorConfig[API_OUTBOUND_PROXY].value1) };
186 m_properties[OUTBOUND_PROXY_DOWN] = { tr("Proxy") + " ↓", getPen(Graph.colorConfig[API_OUTBOUND_PROXY].value2) };
187 if (Graph.hasDirectStats)
188 {
189 m_properties[OUTBOUND_DIRECT_UP] = { tr("Direct") + " ↑", getPen(Graph.colorConfig[API_OUTBOUND_DIRECT].value1) };
190 m_properties[OUTBOUND_DIRECT_DOWN] = { tr("Direct") + " ↓", getPen(Graph.colorConfig[API_OUTBOUND_DIRECT].value2) };
191 }
192 }
193 else
194 {
195 m_properties[INBOUND_UP] = { tr("Total") + " ↑", getPen(Graph.colorConfig[API_INBOUND].value1) };
196 m_properties[INBOUND_DOWN] = { tr("Total") + " ↓", getPen(Graph.colorConfig[API_INBOUND].value2) };
197 }
198}

212quint64 SpeedWidget::maxYValue()
213{
214 quint64 maxYValue = 0;
215
216 for (int id = 0; id < NB_GRAPHS; ++id)
217 for (int i = dataCollection.size() - 1, j = 0; (i >= 0) && (j <= VIEWABLE); --i, ++j)
218 if (dataCollection[i].y[id] > maxYValue)
219 maxYValue = dataCollection[i].y[id];
220
221 return maxYValue;
222}

224void SpeedWidget::paintEvent(QPaintEvent *)
225{
226 const auto fullRect = viewport()->rect();
227 auto rect = viewport()->rect();
228 rect.adjust(4, 4, 0, -4); // Add padding
229 const auto niceScale = getRoundedYScale(maxYValue());
230 // draw Y axis speed labels
231 const QVector<QString> speedLabels = {
232 formatLabel(niceScale.arg, niceScale.unit),
233 formatLabel((0.75 * niceScale.arg), niceScale.unit),
234 formatLabel((0.50 * niceScale.arg), niceScale.unit),
235 formatLabel((0.25 * niceScale.arg), niceScale.unit),
236 formatLabel(0.0, niceScale.unit),
237 };
238
239 QPainter painter(viewport());
240 painter.setRenderHints(QPainter::Antialiasing);
241 //
242 const auto fontMetrics = painter.fontMetrics();
243 rect.adjust(0, fontMetrics.height(), 0, 0); // Add top padding for top speed text
244 //
245 int yAxisWidth = 0;
246 for (const auto &label : speedLabels)
247 if (fontMetrics.horizontalAdvance(label) > yAxisWidth)
248 yAxisWidth = fontMetrics.horizontalAdvance(label);
249
250 int i = 0;
251 for (const auto &label : speedLabels)
252 {
253 QRectF labelRect(rect.topLeft() + QPointF(-yAxisWidth, (i++) * 0.25 * rect.height() - fontMetrics.height()),
254 QSizeF(2 * yAxisWidth, fontMetrics.height()));
255 painter.drawText(labelRect, label, Qt::AlignRight | Qt::AlignTop);
256 }
257
258 // draw grid lines
259 rect.adjust(yAxisWidth + 4, 0, 0, 0);
260 QPen gridPen;
261 gridPen.setStyle(Qt::DashLine);
262 gridPen.setWidthF(1);
263 gridPen.setColor(QColor(128, 128, 128, 128));
264 // Set antialiasing for graphs
265 painter.setPen(gridPen);
266 painter.drawLine(fullRect.left(), rect.top(), rect.right(), rect.top());
267 painter.drawLine(fullRect.left(), rect.top() + 0.25 * rect.height(), rect.right(), rect.top() + 0.25 * rect.height());
268 painter.drawLine(fullRect.left(), rect.top() + 0.50 * rect.height(), rect.right(), rect.top() + 0.50 * rect.height());
269 painter.drawLine(fullRect.left(), rect.top() + 0.75 * rect.height(), rect.right(), rect.top() + 0.75 * rect.height());
270 painter.drawLine(fullRect.left(), rect.bottom(), rect.right(), rect.bottom());
271
272 constexpr auto TIME_AXIS_DIVISIONS = 6;
273 for (int i = 0; i < TIME_AXIS_DIVISIONS; ++i)
274 {
275 const int x = rect.left() + (i * rect.width()) / TIME_AXIS_DIVISIONS;
276 painter.drawLine(x, fullRect.top(), x, fullRect.bottom());
277 }
278
279 // draw graphs
280 // Need, else graphs cross left gridline
281 rect.adjust(3, 0, 0, 0);
282 //
283 const double yMultiplier = (niceScale.arg == 0.0) ? 0.0 : (static_cast<double>(rect.height()) / niceScale.sizeInBytes());
284 const double xTickSize = static_cast<double>(rect.width()) / VIEWABLE;
285
286 for (const auto &id : m_properties.keys())
287 {
288 QVector<QPoint> points;
289
290 for (int i = static_cast<int>(dataCollection.size()) - 1, j = 0; (i >= 0) && (j <= VIEWABLE); --i, ++j)
291 {
292 const int newX = rect.right() - j * xTickSize;
293 const int newY = rect.bottom() - dataCollection[i].y[id] * yMultiplier;
294 points.push_back({ newX, newY });
295 }
296
297 painter.setPen(m_properties[id].pen);
298 painter.drawPolyline(points.data(), points.size());
299 }
300
301 // draw legend
302 double legendHeight = 0;
303 int legendWidth = 0;
304
305 for (const auto &property : m_properties)
306 {
307 if (fontMetrics.horizontalAdvance(property.name) > legendWidth)
308 legendWidth = fontMetrics.horizontalAdvance(property.name);
309
310 legendHeight += 1.5 * fontMetrics.height();
311 }
312
313 {
314 QPoint legendTopLeft(rect.left() + 4, fullRect.top() + 4);
315 QRectF legendBackgroundRect(QPoint(legendTopLeft.x() - 4, legendTopLeft.y() - 4), QSizeF(legendWidth + 8, legendHeight + 8));
316 auto legendBackgroundColor = QWidget::palette().color(QWidget::backgroundRole());
317 legendBackgroundColor.setAlpha(128); // 50% transparent
318 painter.fillRect(legendBackgroundRect, legendBackgroundColor);
319
320 int i = 0;
321 for (const auto &property : m_properties)
322 {
323 int nameSize = fontMetrics.horizontalAdvance(property.name);
324 double indent = 1.5 * (i++) * fontMetrics.height();
325 painter.setPen(property.pen);
326 painter.drawLine(legendTopLeft + QPointF(0, indent + fontMetrics.height()),
327 legendTopLeft + QPointF(nameSize, indent + fontMetrics.height()));
328 painter.drawText(QRectF(legendTopLeft + QPointF(0, indent), QSizeF(2 * nameSize, fontMetrics.height())), property.name,
329 QTextOption(Qt::AlignVCenter));
330 }
331 }
