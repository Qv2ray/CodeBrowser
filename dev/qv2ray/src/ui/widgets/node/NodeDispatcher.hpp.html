78 {
79#define PROCESS(type) \
80 bool hasExisting = type##s.contains(newTag); \
81 if (hasExisting) \
82 return false; \
83 type##s[newTag] = type##s.take(originalTag); \
84 type##Nodes[newTag] = type##Nodes.take(originalTag);

86 if constexpr (t == NODE_INBOUND)

90 PROCESS(inbound);

100 PROCESS(rule)
101 }

104 Q_UNREACHABLE();

108#undef PROCESS
