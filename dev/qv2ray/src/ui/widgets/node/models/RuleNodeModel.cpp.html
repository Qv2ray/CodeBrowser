8RuleNodeModel::RuleNodeModel(std::shared_ptr<NodeDispatcher> _dispatcher, std::shared_ptr<node_data_t> data) : NodeDataModel()
9{

18 const auto renameFunc = [this](ComplexTagNodeMode mode, const QString originalTag, const QString newTag) {
19 if (mode == NODE_INBOUND)
20 {
21 if (dataptr->inboundTag.contains(originalTag))
22 {
23 dataptr->inboundTag.removeAll(originalTag);
24 dataptr->inboundTag.push_back(newTag);
25 }
26 }
27 else if (mode == NODE_OUTBOUND)
28 {
29 if (dataptr->outboundTag == originalTag)
30 dataptr->outboundTag = newTag;
31 if (dataptr->balancerTag == originalTag)
32 dataptr->balancerTag = newTag;
33 }
34 };
35 connect(dispatcher.get(), &NodeDispatcher::OnObjectTagChanged, renameFunc);
36}

57void RuleNodeModel::setInData(std::vector<std::shared_ptr<NodeData>> indata, PortIndex)
58{
59 if (dispatcher->IsNodeConstructing())
60 return;
61 dataptr->inboundTag.clear();
62 for (const auto d : indata)
63 {
64 if (!d)
65 {
66 LOG("Invalid inbound nodedata to rule.");
67 continue;
68 }
69 const auto inboundTag = getTag(*static_cast<InboundNodeData *>(d.get())->GetData());
70 dataptr->inboundTag.push_back(inboundTag);
71 DEBUG("Connecting inbound:", inboundTag, "to", dataptr->QV2RAY_RULE_TAG);
72 }
73}
