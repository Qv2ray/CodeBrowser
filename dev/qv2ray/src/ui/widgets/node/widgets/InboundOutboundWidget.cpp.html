49void InboundOutboundWidget::on_editBtn_clicked()
50{
51 if (workingMode == NODE_INBOUND)
52 {
53 InboundEditor editor{ *inboundObject, parentWidget() };
54 *inboundObject = editor.OpenEditor();
55 // Set tag
56 const auto newTag = getTag(*inboundObject);
57 tagTxt->setText(newTag);
58 (*inboundObject)["tag"] = newTag;
59 }
60 else
61 {
62 if (isExternalOutbound)
63 {
64 if (QvMessageBoxAsk(parentWidget(), tr("Edit Outbound"), editExternalMsg) != Yes)
65 {
66 return;
67 }
68 const auto externalId = outboundObject->connectionId;
69 const auto root = ConnectionManager->GetConnectionRoot(externalId);
70 if (IsComplexConfig(root))
71 {
72 if (QvMessageBoxAsk(parentWidget(), tr("Trying to edit an Complex Config"), editExternalComplexMsg) != Yes)
73 {
74 return;
75 }
76 }
77 OUTBOUND outbound{ QJsonIO::GetValue(root, "outbounds", 0).toObject() };
78 OutboundEditor editor{ outbound, parentWidget() };
79 outbound = editor.OpenEditor();
80 if (editor.result() == QDialog::Accepted)
81 ConnectionManager->UpdateConnection(externalId, CONFIGROOT{ QJsonObject{ { "outbounds", QJsonArray{ outbound } } } });
82 }
83 else
84 {
85 OutboundEditor editor{ outboundObject->realOutbound, parentWidget() };
86 outboundObject->realOutbound = editor.OpenEditor();
87 // Set tag
88 const auto newTag = getTag(outboundObject->realOutbound);
89 tagTxt->setText(newTag);
90 outboundObject->realOutbound["tag"] = newTag;
91 }
92 }
93}

95void InboundOutboundWidget::on_editJsonBtn_clicked()
96{
97 if (workingMode == NODE_INBOUND)
98 {
99 JsonEditor editor{ *inboundObject, parentWidget() };
100 *inboundObject = INBOUND{ editor.OpenEditor() };
101 const auto newTag = getTag(*inboundObject);
102 // Set tag
103 tagTxt->setText(newTag);
104 (*inboundObject)["tag"] = newTag;
105 }
106 else
107 {
108 if (isExternalOutbound)
109 {
110 if (QvMessageBoxAsk(parentWidget(), tr("Edit Outbound"), editExternalMsg) != Yes)
111 return;
112 const auto externalId = outboundObject->connectionId;
113 const auto root = ConnectionManager->GetConnectionRoot(externalId);
114
115 OUTBOUND outbound{ QJsonIO::GetValue(root, "outbounds", 0).toObject() };
116 JsonEditor editor{ outbound, parentWidget() };
117 outbound = OUTBOUND{ editor.OpenEditor() };
118 if (editor.result() == QDialog::Accepted)
119 ConnectionManager->UpdateConnection(externalId, CONFIGROOT{ QJsonObject{ { "outbounds", QJsonArray{ outbound } } } });
120 }
121 else
122 {
123 // Open Editor
124 JsonEditor editor{ outboundObject->realOutbound, parentWidget() };
125 outboundObject->realOutbound = OUTBOUND{ editor.OpenEditor() };
126 //
127 // Set tag (only for local connections)
128 const auto newTag = getTag(outboundObject->realOutbound);
129 tagTxt->setText(newTag);
130 outboundObject->realOutbound["tag"] = newTag;
131 }
132 }

135void InboundOutboundWidget::on_tagTxt_textEdited(const QString &arg1)
136{
137 if (workingMode == NODE_INBOUND)
138 {
139 const auto originalTag = (*inboundObject)["tag"].toString();
140 if (originalTag == arg1 || dispatcher->RenameTag<NODE_INBOUND>(originalTag, arg1))
141 {
142 BLACK(tagTxt);
143 (*inboundObject)["tag"] = arg1;
144 return;
145 }
146 RED(tagTxt);
147 }
148 else
149 {
150 const auto originalTag = outboundObject->getDisplayName();
151 if (originalTag == arg1 || dispatcher->RenameTag<NODE_OUTBOUND>(originalTag, arg1))
152 {
153 BLACK(tagTxt);
154 outboundObject->realOutbound["tag"] = arg1;
155 return;
156 }
157 RED(tagTxt);
158 }
