42void Qv2rayWidgetApplication::onMessageReceived(quint32 clientId, QByteArray _msg)
43{
44 // Sometimes SingleApplication will send message with clientId == 0, ignore them.
45 if (clientId == instanceId())
46 return;
47
48 if (!isInitialized)
49 return;
50
51 const auto msg = Qv2rayStartupArguments::fromJson(JsonFromString(_msg));
52 LOG("Client ID:", clientId, ", message received, version:", msg.buildVersion);
53 DEBUG(_msg);
54 //
55 if (msg.buildVersion > QV2RAY_VERSION_BUILD)
56 {
57 const auto newPath = msg.fullArgs.first();
58 QString message;
59 message += tr("A new version of Qv2ray is starting:") + NEWLINE;
60 message += NEWLINE;
61 message += tr("New version information: ") + NEWLINE;
62 message += tr("Version: %1:%2").arg(msg.version).arg(msg.buildVersion) + NEWLINE;
63 message += tr("Path: %1").arg(newPath) + NEWLINE;
64 message += NEWLINE;
65 message += tr("Do you want to exit and launch that new version?");
66
67 const auto result = QvMessageBoxAsk(nullptr, tr("New version detected"), message);
68 if (result == Yes)
69 {
70 StartupArguments._qvNewVersionPath = newPath;
71 SetExitReason(EXIT_NEW_VERSION_TRIGGER);
72 QCoreApplication::quit();
73 }
74 }
75
76 for (const auto &argument : msg.arguments)
77 {
78 switch (argument)
79 {
80 case Qv2rayStartupArguments::EXIT:
81 {
82 SetExitReason(EXIT_NORMAL);
83 quit();
84 break;
85 }
86 case Qv2rayStartupArguments::NORMAL:
87 {
88 mainWindow->show();
89 mainWindow->raise();
90 mainWindow->activateWindow();
91 break;
92 }
93 case Qv2rayStartupArguments::RECONNECT:
94 {
95 ConnectionManager->RestartConnection();
96 break;
97 }
98 case Qv2rayStartupArguments::DISCONNECT:
99 {
100 ConnectionManager->StopConnection();
101 break;
102 }
103 case Qv2rayStartupArguments::QV2RAY_LINK:
104 {
105 for (const auto &link : msg.links)
106 {
107 const auto url = QUrl::fromUserInput(link);
108 const auto command = url.host();
109 auto subcommands = url.path().split("/");
110 subcommands.removeAll("");
111 QMap<QString, QString> args;
112 for (const auto &kvp : QUrlQuery(url).queryItems())
113 {
114 args.insert(kvp.first, kvp.second);
115 }
116 if (command == "open")
117 {
118 emit mainWindow->ProcessCommand(command, subcommands, args);
119 }
120 }
121 break;
122 }
123 }
124 }
125}

128Qv2rayExitReason Qv2rayWidgetApplication::runQv2rayInternal()
129{
130 setQuitOnLastWindowClosed(false);
131 hTray = new QSystemTrayIcon();
132 StyleManager = new QvStyleManager();
133 StyleManager->ApplyStyle(GlobalConfig.uiConfig.theme);
134 // Show MainWindow
135 UIStates = JsonFromString(StringFromFile(QV2RAY_CONFIG_DIR + QV2RAY_WIDGETUI_STATE_FILENAME));
136 mainWindow = new MainWindow();
137 if (StartupArguments.arguments.contains(Qv2rayStartupArguments::QV2RAY_LINK))
138 {
139 for (const auto &link : StartupArguments.links)
140 {
141 const auto url = QUrl::fromUserInput(link);
142 const auto command = url.host();
143 auto subcommands = url.path().split("/");
144 subcommands.removeAll("");
145 QMap<QString, QString> args;
146 for (const auto &kvp : QUrlQuery(url).queryItems())
147 {
148 args.insert(kvp.first, kvp.second);
149 }
150 if (command == "open")
151 {
152 emit mainWindow->ProcessCommand(command, subcommands, args);
153 }
154 }
155 }
156#ifdef Q_OS_MACOS
157 connect(this, &QApplication::applicationStateChanged, [this](Qt::ApplicationState state) {
158 switch (state)
159 {
160 case Qt::ApplicationActive:
161 {
162 mainWindow->show();
163 mainWindow->raise();
164 mainWindow->activateWindow();
165 break;
166 }
167 case Qt::ApplicationHidden:
168 case Qt::ApplicationInactive:
169 case Qt::ApplicationSuspended: break;
170 }
171 });
172#endif
173 isInitialized = true;
174 return (Qv2rayExitReason) exec();
175}
