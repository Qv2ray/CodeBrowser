66 AutoCompleteTextEdit::AutoCompleteTextEdit(const QString &prefix, const QStringList &sourceStrings, QWidget *parent) : QPlainTextEdit(parent)
67 {
68 this->prefix = prefix;
69 this->setLineWrapMode(QPlainTextEdit::NoWrap);
70 c = new QCompleter(this);
71 c->setModel(new QStringListModel(sourceStrings, c));
72 c->setWidget(this);
73 c->setCompletionMode(QCompleter::PopupCompletion);
74 c->setCaseSensitivity(Qt::CaseInsensitive);
75 QObject::connect(c, QOverload<const QString &>::of(&QCompleter::activated), this, &AutoCompleteTextEdit::insertCompletion);
76 }
77
78 AutoCompleteTextEdit::~AutoCompleteTextEdit()
79 {
80 }

82 void AutoCompleteTextEdit::insertCompletion(const QString &completion)
83 {
84 QTextCursor tc = textCursor();
85 int extra = completion.length() - c->completionPrefix().length();
86 tc.movePosition(QTextCursor::Left);
87 tc.movePosition(QTextCursor::EndOfWord);
88 tc.insertText(completion.right(extra));
89 setTextCursor(tc);
90 }
91
92 QString AutoCompleteTextEdit::lineUnderCursor() const
93 {
94 QTextCursor tc = textCursor();
95 tc.select(QTextCursor::LineUnderCursor);
96 return tc.selectedText();
97 }
98
99 QString AutoCompleteTextEdit::wordUnderCursor() const
100 {
101 QTextCursor tc = textCursor();
102 tc.select(QTextCursor::WordUnderCursor);
103 return tc.selectedText();
104 }

114 void AutoCompleteTextEdit::keyPressEvent(QKeyEvent *e)
115 {
116 const bool hasCtrlOrShiftModifier = e->modifiers().testFlag(Qt::ControlModifier) || e->modifiers().testFlag(Qt::ShiftModifier);
117 const bool hasOtherModifiers = (e->modifiers() != Qt::NoModifier) && !hasCtrlOrShiftModifier; // has other modifiers
118 //
119 const bool isSpace = (e->modifiers().testFlag(Qt::ShiftModifier) || e->modifiers().testFlag(Qt::NoModifier)) //
120 && e->key() == Qt::Key_Space;
121 const bool isTab = (e->modifiers().testFlag(Qt::NoModifier) && e->key() == Qt::Key_Tab);
122 const bool isOtherSpace = e->text() == "　";
123 //
124 if (isSpace || isTab || isOtherSpace)
125 {
126 QToolTip::showText(this->mapToGlobal(QPoint(0, 0)), tr("You can not input space characters here."), this, QRect{}, 2000);
127 return;
128 }
129 //
130 if (c && c->popup()->isVisible())
131 {
132 // The following keys are forwarded by the completer to the widget
133 switch (e->key())
134 {
135 case Qt::Key_Enter:
136 case Qt::Key_Return:
137 case Qt::Key_Escape:
138 case Qt::Key_Tab:
139 case Qt::Key_Backtab: e->ignore(); return; // let the completer do default behavior
140
141 default: break;
142 }
143 }
144
145 QPlainTextEdit::keyPressEvent(e);
146
147 if (!c || (hasCtrlOrShiftModifier && e->text().isEmpty()))
148 return;
149
150 // if we have other modifiers, or the text is empty, or the line does not start with our prefix.
151 if (hasOtherModifiers || e->text().isEmpty() || !lineUnderCursor().startsWith(prefix))
152 {
153 c->popup()->hide();
154 return;
155 }
156
157 if (auto word = wordUnderCursor(); word != c->completionPrefix())
158 {
159 c->setCompletionPrefix(word);
160 c->popup()->setCurrentIndex(c->completionModel()->index(0, 0));
161 }
162
163 QRect cr = cursorRect();
164 cr.setWidth(c->popup()->sizeHintForColumn(0) + c->popup()->verticalScrollBar()->sizeHint().width());
165 c->complete(cr); // popup it up!
166 }
