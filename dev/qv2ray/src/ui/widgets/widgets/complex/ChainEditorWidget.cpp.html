54void ChainEditorWidget::OnDispatcherChainedOutboundCreated(std::shared_ptr<OutboundObjectMeta> data, QtNodes::Node &node)
55{
56 outboundNodes[data->getDisplayName()] = node.id();
57 const auto outboundCount = outboundNodes.count();
58 const static int offsets[]{ 300, 0, -300 };
59 auto pos = this->pos();
60 pos.setX(pos.x() + GRAPH_GLOBAL_OFFSET_X + offsets[outboundCount % 3]);
61 pos.setY(pos.y() + outboundCount * 100 + GRAPH_GLOBAL_OFFSET_Y);
62 scene->setNodePosition(node, pos);
63}

85void ChainEditorWidget::ShowChainLinkedList()
86{
87 if (dispatcher->IsNodeConstructing())
88 return;
89 connectionSignalBlocked = true;
90 const auto connections = scene->connections();
91 for (const auto &connection : connections)
92 {
93 scene->deleteConnection(*connection.second);
94 }
95 //
96 // Warn: Some outbound node MAY NOT be created YET!
97 if (!currentChain)
98 return;
99 const auto &outbounds = currentChain->outboundTags;
100 for (auto index = 0; index < outbounds.count() - 1; index++)
101 {
102 const auto &inTag = outbounds[index + 1];
103 const auto &outTag = outbounds[index];
104 bool hasErrorOccured = false;
105 for (const auto &tag : { inTag, outTag })
106 {
107 if (!outboundNodes.contains(tag))
108 {
109 QvMessageBoxWarn(this, tr("Chain Editor"), tr("Could not find outbound tag: %1, The chain may be corrupted").arg(tag));
110 hasErrorOccured = true;
111 }
112 }
113 if (!hasErrorOccured)
114 {
115 const auto &nodeIn = scene->node(outboundNodes[inTag]);
116 const auto &nodeOut = scene->node(outboundNodes[outTag]);
117 scene->createConnection(*nodeIn, 0, *nodeOut, 0);
118 }
119 }
120 connectionSignalBlocked = false;
121}

123std::tuple<bool, QString, QStringList> ChainEditorWidget::VerifyChainLinkedList(const QUuid &ignoredConnectionId)
124{
125 QList<QString> resultList;
126 auto conns = scene->connections();
127 bool needReIterate = true;
128 const auto expectedChainLength = conns.size() + uint(ignoredConnectionId.isNull());
129 for (auto i = 0u; i < expectedChainLength; i++)
130 {
131 auto iter = conns.begin();
132 // Loop against all items.
133 while (iter != conns.end())
134 {
135#define NEED_ITERATE(x) \
136 needReIterate = x; \
137 continue
138 const auto &id = iter->first;
139 const auto &connection = iter->second;
140 const auto &nodeInId = connection->getNode(QtNodes::PortType::In)->id();
141 const auto &nodeOutId = connection->getNode(QtNodes::PortType::Out)->id();
142 //
143 const auto &nextTag = outboundNodes.key(nodeInId);
144 const auto &previousTag = outboundNodes.key(nodeOutId);
145 //
146 // If we are ignoring this node.
147 if (id == ignoredConnectionId)
148 {
149 iter++;
150 NEED_ITERATE(false);
151 }
152 //
153 if (resultList.isEmpty())
154 {
155 iter++;
156 resultList << previousTag << nextTag;
157 NEED_ITERATE(false);
158 }
159
160 if (resultList.last() != previousTag && resultList.first() != nextTag)
161 {
162 iter++;
163 NEED_ITERATE(true);
164 // return std::make_tuple(false, tr("Two different chains detected."), QStringList());
165 }
166
167 if (resultList.contains(previousTag) && resultList.contains(nextTag))
168 {
169 iter++;
170 NEED_ITERATE(true);
171 // return std::make_tuple(false, tr("Looped chain detected."), QStringList());
172 }
173
174 if (resultList.last() == previousTag)
175 {
176 resultList << nextTag;
177 iter = conns.erase(iter);
178 NEED_ITERATE(false);
179 }
180
181 if (resultList.front() == nextTag)
182 {
183 resultList.prepend(previousTag);
184 iter = conns.erase(iter);
185 NEED_ITERATE(false);
186 }
187#undef NEED_ITERATE
188 }
189 if ((ulong) resultList.count() == expectedChainLength)
190 {
191 return { true, tr("OK"), resultList };
192 }
193 }
194 return { false, tr("There's an error in your connection."), resultList };
195}

203void ChainEditorWidget::TrySaveChainOutboudData(const QUuid &ignoredConnectionId)
204{
205 if (!currentChain)
206 {
207 QvMessageBoxWarn(this, tr("Chain Editor"), tr("Please Select a Chain"));
208 return;
209 }
210 const auto &[result, errMessage, list] = VerifyChainLinkedList(ignoredConnectionId);
211 if (!result)
212 {
213 RED(statusLabel);
214 statusLabel->setText(errMessage);
215 }
216 else
217 {
218 BLACK(statusLabel);
219 statusLabel->setText(list.join(" >> "));
220 currentChain->outboundTags = list;
221 }
222}

263void ChainEditorWidget::OnDispatcherObjectTagChanged(ComplexTagNodeMode mode, const QString originalTag, const QString newTag)
264{
265 if (mode == NODE_OUTBOUND)
266 {
267 // Simply compare if there is a match (Since no duplication of DisplayName should occur.)
268 if (outboundNodes.contains(originalTag))
269 outboundNodes[newTag] = outboundNodes.take(originalTag);
270
271 // Check Chains.
272 if (chains.contains(originalTag))
273 chains[newTag] = chains.take(originalTag);
274
275 const auto index = chainComboBox->findText(originalTag);
276 if (index >= 0)
277 {
278 chainComboBox->setItemText(index, newTag);
279 }
280
281 for (const auto &chain : chains)
282 {
283 if (chain->outboundTags.contains(originalTag))
284 chain->outboundTags.replace(chain->outboundTags.indexOf(originalTag), newTag);
285 }
286 }
