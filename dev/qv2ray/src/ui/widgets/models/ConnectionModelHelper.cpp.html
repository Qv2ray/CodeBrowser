8ConnectionListHelper::ConnectionListHelper(QTreeView *view, QObject *parent) : QObject(parent)
9{
10 parentView = view;
11 model = new QStandardItemModel();
12 view->setModel(model);
13 for (const auto &group : ConnectionManager->AllGroups())
14 {
15 addGroupItem(group);
16 for (const auto &connection : ConnectionManager->GetConnections(group))
17 {
18 addConnectionItem({ connection, group });
19 }
20 }
21 const auto renamedLambda = [&](const ConnectionId &id, const QString &, const QString &newName) {
22 for (const auto &gid : ConnectionManager->GetConnectionContainedIn(id))
23 {
24 ConnectionGroupPair pair{ id, gid };
25 if (pairs.contains(pair))
26 pairs[pair]->setData(newName, ROLE_DISPLAYNAME);
27 }
28 };
29
30 const auto latencyLambda = [&](const ConnectionId &id, const int avg) {
31 for (const auto &gid : ConnectionManager->GetConnectionContainedIn(id))
32 {
33 ConnectionGroupPair pair{ id, gid };
34 if (pairs.contains(pair))
35 pairs[pair]->setData(NumericString(avg), ROLE_LATENCY);
36 }
37 };
38
39 const auto statsLambda = [&](const ConnectionGroupPair &id, const QMap<StatisticsType, QvStatsSpeedData> &data) {
40 if (connections.contains(id.connectionId))
41 {
42 for (const auto &index : connections[id.connectionId])
43 index->setData(NumericString(GetConnectionTotalData(id.connectionId)), ROLE_DATA_USAGE);
44 }
45 };

47 connect(ConnectionManager, &QvConfigHandler::OnConnectionRemovedFromGroup, this, &ConnectionListHelper::OnConnectionDeleted);
48 connect(ConnectionManager, &QvConfigHandler::OnConnectionCreated, this, &ConnectionListHelper::OnConnectionCreated);

68void ConnectionListHelper::Filter(const QString &key)
69{
70 for (const auto &groupId : ConnectionManager->AllGroups())
71 {
72 const auto groupItem = model->indexFromItem(groups[groupId]);
73 bool isTotallyHide = true;
74 for (const auto &connectionId : ConnectionManager->GetConnections(groupId))
75 {
76 const auto connectionItem = model->indexFromItem(pairs[{ connectionId, groupId }]);
77 const auto willTotallyHide = static_cast<ConnectionItemWidget *>(parentView->indexWidget(connectionItem))->NameMatched(key);
78 parentView->setRowHidden(connectionItem.row(), connectionItem.parent(), !willTotallyHide);
79 isTotallyHide &= willTotallyHide;
80 }
81 parentView->indexWidget(groupItem)->setHidden(isTotallyHide);
82 if (!isTotallyHide)
83 parentView->expand(groupItem);
84 }
85}

87QStandardItem *ConnectionListHelper::addConnectionItem(const ConnectionGroupPair &id)
88{
89 // Create Standard Item
90 auto connectionItem = new QStandardItem();
91 connectionItem->setData(GetDisplayName(id.connectionId), ConnectionInfoRole::ROLE_DISPLAYNAME);
92 connectionItem->setData(NumericString(GetConnectionLatency(id.connectionId)), ConnectionInfoRole::ROLE_LATENCY);
93 connectionItem->setData(NumericString(GetConnectionTotalData(id.connectionId)), ConnectionInfoRole::ROLE_DATA_USAGE);
94 //
95 // Find groups
96 const auto groupIndex = groups.contains(id.groupId) ? groups[id.groupId] : addGroupItem(id.groupId);
97 // Append into model
98 groupIndex->appendRow(connectionItem);
99 const auto connectionIndex = connectionItem->index();
100 //
101 auto widget = new ConnectionItemWidget(id, parentView);
102 connect(widget, &ConnectionItemWidget::RequestWidgetFocus, [widget, connectionIndex, this]() {
103 parentView->setCurrentIndex(connectionIndex);
104 parentView->scrollTo(connectionIndex);
105 parentView->clicked(connectionIndex);
106 });
107 //
108 parentView->setIndexWidget(connectionIndex, widget);
109 pairs[id] = connectionItem;
110 connections[id.connectionId].append(connectionItem);
111 return connectionItem;
112}

114QStandardItem *ConnectionListHelper::addGroupItem(const GroupId &groupId)
115{
116 // Create Item
117 const auto item = new QStandardItem();
118 // Set item into model
119 model->appendRow(item);
120 // Get item index
121 const auto index = item->index();
122 parentView->setIndexWidget(index, new ConnectionItemWidget(groupId, parentView));
123 groups[groupId] = item;
124 return item;
125}

132void ConnectionListHelper::OnConnectionDeleted(const ConnectionGroupPair &id)
133{
134 auto item = pairs.take(id);
135 const auto index = model->indexFromItem(item);
136 if (!index.isValid())
137 return;
138 model->removeRow(index.row(), index.parent());
139 connections[id.connectionId].removeAll(item);
140}

152void ConnectionListHelper::OnGroupDeleted(const GroupId &id, const QList<ConnectionId> &connections)
153{
154 for (const auto &conn : connections)
155 {
156 const ConnectionGroupPair pair{ conn, id };
157 OnConnectionDeleted(pair);
158 }
159 const auto item = groups.take(id);
160 const auto index = model->indexFromItem(item);
161 if (!index.isValid())
162 return;
163 model->removeRow(index.row(), index.parent());
164}
