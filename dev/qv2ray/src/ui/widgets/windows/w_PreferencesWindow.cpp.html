310std::optional<QString> PreferencesWindow::checkTProxySettings() const
311{
312 if (CurrentConfig.inboundConfig.useTPROXY)
313 {
314 if (!IsIPv4Address(CurrentConfig.inboundConfig.tProxySettings.tProxyIP))
315 {
316 return tr("Invalid tproxy listening ipv4 address.");
317 }
318 else if (CurrentConfig.inboundConfig.tProxySettings.tProxyV6IP != "" && !IsIPv6Address(CurrentConfig.inboundConfig.tProxySettings.tProxyV6IP))
319 {
320 return tr("Invalid tproxy listening ipv6 address.");
321 }
322 }
323 return std::nullopt;
324}

326void PreferencesWindow::on_buttonBox_accepted()
327{
328 // Note:
329 // A signal-slot connection from buttonbox_accpted to QDialog::accepted()
330 // has been removed. To prevent closing this Dialog.
331 QSet<int> ports;
332 auto size = 0;
333
334 if (CurrentConfig.inboundConfig.useHTTP)
335 {
336 size++;
337 ports << CurrentConfig.inboundConfig.httpSettings.port;
338 }
339
340 if (CurrentConfig.inboundConfig.useSocks)
341 {
342 size++;
343 ports << CurrentConfig.inboundConfig.socksSettings.port;
344 }
345
346 if (CurrentConfig.inboundConfig.useTPROXY)
347 {
348 size++;
349 ports << CurrentConfig.inboundConfig.tProxySettings.port;
350 }
351
352 if (!QvCoreApplication->StartupArguments.noAPI)
353 {
354 size++;
355 ports << CurrentConfig.kernelConfig.statsPort;
356 }
357
358 if (ports.size() != size)
359 {
360 // Duplicates detected.
361 QvMessageBoxWarn(this, tr("Preferences"), tr("Duplicated port numbers detected, please check the port number settings."));
362 }
363 else if (!IsValidIPAddress(CurrentConfig.inboundConfig.listenip))
364 {
365 QvMessageBoxWarn(this, tr("Preferences"), tr("Invalid inbound listening address."));
366 }
367 else if (const auto err = checkTProxySettings(); err.has_value())
368 {
369 QvMessageBoxWarn(this, tr("Preferences"), *err);
370 }
371 else if (!dnsSettingsWidget->CheckIsValidDNS())
372 {
373 QvMessageBoxWarn(this, tr("Preferences"), tr("Invalid DNS settings."));
374 }
375 else
376 {
377 if (CurrentConfig.uiConfig.language != GlobalConfig.uiConfig.language)
378 {
379 // Install translator
380 if (Qv2rayTranslator->InstallTranslation(CurrentConfig.uiConfig.language))
381 {
382 UIMessageBus.EmitGlobalSignal(QvMBMessage::RETRANSLATE);
383 QApplication::processEvents();
384 }
385 }
386 CurrentConfig.defaultRouteConfig.routeConfig = routeSettingsWidget->GetRouteConfig();
387 if (!(CurrentConfig.defaultRouteConfig.routeConfig == GlobalConfig.defaultRouteConfig.routeConfig))
388 {
389 NEEDRESTART
390 }
391 const auto &[dns, fakedns] = dnsSettingsWidget->GetDNSObject();
392 CurrentConfig.defaultRouteConfig.dnsConfig = dns;
393 CurrentConfig.defaultRouteConfig.fakeDNSConfig = fakedns;
394 if (!(CurrentConfig.defaultRouteConfig.dnsConfig == GlobalConfig.defaultRouteConfig.dnsConfig))
395 {
396 NEEDRESTART
397 }
398 //
399 //
400 if (CurrentConfig.uiConfig.theme != GlobalConfig.uiConfig.theme)
401 {
402 StyleManager->ApplyStyle(CurrentConfig.uiConfig.theme);
403 }
404 GlobalConfig.loadJson(CurrentConfig.toJson());
405 SaveGlobalSettings();
406 UIMessageBus.EmitGlobalSignal(QvMBMessage::UPDATE_COLORSCHEME);
407 if (NeedRestart && !KernelInstance->CurrentConnection().isEmpty())
408 {
409 const auto message = tr("You may need to reconnect to apply the settings now.") + NEWLINE + //
410 tr("Otherwise they will be applied next time you connect to a server.") + NEWLINE + //
411 NEWLINE + //
412 tr("Do you want to reconnect now?");
413 const auto askResult = QvMessageBoxAsk(this, tr("Reconnect Required"), message);
414 if (askResult == Yes)
415 {
416 ConnectionManager->RestartConnection();
417 this->setEnabled(false);
418 }
419 }
420 accept();
421 }
422}

760 QvMessageBoxWarn(this, tr("Watch Out!"), content);

766 QvMessageBoxWarn(this, tr("V2Ray Core Settings"), *msg);

768#if QV2RAY_FEATURE(kernel_check_output)

1101void PreferencesWindow::on_pushButton_clicked()
1102{
1103#if QV2RAY_FEATURE(util_has_ntp)
1104 const auto ntpTitle = tr("NTP Checker");
1105 const auto ntpHint = tr("Check date and time from server:");
1106 const static QStringList ntpServerList = { "cn.pool.ntp.org", "cn.ntp.org.cn", "edu.ntp.org.cn",
1107 "time.pool.aliyun.com", "time1.cloud.tencent.com", "ntp.neu.edu.cn" };
1108 bool ok = false;
1109 const auto ntpServer = QInputDialog::getItem(this, ntpTitle, ntpHint, ntpServerList, 0, true, &ok).trimmed();
1110 if (!ok)
1111 return;
1112
1113 auto client = new ntp::NtpClient(this);
1114 connect(client, &ntp::NtpClient::replyReceived, [&](const QHostAddress &, quint16, const ntp::NtpReply &reply) {
1115 const int offsetSecTotal = reply.localClockOffset() / 1000;
1116 if (offsetSecTotal >= 90 || offsetSecTotal <= -90)
1117 {
1118 const auto inaccurateWarning = tr("Your time offset is %1 seconds, which is too high.") + NEWLINE + //
1119 tr("Please synchronize your system to use the VMess protocol.");
1120 QvMessageBoxWarn(this, tr("Time Inaccurate"), inaccurateWarning.arg(offsetSecTotal));
1121 }
1122 else if (offsetSecTotal > 15 || offsetSecTotal < -15)
1123 {
1124 const auto smallErrorWarning = tr("Your time offset is %1 seconds, which is a little high.") + NEWLINE + //
1125 tr("VMess protocol may still work, but we suggest you synchronize your clock.");
1126 QvMessageBoxInfo(this, tr("Time Somewhat Inaccurate"), smallErrorWarning.arg(offsetSecTotal));
1127 }
1128 else
1129 {
1130 const auto accurateInfo = tr("Your time offset is %1 seconds, which looks good.") + NEWLINE + //
1131 tr("VMess protocol may not suffer from time inaccuracy.");
1132 QvMessageBoxInfo(this, tr("Time Accurate"), accurateInfo.arg(offsetSecTotal));
1133 }
1134 });
1135
1136 const auto hostInfo = QHostInfo::fromName(ntpServer);
1137 if (hostInfo.error() == QHostInfo::NoError)
1138 client->sendRequest(hostInfo.addresses().first(), 123);
1139 else
1140 QvMessageBoxWarn(this, ntpTitle, tr("Failed to lookup server: %1").arg(hostInfo.errorString()));
1141#else
1142 QvMessageBoxWarn(this, tr("No NTP Backend"), tr("Qv2ray was not built with NTP support."));
1143#endif
1144}
