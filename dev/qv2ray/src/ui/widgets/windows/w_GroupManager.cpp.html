116void GroupManager::onRCMExportConnectionTriggered()
117{
118 const auto &list = GET_SELECTED_CONNECTION_IDS(SELECTED_ROWS_INDEX);
119 QFileDialog d;
120 switch (list.count())
121 {
122 case 0: return;
123 case 1:
124 {
125 const auto id = ConnectionId(list.first());
126 auto filePath = d.getSaveFileName(this, GetDisplayName(id));
127 if (filePath.isEmpty())
128 return;
129 auto root = RouteManager->GenerateFinalConfig({ id, currentGroupId }, false);
130 //
131 // Apply export filter
132 exportConnectionFilter(root);
133 //
134 if (filePath.endsWith(".json"))
135 {
136 filePath += ".json";
137 }
138 //
139 StringToFile(JsonToString(root), filePath);
140 QDesktopServices::openUrl(QUrl::fromLocalFile(QFileInfo(filePath).absoluteDir().absolutePath()));
141 break;
142 }
143 default:
144 {
145 const auto path = d.getExistingDirectory();
146 if (path.isEmpty())
147 return;
148 for (const auto &connId : list)
149 {
150 ConnectionId id(connId);
151 auto root = RouteManager->GenerateFinalConfig({ id, currentGroupId });
152 //
153 // Apply export filter
154 exportConnectionFilter(root);
155 //
156 const auto fileName = RemoveInvalidFileName(GetDisplayName(id)) + ".json";
157 StringToFile(JsonToString(root), path + "/" + fileName);
158 }
159 QDesktopServices::openUrl(QUrl::fromLocalFile(path));
160 break;
161 }
162 }

190void GroupManager::reloadConnectionsList(const GroupId &group)
191{
192 if (group == NullGroupId)
193 return;
194 connectionsTable->clearContents();
195 connectionsTable->model()->removeRows(0, connectionsTable->rowCount());
196 const auto &connections = ConnectionManager->GetConnections(group);
197 for (auto i = 0; i < connections.count(); i++)
198 {
199 const auto &conn = connections.at(i);
200 connectionsTable->insertRow(i);
201 //
202 auto displayNameItem = new QTableWidgetItem(GetDisplayName(conn));
203 displayNameItem->setData(Qt::UserRole, conn.toString());
204 auto typeItem = new QTableWidgetItem(GetConnectionProtocolString(conn));
205 //
206 const auto [type, host, port] = GetConnectionInfo(conn);
207 auto hostPortItem = new QTableWidgetItem(host + ":" + QSTRN(port));
208 //
209 QStringList groupsNamesString;
210 for (const auto &group : ConnectionManager->GetConnectionContainedIn(conn))
211 {
212 groupsNamesString.append(GetDisplayName(group));
213 }
214 auto groupsItem = new QTableWidgetItem(groupsNamesString.join(";"));
215 connectionsTable->setItem(i, 0, displayNameItem);
216 connectionsTable->setItem(i, 1, typeItem);
217 connectionsTable->setItem(i, 2, hostPortItem);
218 connectionsTable->setItem(i, 3, groupsItem);
219 }
220 connectionsTable->resizeColumnsToContents();
221}

292void GroupManager::on_updateButton_clicked()
293{
294 const auto address = subAddrTxt->text().trimmed();
295 if (address.isEmpty())
296 {
297 QvMessageBoxWarn(this, tr("Update Subscription"), tr("The subscription link is empty."));
298 return;
299 }
300 if (!QUrl(address).isValid())
301 {
302 QvMessageBoxWarn(this, tr("Update Subscription"), tr("The subscription link is invalid."));
303 return;
304 }
305 if (QvMessageBoxAsk(this, tr("Update Subscription"), tr("Would you like to update the subscription?")) == Yes)
306 {
307 this->setEnabled(false);
308 qApp->processEvents();
309 ConnectionManager->UpdateSubscription(currentGroupId);
310 this->setEnabled(true);
311 on_groupList_itemClicked(groupList->currentItem());
312 }

356void GroupManager::on_groupList_itemClicked(QListWidgetItem *item)
357{
358 if (item == nullptr)
359 {
360 return;
361 }
362 groupInfoGroupBox->setEnabled(true);
363 currentGroupId = GroupId(item->data(Qt::UserRole).toString());
364 groupNameTxt->setText(GetDisplayName(currentGroupId));
365 //
366 const auto _group = ConnectionManager->GetGroupMetaObject(currentGroupId);
367 groupIsSubscriptionGroup->setChecked(_group.isSubscription);
368 subAddrTxt->setText(_group.subscriptionOption.address);
369 lastUpdatedLabel->setText(timeToString(_group.lastUpdatedDate));
370 createdAtLabel->setText(timeToString(_group.creationDate));
371 updateIntervalSB->setValue(_group.subscriptionOption.updateInterval);
372 {
373 const auto &type = _group.subscriptionOption.type;
374 const auto index = subscriptionTypeCB->findData(type);
375 if (index < 0)
376 QvMessageBoxWarn(this, tr("Unknown Subscription Type"), tr("Unknown subscription type \"%1\", a plugin may be missing.").arg(type));
377 else
378 subscriptionTypeCB->setCurrentIndex(index);
379 }
380 //
381 // Import filters
382 {
383 IncludeRelation->setCurrentIndex(_group.subscriptionOption.IncludeRelation);
384 IncludeKeywords->clear();
385 for (const auto &key : _group.subscriptionOption.IncludeKeywords)
386 {
387 auto str = key.trimmed();
388 if (!str.isEmpty())
389 {
390 IncludeKeywords->appendPlainText(str);
391 }
392 }
393 ExcludeRelation->setCurrentIndex(_group.subscriptionOption.ExcludeRelation);
394 ExcludeKeywords->clear();
395 for (const auto &key : _group.subscriptionOption.ExcludeKeywords)
396 {
397 auto str = key.trimmed();
398 if (!str.isEmpty())
399 {
400 ExcludeKeywords->appendPlainText(str);
401 }
402 }
403 }
404 //
405 // Load DNS / Route config
406 const auto routeId = ConnectionManager->GetGroupRoutingId(currentGroupId);
407 {
408 const auto &[overrideDns, dns, fakedns] = RouteManager->GetDNSSettings(routeId);
409 dnsSettingsWidget->SetDNSObject(dns, fakedns);
410 dnsSettingsGB->setChecked(overrideDns);
411 //
412 const auto &[overrideRoute, route] = RouteManager->GetAdvancedRoutingSettings(routeId);
413 routeSettingsWidget->SetRouteConfig(route);
414 routeSettingsGB->setChecked(overrideRoute);
415 }
416 reloadConnectionsList(currentGroupId);
417}

500void GroupManager::exportConnectionFilter(CONFIGROOT &root)
501{
502 root.remove("api");
503 root.remove("stats");
504 auto inbounds = root["inbounds"].toArray();
505 for (auto iter = inbounds.begin(); iter != inbounds.end();)
506 {
507 auto obj = iter->toObject();
508 if (obj["tag"] == API_TAG_INBOUND)
509 iter = inbounds.erase(iter);
510 iter++;
511 }
512 root["inbounds"] = inbounds;
513}
