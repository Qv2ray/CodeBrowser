37QStringList Qv2rayApplicationInterface::GetAssetsPaths(const QString &dirName) const
38{
39 // Configuration Path
40 QStringList list;
41
42 if (qEnvironmentVariableIsSet("QV2RAY_RESOURCES_PATH"))
43 list << makeAbs(qEnvironmentVariable("QV2RAY_RESOURCES_PATH") + "/" + dirName);
44
45 // Default behavior on Windows
46 list << makeAbs(QCoreApplication::applicationDirPath() + "/" + dirName);
47 list << makeAbs(QV2RAY_CONFIG_DIR + dirName);
48 list << ":/" + dirName;
49
50 list << QStandardPaths::locateAll(QStandardPaths::AppDataLocation, dirName, QStandardPaths::LocateDirectory);
51 list << QStandardPaths::locateAll(QStandardPaths::AppConfigLocation, dirName, QStandardPaths::LocateDirectory);
52
53#ifdef Q_OS_UNIX
54 if (qEnvironmentVariableIsSet("APPIMAGE"))
55 list << makeAbs(QCoreApplication::applicationDirPath() + "/../share/qv2ray/" + dirName);
56
57 if (qEnvironmentVariableIsSet("SNAP"))
58 list << makeAbs(qEnvironmentVariable("SNAP") + "/usr/share/qv2ray/" + dirName);
59
60 if (qEnvironmentVariableIsSet("XDG_DATA_DIRS"))
61 list << makeAbs(qEnvironmentVariable("XDG_DATA_DIRS") + "/" + dirName);
62
63 list << makeAbs("/usr/local/share/qv2ray/" + dirName);
64 list << makeAbs("/usr/local/lib/qv2ray/" + dirName);
65 list << makeAbs("/usr/share/qv2ray/" + dirName);
66 list << makeAbs("/usr/lib/qv2ray/" + dirName);
67 list << makeAbs("/lib/qv2ray/" + dirName);
68#endif
69
70#ifdef Q_OS_MAC
71 // macOS platform directories.
72 list << QDir(QCoreApplication::applicationDirPath() + "/../Resources/" + dirName).absolutePath();
73#endif
74
75 list.removeDuplicates();
76 return list;
77}
