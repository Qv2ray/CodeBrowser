55 template<typename enumKey, typename TValue, typename = typename std::enable_if_t<std::is_enum_v<enumKey>>>
56 struct QvEnumMap : QMap<enumKey, TValue>
57 {
58 // WARN: Changing this will break all existing JSON.
59 static constexpr auto ENUM_JSON_KEY_PREFIX = "$";
60 void loadJson(const QJsonValue &json_object)
61 {
62 QMap<QString, TValue> data;
63 JsonStructHelper::Deserialize(data, json_object);
64 this->clear();
65 for (QString k_str : data.keys())
66 {
67 enumKey k = static_cast<enumKey>(k_str.remove(ENUM_JSON_KEY_PREFIX).toInt());
68 this->insert(k, data[ENUM_JSON_KEY_PREFIX + k_str]);
69 }
70 }
71 [[nodiscard]] static auto fromJson(const QJsonValue &json)
72 {
73 QvEnumMap t;
74 t.loadJson(json);
75 return t;
76 }
77 [[nodiscard]] const QJsonObject toJson() const
78 {
79 QMap<QString, TValue> data;
80 for (const auto &k : this->keys())
81 {
82 data[ENUM_JSON_KEY_PREFIX + QString::number(k)] = this->value(k);
83 }
84 return JsonStructHelper::Serialize(data).toObject();
85 }
86 };
