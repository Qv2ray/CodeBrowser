60// This class is for internal use by the protocol buffer library and by
61// protocol-complier-generated message classes. It must not be called
62// directly by clients.
63//
64// This class contains code for implementing the binary protocol buffer
65// wire format via reflection. The WireFormatLite class implements the
66// non-reflection based routines.
67//
68// This class is really a namespace that contains only static methods
69class LIBPROTOBUF_EXPORT WireFormat {
70 public:
71
72 // Given a field return its WireType
73 static inline WireFormatLite::WireType WireTypeForField(
74 const FieldDescriptor* field);
75
76 // Given a FieldDescriptor::Type return its WireType
77 static inline WireFormatLite::WireType WireTypeForFieldType(
78 FieldDescriptor::Type type);
79
80 // Compute the byte size of a tag. For groups, this includes both the start
81 // and end tags.
82 static inline size_t TagSize(int field_number, FieldDescriptor::Type type);
83
84 // These procedures can be used to implement the methods of Message which
85 // handle parsing and serialization of the protocol buffer wire format
86 // using only the Reflection interface. When you ask the protocol
87 // compiler to optimize for code size rather than speed, it will implement
88 // those methods in terms of these procedures. Of course, these are much
89 // slower than the specialized implementations which the protocol compiler
90 // generates when told to optimize for speed.
91
92 // Read a message in protocol buffer wire format.
93 //
94 // This procedure reads either to the end of the input stream or through
95 // a WIRETYPE_END_GROUP tag ending the message, whichever comes first.
96 // It returns false if the input is invalid.
97 //
98 // Required fields are NOT checked by this method. You must call
99 // IsInitialized() on the resulting message yourself.
100 static bool ParseAndMergePartial(io::CodedInputStream* input,
101 Message* message);
102
103 // Serialize a message in protocol buffer wire format.
104 //
105 // Any embedded messages within the message must have their correct sizes
106 // cached. However, the top-level message need not; its size is passed as
107 // a parameter to this procedure.
108 //
109 // These return false iff the underlying stream returns a write error.
110 static void SerializeWithCachedSizes(
111 const Message& message,
112 int size, io::CodedOutputStream* output);
113
114 // Implements Message::ByteSize() via reflection. WARNING: The result
115 // of this method is *not* cached anywhere. However, all embedded messages
116 // will have their ByteSize() methods called, so their sizes will be cached.
117 // Therefore, calling this method is sufficient to allow you to call
118 // WireFormat::SerializeWithCachedSizes() on the same object.
119 static size_t ByteSize(const Message& message);
120
121 // -----------------------------------------------------------------
122 // Helpers for dealing with unknown fields
123
124 // Skips a field value of the given WireType. The input should start
125 // positioned immediately after the tag. If unknown_fields is non-NULL,
126 // the contents of the field will be added to it.
127 static bool SkipField(io::CodedInputStream* input, uint32 tag,
128 UnknownFieldSet* unknown_fields);
129
130 // Reads and ignores a message from the input. If unknown_fields is non-NULL,
131 // the contents will be added to it.
132 static bool SkipMessage(io::CodedInputStream* input,
133 UnknownFieldSet* unknown_fields);
134
135 // Read a packed enum field. If the is_valid function is not NULL, values for
136 // which is_valid(value) returns false are appended to unknown_fields_stream.
137 static bool ReadPackedEnumPreserveUnknowns(io::CodedInputStream* input,
138 uint32 field_number,
139 bool (*is_valid)(int),
140 UnknownFieldSet* unknown_fields,
141 RepeatedField<int>* values);
142
143 // Write the contents of an UnknownFieldSet to the output.
144 static void SerializeUnknownFields(const UnknownFieldSet& unknown_fields,
145 io::CodedOutputStream* output);
146 // Same as above, except writing directly to the provided buffer.
147 // Requires that the buffer have sufficient capacity for
148 // ComputeUnknownFieldsSize(unknown_fields).
149 //
150 // Returns a pointer past the last written byte.
151 static uint8* SerializeUnknownFieldsToArray(
152 const UnknownFieldSet& unknown_fields,
153 uint8* target);
154
155 // Same thing except for messages that have the message_set_wire_format
156 // option.
157 static void SerializeUnknownMessageSetItems(
158 const UnknownFieldSet& unknown_fields,
159 io::CodedOutputStream* output);
160 // Same as above, except writing directly to the provided buffer.
161 // Requires that the buffer have sufficient capacity for
162 // ComputeUnknownMessageSetItemsSize(unknown_fields).
163 //
164 // Returns a pointer past the last written byte.
165 static uint8* SerializeUnknownMessageSetItemsToArray(
166 const UnknownFieldSet& unknown_fields,
167 uint8* target);
168
169 // Compute the size of the UnknownFieldSet on the wire.
170 static size_t ComputeUnknownFieldsSize(const UnknownFieldSet& unknown_fields);
171
172 // Same thing except for messages that have the message_set_wire_format
173 // option.
174 static size_t ComputeUnknownMessageSetItemsSize(
175 const UnknownFieldSet& unknown_fields);
176
177
178 // Helper functions for encoding and decoding tags. (Inlined below and in
179 // _inl.h)
180 //
181 // This is different from MakeTag(field->number(), field->type()) in the case
182 // of packed repeated fields.
183 static uint32 MakeTag(const FieldDescriptor* field);
184
185 // Parse a single field. The input should start out positioned immediately
186 // after the tag.
187 static bool ParseAndMergeField(
188 uint32 tag,
189 const FieldDescriptor* field, // May be NULL for unknown
190 Message* message,
191 io::CodedInputStream* input);
192
193 // Serialize a single field.
194 static void SerializeFieldWithCachedSizes(
195 const FieldDescriptor* field, // Cannot be NULL
196 const Message& message,
197 io::CodedOutputStream* output);
198
199 // Compute size of a single field. If the field is a message type, this
200 // will call ByteSize() for the embedded message, insuring that it caches
201 // its size.
202 static size_t FieldByteSize(
203 const FieldDescriptor* field, // Cannot be NULL
204 const Message& message);
205
206 // Parse/serialize a MessageSet::Item group. Used with messages that use
207 // opion message_set_wire_format = true.
208 static bool ParseAndMergeMessageSetItem(
209 io::CodedInputStream* input,
210 Message* message);
211 static void SerializeMessageSetItemWithCachedSizes(
212 const FieldDescriptor* field,
213 const Message& message,
214 io::CodedOutputStream* output);
215 static size_t MessageSetItemByteSize(
216 const FieldDescriptor* field,
217 const Message& message);
218
219 // Computes the byte size of a field, excluding tags. For packed fields, it
220 // only includes the size of the raw data, and not the size of the total
221 // length, but for other length-delimited types, the size of the length is
222 // included.
223 static size_t FieldDataOnlyByteSize(
224 const FieldDescriptor* field, // Cannot be NULL
225 const Message& message);
226
227 enum Operation {
228 PARSE = 0,
229 SERIALIZE = 1,
230 };
231
232 // Verifies that a string field is valid UTF8, logging an error if not.
233 // This function will not be called by newly generated protobuf code
234 // but remains present to support existing code.
235 static void VerifyUTF8String(const char* data, int size, Operation op);
236 // The NamedField variant takes a field name in order to produce an
237 // informative error message if verification fails.
238 static void VerifyUTF8StringNamedField(const char* data,
239 int size,
240 Operation op,
241 const char* field_name);
242
243 private:
244 // Skip a MessageSet field.
245 static bool SkipMessageSetField(io::CodedInputStream* input,
246 uint32 field_number,
247 UnknownFieldSet* unknown_fields);
248
249 // Parse a MessageSet field.
250 static bool ParseAndMergeMessageSetField(uint32 field_number,
251 const FieldDescriptor* field,
252 Message* message,
253 io::CodedInputStream* input);
254
255 GOOGLE_DISALLOW_EVIL_CONSTRUCTORS(WireFormat);
256};
257
258// Subclass of FieldSkipper which saves skipped fields to an UnknownFieldSet.
259class LIBPROTOBUF_EXPORT UnknownFieldSetFieldSkipper : public FieldSkipper {
260 public:
261 UnknownFieldSetFieldSkipper(UnknownFieldSet* unknown_fields)
262 : unknown_fields_(unknown_fields) {}
263 virtual ~UnknownFieldSetFieldSkipper() {}
264
265 // implements FieldSkipper -----------------------------------------
266 virtual bool SkipField(io::CodedInputStream* input, uint32 tag);
267 virtual bool SkipMessage(io::CodedInputStream* input);
268 virtual void SkipUnknownEnum(int field_number, int value);
269
270 protected:
271 UnknownFieldSet* unknown_fields_;
272};

274// inline methods ====================================================
275
276inline WireFormatLite::WireType WireFormat::WireTypeForField(
277 const FieldDescriptor* field) {
278 if (field->is_packed()) {
279 return WireFormatLite::WIRETYPE_LENGTH_DELIMITED;
280 } else {
281 return WireTypeForFieldType(field->type());
282 }
283}
284
285inline WireFormatLite::WireType WireFormat::WireTypeForFieldType(
286 FieldDescriptor::Type type) {
287 // Some compilers don't like enum -> enum casts, so we implicit_cast to
288 // int first.
289 return WireFormatLite::WireTypeForFieldType(
290 static_cast<WireFormatLite::FieldType>(
291 implicit_cast<int>(type)));
292}
293
294inline uint32 WireFormat::MakeTag(const FieldDescriptor* field) {
295 return WireFormatLite::MakeTag(field->number(), WireTypeForField(field));
296}
297
298inline size_t WireFormat::TagSize(int field_number,
299 FieldDescriptor::Type type) {
300 // Some compilers don't like enum -> enum casts, so we implicit_cast to
301 // int first.
302 return WireFormatLite::TagSize(field_number,
303 static_cast<WireFormatLite::FieldType>(
304 implicit_cast<int>(type)));
305}
306
307inline void WireFormat::VerifyUTF8String(const char* data, int size,
308 WireFormat::Operation op) {
309#ifdef GOOGLE_PROTOBUF_UTF8_VALIDATION_ENABLED
310 WireFormatLite::VerifyUtf8String(
311 data, size, static_cast<WireFormatLite::Operation>(op), NULL);
312#else
313 // Avoid the compiler warning about unused variables.
314 (void)data; (void)size; (void)op;
315#endif
316}
317
318inline void WireFormat::VerifyUTF8StringNamedField(
319 const char* data, int size, WireFormat::Operation op,
320 const char* field_name) {
321#ifdef GOOGLE_PROTOBUF_UTF8_VALIDATION_ENABLED
322 WireFormatLite::VerifyUtf8String(
323 data, size, static_cast<WireFormatLite::Operation>(op), field_name);
324#else
325 // Avoid the compiler warning about unused variables.
326 (void)data; (void)size; (void)op; (void)field_name;
327#endif
328}
