91/** A single argument... each argument has a key and a value
92
93 A note on naming keys:
94 Keys are namespaced into groups, usually grouped by library, and are
95 keys for module XYZ are named XYZ.key1, XYZ.key2, etc. Module names must
96 be restricted to the regex [A-Za-z][_A-Za-z0-9]{,15}.
97 Key names must be restricted to the regex [A-Za-z][_A-Za-z0-9]{,47}.
98
99 GRPC core library keys are prefixed by grpc.
100
101 Library authors are strongly encouraged to \#define symbolic constants for
102 their keys so that it's possible to change them in the future. */
103typedef struct {
104 grpc_arg_type type;
105 char* key;
106 union grpc_arg_value {
107 char* string;
108 int integer;
109 struct grpc_arg_pointer {
110 void* p;
111 const grpc_arg_pointer_vtable* vtable;
112 } pointer;
113 } value;
114} grpc_arg;

116/** An array of arguments that can be passed around.
117
118 Used to set optional channel-level configuration.
119 These configuration options are modelled as key-value pairs as defined
120 by grpc_arg; keys are strings to allow easy backwards-compatible extension
121 by arbitrary parties. All evaluation is performed at channel creation
122 time (i.e. the keys and values in this structure need only live through the
123 creation invocation).
124
125 However, if one of the args has grpc_arg_type==GRPC_ARG_POINTER, then the
126 grpc_arg_pointer_vtable must live until the channel args are done being
127 used by core (i.e. when the object for use with which they were passed
128 is destroyed).
129
130 See the description of the \ref grpc_arg_keys "available args" for more
131 details. */
132typedef struct {
133 size_t num_args;
134 grpc_arg* args;
135} grpc_channel_args;

352/** Result of a grpc call. If the caller satisfies the prerequisites of a
353 particular operation, the grpc_call_error returned will be GRPC_CALL_OK.
354 Receiving any other value listed here is an indication of a bug in the
355 caller. */
356typedef enum grpc_call_error {
357 /** everything went ok */
358 GRPC_CALL_OK = 0,
359 /** something failed, we don't know what */
360 GRPC_CALL_ERROR,
361 /** this method is not available on the server */
362 GRPC_CALL_ERROR_NOT_ON_SERVER,
363 /** this method is not available on the client */
364 GRPC_CALL_ERROR_NOT_ON_CLIENT,
365 /** this method must be called before server_accept */
366 GRPC_CALL_ERROR_ALREADY_ACCEPTED,
367 /** this method must be called before invoke */
368 GRPC_CALL_ERROR_ALREADY_INVOKED,
369 /** this method must be called after invoke */
370 GRPC_CALL_ERROR_NOT_INVOKED,
371 /** this call is already finished
372 (writes_done or write_status has already been called) */
373 GRPC_CALL_ERROR_ALREADY_FINISHED,
374 /** there is already an outstanding read/write operation on the call */
375 GRPC_CALL_ERROR_TOO_MANY_OPERATIONS,
376 /** the flags value was illegal for this call */
377 GRPC_CALL_ERROR_INVALID_FLAGS,
378 /** invalid metadata was passed to this call */
379 GRPC_CALL_ERROR_INVALID_METADATA,
380 /** invalid message was passed to this call */
381 GRPC_CALL_ERROR_INVALID_MESSAGE,
382 /** completion queue for notification has not been registered
383 * with the server */
384 GRPC_CALL_ERROR_NOT_SERVER_COMPLETION_QUEUE,
385 /** this batch of operations leads to more operations than allowed */
386 GRPC_CALL_ERROR_BATCH_TOO_BIG,
387 /** payload type requested is not the type registered */
388 GRPC_CALL_ERROR_PAYLOAD_TYPE_MISMATCH,
389 /** completion queue has been shutdown */
390 GRPC_CALL_ERROR_COMPLETION_QUEUE_SHUTDOWN
391} grpc_call_error;

399/** Hint that the write may be buffered and need not go out on the wire

407#define GRPC_WRITE_THROUGH (0x00000004u)

423#define GRPC_INITIAL_METADATA_CORKED (0x00000100u)

425/** Mask of all valid flags */

491typedef enum {
492 /** Send initial metadata: one and only one instance MUST be sent for each
493 call, unless the call was cancelled - in which case this can be skipped.
494 This op completes after all bytes of metadata have been accepted by
495 outgoing flow control. */
496 GRPC_OP_SEND_INITIAL_METADATA = 0,
497 /** Send a message: 0 or more of these operations can occur for each call.
498 This op completes after all bytes for the message have been accepted by
499 outgoing flow control. */
500 GRPC_OP_SEND_MESSAGE,
501 /** Send a close from the client: one and only one instance MUST be sent from
502 the client, unless the call was cancelled - in which case this can be
503 skipped. This op completes after all bytes for the call
504 (including the close) have passed outgoing flow control. */
505 GRPC_OP_SEND_CLOSE_FROM_CLIENT,
506 /** Send status from the server: one and only one instance MUST be sent from
507 the server unless the call was cancelled - in which case this can be
508 skipped. This op completes after all bytes for the call
509 (including the status) have passed outgoing flow control. */
510 GRPC_OP_SEND_STATUS_FROM_SERVER,
511 /** Receive initial metadata: one and only one MUST be made on the client,
512 must not be made on the server.
513 This op completes after all initial metadata has been read from the
514 peer. */
515 GRPC_OP_RECV_INITIAL_METADATA,
516 /** Receive a message: 0 or more of these operations can occur for each call.
