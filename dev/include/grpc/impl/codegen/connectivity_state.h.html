26/** Connectivity state of a channel. */
27typedef enum {
28 /** channel is idle */
29 GRPC_CHANNEL_IDLE,
30 /** channel is connecting */
31 GRPC_CHANNEL_CONNECTING,
32 /** channel is ready for work */
33 GRPC_CHANNEL_READY,
34 /** channel has seen a failure but expects to recover */
35 GRPC_CHANNEL_TRANSIENT_FAILURE,
36 /** channel has seen a failure that it cannot recover from */
37 GRPC_CHANNEL_SHUTDOWN
38} grpc_connectivity_state;
